 20 /**
 21  * This object implements nsIStreamListener interface and is intended to monitor all network
 22  * channels (nsIHttpChannel). A new instance of this object is created and registered an HTTP
 23  * channel. See Firebug.TabCacheModel.onExamineResponse method.
 24  */
 25 
 26 function ChannelListener()
 27 {
 28     this.window = null;
 29     this.request = null;
 30 
 31     this.endOfLine = false;
 32     this.ignore = false;
 33 
 34     // The original channel listener (see nsITraceableChannel for more).
 35     this.listener = null;
 36 
 37     // The proxy listener is used to send events to possible listeners (e.g. net panel)
 38     // and properly pass the request object through nsIStreamListner to chrome-window space.
 39     this.proxyListener = null;
 40 
 41     // The response will be written into the outputStream of this pipe.
 42     // Both ends of the pipe must be blocking. Initialized in TabCacheModel.registerStreamListener.
 43     this.sink = null;
 44 
 45     if (FBTrace.DBG_FAKE)  // cause the detrace to remove this statement and check for cached tracer
 46     {
 47         Components.utils.import("resource://firebug/firebug-trace-service.js");
 48         FBTrace = traceConsoleService.getTracer("extensions.firebug");
 49     }
 50 }
 51 
 52 ChannelListener.prototype =
 53 {
 54 
 55     setAsyncListener: function(request, stream, listener)
 56     {
 57         try
 58         {
 59             // xxxHonza: is there any other way how to find out the stream is closed?
 60             // Throws NS_BASE_STREAM_CLOSED if the stream is closed normally or at end-of-file.
 61             var available = stream.available();
 62         }
 63         catch (err)
 64         {
 65             if (err.name == "NS_BASE_STREAM_CLOSED")
 66             {
 67                 if (FBTrace.DBG_CACHE)
 68                     FBTrace.sysout("tabCache.ChannelListener.setAsyncListener; " +
 69                         "Don't set, the stream is closed.");
 70                 return;
 71             }
 72 
 73             if (FBTrace.DBG_CACHE || FBTrace.DBG_ERRORS)
 74                 FBTrace.sysout("tabCache.ChannelListener.setAsyncListener; EXCEPTION " +
 75                     safeGetName(request), err);
 76             return;
 77         }
 78 
 79         try
 80         {
 81             // Asynchronously wait for the stream to be readable or closed.
 82             stream.asyncWait(listener, 0, 0, null);
 83         }
 84         catch (err)
 85         {
 86             if (FBTrace.DBG_CACHE || FBTrace.DBG_ERRORS)
 87                 FBTrace.sysout("tabCache.ChannelListener.setAsyncListener; EXCEPTION " +
 88                     safeGetName(request), err);
 89         }
 90     },
 91 
 92     onCollectData: function(request, context, inputStream, offset, count)
 93     {
 94         if (FBTrace.DBG_CACHE && this.ignore)
 95             FBTrace.sysout("tabCache.ChannelListener.onCollectData; IGNORE stopping further onCollectData");
 96 
 97         try
 98         {
 99             if (this.sink)
100             {
101                 var bis = CCIN("@mozilla.org/binaryinputstream;1", "nsIBinaryInputStream");
102                 bis.setInputStream(inputStream);
103                 var data = bis.readBytes(count);
104 
105                 // Data from the pipe has been consumed (to avoid mem leaks) so, we can end now.
106                 if (this.ignore)
107                     return;
108             }
109             else
110             {
111                 // In this case, we don't need to read the data.
112                 if (this.ignore)
113                     return;
114 
115                 var binaryInputStream = CCIN("@mozilla.org/binaryinputstream;1", "nsIBinaryInputStream");
116                 var storageStream = CCIN("@mozilla.org/storagestream;1", "nsIStorageStream");
117                 var binaryOutputStream = CCIN("@mozilla.org/binaryoutputstream;1", "nsIBinaryOutputStream");
118 
119                 binaryInputStream.setInputStream(inputStream);
120                 storageStream.init(8192, count, null);
121                 binaryOutputStream.setOutputStream(storageStream.getOutputStream(0));
122 
123                 var data = binaryInputStream.readBytes(count);
124                 binaryOutputStream.writeBytes(data, count);
125             }
126 
127             // Avoid creating additional empty line if response comes in more pieces
128             // and the split is made just between "\r" and "\n" (Win line-end).
129             // So, if the response starts with "\n" while the previous part ended with "\r",
130             // remove the first character.
131             if (this.endOfLine && data.length && data[0] == "\n")
132                 data = data.substring(1);
133 
134             if (data.length)
135                 this.endOfLine = data[data.length-1] == "\r";
136 
137             // Store received data into the cache as they come. If the method returns
138             // false, the rest of the response is ignored (not cached). This is used
139             // to limit size of a cached response.
140             if (!context.sourceCache.storePartialResponse(request, data, this.window))
141             {
142                 this.ignore = true;
143                 if (FBTrace.DBG_CACHE)
144                     FBTrace.sysout("tabCache.ChannelListener.onCollectData IGNORE SET because of storePartialResponse");
145             }
146 
147             // Let other listeners use the stream.
148             if (storageStream)
149                 return storageStream.newInputStream(0);
150         }
151         catch (err)
152         {
153             if (FBTrace.DBG_CACHE || FBTrace.DBG_ERRORS)
154                 FBTrace.sysout("tabCache.ChannelListener.onCollectData EXCEPTION\n", err);
155         }
156 
157         return null;
158     },
159 
160     /* nsIStreamListener */
161     onDataAvailable: function(request, requestContext, inputStream, offset, count)
162     {
163         try
164         {
165             // Use wrappedJSObject to bypass IDL definition that doesn't return any value.
166             var newStream = this.proxyListener.wrappedJSObject.onDataAvailable(request, requestContext,
167                 inputStream, offset, count);
168 
169             if (newStream)
170                 inputStream = newStream;
171 
172             var context = this.getContext(this.window);
173             if (context)
174             {
175                 newStream = this.onCollectData(request, context, inputStream, offset, count);
176                 if (newStream)
177                     inputStream = newStream;
178             }
179         }
180         catch (err)
181         {
182             if (FBTrace.DBG_CACHE || FBTrace.DBG_ERRORS)
183                 FBTrace.sysout("tabCache.ChannelListener.onDataAvailable onCollectData FAILS " +
184                     "(" + offset + ", " + count + ") EXCEPTION: " +
185                     safeGetName(request), err);
186         }
187 
188         if (this.listener)
189         {
190             try  // https://bugzilla.mozilla.org/show_bug.cgi?id=492534
191             {
192                 this.listener.onDataAvailable(request, requestContext, inputStream, offset, count);
193             }
194             catch(exc)
195             {
196                 if (FBTrace.DBG_CACHE)
197                     FBTrace.sysout("tabCache.ChannelListener.onDataAvailable cancelling request at " +
198                     "(" + offset + ", " + count + ") EXCEPTION: " +
199                     safeGetName(request), exc);
200 
201                 request.cancel(exc.result);
202             }
203         }
204     },
205 
206     onStartRequest: function(request, requestContext)
207     {
208         try
209         {
210             this.request = request.QueryInterface(Ci.nsIHttpChannel);
211 
212             if (FBTrace.DBG_CACHE)
213                 FBTrace.sysout("tabCache.ChannelListener.onStartRequest; " +
214                     this.request.contentType + ", " + safeGetName(this.request));
215 
216             // Don't register listener twice (redirects, see also bug529536).
217             // xxxHonza: I don't know any way how to find out that a listener
218             // has been already registered for the channel. So, use the redirection limit
219             // to see that the channel has been redirected and so, listener is there.
220             if (request.redirectionLimit < redirectionLimit)
221             {
222                 if (FBTrace.DBG_CACHE)
223                     FBTrace.sysout("tabCache.ChannelListener.onStartRequest; redirected request " +
224                         request.redirectionLimit + " (max=" + redirectionLimit + ")");
225                 return;
226             }
227 
228             // Pass to the proxy only if the associated context exists (the window is not unloaded)
229             var context = this.getContext(this.window);
230             if (context)
231             {
232                 // Due to #489317, the check whether this response should be cached
233                 // must be done here (the content type is not valid before calling
234                 // onStartRequest). Let's ignore the response if it should not be cached.
235                 this.ignore = !this.shouldCacheRequest(request);
236 
237                 // Notify proxy listener.
238                 this.proxyListener.onStartRequest(request, requestContext);
239 
240                 // Listen for incoming data.
241                 if (FBTrace.DBG_CACHE && !this.sink)
242                     FBTrace.sysout("tabCache.ChannelListener.onStartRequest NO SINK stopping setAsyncListener");
243                 if (FBTrace.DBG_CACHE && this.ignore && this.sink)
244                     FBTrace.sysout("tabCache.ChannelListener.onStartRequest IGNORE(shouldCacheRequest) stopping setAsyncListener");
245 
246                 // Even if the response is marked as ignored we need to read the sink
247                 // to avoid mem leaks.
248                 if (this.sink)
249                     this.setAsyncListener(request, this.sink.inputStream, this);
250             }
251         }
252         catch (err)
253         {
254             if (FBTrace.DBG_CACHE || FBTrace.DBG_ERRORS)
255                 FBTrace.sysout("tabCache.ChannelListener.onStartRequest EXCEPTION\n", err);
256         }
257 
258         if (this.listener)
259         {
260             try  // https://bugzilla.mozilla.org/show_bug.cgi?id=492534
261             {
262                 this.listener.onStartRequest(request, requestContext);
263             }
264             catch(exc)
265             {
266                 if (FBTrace.DBG_CACHE)
267                     FBTrace.sysout("tabCache.ChannelListener.onStartRequest cancelling request " +
268                     "EXCEPTION: " + safeGetName(request), exc);
269 
270                 request.cancel(exc.result);
271             }
272         }
273     },
274 
275     onStopRequest: function(request, requestContext, statusCode)
276     {
277         try
278         {
279             if (FBTrace.DBG_CACHE)
280                 FBTrace.sysout("tabCache.ChannelListener.onStopRequest; " +
281                     request.contentType + ", " + safeGetName(request));
282 
283             var context = this.getContext(this.window);
284             if (context)
285                 this.proxyListener.onStopRequest(request, requestContext, statusCode);
286         }
287         catch (err)
288         {
289             if (FBTrace.DBG_CACHE || FBTrace.DBG_ERRORS)
290                 FBTrace.sysout("tabCache.ChannelListener.onStopRequest EXCEPTION\n", err);
291         }
292 
293         // The request body has been downloaded. Remove the listener (the last parameter
294         // is null) since it's not needed now.
295         if (this.sink)
296             this.setAsyncListener(request, this.sink.inputStream, null);
297 
298         if (this.listener)
299             this.listener.onStopRequest(request, requestContext, statusCode);
300     },
301 
302     /* nsITraceableChannel */
303     setNewListener: function(listener)
304     {
305         this.proxyListener = listener;
306         return null;
307     },
308 
309     /* nsIInputStreamCallback */
310     onInputStreamReady : function(stream)
311     {
312         try
313         {
314             if (FBTrace.DBG_CACHE)
315                 FBTrace.sysout("tabCache.ChannelListener.onInputStreamReady " +
316                     safeGetName(this.request));
317 
318             if (stream instanceof Ci.nsIAsyncInputStream)
319             {
320                 try
321                 {
322                     var available = stream.available();
323                     this.onDataAvailable(this.request, null, stream, 0, available);
324                 }
325                 catch (err)
326                 {
327                     // stream.available throws an exception if the stream is closed,
328                     // which is ok, since this callback can be called even in this
329                     // situations.
330                     if (FBTrace.DBG_CACHE)
331                         FBTrace.sysout("tabCache.ChannelListener.onInputStreamReady EXCEPTION calling onDataAvailable:  " +
332                             safeGetName(this.request), err);
333                 }
334 
335                 // Listen for further incoming data.
336                 if (FBTrace.DBG_CACHE && this.ignore)
337                     FBTrace.sysout("tabCache.ChannelListener.onInputStreamReady IGNORE stopping setAsyncListener");
338 
339                 this.setAsyncListener(this.request, stream, this);
340             }
341             else
342             {
343                 if (FBTrace.DBG_CACHE)
344                     FBTrace.sysout("tabCache.ChannelListener.onInputStreamReady NOT a nsIAsyncInputStream",stream);
345             }
346         }
347         catch (err)
348         {
349             if (FBTrace.DBG_CACHE || FBTrace.DBG_ERRORS)
350                 FBTrace.sysout("tabCache.ChannelListener.onInputStreamReady EXCEPTION " +
351                     safeGetName(this.request), err);
352         }
353     },
354 
355     /* nsISupports */
356     QueryInterface: function(iid)
357     {
358         if (iid.equals(Ci.nsIStreamListener) ||
359             iid.equals(Ci.nsIInputStreamCallback) ||
360             iid.equals(Ci.nsISupportsWeakReference) ||
361             iid.equals(Ci.nsITraceableChannel) ||
362             iid.equals(Ci.nsISupports))
363         {
364             return this;
365         }
366 
367         throw Components.results.NS_NOINTERFACE;
368     },
369 
370     getContext: function(win)
371     {
372         // This must be overridden in tabCache. This scope doesn't have an
373         // access to TabWatcher and its getContextByWindow method.
374         return null;
375     }

380 function safeGetName(request)
381 {
382     try
383     {
384         return request.name;
385     }
386     catch (exc)
387     {
388         return null;
389     }
