  7 const DBG_ = "DBG_";

 11 const Cr = Components.results;

 13 var EXPORTED_SYMBOLS = ["traceConsoleService"];

 15 const PrefService = Cc["@mozilla.org/preferences-service;1"];
 16 const prefs = PrefService.getService(Ci.nsIPrefBranch2);
 17 const prefService = PrefService.getService(Ci.nsIPrefService);

 22 // ************************************************************************************************
 23 // Service implementation
 24 
 25 
 26 var toOSConsole = false;
 27 
 28 var traceConsoleService =
 29 {
 30     initialize: function() {
 31         this.observers = [];
 32         this.optionMaps = {};
 33 
 34         // Listen for preferences changes. Trace Options can be changed at run time.
 35         prefs.addObserver("extensions", this, false);
 36 
 37         return this;
 38     },
 39 
 40     osOut: function(str)
 41     {
 42         if (!this.outChannel)
 43         {
 44             try
 45             {
 46                 var appShellService = Components.classes["@mozilla.org/appshell/appShellService;1"].
 47                     getService(Components.interfaces.nsIAppShellService);
 48                 this.hiddenWindow = appShellService.hiddenDOMWindow;
 49                 this.outChannel = "hidden";
 50             }
 51             catch(exc)
 52             {
 53                 var consoleService = Cc["@mozilla.org/consoleservice;1"].getService(Ci.nsIConsoleService);
 54                 this.outChannel = "service"
 55                 this.outChannel("Using consoleService because nsIAppShellService.hiddenDOMWindow not available "+exc);
 56             }
 57         }
 58         if (this.outChannel === "hidden")  // apparently can't call via JS function
 59             this.hiddenWindow.dump(str);
 60         else
 61             consoleService.logStringMessage(str);
 62     },
 63 
 64     getTracer: function(prefDomain)
 65     {
 66         if (this.getPref("extensions.firebug-tracing-service.DBG_toOSConsole"))
 67         {
 68              toOSConsole = true;  // also need browser.dom.window.dump.enabled true
 69              traceConsoleService.osOut("traceConsoleService.getTracer, prefDomain: "+prefDomain+"\n");
 70         }
 71 
 72         if (!this.optionMaps[prefDomain])
 73             this.optionMaps[prefDomain] = this.createManagedOptionMap(prefDomain);
 74 
 75         return this.optionMaps[prefDomain];
 76     },
 77 
 78     createManagedOptionMap: function(prefDomain)
 79     {
 80         var optionMap = new TraceBase(prefDomain);
 81 
 82         var branch = prefService.getBranch ( prefDomain );
 83         var arrayDesc = {};
 84         var children = branch.getChildList("", arrayDesc);
 85         for (var i = 0; i < children.length; i++)
 86         {
 87             var p = children[i];
 88             var m = p.indexOf("DBG_");
 89             if (m != -1)
 90             {
 91                 var optionName = p.substr(1); // drop leading .
 92                 optionMap[optionName] = this.getPref(prefDomain+p);
 93                 if (toOSConsole)
 94                     this.osOut("traceConsoleService.createManagedOptionMap "+optionName+"="+optionMap[optionName]+"\n");
 95             }
 96         }
 97 
 98         return optionMap;
 99     },
100 
101     /* nsIObserve */
102     observe: function(subject, topic, data)
103     {
104         if (data.substr(0,EXTENSIONS.length) == EXTENSIONS)
105         {
106             for (var prefDomain in traceConsoleService.optionMaps)
107             {
108                 if (data.substr(0, prefDomain.length) == prefDomain)
109                 {
110                     var optionName = data.substr(prefDomain.length+1); // skip dot
111                     if (optionName.substr(0, DBG_.length) == DBG_)
112                         traceConsoleService.optionMaps[prefDomain][optionName] = this.getPref(data);
113                     if (toOSConsole)
114                         traceConsoleService.osOut("traceConsoleService.observe, prefDomain: "+prefDomain+" optionName "+optionName+"\n");
115                 }
116             }
117         }
118     },
119 
120     getPref: function(prefName)
121     {
122         var type = prefs.getPrefType(prefName);
123         if (type == Ci.nsIPrefBranch.PREF_STRING)
124             return prefs.getCharPref(prefName);
125         else if (type == Ci.nsIPrefBranch.PREF_INT)
126             return prefs.getIntPref(prefName);
127         else if (type == Ci.nsIPrefBranch.PREF_BOOL)
128             return prefs.getBoolPref(prefName);
129     },
130 
131     // Prepare trace-object and dispatch to all observers.
132     dispatch: function(messageType, message, obj, scope)
133     {
134         // Translate string object.
135         if (typeof(obj) == "string") {
136             var string = Cc["@mozilla.org/supports-cstring;1"].createInstance(Ci.nsISupportsCString);
137             string.data = obj;
138             obj = string;
139         }
140 
141         // Create wrapper with message type info.
142         var messageInfo = {
143             obj: obj,
144             type: messageType,
145             scope: scope,
146             time: (new Date()).getTime()
147         };
148         if (toOSConsole)
149             traceConsoleService.osOut(messageType+": "+message+"\n");
150         // Pass JS object properly through XPConnect.
151         var wrappedSubject = {wrappedJSObject: messageInfo};
152         traceConsoleService.notifyObservers(wrappedSubject, "firebug-trace-on-message", message);
153     },
154 
155     /* nsIObserverService */
156     addObserver: function(observer, topic, weak)
157     {
158         if (topic != "firebug-trace-on-message")
159             throw Cr.NS_ERROR_INVALID_ARG;
160 
161         if (this.observers.length == 0) // mark where trace begins.
162             lastResort(this.observers, topic, "addObserver");
163 
164         this.observers.push(observer);
165     },
166 
167     removeObserver: function(observer, topic)
168     {
169         if (topic != "firebug-trace-on-message")
170             throw Cr.NS_ERROR_INVALID_ARG;
171 
172         for (var i=0; i < this.observers.length; i++) {
173             if (this.observers[i] == observer) {
174                 this.observers.splice(i, 1);
175                 break;
176             }
177         }
178     },
179 
180     notifyObservers: function(subject, topic, someData)
181     {
182         if (this.observers.length > 0)
183         {
184             for (var i=0; i < this.observers.length; i++)
185             {
186                 try
187                 {
188                     this.observers[i].observe(subject, topic, someData);
189                 }
190                 catch (err)
191                 {
192                     // If it's not possible to distribute the log through registered observers,
193                     // use Firefox ErrorConsole. Ultimately the trace-console listens for it
194                     // too and so, will display that.
195                     var scriptError = Cc["@mozilla.org/scripterror;1"].createInstance(Ci.nsIScriptError);
196                     scriptError.init("[JavaScript Error: Failed to notify firebug-trace observers!] " +
197                         err.toString(), err.sourceName,
198                         err.sourceLine, err.lineNumber, err.columnNumber, err.flags, err.category);
199                     consoleService.logMessage(scriptError);
200                 }
201             }
202         }
203         else
204         {
205             lastResort(this.observers, subject, someData);
206         }
207     },
208 
209     enumerateObservers: function(topic)
210     {
211         return null;
212     },
213 
214     /* nsISupports */
215     QueryInterface: function(iid)
216     {
217         if (iid.equals(Ci.nsISupports) ||
218             iid.equals(Ci.nsIObserverService))
219              return this;
220 
221         throw Cr.NS_ERROR_NO_INTERFACE;
222     }
223 };
224 
225 function lastResort(listeners, subject, someData)
226 {
227     var unwrapped = subject.wrappedJSObject;
228     if (unwrapped)
229         var objPart = unwrapped.obj ? (" obj: "+unwrapped.obj) : "";
230     else
231         var objPart = subject;
232 
233     traceConsoleService.osOut("FTS"+listeners.length+": "+someData+" "+objPart+"\n");
234 }

238 var TraceAPI = {
239     dump: function(messageType, message, obj) {
240         if (this.noTrace)
241             return;
242 
243         this.noTrace = true;
244         try
245         {
246             traceConsoleService.dispatch(messageType, message, obj);
247         }
248         catch(exc)
249         {
250         }
251         finally
252         {
253             this.noTrace = false;
254         }
255     },
256 
257     sysout: function(message, obj) {
258         this.dump("no-message-type", message, obj);
259     },
260 
261     setScope: function(scope)
262     {
263         this.scopeOfFBTrace = scope;
264     },
265 
266     matchesNode: function(node)
267     {
268         return (node.getAttribute('anonid')=="title-box");
269     },
270 
271 };
272 
273 var TraceBase = function(prefDomain) {
274     this.prefDomain = prefDomain;
275 }
276 //Derive all properties from TraceAPI
277 for (var p in TraceAPI)
278     TraceBase.prototype[p] = TraceAPI[p];
279 
280 TraceBase.prototype.sysout = function(message, obj) {
281         if (this.noTrace)
282             return;
283 
284         this.noTrace = true;
285 
286         try
287         {
288             traceConsoleService.dispatch(this.prefDomain, message, obj, this.scopeOfFBTrace);
289         }
290         catch(exc)
291         {
292             if (toOSConsole)
293                 traceConsoleService.osOut("traceConsoleService.dispatch FAILS "+exc+"\n");
294         }
295         finally
296         {
297             this.noTrace = false;
298         }
299 }
