  3 (function() {
  4 
  5 // ************************************************************************************************
  6 // Constants
  7 
  8 const Cc = Components.classes;
  9 const Ci = Components.interfaces;
 10 
 11 var sss = Cc["@mozilla.org/content/style-sheet-service;1"].getService(Ci.nsIStyleSheetService);
 12 var ios = Cc["@mozilla.org/network/io-service;1"].getService(Ci.nsIIOService);
 13 
 14 Components.utils.import("resource://firebug/firebug-trace-service.js");
 15 var FBTrace = traceConsoleService.getTracer("extensions.firebug");
 16 
 17 // ************************************************************************************************
 18 // Overlay
 19 
 20 var uri = ios.newURI("chrome://firebug/skin/about.css", null, null);
 21 
 22 var FirebugAboutOverlay =
 23 {
 24     onLoad: function()
 25     {
 26         try
 27         {
 28             var extension = window.arguments[0];
 29             if (extension == "urn:mozilla:item:firebug@software.joehewitt.com" ||
 30                 extension.id && extension.id == "firebug@software.joehewitt.com")
 31             {
 32                 sss.loadAndRegisterSheet(uri, sss.USER_SHEET);
 33             }
 34 
 35             if (FBTrace.DBG_WINDOWS)
 36                 FBTrace.sysout("FirebugAboutOverlay.onLoad; ", extension);
 37         }
 38         catch (e)
 39         {
 40             if (FBTrace.DBG_ERRORS)
 41                 FBTrace.sysout("FirebugAboutOverlay.onLoad; EXCEPTION " + e, e);
 42         }
 43     },
 44 
 45     onUnload: function()
 46     {
 47         if (sss.sheetRegistered(uri, sss.USER_SHEET))
 48             sss.unregisterSheet(uri, sss.USER_SHEET);
 49 
 50         window.removeEventListener("load", FirebugAboutOverlay.onLoad, false);
 51         window.removeEventListener("unload", FirebugAboutOverlay.onUnload, false);
 52     }
 53 };
 54 
 55 // ************************************************************************************************
 56 // Registration
 57 
 58 window.addEventListener("load", FirebugAboutOverlay.onLoad, false);
 59 window.addEventListener("unload", FirebugAboutOverlay.onUnload, false);
 60 
 61 // ************************************************************************************************
 62 })();
