 34 /**
 35  * Describes a variable visible in a stack frame of an execution context, or a property
 36  * of an object. A variable has a name and a value.
 37  * 
 38  * @constructor
 39  * @param name variable name as a {@link String}
 40  * @type Variable
 41  * @return a new Variable
 42  * @version 1.0
 43  */
 44 function Variable(name) {
 45 	this.name = name;
 46 }
 47 
 48 /**
 49  * Returns the name of this variable as a {@link String}.
 50  * <p>
 51  * This function does not require communication with
 52  * the browser.
 53  * </p>
 54  * @function
 55  * @returns the name of this variable as a {@link String}
 56  */
 57 Variable.prototype.getName = function() {
 58 	return this.name;
 59 };
