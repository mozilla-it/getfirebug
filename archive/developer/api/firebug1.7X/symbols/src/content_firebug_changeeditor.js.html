  9 var item;
 10 var FBL;
 11 var internalFilefieldTextbox;
 12 var browseButton;
 13 
 14 function onLoad()
 15 {
 16     var args = window.arguments[0];
 17     item = args.item;
 18     FBL = args.FBL;
 19 
 20     browseButton = document.getElementById("browse-button");
 21 
 22     document.getElementById("name").value = item.label;
 23     if (item.executable)
 24     {
 25         try
 26         {
 27             var file = fbXPCOMUtils.CCIN("@mozilla.org/file/local;1", "nsILocalFile");
 28             file.initWithPath(item.executable);
 29             document.getElementById("executable").file = file;
 30         }
 31         catch(exc) {}
 32     }
 33 
 34     if (item.cmdline)
 35         document.getElementById("cmdline").value = item.cmdline;
 36 
 37     onChange();
 38 
 39     // Localization
 40     internationalizeUI(document);
 41 
 42     if (document.getAnonymousElementByAttribute && !document.getElementById("executable").file)
 43     {
 44         setTimeout(function()
 45         {
 46             internalFilefieldTextbox = document.getAnonymousElementByAttribute(
 47                 document.getElementById("executable"), "class", "fileFieldLabel");
 48 
 49             if (internalFilefieldTextbox)
 50             {
 51                 internalFilefieldTextbox.readOnly = false;
 52                 internalFilefieldTextbox.addEventListener("input", function(e) {
 53                     browseButton.disabled = (this.value != "");
 54                     onChange();
 55                 }, false);
 56             }
 57         }, 100);
 58     }
 59 }
 60 
 61 function internationalizeUI(doc)
 62 {
 63     var elements = ["firebug-external-editors-change", "fbNameLabel", "fbExecutableLabel",
 64         "browse-button", "fbCmdLineLabel"];
 65 
 66     for (var i=0; i<elements.length; i++)
 67     {
 68         var element = doc.getElementById(elements[i]);
 69         if (!element)
 70             continue;
 71 
 72         if (element.hasAttribute("title"))
 73             FBL.internationalize(element, "title");
 74 
 75         if (element.hasAttribute("label"))
 76             FBL.internationalize(element, "label");
 77 
 78         if (element.hasAttribute("value"))
 79             FBL.internationalize(element, "value");
 80     }
 81 }
 82 
 83 function onAccept()
 84 {
 85     item.label = document.getElementById("name").value;
 86     if (!browseButton.disabled)
 87     {
 88         var file = document.getElementById("executable").file;
 89         item.executable = "";
 90         if (file)
 91             item.executable = file.path;
 92     }
 93     else
 94     {
 95         item.executable = internalFilefieldTextbox.value.replace(/^\s+|\s+$/g, '');
 96     }
 97 
 98     item.cmdline = document.getElementById("cmdline").value;
 99 
100     try
101     {
102         var file = fbXPCOMUtils.CCIN("@mozilla.org/file/local;1", "nsILocalFile");
103         file.initWithPath(item.executable);
104         if (!file.isExecutable())
105            throw "NotAnExecutable"; 
106 
107         window.arguments[1].saveChanges = true;
108         return true;
109     }
110     catch (exc)
111     {
112         const Ci = Components.interfaces;
113         const nsIPromptService = nsIPromptService;
114         var promptService = fbXPCOMUtils.CCIN("@mozilla.org/embedcomp/prompt-service;1",
115             "nsIPromptService");
116 
117         if (exc == "NotAnExecutable")
118         {
119             promptService.alert(null, FBL.$STR("changeEditor.Invalid_Application_Path"),
120                 FBL.$STR("changeEditor.Path_is_not_an_executable"));
121         }
122         else
123         {
124             promptService.alert(null, FBL.$STR("changeEditor.Invalid_Application_Path"),
125                 FBL.$STR("changeEditor.Application_does_not_exist"));
126         }
127 
128         return false;
129     }
130 }
131 
132 function onChange()
133 {
134     document.documentElement.getButton("accept").disabled = !(
135         document.getElementById("name").value && (
136             (browseButton.disabled && internalFilefieldTextbox &&
137                 internalFilefieldTextbox.value &&
138                 internalFilefieldTextbox.value.replace(/^\s+|\s+$/g, '')) ||
139             (!browseButton.disabled && document.getElementById("executable").file)
140         )
141     );
142 }
143 
144 function onBrowse()
145 {
146     const Ci = Components.interfaces;
147     const nsIFilePicker = Ci.nsIFilePicker;
148     var picker = fbXPCOMUtils.CCIN("@mozilla.org/filepicker;1", "nsIFilePicker");
149     picker.init(window, "", nsIFilePicker.modeOpen);
150     picker.appendFilters(nsIFilePicker.filterApps);
151 
152     if (picker.show() == nsIFilePicker.returnOK && picker.file)
153     {
154         var nameField = document.getElementById("name");
155         var execField = document.getElementById("executable");
156         execField.file = picker.file;
157 
158         if (internalFilefieldTextbox)
159             internalFilefieldTextbox.readOnly = true;
160 
161         if (nameField.value == "")
162             nameField.value = execField.file.leafName.replace(".exe","");
163 
164         onChange();
165         return true;
166     }
167 
168     return false;
169 }
