 34 /**
 35  * Describes a value referenced by a variable.
 36  * A value represents data in a program - for example a number, string or object.
 37  * 
 38  * @constructor
 39  * @param type type of value
 40  * @param data underlying data value, based on the type of value
 41  * @type Value
 42  * @return a new Value
 43  * @version 1.0
 44  */
 45 function Value(type, data) {
 46 	this.type = type;
 47 	this.data = data;
 48 }
 49 
 50 /**
 51  * Returns the type of data this value contains as a {@link String}.
 52  * One of the following is returned:
 53  * <ul>
 54  * <li><code>object</code></li>
 55  * <li><code>function</code></li>
 56  * <li><code>boolean</code></li>
 57  * <li><code>number</code></li>
 58  * <li><code>string</code></li>
 59  * <li><code>undefined</code></li>
 60  * </ul>
 61  * <p>
 62  * <ul>
 63  * <li>TODO: is an array also a special data type?</li>
 64  * <li>TODO: do we need to distinguish between null and undefined?</li>
 65  * </ul>
 66  * </p>
 67  * @function
 68  * @returns the type of data this value contains
 69  */
 70 Value.prototype.getType = function() {
 71 	return this.type;
 72 };
 73 
 74 /**
 75  * Returns the underlying data associated with this value.
 76  * <table border="1">
 77  * 	<tr>
 78  * 		<th>Value Type</th>
 79  * 		<th>Return Type</th>
 80  *  </tr>
 81  *  <tr>
 82  *		<td>object</td>
 83  *		<td>an array of {@link Variable}'s representing the properties of the object</td>
 84  *	</tr>
 85  *	<tr>
 86  *		<td>function</td>
 87  *		<td>TODO</td>
 88  *	</tr>
 89  *	<tr>
 90  *		<td>boolean</td>
 91  *		<td>a boolean value</td>
 92  *	</tr>
 93  *	<tr>
 94  *		<td>number</td>
 95  *		<td>a number value</td>
 96  *	</tr>
 97  *	<tr>
 98  *		<td>string</td>
 99  *		<td>returns a {@link String}</td>
100  *	</tr>
101  *	<tr>
102  *		<td>undefined</td>
103  *		<td>returns null</td>
104  *	</tr>
105  * </table>
106  * 
107  * @function
108  * @returns the underlying data value or <code>null</code> if undefined
109  */
110 Value.prototype.getData = function() {
111 	return this.data;
112 };
