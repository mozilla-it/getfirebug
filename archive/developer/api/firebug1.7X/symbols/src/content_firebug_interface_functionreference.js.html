 34 /**
 35  * Describes an instance of a function object in a JavaScript program.
 36  * 
 37  * @constructor
 38  * @param id unique object identifier of this function (a number)
 39  * @param name function name
 40  * @type FunctionReference
 41  * @augments ObjectReference
 42  * @return a new {@link FunctionReference}
 43  * @version 1.0
 44  */
 45 function FunctionReference(id, name) {
 46 	ObjectReference.call(this, "function", id);
 47 	this.name = name;
 48 }
 49 
 50 /**
 51  * Subclass of {@link ObjectReference}
 52  */
 53 FunctionReference.prototype = subclass(ObjectReference.prototype);
 54 
 55 /**
 56  * Returns the name of this function.
 57  * <p>
 58  * This function does not require communication with
 59  * the browser.
 60  * </p>
 61  * @function
 62  * @returns the name of this function
 63  */
 64 FunctionReference.prototype.getName = function() {
 65 	return this.name;
 66 };
