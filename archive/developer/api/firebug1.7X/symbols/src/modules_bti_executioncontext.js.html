 11 /**
 12  * Describes the execution of JavaScript within a browser context. An execution
 13  * context pertains to one or more compilation units (JavaScript scripts)
 14  * that may contain breakpoints. An execution context can suspend and resume
 15  * and provides stack frames when suspended.
 16  * 
 17  * @constructor
 18  * @param browser the browser context the execution context is contained in
 19  * @type JavaScriptContext
 20  * @return a new JavaScriptContext
 21  * @version 1.0
 22  */
 23 function JavaScriptContext(browserContext)
 24 {
 25     this.browserContext= browserContext;
 26     this.is_suspended = false;
 27     this.frames = [];
 28 }

132 /**
133  * Sets this execution context as currently suspended. Fires notification
134  * of the suspend to registered listeners. Subclasses may call this method
135  * when a suspend occurs.
136  * <p>
137  * Has no effect if this execution context is already suspended.
138  * </p>
139  * 
140  * @function
141  * @param compilationUnit the compilation unit where the suspend occurred
142  * @param lineNumber the line number the suspend occurred at
143  */
144 JavaScriptContext.prototype._suspended = function(compilationUnit, lineNumber)
145 {
146     if (!this.is_suspended)
147     {
148         this.is_suspended = true;
149         this.getBrowserContext().getBrowser()._dispatch("onBreak", [compilationUnit, lineNumber]);
150     }
151 };
152 
153 /**
154  * Sets this execution context as currently running. Fires notification
155  * of the resume to registered listeners. Subclasses may call this method
156  * when a resume occurs.
157  * <p>
158  * Has no effect if this execution context is already running.
159  * </p>
160  * 
161  * @function
162  */
163 JavaScriptContext.prototype._resumed = function()
164 {
165     if (this.is_suspended)
166     {
167         this.is_suspended = false;
168         this.getBrowserContext().getBrowser()._dispatch("onResume", [this]);
169     }
170 };
