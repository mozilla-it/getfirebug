 33 /**
 34  * Describes the event listener functions supported by a {@link Browser}.
 35  *
 36  * @constructor
 37  * @type BrowserEventListener
 38  * @return a new {@link BrowserEventListener}
 39  * @version 1.0
 40  */
 41 Browser.EventListener = {
 42 
 43     /**
 44      * Notification that execution has suspended in the specified
 45      * compilation unit.
 46      *
 47      * @function
 48      * @param compilationUnit the {@link CompilationUnit} execution has suspended in
 49      * @param lineNumber the line number execution has suspended at
 50      */
 51     onBreak: function(compilationUnit, lineNumber) {},
 52 
 53     /**
 54      * TODO:
 55      */
 56     onConsoleDebug: function() {},
 57 
 58     /**
 59      * TODO:
 60      */
 61     onConsoleError: function() {},
 62 
 63     /**
 64      * Notification the specified information messages have been logged.
 65      *
 66      * @function
 67      * @param browserContext the {@link BrowserContext} the messages were logged from
 68      * @param messages array of messages as {@link String}'s
 69      */
 70     onConsoleInfo: function(browserContext, messages) {},
 71 
 72     /**
 73      * Notification the specified messages have been logged.
 74      *
 75      * @function
 76      * @param browserContext the {@link BrowserContext} the messages were logged from
 77      * @param messages array of messages as {@link String}'s
 78      */
 79     onConsoleLog: function(browserContext, messages) {},
 80 
 81     /**
 82      * Notification the specified warning messages have been logged.
 83      *
 84      * @function
 85      * @param browserContext the {@link BrowserContext} the messages were logged from
 86      * @param messages array of messages as {@link String}'s
 87      */
 88     onConsoleWarn: function(browserContext, messages) {},
 89 
 90     /**
 91      * Notification the specified browser context has been created. This notification
 92      * is sent when a new context is created and before any scripts are compiled in
 93      * the new context.
 94      *
 95      * @function
 96      * @param browserContext the {@link BrowserContext} that was created
 97      */
 98     onContextCreated: function(browserContext) {},
 99 
100     /**
101      * Notification the focus browser context has been changed.
102      *
103      * @function
104      * @param fromContext the previous {@link BrowserContext} that had focus or <code>null</code>
105      * @param toContext the {@link BrowserContext} that now has focus or <code>null</code>
106      */
107     onContextChanged: function(fromContext, toContext) {},
108 
109     /**
110      * Notification the specified browser context has been destroyed.
111      *
112      * @function
113      * @param browserContext the {@link BrowserContext} that was destroyed
114      */
115     onContextDestroyed: function(browserContext) {},
116 
117     /**
118      * Notification the specified browser context has completed loading.
119      *
120      * @function
121      * @param browserContext the {@link BrowserContext} that has completed loading
122      */
123     onContextLoaded: function(browserContext) {},
124 
125     /**
126      * Notification the connection to the remote browser has been closed.
127      *
128      * @function
129      * @param browser the {@link Browser} that has been disconnected
130      */
131     onDisconnect: function(browser) {},
132 
133     /**
134      * TODO:
135      */
136     onInspectNode: function() {},
137 
138     /**
139      * Notification the specified execution context has resumed execution.
140      *
141      * @function
142      * @param javaScriptContext the {@link JavaScriptContext} that has resumed
143      */
144     onResume: function(javaScriptContext) {},
145 
146     /**
147      * Notification the specified compilation unit has been compiled (loaded)
148      * in its browser context.
149      *
150      * @function
151      * @param compilationUnit the {@link CompilationUnit} that has been compiled
152      */
153     onScript: function(compilationUnit) {},
154 
155     /**
156      * Notification the specified breakpoint has been installed or cleared.
157      *
158      * @function
159      * @param breakpoint the {@link Breakpoint} that has been toggled
160      */
161     onToggleBreakpoint: function(breakpoint) {}
162 
163 }
