 34 /**
 35  * Describes a breakpoint in a compilation unit. A breakpoint is specific to a compilation unit and
 36  * execution context.
 37  * <p>
 38  * A breakpoint proceeds through states in its lifecycle.
 39  * <table border="1">
 40  *   <tr>
 41  *     <th>State</th>
 42  *     <th>isInstalled()</th>
 43  *     <th>isCleared()</th>
 44  *   </tr>
 45  *   <tr>
 46  *     <td>Created, but not yet installed</td>
 47  *     <td>false</td>
 48  *     <td>false</td>
 49  *   </tr>
 50  *   <tr>
 51  *     <td>Created and installed</td>
 52  *     <td>true</td>
 53  *     <td>false</td>
 54  *   </tr>
 55  *   <tr>
 56  *     <td>Cleared, but not yet uninstalled</td>
 57  *     <td>true</td>
 58  *     <td>true</td>
 59  *   </tr>
 60  *   <tr>
 61  *     <td>Cleared and uninstalled</td>
 62  *     <td>false</td>
 63  *     <td>true</td>
 64  *   </tr>
 65  * </table>
 66  * </p>
 67  * 
 68  * @constructor
 69  * @param compilationUnit the {@link CompilationUnit} unit that contains this breakpoint
 70  * @param lineNumber the source code line number the breakpoint is set on
 71  * @type Breakpoint
 72  * @return a new Breakpoint
 73  * @version 1.0
 74  */
 75 function Breakpoint(compilationUnit, lineNumber) {
 76 	this.compilationUnit = compilationUnit;
 77 	this.lineNumber = lineNumber;
 78 	this.installed = false;
 79 	this.cleared = false;
 80 }

 91  * @function
 92  * @returns a {@link CompilationUnit}
 93  */
 94 Breakpoint.prototype.getCompilationUnit = function() {
 95 	return this.compilationUnit;
 96 };

115 /**
116  * Returns whether this breakpoint is installed in its {@link CompilationUnit}.
117  * <p>
118  * This function does not require communication with
119  * the browser.
120  * </p>
121  * @function
122  * @returns <code>true</code> if the breakpoint is installed
123  *  and <code>false</code> if it has been cleared
124  */
125 Breakpoint.prototype.isInstalled = function() {
126 	return this.installed;
127 };
128 
129 /**
130  * Returns whether this breakpoint has been cleared. A breakpoint is considered cleared
131  * as soon as its clear function has been called regardless of the browser's knowledge of
132  * the clear.
133  * <p>
134  * This function does not require communication with
135  * the browser.
136  * </p>
137  * @function
138  * @returns whether this breakpoint has been cleared
139  */
140 Breakpoint.prototype.isCleared = function() {
141 	return this.cleared;
142 };
143 
144 /**

148  * the browser.
149  * </p>
150  * @function
151  * @returns line number
152  * 
153  */
154 Breakpoint.prototype.getLineNumber = function() {
155 	return this.lineNumber;
156 };

158 // ---- PRIVATE ----

160 /**
161  * Implementations must call this method when a breakpoint gets installed in the browser.
162  * Notification will be sent to registered listeners that the breakpoint has been installed.
163  * Updates the installed property of this breakpoint. This method should only be called once when
164  * a breakpoint becomes installed.
165  * 
166  * @function
167  */
168 Breakpoint.prototype._installed = function() {
169 	if (!this.installed) {
170 		this.installed = true;
171 		this.getCompilationUnit().getBrowserContext().getBrowser()._dispatch("onToggleBreakpoint", [this]);	
172 	}
173 }
174 
175 /**
176  * Implementations must call this method when a breakpoint is cleared from the browser.
177  * Notification will be sent to registered listeners that the breakpoint has been cleared.
178  * Updates the installed and cleared properties of this breakpoint. This method should only
179  * be called once when a breakpoint is cleared.
180  * 
181  * @function
182  */
183 Breakpoint.prototype._cleared = function() {
184 	if (!this.cleared) {
185 		this.cleared = true;
186 		this.getCompilationUnit().getBrowserContext().getBrowser()._dispatch("onToggleBreakpoint", [this]);
187 	}
188 }
