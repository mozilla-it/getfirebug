 29 Firebug.SourceCache = function(context)
 30 {
 31     this.context = context;
 32     this.cache = {};
 33 };
 34 
 35 Firebug.SourceCache.prototype = extend(new Firebug.Listener(),
 36 {
 37     isCached: function(url)
 38     {
 39         return (this.cache[url] ? true : false);
 40     },
 41 
 42     loadText: function(url, method, file)
 43     {
 44         var lines = this.load(url, method, file);
 45         return lines ? lines.join("") : null;
 46     },
 47 
 48     load: function(url, method, file)
 49     {
 50         if (FBTrace.DBG_CACHE)
 51         {
 52             FBTrace.sysout("sourceCache.load: " + url);
 53 
 54             if (!this.cache.hasOwnProperty(url) && this.cache[url])
 55                 FBTrace.sysout("sourceCache.load; ERROR - hasOwnProperty returns false, " +
 56                     "but the URL is cached: " + url, this.cache[url]);
 57         }
 58 
 59         // xxxHonza: sometimes hasOwnProperty return false even if the URL is obviously there.
 60         //if (this.cache.hasOwnProperty(url))
 61         var response = this.cache[this.removeAnchor(url)];
 62         if (response)
 63             return response;
 64 
 65         if (FBTrace.DBG_CACHE)
 66         {
 67             var urls = [];
 68             for (var prop in this.cache)
 69                 urls.push(prop);
 70 
 71             FBTrace.sysout("sourceCache.load: Not in the Firebug internal cache", urls);
 72         }
 73 
 74         var d = FBL.splitDataURL(url);  //TODO the RE should not have baseLine
 75         if (d)
 76         {
 77             var src = d.encodedContent;
 78             var data = decodeURIComponent(src);
 79             var lines = splitLines(data)
 80             this.cache[url] = lines;
 81 
 82             return lines;
 83         }
 84 
 85         var j = FBL.reJavascript.exec(url);
 86         if (j)
 87         {
 88             var src = url.substring(FBL.reJavascript.lastIndex);
 89             var lines = splitLines(src);
 90             this.cache[url] = lines;
 91 
 92             return lines;
 93         }
 94 
 95         var c = FBL.reChrome.test(url);
 96         if (c)
 97         {
 98             if (Firebug.filterSystemURLs)
 99                 return ["Filtered chrome url "+url];  // ignore chrome
100 
101             // If the chrome.manifest has  xpcnativewrappers=no, platform munges the url
102             var reWrapperMunge = /(\S*)\s*->\s*(\S*)/;
103             var m = reWrapperMunge.exec(url);
104             if (m)
105             {
106                 url = m[2];
107                 if (FBTrace.DBG_CACHE)
108                     FBTrace.sysout("sourceCache found munged xpcnativewrapper url and set it to "+url+" m "+m+" m[0]:"+m[0]+" [1]"+m[1], m);
109             }
110 
111             var chromeURI = makeURI(url);
112             if (!chromeURI)
113             {
114                 if (FBTrace.DBG_CACHE)
115                     FBTrace.sysout("sourceCache.load failed to convert chrome to local: "+url);
116                 return ["sourceCache failed to make URI from "+url];
117             }
118 
119             var localURI = chromeReg.convertChromeURL(chromeURI);
120             if (FBTrace.DBG_CACHE)
121                 FBTrace.sysout("sourceCache.load converting chrome to local: "+url, " -> "+localURI.spec);
122             return this.loadFromLocal(localURI.spec);
123         }
124 
125         c = FBL.reFile.test(url);
126         if (c)
127         {
128             return this.loadFromLocal(url);
129         }
130 
131         if ( url.indexOf('resource://') === 0)
132         {
133             var fileURL = FBL.resourceToFile(url);
134             return this.loadFromLocal(url);
135         }
136 
137         // Unfortunately, the URL isn't available so, let's try to use FF cache.
138         // Notice that additional network request to the server can be made in
139         // this method (double-load).
140         return this.loadFromCache(url, method, file);
141     },
142 
143     store: function(url, text)
144     {
145         var tempURL = this.removeAnchor(url);
146 
147         if (FBTrace.DBG_CACHE)
148             FBTrace.sysout("sourceCache for " + this.context.getName() + " store url=" +
149                 url + ((tempURL != url) ? " -> " + tempURL : ""), text);
150 
151         var lines = splitLines(text);
152         return this.storeSplitLines(tempURL, lines);
153     },
154 
155     removeAnchor: function(url)
156     {
157         var index = url.indexOf("#");
158         if (index < 0)
159             return url;
160 
161         return url.substr(0, index);
162     },
163 
164     loadFromLocal: function(url)
165     {
166         // if we get this far then we have either a file: or chrome: url converted to file:
167         var src = getResource(url);
168         if (src)
169         {
170             var lines = splitLines(src);
171             this.cache[url] = lines;
172 
173             return lines;
174         }
175     },
176 
177     loadFromCache: function(url, method, file)
178     {
179         if (FBTrace.DBG_CACHE) FBTrace.sysout("sourceCache.loadFromCache url:"+url);
180 
181         var doc = this.context.window.document;
182         if (doc)
183             var charset = doc.characterSet;
184         else
185             var charset = "UTF-8";
186 
187         var channel;
188         try
189         {
190             channel = ioService.newChannel(url, null, null);
191             channel.loadFlags |= LOAD_FROM_CACHE | LOAD_BYPASS_LOCAL_CACHE_IF_BUSY;
192 
193             if (method && (channel instanceof nsIHttpChannel))
194             {
195                 var httpChannel = QI(channel, nsIHttpChannel);
196                 httpChannel.requestMethod = method;
197             }
198         }
199         catch (exc)
200         {
201             if (FBTrace.DBG_CACHE)
202                 FBTrace.sysout("sourceCache for url:"+url+" window="+this.context.window.location.href+" FAILS:", exc);
203             return;
204         }
205 
206         if (url == this.context.browser.contentWindow.location.href)
207         {
208             if (FBTrace.DBG_CACHE) FBTrace.sysout("sourceCache.load content window href\n");
209             if (channel instanceof nsIUploadChannel)
210             {
211                 var postData = getPostStream(this.context);
212                 if (postData)
213                 {
214                     var uploadChannel = QI(channel, nsIUploadChannel);
215                     uploadChannel.setUploadStream(postData, "", -1);
216                     if (FBTrace.DBG_CACHE) FBTrace.sysout("sourceCache.load uploadChannel set\n");
217                 }
218             }
219 
220             if (channel instanceof nsICachingChannel)
221             {
222                 var cacheChannel = QI(channel, nsICachingChannel);
223                 cacheChannel.cacheKey = getCacheKey(this.context);
224                 if (FBTrace.DBG_CACHE) FBTrace.sysout("sourceCache.load cacheChannel key"+cacheChannel.cacheKey+"\n");
225             }
226         }
227         else if ((method == "PUT" || method == "POST") && file)
228         {
229             if (channel instanceof nsIUploadChannel)
230             {
231                 // In case of PUT and POST, don't forget to use the original body.
232                 var postData = getPostText(file, this.context);
233                 if (postData)
234                 {
235                     var postDataStream = getInputStreamFromString(postData);
236                     var uploadChannel = QI(channel, nsIUploadChannel);
237                     uploadChannel.setUploadStream(postDataStream, "application/x-www-form-urlencoded", -1);
238                     if (FBTrace.DBG_CACHE) FBTrace.sysout("sourceCache.load uploadChannel set\n");
239                 }
240             }
241         }
242 
243         var stream;
244         try
245         {
246             if (FBTrace.DBG_CACHE) FBTrace.sysout("sourceCache.load url:"+url+" with charset"+charset+"\n");
247             stream = channel.open();
248         }
249         catch (exc)
250         {
251             if (FBTrace.DBG_ERRORS)
252             {
253                 var isCache = (channel instanceof nsICachingChannel)?"nsICachingChannel":"NOT caching channel";
254                 var isUp = (channel instanceof nsIUploadChannel)?"nsIUploadChannel":"NOT nsIUploadChannel";
255                 FBTrace.sysout(url+" vs "+this.context.browser.contentWindow.location.href+" and "+isCache+" "+isUp+"\n");
256                 FBTrace.sysout("sourceCache.load fails channel.open for url="+url+ " cause:", exc);
257                 FBTrace.sysout("sourceCache.load fails channel=", channel);
258             }
259             return ["sourceCache.load FAILS for url="+url, exc.toString()];
260         }
261 
262         try
263         {
264             var data = readFromStream(stream, charset);
265             var lines = splitLines(data);
266             this.cache[url] = lines;
267             return lines;
268         }
269         catch (exc)
270         {
271             if (FBTrace.DBG_ERRORS)
272                 FBTrace.sysout("sourceCache.load FAILS, url="+url, exc);
273             return ["sourceCache.load FAILS for url="+url, exc.toString()];
274         }
275         finally
276         {
277             stream.close();
278         }
279     },
280 
281     storeSplitLines: function(url, lines)
282     {
283         if (FBTrace.DBG_CACHE)
284             FBTrace.sysout("sourceCache for window="+this.context.getName()+" store url="+url+"\n");
285         return this.cache[url] = lines;
286     },
287 
288     invalidate: function(url)
289     {
290         url = this.removeAnchor(url);
291 
292         if (FBTrace.DBG_CACHE)
293             FBTrace.sysout("sourceCache.invalidate; " + url);
294 
295         delete this.cache[url];
296     },
297 
298     getLine: function(url, lineNo)
299     {
300         var lines = this.load(url);
301         if (lines)
302         {
303             if (lineNo <= lines.length)
304                 return lines[lineNo-1];
305             else
306                 return (lines.length == 1) ? lines[0] : "("+lineNo+" out of range "+lines.length+")";
307         }
308         else
309             return "(no source for "+url+")";
310     }
311 });

313 // xxxHonza getPostText and readPostTextFromRequest are copied from
314 // net.js. These functions should be removed when this cache is
315 // refactored due to the double-load problem.
316 function getPostText(file, context)
317 {
318     if (!file.postText)
319         file.postText = readPostTextFromPage(file.href, context);
320 
321     if (!file.postText)
322         file.postText = readPostTextFromRequest(file.request, context);
323 
324     return file.postText;
325 }

329 function getPostStream(context)
330 {
331     try
332     {
333         var webNav = context.browser.webNavigation;
334         var descriptor = QI(webNav, Ci.nsIWebPageDescriptor).currentDescriptor;
335         var entry = QI(descriptor, Ci.nsISHEntry);
336 
337         if (entry.postData)
338         {
339             // Seek to the beginning, or it will probably start reading at the end
340             var postStream = QI(entry.postData, Ci.nsISeekableStream);
341             postStream.seek(0, 0);
342             return postStream;
343         }
344      }
345      catch (exc)
346      {
347      }
348 }
349 
350 function getCacheKey(context)
351 {
352     try
353     {
354         var webNav = context.browser.webNavigation;
355         var descriptor = QI(webNav, Ci.nsIWebPageDescriptor).currentDescriptor;
356         var entry = QI(descriptor, Ci.nsISHEntry);
357         return entry.cacheKey;
358      }
359      catch (exc)
360      {
361      }
362 }
