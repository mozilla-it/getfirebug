178 var total_handlers = 0;
179 function createConsoleHandler(context, win)
180 {
181     var handler = {};
182     handler.console = Firebug.Console.createConsole(context, win),
183 
184     handler.detach = function()
185     {
186         win.document.removeEventListener('firebugAppendConsole', this.boundHandler, true);
187 
188         if (FBTrace.DBG_CONSOLE)
189             FBTrace.sysout("consoleInjector FirebugConsoleHandler removeEventListener "+this.handler_name);
190     };
191 
192     handler.handler_name = ++total_handlers;
193     handler.token = Math.random();
194 
195     handler.handleEvent = function(event)
196     {
197         if (FBTrace.DBG_CONSOLE)
198             FBTrace.sysout("FirebugConsoleHandler("+this.handler_name+") "+win.document.getUserData("firebug-methodName")+", event", event);
199         if (!Firebug.CommandLine.CommandHandler.handle(event, this.console, win))
200         {
201             if (FBTrace.DBG_CONSOLE)
202                 FBTrace.sysout("FirebugConsoleHandler", this);
203 
204             var methodName = win.document.getUserData("firebug-methodName");
205             Firebug.Console.log($STRF("console.MethodNotSupported", [methodName]));
206         }
207     };
208 
209     handler.setEvaluatedCallback = function( fnOfResult )
210     {
211         this.console.evaluated = fnOfResult;
212     };
213 
214     handler.setEvaluateErrorCallback = function( fnOfResultAndContext )
215     {
216         this.console.evaluateError = fnOfResultAndContext;
217     };
218 
219     handler.win = win;
220     handler.context = context;
221 
222     handler.onUnload = function()
223     {
224         Firebug.Console.injector.detachConsole(this.context, this.win);
225     };
226 
227     win.addEventListener("unload", bind(handler.onUnload, handler), true);
228 
229     // When raised on our injected element, callback to Firebug and append to console
230     handler.boundHandler = bind(handler.handleEvent, handler);
231     win.document.addEventListener('firebugAppendConsole', handler.boundHandler, true); // capturing
232 
233     if (FBTrace.DBG_CONSOLE)
234         FBTrace.sysout("consoleInjector FirebugConsoleHandler addEventListener "+handler.handler_name);
235 
236     return handler;
237 }

360             {

362 
363                 frameCounter = {logRow: logRow, count: 1};
364                 context.frameCounters[frameId] = frameCounter;
365             }
366             else
367                 ++frameCounter.count;
368 
369             var label = key == undefined
370                 ? frameCounter.count
371                 : key + " " + frameCounter.count;

374         }
375         return "_firebugIgnore";

411 
412         var timeCounter = this.timeCounters[key];
413         if (timeCounter)

415             var diff = time - timeCounter;
416             var label = name + ": " + diff + "ms";

420             delete this.timeCounters[key];
421         }
422         return diff;

456     function logAssert(category, args)
457     {
458         Firebug.Errors.increaseCount(context);
459 
460         if (!args || !args.length || args.length == 0)
461             var msg = [FBL.$STR("Assertion")];
462         else
463             var msg = args[0];
464 
465         if (msg.stack)
466         {
467             var trace = parseToStackTrace(msg.stack, context);
468             if (FBTrace.DBG_CONSOLE)
469                 FBTrace.sysout("logAssert trace from msg.stack", trace);
470         }
471         else if (context.stackTrace)
472         {
473             var trace = context.stackTrace
474             if (FBTrace.DBG_CONSOLE)
475                 FBTrace.sysout("logAssert trace from context.window.stackTrace", trace);
476         }
477         else
478         {
479             var trace = getJSDUserStack();
480             if (FBTrace.DBG_CONSOLE)
481                 FBTrace.sysout("logAssert trace from getJSDUserStack", trace);
482         }
483 
484         trace = cleanStackTraceOfFirebug(trace);
485 
486         var url = msg.fileName ? msg.fileName : win.location.href;
487         var lineNo = (trace && msg.lineNumber) ? msg.lineNumber : 0; // we may have only the line popped above
488         var errorObject = new FBL.ErrorMessage(msg, url, lineNo, "", category, context, trace);
489 
490         if (trace && trace.frames && trace.frames[0])
491            errorObject.correctWithStackTrace(trace);
492 
493         errorObject.resetSource();
494 
495         if (args.length > 1)
496         {
497             errorObject.objects = []
498             for (var i = 1; i < args.length; i++)
499                 errorObject.objects.push(args[i]);
500         }
501 
502         var row = Firebug.Console.log(errorObject, context, "errorMessage", null, true); // noThrottle
503         row.scrollIntoView();
504 
505         return "_firebugIgnore";
506     }

508     function getComponentsStackDump()
509     {
510         // Starting with our stack, walk back to the user-level code
511         var frame = Components.stack;
512         var userURL = win.location.href.toString();
513 
514         if (FBTrace.DBG_CONSOLE)
515             FBTrace.sysout("consoleInjector.getComponentsStackDump initial stack for userURL "+userURL, frame);
516 
517         // Drop frames until we get into user code.
518         while (frame && FBL.isSystemURL(frame.filename) )
519             frame = frame.caller;
520 
521         // Drop two more frames, the injected console function and firebugAppendConsole()
522         if (frame)
523             frame = frame.caller;
524         if (frame)
525             frame = frame.caller;
526 
527         if (FBTrace.DBG_CONSOLE)
528             FBTrace.sysout("consoleInjector.getComponentsStackDump final stack for userURL "+userURL, frame);
529 
530         return frame;
531     }

538     function getJSDUserStack()
539     {
540         var trace = Firebug.Debugger.getCurrentStackTrace(context);
541 
542         var frames = trace ? trace.frames : null;
543         if (frames && (frames.length > 0) )
544         {
545             var oldest = frames.length - 1;  // 6 - 1 = 5
546             for (var i = 0; i < frames.length; i++)
547             {
548                 if (frames[oldest - i].href.indexOf("chrome:") == 0)
549                     break;
550 
551                 // firebug-service scope reached, in some cases the url starts with file://
552                 if (frames[oldest - i].href.indexOf("modules/firebug-service.js") != -1)
553                     break;
554 
555                 // command line
556                 var fn = frames[oldest - i].getFunctionName() + "";
557                 if (fn && (fn.indexOf("_firebugEvalEvent") != -1))
558                     break;
559             }
560 
561             // take the oldest frames, leave 2 behind they are injection code
562             trace.frames = trace.frames.slice(2 - i);
563 
564             if (FBTrace.DBG_CONSOLE)
565                 FBTrace.sysout("consoleInjector getJSDUserStack: "+frames.length+" oldest: "+
566                     oldest+" i: "+i+" i - oldest + 2: "+(i - oldest + 2), trace.toString().split('\n'));
567 
568             return trace;
569         }
570         else
571         {
572             return "Firebug failed to get stack trace with any frames";
573         }
574     }
