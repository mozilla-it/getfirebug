  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 // Constants
  7 
  8 const Cc = Components.classes;
  9 const Ci = Components.interfaces;
 10 
 11 const MAX_HISTORY_MENU_ITEMS = 15;
 12 
 13 // ************************************************************************************************
 14 
 15 /**
 16  * @class Support for back and forward pattern for navigating within Firebug UI (panels).
 17  */
 18 Firebug.NavigationHistory = extend(Firebug.Module,
 19 {
 20     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
 21     // Extending Module
 22 
 23     initContext: function(context, persistedState)
 24     {
 25         Firebug.Module.initContext.apply(this, arguments);
 26 
 27         // Initialize context members.
 28         context.navigationHistory = [];
 29         context.navigationHistoryIndex = 0;
 30 
 31         if (persistedState && persistedState.navigationHistory)
 32         {
 33             context.navigationHistory = persistedState.navigationHistory;
 34             context.navigationHistoryIndex = persistedState.navigationHistoryIndex;
 35         }
 36     },
 37 
 38     reattachContext: function(browser, context)
 39     {
 40         Firebug.Module.reattachContext.apply(this, arguments);
 41 
 42         this.updateButtons(context);
 43     },
 44 
 45     destroyContext: function(context, persistedState)
 46     {
 47         Firebug.Module.destroyContext.apply(this, arguments);
 48 
 49         if (persistedState)
 50         {
 51             persistedState.navigationHistory = context.navigationHistory;
 52             persistedState.navigationHistoryIndex = context.navigationHistoryIndex;
 53         }
 54     },
 55 
 56     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
 57     // History popup menu
 58 
 59     onPopupShowing: function(popup, context)
 60     {
 61         var currIndex = this.getCurrentIndex(context);
 62 
 63         if (FBTrace.DBG_HISTORY)
 64             FBTrace.sysout("history.onPopupShowing; " + currIndex + ", " + context.getName(), context);
 65 
 66         FBL.eraseNode(popup);
 67 
 68         var list = this.getHistory(context);
 69 
 70         // Don't display the popup for a single item.
 71         var count = list.length;
 72         if (count <= 1)
 73             return false;
 74 
 75         var maxItems = MAX_HISTORY_MENU_ITEMS;
 76         var half = Math.floor(maxItems / 2);
 77         var start = Math.max(currIndex - half, 0);
 78         var end = Math.min(start == 0 ? maxItems : currIndex + half + 1, count);
 79 
 80         if (end == count)
 81             start = Math.max(count - maxItems, 0);
 82 
 83         var tooltipBack = $STR("firebug.history.Go back to this panel");
 84         var tooltipCurrent = $STR("firebug.history.Stay on this panel");
 85         var tooltipForward = $STR("firebug.history.Go forward to this panel");
 86 
 87         for (var i=end-1; i>=start; i--)
 88         {
 89             var historyItem = list[i];
 90             var panelType = Firebug.getPanelType(historyItem.panelName);
 91             var label = Firebug.getPanelTitle(panelType);
 92             if (historyItem.location && historyItem.location.href)
 93                 label += " - " + historyItem.location.href;
 94 
 95             var menuInfo = {
 96                 label: label,
 97                 nol10n: true,
 98                 className: "menuitem-iconic fbURLMenuItem",
 99             };
100 
101             if (i < currIndex)
102             {
103                 menuInfo.className += " navigationHistoryMenuItemBack";
104                 menuInfo.tooltiptext = tooltipBack;
105             }
106             else if (i == currIndex)
107             {
108                 menuInfo.type = "radio";
109                 menuInfo.checked = "true";
110                 menuInfo.className = "navigationHistoryMenuItemCurrent";
111                 menuInfo.tooltiptext = tooltipCurrent;
112             }
113             else
114             {
115                 menuInfo.className += " navigationHistoryMenuItemForward";
116                 menuInfo.tooltiptext = tooltipForward;
117             }
118 
119             var menuItem = FBL.createMenuItem(popup, menuInfo);
120             menuItem.repObject = location;
121             menuItem.setAttribute("index", i);
122         }
123 
124         return true;
125     },
126 
127     onHistoryCommand: function(event, context)
128     {
129         var menuItem = event.target;
130         var index = menuItem.getAttribute("index");
131         if (!index)
132             return false;
133 
134         this.gotoHistoryIndex(context, index);
135         return true;
136     },
137 
138     goBack: function(context)
139     {
140         var currIndex = this.getCurrentIndex(context);
141 
142         if (FBTrace.DBG_HISTORY)
143             FBTrace.sysout("history.goBack; " + currIndex + ", " + context.getName(), context);
144 
145         this.gotoHistoryIndex(context, currIndex - 1);
146     },
147 
148     goForward: function(context)
149     {
150         var currIndex = this.getCurrentIndex(context);
151 
152         if (FBTrace.DBG_HISTORY)
153             FBTrace.sysout("history.goForward; " + currIndex + ", " + context.getName(), context);
154 
155         this.gotoHistoryIndex(context, currIndex + 1);
156     },
157 
158     gotoHistoryIndex: function(context, index)
159     {
160         var list = this.getHistory(context);
161         if (index < 0 || index >= list.length)
162             return;
163 
164         var historyItem = list[index];
165 
166         try
167         {
168             this.navInProgress = true;
169             Firebug.chrome.navigate(historyItem.location, historyItem.panelName);
170             context.navigationHistoryIndex = index;
171         }
172         catch (e)
173         {
174         }
175         finally
176         {
177             this.navInProgress = false;
178         }
179 
180         this.updateButtons(context);
181     },
182 
183 
184     updateButtons: function(context)
185     {
186         var list = this.getHistory(context);
187 
188         var backButton = Firebug.chrome.$("fbNavigateBackButton");
189         var forwardButton = Firebug.chrome.$("fbNavigateForwardButton");
190 
191         backButton.setAttribute("disabled", "true");
192         forwardButton.setAttribute("disabled", "true");
193 
194         if (list.length <= 1)
195             return;
196 
197         var currIndex = this.getCurrentIndex(context);
198 
199         if (currIndex > 0)
200             backButton.removeAttribute("disabled");
201 
202         if (currIndex < list.length-1)
203             forwardButton.removeAttribute("disabled");
204     },
205 
206     getHistory: function(context)
207     {
208         if (!context.navigationHistory)
209             context.navigationHistory = [];
210 
211         return context.navigationHistory;
212     },
213 
214     getCurrentIndex: function(context)
215     {
216         if (typeof(context.navigationHistoryIndex) == "undefined")
217             context.navigationHistoryIndex = 0;
218 
219         return context.navigationHistoryIndex;
220     },
221 
222     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
223     // UI Listener
224 
225     onPanelNavigate: function(location, panel)
226     {
227         var context = panel.context;
228         var currIndex = this.getCurrentIndex(context);
229 
230         if (FBTrace.DBG_HISTORY)
231             FBTrace.sysout("history.onPanelNavigate; " + currIndex + ", " +
232                 "Panel: " + (panel ? panel.name : "Unknown Panel") + ", " +
233                 "Location: " + (location ? location.href : "No Location") + ", " +
234                 context.getName());
235 
236         // The panel must be always there
237         if (!panel)
238             return;
239 
240         // Ignore side panel navigation.
241         if (panel.parentPanel)
242             return;
243 
244         // The user is navigating using the history UI, this action doesn't affect
245         // the history list.
246         if (this.navInProgress)
247             return;
248 
249         var list = this.getHistory(context);
250 
251         // If the last item in the history is the same bail out.
252         var lastHistoryItem = list.length ? list[list.length-1] : null;
253         if (lastHistoryItem && lastHistoryItem.panelName == panel.name &&
254             lastHistoryItem.location == location)
255             return;
256 
257         if (lastHistoryItem && lastHistoryItem.location && location &&
258             lastHistoryItem.location.href == location.href)
259             return;
260 
261         // If the panel is the same, bail out.
262         var currHistoryItem = list.length ? list[currIndex] : null;
263         if (currHistoryItem && currHistoryItem.panelName == panel.name &&
264             currHistoryItem.location == location)
265             return;
266 
267         // Remove forward history.
268         list.splice(currIndex+1, list.length-(currIndex+1));
269 
270         // New back history record.
271         list.push({panelName: panel.name, location: location});
272         context.navigationHistoryIndex = list.length-1;
273 
274         if (FBTrace.DBG_HISTORY)
275             FBTrace.sysout("history.onPanelNavigate; New history record created " + currIndex +
276                 ", " + panel.name + ", " + (location ? location.href : "No Location"), list);
277 
278         // Update back and forward buttons in the UI.
279         this.updateButtons(context);
280     }
281 });
282 
283 // ************************************************************************************************
284 // Registration
285 
286 Firebug.registerModule(Firebug.NavigationHistory);
287 Firebug.registerUIListener(Firebug.NavigationHistory);
288 
289 // ************************************************************************************************
290 return Firebug.NavigationHistory;
291 }});
