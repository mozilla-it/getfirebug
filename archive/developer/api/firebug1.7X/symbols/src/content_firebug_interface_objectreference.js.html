 34 /**
 35  * Describes an instance of an object in a JavaScript program - for example a number,
 36  * string or object.
 37  * 
 38  * @constructor
 39  * @param type type of object
 40  * @param id unique object identifier (a number)
 41  * @type ObjectReference
 42  * @return a new {@link ObjectReference}
 43  * @version 1.0
 44  */
 45 function ObjectReference(type, id) {
 46 	this.type = type;
 47 	this.id = id;
 48 }
 49 
 50 /**
 51  * Returns the unique identifier of this object as a number.
 52  * <p>
 53  * This function does not require communication with
 54  * the browser.
 55  * </p>
 56  * @function
 57  * @returns unique object identifier (number)
 58  */
 59 ObjectReference.prototype.getId = function() {
 60 	return this.id;
 61 };
 62 
 63 /**
 64  * Returns the type of data this object contains as a {@link String}.
 65  * One of the following is returned:
 66  * <ul>
 67  * <li><code>object</code> - implies this object is an instance of {@link ObjectReference}</li>
 68  * <li><code>function</code> - implies this object is an instance of {@link FunctionReference}</li>
 69  * <li><code>boolean</code> - implies this object is an instance of {@link Primitive}</li>
 70  * <li><code>number</code> - implies this object is an instance of {@link Primitive}</li>
 71  * <li><code>string</code> - implies this object is an instance of {@link Primitive}</li>
 72  * <li><code>array</code> - implies this object is an instance of {@link ArrayReference}</li>
 73  * </ul>
 74  * <p>
 75  * This function does not require communication with
 76  * the browser.
 77  * </p>
 78  * @function
 79  * @returns the type of data this object contains
 80  */
 81 ObjectReference.prototype.getType = function() {
 82 	return this.type;
 83 };
