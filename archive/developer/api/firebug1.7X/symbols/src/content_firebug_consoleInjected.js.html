  2 function _createFirebugConsole()
  3 {
  4     var console = {};
  5     console.log = function log() { return console.notifyFirebug(arguments, 'log', 'firebugAppendConsole'); }
  6     console.debug = function debug() { return console.notifyFirebug(arguments, 'debug', 'firebugAppendConsole'); }
  7     console.info = function info() { return console.notifyFirebug(arguments, 'info', 'firebugAppendConsole'); }
  8     console.warn = function warn() { return console.notifyFirebug(arguments, 'warn', 'firebugAppendConsole'); }
  9     console.exception = function exception() { return console.notifyFirebug(arguments, 'exception', 'firebugAppendConsole'); }
 10     console.assert = function assert() { return console.notifyFirebug(arguments, 'assert', 'firebugAppendConsole'); }
 11     console.dir = function dir() { return console.notifyFirebug(arguments, 'dir', 'firebugAppendConsole'); }
 12     console.dirxml = function dirxml() { return console.notifyFirebug(arguments, 'dirxml', 'firebugAppendConsole'); }
 13     console.trace = function firebugDebuggerTracer() { debugger; return "_firebugIgnore"; }
 14     console.group = function group() { return console.notifyFirebug(arguments, 'group', 'firebugAppendConsole'); }
 15     console.groupEnd = function groupEnd() { return console.notifyFirebug(arguments, 'groupEnd', 'firebugAppendConsole'); }
 16     console.groupCollapsed = function groupCollapsed() { return console.notifyFirebug(arguments, 'groupCollapsed', 'firebugAppendConsole'); }
 17     console.time = function time() { return console.notifyFirebug(arguments, 'time', 'firebugAppendConsole'); }
 18     console.timeEnd = function timeEnd() { return console.notifyFirebug(arguments, 'timeEnd', 'firebugAppendConsole'); }
 19     console.profile = function profile() { return console.notifyFirebug(arguments, 'profile', 'firebugAppendConsole'); }
 20     console.profileEnd = function profileEnd() { return console.notifyFirebug(arguments, 'profileEnd', 'firebugAppendConsole'); }
 21     console.count = function count() { return console.notifyFirebug(arguments, 'count', 'firebugAppendConsole'); }
 22     console.clear = function clear() { return console.notifyFirebug(arguments, 'clear', 'firebugAppendConsole'); }
 23     console.table = function table() { return console.notifyFirebug(arguments, 'table', 'firebugAppendConsole'); }
 24 
 25     console.error = function error()
 26     {
 27         window.top._firebugStackTrace = "requested"; // flag to cause trace to store trace in context.stackTrace
 28         console.trace(); // set the context.stackTrace
 29         var rc = console.notifyFirebug(arguments, 'error', 'firebugAppendConsole');
 30         delete window.top._firebugStackTrace;
 31         return rc;
 32     }
 33     // DBG console.uid = Math.random();
 34 
 35     console.notifyFirebug = function notifyFirebug(objs, methodName, eventID)
 36     {
 37         var event = document.createEvent("Events");
 38         event.initEvent(eventID, true, false);
 39 
 40         console.userObjects = [];
 41         for (var i=0; i<objs.length; i++)
 42             console.userObjects.push(objs[i]);
 43 
 44         var length = console.userObjects.length;
 45         document.setUserData("firebug-methodName", methodName, null);
 46 
 47         document.dispatchEvent(event);
 48 
 49         // DBG dump("FirebugConsole dispatched event "+methodName+" via "+eventID+" with "+length+ " user objects, [0]:"+console.userObjects[0]+"\n");
 50 
 51         var result;
 52         if (document.getUserData("firebug-retValueType") == "array")
 53             result = [];
 54 
 55         if (!result && console.userObjects.length == length+1)
 56             return console.userObjects[length];
 57 
 58         for (var i=length; i<console.userObjects.length && result; i++)
 59             result.push(console.userObjects[i]);
 60 
 61         return result;
 62     };
 63 
 64     // ***********************************************************************
 65     // Console API
 66 
 67     console.__defineGetter__("firebug", function firebug() {
 68         return document.getUserData("firebug-Version");
 69     });
 70 
 71     return console;
 72 }
