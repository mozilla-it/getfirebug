 34 /**
 35  * Describes a compilation unit in a browser context. A compilation unit
 36  * may originate from a JavaScript source file or a script element in HTML.
 37  * 
 38  * @constructor
 39  * @param url compilation unit URL - a {@link String} or <code>null</code> if none
 40  * @param context the {@link BrowserContext} this compilation unit is contained in
 41  * @type CompilationUnit
 42  * @return a new CompilationUnit
 43  * @version 1.0
 44  */
 45 function CompilationUnit(url, context) {
 46 	this.url = url;
 47 	this.context = context;
 48 	this.breakpoints = [];
 49 }
 50 
 51 // ---- API ----
 52 
 53 /**
 54  * Returns the URL of this compilation unit.
 55  * <p>
 56  * This function does not require communication with
 57  * the browser.
 58  * </p>
 59  * 
 60  * @function
 61  * @returns compilation unit identifier as a {@link String}
 62  */
 63 CompilationUnit.prototype.getURL = function() {
 64 	return this.url;
 65 };
 66 
 67 /**
 68  * Returns the browser context this compilation unit was compiled in.
 69  * <p>
 70  * This function does not require communication with
 71  * the browser.
 72  * </p>
 73  * 
 74  * @function
 75  * @returns a {@link BrowserContext}
 76  */
 77 CompilationUnit.prototype.getBrowserContext = function() {
 78 	return this.context;
 79 };
 80 
 81 /**
 82  * Returns the breakpoints that have been created in this compilation unit and
 83  * have not been cleared.
 84  * <p>
 85  * This function does not require communication with
 86  * the browser.
 87  * </p>
 88  * @function
 89  * @returns an array of {@link Breakpoint}'s
 90  */
 91 CompilationUnit.prototype.getBreakpoints = function() {
 92 	// return a copy of scripts so the master copy is not corrupted
 93 	var bps = [];
 94 	for ( var i = 0; i < this.breakpoints.length; i++) {
 95 		bps.push(this.breakpoints[i]);
 96 	}
 97 	return bps;
 98 };

136 /**
137  * Adds the specified breakpoint to this compilation unit's collection of breakpoints.
138  * Implementation should call this method when a breakpoint is created in a compilation
139  * unit.
140  * 
141  * @param breakpoint the breakpoint that was created
142  * @function
143  */
144 CompilationUnit.prototype._addBreakpoint = function(breakpoint) {
145 	this.breakpoints.push(breakpoint);
146 };
147 
148 /**
149  * Removes the specified breakpoint from this compilation unit's collection of breakpoints.
150  * Implementation should call this method when a breakpoint is cleared from a compilation
151  * unit.
152  * 
153  * @param breakpoint the breakpoint that was removed
154  * @function
155  */
156 CompilationUnit.prototype._removeBreakpoint = function(breakpoint) {
157 	for ( var i = 0; i < this.breakpoints.length; i++) {
158 		if (this.breakpoints[i] == breakpoint) {
159 			this.breakpoints.splice(i, 1);
160 			return;
161 		}
162 	}
163 };
