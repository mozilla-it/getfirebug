 34 /**
 35  * Describes an instance of an array object in a JavaScript program.
 36  * 
 37  * @constructor
 38  * @param id unique object identifier (number)
 39  * @param length of the array
 40  * @type ArrayReference
 41  * @augments ObjectReference
 42  * @return a new {@link ArrayReference}
 43  * @version 1.0
 44  */
 45 function ArrayReference(id, length) {
 46 	ObjectReference.call(this, "array", id);
 47 	this.length = length;
 48 }
 49 
 50 /**
 51  * Subclass of {@link ObjectReference}
 52  */
 53 ArrayReference.prototype = subclass(ObjectReference.prototype);
 54 
 55 /**
 56  * Returns the length of this array.
 57  * <p>
 58  * This function does not require communication with
 59  * the browser.
 60  * </p>
 61  * @function
 62  * @returns the length of this array.
 63  */
 64 ArrayReference.prototype.getLength = function() {
 65 	return this.length;
 66 };
