 34 /**
 35  * Describes a root context in a browser - the content that has been served up
 36  * and is being rendered for a location (URL) that has been navigated to. 
 37  * 
 38  * @constructor
 39  * @param id unique context identifier, a {@link String} that cannot be <code>null</code>
 40  * @param url the URL associated with this context
 41  * @param browser the browser that contains the context
 42  * @type BrowserContext
 43  * @return a new {@link BrowserContext}
 44  * @version 1.0
 45  */
 46 function BrowserContext(id, url, browser) {
 47 	this.id = id;
 48 	this.url = url;
 49 	this.browser = browser;
 50 	this.is_destroyed = false;
 51 	this.is_loaded = false;
 52 	this.compilationUnits = {}; // map of URL to compilation unit
 53 }
 54

 57 /**
 58  * Returns the unique identifier of this context.
 59  * <p>
 60  * This function does not require communication with
 61  * the browser.
 62  * </p>
 63  * @function
 64  * @returns context identifier as a {@link String}
 65  */
 66 BrowserContext.prototype.getId = function() {
 67 	return this.id;
 68 };

 96 /**
 97  * Returns whether this browser context currently exists. Returns <code>false</code>
 98  * if this context has been destroyed.
 99  * <p>
100  * This function does not require communication with
101  * the browser.
102  * </p>
103  * @function
104  * @returns a boolean indicating whether this context currently exists
105  */
106 BrowserContext.prototype.exists = function() {
107 	return !this.is_destroyed;
108 };

166 /**
167  * Notification this context has been destroyed. Clients should not call
168  * this function. This function is called by the {@link Browser} implementation
169  * of _contextDestroyed(..). Clients should call Browser._contextDestroyed(...)
170  * when a context is destroyed.
171  * 
172  * @function
173  */
174 BrowserContext.prototype._destroyed = function() {
175 	this.is_destroyed = true;
176 }
177 
178 /**
179  * Notification this context has been destroyed. Clients should not call
180  * this function. This function is called by the {@link Browser} implementation
181  * of _contextLoaded(..). Clients should call Browser._contextLoaded(...)
182  * when a context has completed loading.
183  * 
184  * @function
185  */
186 BrowserContext.prototype._loaded = function() {
187 	this.is_loaded = true;
188 }
189 
190 /**
191  * Adds the given compilation unit to the collection of compilation units in this execution context.
192  * Sends 'onScript' notification. Subclasses should call the method when a script has been
193  * created/added in the context. It should only be called once per script. Has no effect if
194  * a script with an identical URL has already been added.
195  * 
196  * @function
197  * @param compilationUnit a {@link CompilationUnit}
198  */
199 BrowserContext.prototype._addCompilationUnit = function(compilationUnit) {
200 	if (!this.compilationUnits[compilationUnit.getURL()]) {
201 		this.compilationUnits[compilationUnit.getURL()] = compilationUnit;
202 		this.getBrowser()._dispatch("onScript", [compilationUnit]);
203 	}
204 };
205 
206 /**
207  * Returns a copy of the compilation units known to this execution context in an array.
208  * 
209  * @function
210  * @returns array of {@link CompilationUnit}
211  */
212 BrowserContext.prototype._getCompilationUnits = function() {
213 	var copyScripts = [];
214 	for (var url in this.compilationUnits) {
215 		copyScripts.push(this.compilationUnits[url]);
216 	}
217 	return copyScripts;
218 };
