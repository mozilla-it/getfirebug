 31     initializeCommandLineIfNeeded: function (context, win)
 32     {
 33         if (!context || !win)
 34             return;
 35 
 36       // The command-line requires that the console has been initialized first,
 37       // so make sure that's so.  This call should have no effect if the console
 38       // is already initialized.
 39       var consoleIsReady = Firebug.Console.isReadyElsePreparing(context, win);
 40 
 41       // Make sure the command-line is initialized.  This call should have no
 42       // effect if the command-line is already initialized.
 43       var commandLineIsReady = Firebug.CommandLine.isReadyElsePreparing(context, win);
 44 
 45       if (FBTrace.DBG_COMMANDLINE)
 46           FBTrace.sysout("commandLine.initializeCommandLineIfNeeded console ready: " +
 47             consoleIsReady + " commandLine ready: " + commandLineIsReady);
 48     },

 50     // returns user-level wrapped object I guess.
 51     evaluate: function(expr, context, thisValue, targetWindow, successConsoleFunction, exceptionFunction)
 52     {
 53         if (!context)
 54             return;
 55 
 56         try
 57         {
 58             var result = null;
 59             var debuggerState = Firebug.Debugger.beginInternalOperation();
 60 
 61             if (this.isSandbox(context))
 62                 result = this.evaluateInSandbox(expr, context, thisValue, targetWindow, successConsoleFunction, exceptionFunction);
 63             else if (Firebug.Debugger.hasValidStack(context))
 64                 result = this.evaluateInDebugFrame(expr, context, thisValue, targetWindow,  successConsoleFunction, exceptionFunction);
 65             else
 66                 result = this.evaluateByEventPassing(expr, context, thisValue, targetWindow, successConsoleFunction, exceptionFunction);
 67 
 68             context.invalidatePanels('dom', 'html');
 69         }
 70         catch (exc)  // XXX jjb, I don't expect this to be taken, the try here is for the finally
 71         {
 72             if (FBTrace.DBG_ERRORS || FBTrace.DBG_COMMANDLINE)
 73                 FBTrace.sysout("commandLine.evaluate with context.stopped:" + context.stopped +
 74                     " EXCEPTION " + exc, exc);
 75         }
 76         finally
 77         {
 78             Firebug.Debugger.endInternalOperation(debuggerState);
 79         }
 80 
 81         return result;
 82     },

 84     evaluateByEventPassing: function(expr, context, thisValue, targetWindow, successConsoleFunction, exceptionFunction)
 85     {
 86         var win = targetWindow ? targetWindow : (context.baseWindow ? context.baseWindow : context.window);
 87 
 88         if (!win)
 89         {
 90             if (FBTrace.DBG_ERRORS || FBTrace.DBG_COMMANDLINE)
 91                 FBTrace.sysout("commandLine.evaluateByEventPassing: no targetWindow!\n");
 92             return;
 93         }
 94 
 95         // We're going to use some command-line facilities, but it may not have initialized yet.
 96         this.initializeCommandLineIfNeeded(context, win);
 97 
 98         // Make sure the command line script is attached.
 99         var attached = win.document.getUserData("firebug-CommandLineAttached");
100         if (!attached)
101         {
102             FBTrace.sysout("commandLine: document does not have command line attached " +
103                 "its too early for command line "+safeGetWindowLocation(win), document);
104 
105             if (isXMLPrettyPrint(context, win))
106             {
107                 var msg = $STR("commandline.disabledForXMLDocs");
108                 var row = Firebug.Console.logFormatted([msg], context, "warn", true);
109                 var objectBox = row.querySelector(".objectBox");
110 
111                 // Log a message with a clickable link that can be used to enable
112                 // the command line - but the page will switch into HTML. The listener
113                 // passed into the function is called when the user clicks the link.
114                 FirebugReps.Description.render(msg, objectBox, bind(function()
115                 {
116                     // Reset the flag that protect script injection into the page.
117                     context.isXMLPrettyPrint = false;
118 
119                     // Now inject the command line.
120                     Firebug.CommandLine.initializeCommandLineIfNeeded(context, win);
121                 }, this));
122             }
123             else
124             {
125                 Firebug.Console.logFormatted(["Firebug cannot find firebug-CommandLineAttached " +
126                     "document.getUserData , its too early for command line",
127                      win], context, "error", true);
128             }
129             return;
130         }
131 
132         var event = document.createEvent("Events");
133         event.initEvent("firebugCommandLine", true, false);
134         win.document.setUserData("firebug-methodName", "evaluate", null);
135 
136         expr = expr.toString();
137         expr = "with(_FirebugCommandLine){" + expr + "\n};";
138         win.document.setUserData("firebug-expr", expr, null);
139 
140         var consoleHandler = Firebug.Console.injector.getConsoleHandler(context, win);
141 
142         if (!consoleHandler)
143         {
144             FBTrace.sysout("commandLine evaluateByEventPassing no consoleHandler "+safeGetWindowLocation(win));
145             return;
146         }
147 
148         if (successConsoleFunction)
149         {
150             consoleHandler.setEvaluatedCallback( function useConsoleFunction(result)
151             {
152                 if (result === "_firebugIgnore")
153                     return;
154                 successConsoleFunction(result, context);  // result will be pass thru this function
155             });
156         }
157 
158         if (exceptionFunction)
159         {
160             consoleHandler.setEvaluateErrorCallback(function useExceptionFunction(result)
161             {
162                 exceptionFunction(result, context, "errorMessage");
163             });
164         }
165         else
166         {
167             consoleHandler.setEvaluateErrorCallback(function useErrorFunction(result)
168             {
169                 if (result)
170                 {
171                     var m = reCmdSource.exec(result.source);
172                     if (m && m.length > 0)
173                         result.source = m[1];
174                 }
175 
176                 Firebug.Console.logFormatted([result], context, "error", true);
177             });
178         }
179 
180         if (FBTrace.DBG_COMMANDLINE)
181             FBTrace.sysout("commandLine.evaluateByEventPassing \'"+expr+"\' using consoleHandler:", consoleHandler);
182         try
183         {
184             win.document.dispatchEvent(event);
185         }
186         catch(exc)
187         {
188             if (FBTrace.DBG_COMMANDLINE || FBTrace.DBG_ERRORS)
189                 FBTrace.sysout("commandLine.evaluateByEventPassing dispatchEvent FAILS "+exc, {exc:exc, event:event});
190         }
191 
192         if (FBTrace.DBG_COMMANDLINE)
193             FBTrace.sysout("commandLine.evaluateByEventPassing return after firebugCommandLine event:", event);
194     },

196     evaluateInDebugFrame: function(expr, context, thisValue, targetWindow,  successConsoleFunction, exceptionFunction)
197     {
198         var result = null;
199 
200         // targetWindow may be frame in HTML
201         var win = targetWindow ? targetWindow : (context.baseWindow ? context.baseWindow : context.window);
202 
203         if (!context.commandLineAPI)
204             context.commandLineAPI = new FirebugCommandLineAPI(context);
205 
206         var htmlPanel = context.getPanel("html", true);
207         var scope = {
208             api       : context.commandLineAPI,
209             vars      : htmlPanel?htmlPanel.getInspectorVars():null,
210             thisValue : thisValue
211         };
212 
213         try
214         {
215             result = Firebug.Debugger.evaluate(expr, context, scope);
216             successConsoleFunction(result, context);  // result will be pass thru this function
217         }
218         catch (e)
219         {
220             exceptionFunction(e, context);
221         }
222         return result;

328     acceptCompletionOrReturnIt: function(context)
329     {
330         var commandLine = this.getCommandLine(context);
331         var completionBox = this.getCompletionBox();
332         if (completionBox.value.length === 0 || commandLine.value.length === completionBox.value.length) // we have nothing to complete
333             return this.autoCompleter.getVerifiedText(commandLine);
334 
335         this.autoCompleter.acceptCompletionInTextBox(commandLine, completionBox);
336         return ""; // next time we will return text
337     },

339     enter: function(context, command)
340     {
341         var expr = command ? command : this.acceptCompletionOrReturnIt(context);
342         if (expr == "")
343             return;
344 
345         var mozJSEnabled = Firebug.getPref("javascript", "enabled");
346         if (mozJSEnabled)
347         {
348             if (!Firebug.largeCommandLine || context.panelName != "console")
349             {
350                 this.clear(context);
351                 Firebug.Console.log(commandPrefix + " " + expr, context, "command", FirebugReps.Text);
352             }
353             else
354             {
355                 var shortExpr = cropString(stripNewLines(expr), 100);
356                 Firebug.Console.log(commandPrefix + " " + shortExpr, context, "command", FirebugReps.Text);
357             }
358 
359             this.commandHistory.appendToHistory(expr);
360 
361             var noscript = getNoScript();
362             if (noscript)
363             {
364                 var noScriptURI = noscript.getSite(Firebug.chrome.getCurrentURI().spec);
365                 if (noScriptURI)
366                     noScriptURI = (noscript.jsEnabled || noscript.isJSEnabled(noScriptURI)) ? null : noScriptURI;
367             }
368 
369             if (noscript && noScriptURI)
370                 noscript.setJSEnabled(noScriptURI, true);
371 
372             var goodOrBad = FBL.bind(Firebug.Console.log, Firebug.Console);
373             this.evaluate(expr, context, null, null, goodOrBad, goodOrBad);
374 
375             if (noscript && noScriptURI)
376                 noscript.setJSEnabled(noScriptURI, false);
377 
378             this.autoCompleter.reset();
379         }
380         else
381         {
382             Firebug.Console.log($STR("console.JSDisabledInFirefoxPrefs"), context, "info");
383         }
384     },

508     {
509         if (Firebug.currentContext && Firebug.currentContext.panelName != "console")

512         collapse(chrome.$("fbCommandBox"), multiLine);
513         collapse(chrome.$("fbPanelSplitter"), !multiLine);
514         collapse(chrome.$("fbSidePanelDeck"), !multiLine);

517             chrome.$("fbSidePanelDeck").selectedPanel = chrome.$("fbLargeCommandBox");

519         var commandLineSmall = this.getCommandLineSmall();
520         var commandLineLarge = this.getCommandLineLarge();
521

523         {
524             commandLineSmall.value = commandLineLarge.value;
525             return;
526         }
527 
528         if (Firebug.currentContext)

531

533                 commandLineLarge.value = cleanIndentation(Firebug.currentContext.commandLineText);

536         }

679     // called by users of command line, currently:
680     // 1) Console on focus command line, 2) Watch onfocus, and 3) debugger loadedContext if watches exist
681     isReadyElsePreparing: function(context, win)
682     {
683         if (FBTrace.DBG_COMMANDLINE)
684             FBTrace.sysout("commandLine.isReadyElsePreparing ", context);
685 
686         if (this.isSandbox(context))
687             return;
688 
689         if (isXMLPrettyPrint(context, win))
690             return false;
691 
692         if (win)
693         {
694             Firebug.CommandLine.injector.attachCommandLine(context, win);
695         }
696         else
697         {
698             Firebug.CommandLine.injector.attachCommandLine(context, context.window);
699             for (var i = 0; i < context.windows.length; i++)
700                 Firebug.CommandLine.injector.attachCommandLine(context, context.windows[i]);
701         }
702 
703         if (!context.window.wrappedJSObject)  // we need to test for the wrappedJSObject because _FirebugCommandLine is evaluated into the page
704         {
705             FBTrace.sysout("context.window with no wrappedJSObject!", context.window);
706             return false;
707         }

1002 function getExpressionOffset(command, offset)
1003 {
1004     // XXXjoe This is kind of a poor-man's JavaScript parser - trying
1005     // to find the start of the expression that the cursor is inside.
1006     // Not 100% fool proof, but hey...
1007 
1008     var bracketCount = 0;
1009 
1010     var start = command.length;
1011     while (start --> 0)
1012     {
1013         var c = command[start];
1014         if (reOpenBracket.test(c))
1015         {
1016             if (bracketCount)
1017                 --bracketCount;
1018             else
1019                 break;
1020         }
1021         else if (reCloseBracket.test(c))
1022         {
1023             if (bracketCount == 0 && command[start + 1] != '.')
1024                 break;
1025             else
1026                 ++bracketCount;
1027         }
1028         else if (bracketCount == 0)
1029         {
1030             if (!/[a-zA-Z0-9$_.]/.test(c))
1031                 break;
1032         }
1033     }
1034 
1035     return start + 1;
1036 }

1047 function autoCompleteEval(preExpr, expr, postExpr, context)
1048 {
1049     var completions;
1050 
1051     try
1052     {
1053         if (preExpr)
1054         {
1055             // Remove the trailing dot (if there is one)
1056             var lastDot = preExpr.lastIndexOf(".");
1057             if (lastDot != -1)
1058                 preExpr = preExpr.substr(0, lastDot);
1059 
1060             var self = this;
1061             self.complete = [];
1062             Firebug.CommandLine.evaluate(preExpr, context, context.thisValue, null,
1063                 function found(result, context)
1064                 {
1065                     if (FBTrace.DBG_COMMANDLINE)
1066                         FBTrace.sysout("commandLine.autoCompleteEval \'"+preExpr+"\' found result", result);
1067 
1068                     if (typeof result === 'string')
1069                         self.complete = [];
1070                     else if (FirebugReps.Arr.isArray(result))
1071                         self.complete = nonNumericKeys(result).sort();
1072                     else
1073                         self.complete = keys(result).sort();
1074                 },
1075                 function failed(result, context)
1076                 {
1077                     if (FBTrace.DBG_COMMANDLINE)
1078                         FBTrace.sysout("commandLine.autoCompleteEval \'"+preExpr+"\' failed result", result);
1079 
1080                     self.complete = [];
1081                 }
1082             );
1083 
1084             if (lastDot !== -1) // if we had no dot, add a keyword that matches exactly
1085                 addMatchingKeyword(preExpr, self.complete);
1086 
1087             return self.complete;
1088         }
1089         else
1090         {
1091             if (context.stopped)
1092                 return Firebug.Debugger.getCurrentFrameKeys(context);
1093 
1094             // Cross window type pseudo-comparison
1095             var innerWindow = context.window.wrappedJSObject;
1096             if (innerWindow && innerWindow.Window && innerWindow.constructor.toString() === innerWindow.Window.toString())
1097                 completions =  keys(context.window.wrappedJSObject);  // return is safe
1098             else  // hopefull sandbox in Chromebug
1099                 completions = keys(context.global);
1100 
1101             addMatchingKeyword(expr, completions);
1102 
1103             return completions.sort();
1104         }
1105     }
1106     catch (exc)
1107     {
1108         if (FBTrace.DBG_ERRORS || FBTrace.DBG_COMMANDLINE)
1109             FBTrace.sysout("commandLine.autoCompleteEval FAILED", exc);
1110         return [];
1111     }

1117     // Use only string props
1118     if (typeof(value) != "string")

1550 function CommandLineHandler(context, win)
1551 {
1552     this.handleEvent = function(event)  // win is the window the handler is bound into
1553     {
1554         context.baseWindow = context.baseWindow || context.window;
1555         this.api = new FirebugCommandLineAPI(context);
1556 
1557         if (FBTrace.DBG_COMMANDLINE)
1558             FBTrace.sysout("commandLine.handleEvent('firebugExecuteCommand') event in context.baseWindow "+
1559                 context.baseWindow.location, event);
1560 
1561         // Appends variables into the api.
1562         var htmlPanel = context.getPanel("html", true);
1563         var vars = htmlPanel ? htmlPanel.getInspectorVars():null;
1564 
1565         for (var prop in vars)
1566         {
1567             function createHandler(p) {
1568                 return function() {
1569                     if (FBTrace.DBG_COMMANDLINE)
1570                         FBTrace.sysout("commandLine.getInspectorHistory: " + p, vars);
1571                     return FBL.unwrapObject(vars[p]);
1572                 }
1573             }
1574             this.api[prop] = createHandler(prop);  // XXXjjb should these be removed?
1575         }
1576 
1577         if (!Firebug.CommandLine.CommandHandler.handle(event, this.api, win))
1578         {
1579             var methodName = win.document.getUserData("firebug-methodName");
1580             Firebug.Console.log($STRF("commandline.MethodNotSupported", [methodName]));
1581         }
1582 
1583         if (FBTrace.DBG_COMMANDLINE)
1584             FBTrace.sysout("commandLine.handleEvent() "+win.document.getUserData("firebug-methodName")+
1585                 " context.baseWindow: "+(context.baseWindow?context.baseWindow.location:"no basewindow"),
1586                 context.baseWindow);
1587     };
