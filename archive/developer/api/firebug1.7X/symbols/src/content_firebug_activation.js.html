 20 /**
 21  * @module Implements Firebug activation logic.
 22  *
 23  * 1) Part of the logic is based on annotation service (see components/firebug-annotations.js)
 24  *    in order to remember whether Firebug is activated for given site or not.
 25  *    If there is "firebugged.showFirebug" annotation for a given site Firbug is activated.
 26  *    If there is "firebugged.closed" annotation for a given site Firbug is not activated.
 27  *
 28  * 2) Other part is based on extensions.firebug.allPagesActivation option. This option
 29  *    can be set to the following values:
 30  *    none: The option isn't used (default value)
 31  *    on:   Firebug is activated for all URLs.
 32  *    off:  Firebug is never activated.
 33  *
 34  *    This logic has higher priority over the URL annotations.
 35  *    If "off" options is selected, all existing URL annotations are removed.
 36  */
 37 Firebug.Activation = extend(Firebug.Module,
 38 {
 39     dispatchName: "activation",
 40 
 41     initializeUI: function()  // called once
 42     {
 43         Firebug.Module.initializeUI.apply(this, arguments);
 44 
 45         Firebug.TabWatcher.addListener(this.TabWatcherListener);
 46 
 47         // The "off" option is removed so make sure to convert previsous prev value
 48         // into "none" if necessary.
 49         if (Firebug.allPagesActivation == "off")
 50             Firebug.allPagesActivation = "none";
 51 
 52         // Update option menu item.
 53         this.updateAllPagesActivation();
 54     },
 55 
 56     getAnnotationService: function()
 57     {
 58         if (!this.annotationSvc)
 59         {
 60             // Import annotation service.
 61             Components.utils.import("resource://firebug/firebug-annotations.js");
 62             this.annotationSvc = annotationService;
 63         }
 64         return this.annotationSvc;
 65     },
 66 
 67     shutdown: function()
 68     {
 69         Firebug.Module.shutdown.apply(this, arguments);
 70 
 71         Firebug.TabWatcher.removeListener(this.TabWatcherListener);
 72 
 73         this.getAnnotationService().flush();
 74     },
 75 
 76     convertToURIKey: function(url, sameOrigin)  // process the URL to canonicalize it. Need not be reversible.
 77     {
 78         // Remove fragment, it shouldn't have any impact on the activation.
 79         url = url.replace(/#.*/, "");
 80 
 81         var uri = makeURI(normalizeURL(url));
 82 
 83         if (Firebug.filterSystemURLs && isSystemURL(url))
 84             return uri;
 85 
 86         if (url == "about:blank")  // avoid exceptions.
 87             return uri;
 88 
 89         if (uri && sameOrigin)
 90         {
 91             try
 92             {
 93                 var prePath = uri.prePath; // returns the string before the path (such as "scheme://user:password@host:port").
 94                 var shortURI = makeURI(prePath);
 95                 if (!shortURI)
 96                     return uri;
 97 
 98                 // annoying "about" URIs throw if you access .host
 99                 if (shortURI.scheme === "about")
100                     return shortURI;
101 
102                 if (shortURI.scheme === "file")
103                     return shortURI;
104 
105                 var host = shortURI.host;
106                 if (host)
107                 {
108                     // Slice the subdomain (if any) from the URL so, activateSameOrigin works for
109                     // domains (including TLD domains). So we want:
110                     // 1) www.google.com -> google.com
111                     // 2) www.stuff.co.nz -> stuff.co.nz
112                     // 3) getfirebug.com -> getfirebug.com
113                     // 4) xxxHonza: what about: mail.cn.mozilla.com -> mozilla.com ?
114                     var levels = host.split('.');
115                     if (levels.length > 2)
116                         levels = levels.slice(1);
117                     shortURI.host = levels.join('.');
118                     return shortURI;
119                 }
120             }
121             catch (exc)
122             {
123                 if (FBTrace.DBG_ERRORS)
124                     FBTrace.sysout("activation.convertToURIKey returning full URI, activateSameOrigin FAILS for shortURI "+shortURI+" because: "+exc, exc);
125                 return uri;
126             }
127         }
128         return uri;
129     },
130 
131     shouldCreateContext: function(browser, url, userCommands)  // true if the Places annotation the URI "firebugged"
132     {
133         if (FBTrace.DBG_ACTIVATION)
134             FBTrace.sysout("shouldCreateContext allPagesActivation " + Firebug.allPagesActivation);
135 
136         if (Firebug.allPagesActivation == "on")
137             return true;
138 
139         if (Firebug.filterSystemURLs && isSystemURL(url)) // if about:blank gets thru, 1483 fails
140             return false;
141 
142         if (userCommands)
143             return true;
144 
145         if (browser.showFirebug && url.substr(0, 8) === "wyciwyg:")  // document.open on a firebugged page
146             return true;
147 
148         try
149         {
150             var uri = this.convertToURIKey(url, Firebug.activateSameOrigin);
151             if (!uri)
152                 return false;
153 
154             var hasAnnotation = this.getAnnotationService().pageHasAnnotation(uri);
155 
156             if (FBTrace.DBG_ACTIVATION)
157                 FBTrace.sysout("shouldCreateContext hasAnnotation "+hasAnnotation +
158                     " for "+uri.spec+" in "+browser.contentWindow.location +
159                     " using activateSameOrigin: "+Firebug.activateSameOrigin);
160 
161             // Annotated so, return the value.
162             if (hasAnnotation)
163                 return this.checkAnnotation(browser, uri);
164 
165             if (browser.FirebugLink) // then Firebug.TabWatcher found a connection
166             {
167                 var dst = browser.FirebugLink.dst;
168                 var dstURI = this.convertToURIKey(dst.spec, Firebug.activateSameOrigin);
169                 if (FBTrace.DBG_ACTIVATION)
170                     FBTrace.sysout("shouldCreateContext found FirebugLink pointing to " +
171                         dstURI.spec, browser.FirebugLink);
172 
173                 if (dstURI && dstURI.equals(uri)) // and it matches us now
174                 {
175                     var srcURI = this.convertToURIKey(browser.FirebugLink.src.spec, Firebug.activateSameOrigin);
176                     if (srcURI)
177                     {
178                         if (FBTrace.DBG_ACTIVATION)
179                             FBTrace.sysout("shouldCreateContext found FirebugLink pointing from " +
180                                 srcURI.spec, browser.FirebugLink);
181 
182                         if (srcURI.schemeIs("file") || (dstURI.host == srcURI.host) ) // and it's on the same domain
183                         {
184                             hasAnnotation = this.getAnnotationService().pageHasAnnotation(srcURI);
185                             if (hasAnnotation) // and the source page was annotated.
186                             {
187                                 var srcShow = this.checkAnnotation(browser, srcURI);
188                                 if (srcShow)  // and the source annotation said show it
189                                     this.watchBrowser(browser);  // so we show dst as well.
190                                 return srcShow;
191                             }
192                         }
193                     }
194                 }
195                 else
196                 {
197                     if (FBTrace.DBG_ACTIVATION)
198                         FBTrace.sysout("shouldCreateContext FirebugLink does not match "+uri.spec, browser.FirebugLink);
199                 }
200             }
201             else if (browser.contentWindow.opener)
202             {
203                 var openerContext = Firebug.TabWatcher.getContextByWindow(browser.contentWindow.opener);
204 
205                 if (FBTrace.DBG_ACTIVATION)
206                     FBTrace.sysout("shouldCreateContext opener found, has "+
207                         (openerContext?"a ":"no ")+" context: "+
208                         browser.contentWindow.opener.location);
209 
210                 if (openerContext)
211                     return true;  // popup windows of Firebugged windows are Firebugged
212             }
213 
214             return false;   // don't createContext
215         }
216         catch (exc)
217         {
218             if (FBTrace.DBG_ERRORS)
219                 FBTrace.sysout("pageHasAnnoation FAILS for url: "+url+" which gave uri "+(uri?uri.spec:"null"), exc);
220         }
221     },
222 
223     checkAnnotation: function(browser, uri)
224     {
225         var annotation = this.getAnnotationService().getPageAnnotation(uri);
226 
227         if (FBTrace.DBG_ACTIVATION)
228             FBTrace.sysout("shouldCreateContext read back annotation "+annotation+" for uri "+uri.spec);
229 
230         // then the user closed Firebug on this page last time
231         if ((Firebug.allPagesActivation != "on") && (annotation.indexOf("closed") > 0))
232             return false; // annotated as 'closed', don't create
233         else
234             return true;    // annotated, createContext
235     },
236 
237     shouldShowContext: function(context)
238     {
239         return this.shouldCreateContext(context.browser, context.getWindowLocation().toString());
240     },
241 
242     watchBrowser: function(browser)  // Firebug is opened in browser
243     {
244         var annotation = "firebugged.showFirebug";
245         this.setPageAnnotation(browser.currentURI.spec, annotation);
246     },
247 
248     unwatchBrowser: function(browser, userCommands)  // Firebug closes in browser
249     {
250         var uri = browser.currentURI.spec;
251         if (userCommands)  // then mark to not open virally.
252             this.setPageAnnotation(uri, "firebugged.closed");
253         else
254             this.removePageAnnotation(uri); // unmark this URI
255     },
256 
257     clearAnnotations: function()
258     {
259         this.getAnnotationService().clear();
260     },
261 
262     setPageAnnotation: function(currentURI, annotation)
263     {
264         if (privateBrowsingEnabled)
265         {
266             Firebug.Console.logFormatted(
267                 [$STR("firebug.activation.privateBrowsingMode")],
268                 Firebug.currentContext, "info");
269             Firebug.chrome.selectPanel('console');
270             return;
271         }
272 
273         var uri = this.convertToURIKey(currentURI, Firebug.activateSameOrigin);
274         if (uri)
275             this.getAnnotationService().setPageAnnotation(uri, annotation);
276 
277         if (FBTrace.DBG_ACTIVATION || FBTrace.DBG_ANNOTATION)
278             FBTrace.sysout("setPageAnnotation currentURI "+currentURI+" becomes URI key "+
279                 (uri?uri.spec:"ERROR"));
280 
281         if (Firebug.activateSameOrigin)
282         {
283             uri = this.convertToURIKey(currentURI, false);
284             if (uri)
285                 this.getAnnotationService().setPageAnnotation(uri, annotation);
286 
287             if (FBTrace.DBG_ACTIVATION || FBTrace.DBG_ANNOTATION)
288                 FBTrace.sysout("setPageAnnotation with activeSameOrigin currentURI "+
289                     currentURI.spec+" becomes URI key "+(uri?uri.spec:"ERROR"));
290         }
291     },
292 
293     removePageAnnotation: function(currentURI)
294     {
295         var uri = this.convertToURIKey(currentURI, Firebug.activateSameOrigin);
296         if (uri)
297             this.getAnnotationService().removePageAnnotation(uri);
298 
299         if (Firebug.activateSameOrigin)
300         {
301             uri = this.convertToURIKey(currentURI, false);
302             if (uri)
303                 this.getAnnotationService().removePageAnnotation(uri);
304         }
305 
306         if (FBTrace.DBG_ACTIVATION)
307             FBTrace.sysout("Firebug.Activation.unwatchBrowser untagged "+uri.spec);
308     },
309 
310     iterateAnnotations: function(fn)  // stops at the first fn(uri) that returns a true value
311     {
312         var annotations = this.getAnnotationService().getAnnotations(this.annotationName);
313         for (var uri in annotations)
314         {
315             var rc = fn(uri, annotations[uri]);
316             if (rc)
317                 return rc;
318         }
319     },
320 
321     toggleAll: function(state)
322     {
323         if (FBTrace.DBG_ACTIVATION)
324             FBTrace.sysout("Firebug.toggleAll("+state+") with allPagesActivation: " +
325                 Firebug.allPagesActivation);
326 
327         if (state == "on")
328         {
329             if (Firebug.allPagesActivation == state) // then we were armed
330                 Firebug.allPagesActivation = "none";
331             else
332                 this.allOn();
333         }
334         else
335         {
336             Firebug.allPagesActivation = "none";
337         }
338 
339         Firebug.setPref(Firebug.prefDomain, "allPagesActivation", Firebug.allPagesActivation);
340         this.updateAllPagesActivation();
341     },
342 
343     updateOption: function(name, value)
344     {
345         if (name = "allPagesActivation")
346             this.updateAllPagesActivation();
347     },
348 
349     updateAllPagesActivation: function()
350     {
351         var allOn = Firebug.allPagesActivation == "on";
352 
353         var menu = $('menu_AllOn');
354         if (menu)
355             menu.setAttribute("checked", allOn);
356 
357         // don't show Off button if we are always on
358         Firebug.chrome.disableOff(allOn);
359     },
360 
361     allOn: function()
362     {
363         Firebug.allPagesActivation = "on";  // In future we always create contexts,
364         Firebug.toggleBar(true);  // and we turn on for the current page
365     }
366 });

385 /**
386  * @module Implements Panel activation logic. A Firebug panel can support activation in order
387  * to avoid performance penalties in cases when panel's features are not necessary at the moment.
388  * Such panel must be derived from {@link Firebug.ActivablePanel} and appropriate activable
389  * module from {@link Firebug.ActivableModule}
390  */
391 Firebug.PanelActivation = extend(Firebug.Module,
392 /** @lends Firebug.PanelActivation */
393 {
394     initialize: function()
395     {
396         prefs.addObserver(Firebug.prefDomain, this, false);
397     },
398 
399     shutdown: function()
400     {
401         prefs.removeObserver(Firebug.prefDomain, this, false);
402     },
403 
404     showPanel: function(browser, panel)
405     {
406         if (FBTrace.DBG_ACTIVATION)
407             FBTrace.sysout("PanelActivation.showPanel; " + (panel ? panel.name : "null panel"));
408 
409         // Panel toolbar is not displayed for disabled panels.
410         var chrome = Firebug.chrome;
411         collapse(chrome.$("fbToolbar"), !panel);
412     },
413 
414     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
415 
416     activatePanelTypes: function(panelTypes)
417     {
418         for (var p in panelTypes)
419         {
420             var panelType = panelTypes[p];
421             if (!this.isPanelActivable(panelType))
422                 continue;
423 
424             if (this.isPanelEnabled(panelType))
425                 panelType.prototype.onActivationChanged(true);
426         }
427     },
428 
429     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
430 
431     isPanelActivable: function(panelType)
432     {
433         return panelType.prototype.activable ? true : false;
434     },
435 
436     isPanelEnabled: function(panelType)
437     {
438         if (!this.isPanelActivable(panelType))
439             return true;
440 
441         // Panel "class" object is used to decide whether a panel is disabled
442         // or not (i.e.: isEnabled is a static method of Firebug.Panel)
443         return panelType ? panelType.prototype.isEnabled() : false;
444     },
445 
446     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
447     // Enable & disable methods.
448 
449     enablePanel: function(panelType)
450     {
451         this.setPanelState(panelType, true);
452     },
453 
454     disablePanel: function(panelType)
455     {
456         this.setPanelState(panelType, false);
457     },
458 
459     enableAllPanels: function()
460     {
461         for (var i = 0; i < Firebug.panelTypes.length; ++i)
462         {
463             var panelType = Firebug.panelTypes[i];
464             this.setPanelState(panelType, true);
465         }
466     },
467 
468     disableAllPanels: function()
469     {
470         for (var i = 0; i < Firebug.panelTypes.length; ++i)
471         {
472             var panelType = Firebug.panelTypes[i];
473             this.setPanelState(panelType, false);
474         }
475     },
476 
477     setPanelState: function(panelType, enable)
478     {
479         if (panelType && panelType.prototype.setEnabled)
480             panelType.prototype.setEnabled(enable);
481 
482         this.updateTab(panelType);
483     },
484 
485     updateTab: function(panelType)
486     {
487         var panelName = panelType.prototype.name;
488         var panelBar = Firebug.chrome.$("fbPanelBar1");
489         var tab = panelBar.updateTab(panelType);
490     },
491 
492     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
493     // Observer activation changes (preference)
494 
495     /**
496      * Observer for activation preferences changes.
497      */
498     observe: function(subject, topic, data)
499     {
500         if (topic != "nsPref:changed")
501             return;
502 
503         if (data.indexOf(".enableSites") == -1)
504             return;
505 
506         var parts = data.split(".");
507         if (parts.length != 4)
508             return;
509 
510         try
511         {
512             var panelName = parts[2];
513             var enable = Firebug.getPref(Firebug.prefDomain, panelName + ".enableSites");
514 
515             var panelType = Firebug.getPanelType(panelName, enable);
516             if (panelType)
517                 this.onActivationChanged(panelType, enable);
518         }
519         catch (e)
520         {
521             if (FBTrace.DBG_ACTIVATION || FBTrace.DBG_ERRORS)
522                 FBTrace.sysout("PanelActivation.observe; EXCEPTION " + e, e);
523         }
524     },
525 
526     onActivationChanged: function(panelType, enable)
527     {
528         if (!enable)
529         {
530             // Iterate all contexts and destroy all instances of the specified panel.
531             var self = this;
532             Firebug.TabWatcher.iterateContexts(function(context) {
533                 context.destroyPanel(panelType, context.persistedState);
534             });
535         }
536 
537         panelType.prototype.onActivationChanged(enable);
538 
539         Firebug.chrome.syncPanel();
540     },
541 });

546  * @domplate This template renders default content for disabled panels.

548 Firebug.DisabledPanelBox = domplate(Firebug.Rep,
549 /** @lends Firebug.DisabledPanelBox */

551     tag:
552         DIV({"class": "disabledPanelBox"},
553             H1({"class": "disabledPanelHead"},
554                 SPAN("$pageTitle")
555             ),
556             P({"class": "disabledPanelDescription", style: "margin-top: 15px;"},
557                 $STR("moduleManager.desc3"),
558                 SPAN(" "),
559                 SPAN({"class": "descImage descImage-$panelName"})
560             ),
561             A({"class": "objectLink", onclick: "$onEnable"},
562                 $STR("moduleManager.Enable")
563             )
564             /* need something here that pushes down any thing appended to the panel */
565         ),
566 
567     onEnable: function(event)
568     {
569         var panelBar = Firebug.chrome.$("fbPanelBar1");
570         var panelType = panelBar.selectedTab.panelType;
571         panelType.prototype.setEnabled(true);
572         panelBar.updateTab(panelType);
573     },
574 
575     /**
576      * Show default content saying that this panel type (specified by name) is disabled.
577      * The parent node is specified in panel.html file.
578      */
579     show: function(browser, panelName)
580     {
581         if (!panelName)
582             return;
583 
584         var panel = Firebug.getPanelType(panelName);
585         var panelTitle = Firebug.getPanelTitle(panel);
586         var args = {
587             pageTitle: $STRF("moduleManager.title", [panelTitle]),
588             panelName: panelName
589         };
590 
591         var parentNode = this.getParentNode(browser);
592         this.tag.replace(args, parentNode, this);
593         parentNode.removeAttribute("collapsed");
594     },
595 
596     /**
597      * Hide currently displayed default content.
598      */
599     hide: function(browser)
600     {
601         var parentNode = this.getParentNode(browser);
602         clearNode(parentNode);
603         parentNode.setAttribute("collapsed", true);
604     },
605 
606     getParentNode: function(browser)
607     {
608         var doc = browser.contentDocument;
609         return doc.documentElement.querySelector(".disabledPanelNode");
610     },
611 });
