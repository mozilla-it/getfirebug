  2 
  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 // Constants
  7 
  8 const Cc = Components.classes;
  9 const Ci = Components.interfaces;
 10 
 11 const versionChecker = Cc["@mozilla.org/xpcom/version-comparator;1"].getService(Ci.nsIVersionComparator);
 12 const appInfo = Cc["@mozilla.org/xre/app-info;1"].getService(Ci.nsIXULAppInfo);
 13 
 14 // ************************************************************************************************
 15 
 16 /**
 17  * This module is responsible for displaying a link to known issues:
 18  * http://getfirebug.com/knownissues
 19  */
 20 Firebug.KnownIssues = extend(Firebug.Module,
 21 /** @lends Firebug.KnownIssues */
 22 {
 23     dispatchName: "knownIssues",
 24 
 25     initialize: function()
 26     {
 27         var popupPrefName = "commandLineShowCompleterPopup";
 28         if (/Linux/.test(window.navigator.platform))
 29             Firebug.registerPreference(popupPrefName, false);
 30         else
 31             Firebug.registerPreference(popupPrefName, true);
 32 
 33         Firebug.commandLineShowCompleterPopup = Firebug.getPref(Firebug.prefDomain, popupPrefName);
 34 
 35         // In Firefox 4.0b7+ the addon-toolbar is not showing up. We'll show it once just
 36         // in case the user overlooks the new Firebug start-button in the toolbar. As soon
 37         // as users will get used to the toolbar button this could be removed completely.
 38         if (!Firebug.addonBarOpened && window.setToolbarVisibility)
 39         {
 40             var addonBar = document.getElementById("addon-bar");
 41             setToolbarVisibility(addonBar, true)
 42 
 43             // This is just one time operation.
 44             Firebug.setPref(Firebug.prefDomain, "addonBarOpened", true);
 45         }
 46 
 47         if (FBTrace.DBG_INITIALIZE)
 48             FBTrace.sysout("Set commandLineShowCompleterPopup " +
 49                 Firebug.commandLineShowCompleterPopup);
 50     },
 51 
 52     showPanel: function(browser, panel)
 53     {
 54         if (!panel)
 55             return;
 56 
 57         // Only display if the console panel is actually visible.
 58         if (panel && panel.name != "console")
 59             return;
 60 
 61         // Don't display if it was already displayed before.
 62         if (panel.panelNode.querySelector(".objectLink.knownIssues"))
 63             return;
 64 
 65         // Display the message only for Firefox 40
 66         if (versionChecker.compare(appInfo.version, "4.0*") < 0)
 67             return;
 68 
 69         try
 70         {
 71             // Log the message into the console.
 72             Firebug.Console.log([], panel.context, "info", Firebug.KnownIssuesRep);
 73         }
 74         catch (e)
 75         {
 76             if (FBTrace.DBG_ERRORS)
 77                 FBTrace.sysout("Firebug.KnownIssues.showPanel; EXCEPTION", e);
 78         }
 79     }
 80 });
 81 
 82 // ************************************************************************************************
 83 
 84 Firebug.KnownIssuesRep = domplate(Firebug.Rep,
 85 {
 86     className: "text",
 87 
 88     tag:
 89         FirebugReps.OBJECTBOX({onclick: ""},
 90             SPAN($STR("message.knownIssues40")),
 91             SPAN(" "),
 92             SPAN({"class": "objectLink knownIssues", style: "color:blue", onclick: "$onclick"},
 93                 "http://getfirebug.com/knownissues"
 94             )
 95         ),
 96 
 97     onclick: function(event)
 98     {
 99         openNewTab("http://getfirebug.com/knownissues");
100     }
101 });
102 
103 // ************************************************************************************************
104 
105 Firebug.registerModule(Firebug.KnownIssues);
106 
107 // ************************************************************************************************
108 return Firebug.KnownIssues;
109 }});
