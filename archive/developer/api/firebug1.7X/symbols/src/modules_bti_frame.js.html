 11 /**
 12  * Describes a stack frame in a JavaScript execution context.
 13  * 
 14  * @constructor
 15  * @param index frame index (0 indicates the top frame)
 16  * @param context the {@link JavaScriptContext} the frame is contained in
 17  * @param compilationUnit the {@link CompilationUnit} this frame is associated with
 18  * @param functionName the name of the function the frame is associated with - a {@link String}
 19  * @param lineNumber the source code line number the frame is associated with 
 20  * @type StackFrame
 21  * @return a new StackFrame
 22  * @version 1.0
 23  */
 24 function StackFrame(index, context, compilationUnit, functionName, lineNumber)
 25 {
 26     this.index = index;
 27     this.context = context;
 28     this.compilationUnit = compilationUnit;
 29     this.functionName = functionName;
 30     this.lineNumber = lineNumber;
 31 }

 36 /**

 38  * <p>
 39  * This function does not require communication with
 40  * the browser.
 41  * </p>
 42  * @function
