  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 // Constants
  7 
  8 // List of XML related content types.
  9 var xmlContentTypes =
 10 [
 11     "text/xml",
 12     "application/xml",
 13     "application/xhtml+xml",
 14     "application/rss+xml",
 15     "application/atom+xml",,
 16     "application/vnd.mozilla.maybe.feed",
 17     "application/rdf+xml",
 18     "application/vnd.mozilla.xul+xml"
 19 ];
 20 
 21 // ************************************************************************************************
 22 // Model implementation
 23 
 24 /**
 25  * @module Implements viewer for XML based network responses. In order to create a new
 26  * tab wihin network request detail, a listener is registered into
 27  * <code>Firebug.NetMonitor.NetInfoBody</code> object.
 28  */
 29 Firebug.XMLViewerModel = extend(Firebug.Module,
 30 {
 31     dispatchName: "xmlViewer",
 32 
 33     initialize: function()
 34     {
 35         Firebug.ActivableModule.initialize.apply(this, arguments);
 36         Firebug.NetMonitor.NetInfoBody.addListener(this);
 37     },
 38 
 39     shutdown: function()
 40     {
 41         Firebug.ActivableModule.shutdown.apply(this, arguments);
 42         Firebug.NetMonitor.NetInfoBody.removeListener(this);
 43     },
 44 
 45     /**
 46      * Check response's content-type and if it's a XML, create a new tab with XML preview.
 47      */
 48     initTabBody: function(infoBox, file)
 49     {
 50         if (FBTrace.DBG_XMLVIEWER)
 51             FBTrace.sysout("xmlviewer.initTabBody", infoBox);
 52 
 53         // If the response is XML let's display a pretty preview.
 54         if (this.isXML(safeGetContentType(file.request)))
 55         {
 56             Firebug.NetMonitor.NetInfoBody.appendTab(infoBox, "XML",
 57                 $STR("xmlviewer.tab.XML"));
 58 
 59             if (FBTrace.DBG_XMLVIEWER)
 60                 FBTrace.sysout("xmlviewer.initTabBody; XML response available");
 61         }
 62     },
 63 
 64     isXML: function(contentType)
 65     {
 66         if (!contentType)
 67             return false;
 68 
 69         // Look if the response is XML based.
 70         for (var i=0; i<xmlContentTypes.length; i++)
 71         {
 72             if (contentType.indexOf(xmlContentTypes[i]) == 0)
 73                 return true;
 74         }
 75 
 76         return false;
 77     },
 78 
 79     /**
 80      * Parse XML response and render pretty printed preview.
 81      */
 82     updateTabBody: function(infoBox, file, context)
 83     {
 84         var tab = infoBox.selectedTab;
 85         var tabBody = infoBox.getElementsByClassName("netInfoXMLText").item(0);
 86         if (!hasClass(tab, "netInfoXMLTab") || tabBody.updated)
 87             return;
 88 
 89         tabBody.updated = true;
 90 
 91         this.insertXML(tabBody, Firebug.NetMonitor.Utils.getResponseText(file, context));
 92     },
 93 
 94     insertXML: function(parentNode, text)
 95     {
 96         var parser = CCIN("@mozilla.org/xmlextras/domparser;1", "nsIDOMParser");
 97         var doc = parser.parseFromString(text, "text/xml");
 98         var root = doc.documentElement;
 99 
100         // Error handling
101         var nsURI = "http://www.mozilla.org/newlayout/xml/parsererror.xml";
102         if (root.namespaceURI == nsURI && root.nodeName == "parsererror")
103         {
104             this.ParseError.tag.replace({error: {
105                 message: root.firstChild.nodeValue,
106                 source: root.lastChild.textContent
107             }}, parentNode);
108             return;
109         }
110 
111         if (FBTrace.DBG_XMLVIEWER)
112             FBTrace.sysout("xmlviewer.updateTabBody; XML response parsed", doc);
113 
114         // Override getHidden in these templates. The parsed XML documen is
115         // hidden, but we want to display it using 'visible' styling.
116         var templates = [
117             Firebug.HTMLPanel.CompleteElement,
118             Firebug.HTMLPanel.Element,
119             Firebug.HTMLPanel.TextElement,
120             Firebug.HTMLPanel.EmptyElement,
121             Firebug.HTMLPanel.XEmptyElement,
122         ];
123 
124         var originals = [];
125         for (var i=0; i<templates.length; i++)
126         {
127             originals[i] = templates[i].getHidden;
128             templates[i].getHidden = function() {
129                 return "";
130             }
131         }
132 
133         // Generate XML preview.
134         Firebug.HTMLPanel.CompleteElement.tag.replace({object: doc.documentElement}, parentNode);
135 
136         for (var i=0; i<originals.length; i++)
137             templates[i].getHidden = originals[i];
138     }
139 });
140 
141 // ************************************************************************************************
142 // Domplate
143 
144 /**
145  * @domplate Represents a template for displaying XML parser errors. Used by
146  * <code>Firebug.XMLViewerModel</code>.
147  */
148 Firebug.XMLViewerModel.ParseError = domplate(Firebug.Rep,
149 {
150     tag:
151         DIV({"class": "xmlInfoError"},
152             DIV({"class": "xmlInfoErrorMsg"}, "$error.message"),
153             PRE({"class": "xmlInfoErrorSource"}, "$error|getSource")
154         ),
155 
156     getSource: function(error)
157     {
158         var parts = error.source.split("\n");
159         if (parts.length != 2)
160             return error.source;
161 
162         var limit = 50;
163         var column = parts[1].length;
164         if (column >= limit) {
165             parts[0] = "..." + parts[0].substr(column - limit);
166             parts[1] = "..." + parts[1].substr(column - limit);
167         }
168 
169         if (parts[0].length > 80)
170             parts[0] = parts[0].substr(0, 80) + "...";
171 
172         return parts.join("\n");
173     }
174 });
175 
176 // ************************************************************************************************
177 // Registration
178 
179 Firebug.registerModule(Firebug.XMLViewerModel);
180 return Firebug.XMLViewerModel;
181 }});
