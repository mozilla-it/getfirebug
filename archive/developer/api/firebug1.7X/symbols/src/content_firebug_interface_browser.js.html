 34 /**
 35  * Proxy to a debuggable web browser. A browser may be remote and contain one or more
 36  * JavaScript execution contexts. Each JavaScript execution context may contain one or
 37  * more compilation units. A browser provides notification to registered listeners describing
 38  * events that occur in the browser.
 39  * 
 40  * @constructor
 41  * @type Browser
 42  * @return a new Browser
 43  * @version 1.0
 44  */
 45 function Browser() {
 46 	this.contexts = {}; // map of contexts, indexed by conext ID
 47 	this.activeContext = null;
 48 	this.handlers = {}; // map of event types to array of handler functions
 49 	this.EVENT_TYPES = ["onBreak", "onConsoleDebug", "onConsoleError", "onConsoleInfo", "onConsoleLog",
 50 	                    "onConsoleWarn", "onContextCreated", "onContextDestroyed", "onContextChanged",
 51 	                    "onContextLoaded", "onInspectNode", "onResume", "onScript", "onSuspend", "onToggleBreakpoint",
 52 	                    "onDisconnect"];
 53 	this.connected = false;
 54 }

 59  * @function
 60  */

 62 	function f(){};
 63 	f.prototype = obj;
 64 	return new f();
 65 }

 69 /**
 70  * Returns the {@link BrowserContext} with the specified id or <code>null</code>
 71  * if none.
 72  * 
 73  * @function
 74  * @param id identifier of an {@link BrowserContext}
 75  * @returns the {@link BrowserContext} with the specified id or <code>null</code>
 76  * 
 77  */
 78 Browser.prototype.getBrowserContext = function(id) {
 79 	var context = this.contexts[id];
 80 	if (context) {
 81 		return context;
 82 	}
 83 	return null;
 84 };
 85 
 86 /**
 87  * Returns the root contexts being browsed. A {@link BrowserContext} represents the
 88  * content that has been served up and is being rendered for a location (URL) that
 89  * has been navigated to. 
 90  * <p>
 91  * This function does not require communication with the remote browser.
 92  * </p>
 93  * @function
 94  * @returns an array of {@link BrowserContext}'s
 95  */
 96 Browser.prototype.getBrowserContexts = function() {
 97 	var knownContexts = [];
 98 	for (var id in this.contexts) {
 99 		knownContexts.push(this.contexts[id]);
100 	}
101 	return knownContexts;
102 }

126 /**
127  * Registers a listener (function) for a specific type of event. Listener
128  * call back functions are specified in {@link BrowserEventListener}.
129  * <p>
130  * The supported event types are:
131  * <ul>
132  *   <li>onBreak</li>
133  *   <li>onConsoleDebug</li>
134  *   <li>onConsoleError</li>
135  *   <li>onConsoleInfo</li>
136  *   <li>onConsoleLog</li>
137  *   <li>onConsoleWarn</li>
138  *   <li>onContextCreated</li>
139  *   <li>onContextChanged</li>
140  *   <li>onContextDestroyed</li>
141  *   <li>onDisconnect</li>
142  *   <li>onInspectNode</li>
143  *   <li>onResume</li>
144  *   <li>onScript</li>
145  *   <li>onToggleBreakpoint</li>
146  * </ul>
147  * <ul>
148  * <li>TODO: how can clients remove (deregister) listeners?</li>
149  * </ul>
150  * </p>
151  * @function
152  * @param eventType an event type ({@link String}) listed above
153  * @param listener a listener (function) that handles the event as specified
154  *   by {@link BrowserEventListener}
155  * @exception Error if an unsupported event type is specified
156  */
157 Browser.prototype.addEventListener = function(eventType, listener) {
158 	var i = this.EVENT_TYPES.indexOf(eventType);
159 	if (i < 0) {
160 		// unsupported event type
161 		throw new Error("eventType '" + eventType + "' is not supported");
162 	}
163 	var list = this.handlers[eventType];
164 	if (!list) {
165 		list = [];
166 		this.handlers[eventType] = list;
167 	}
168 	list.push(listener);
169 };

184 /**
185  * Notification the given context has been added to this browser.
186  * Adds the context to the list of active contexts and notifies context
187  * listeners.
188  * <p>
189  * Has no effect if the context has already been created. For example,
190  * it's possible for a race condition to occur when a remote browser
191  * sends notification of a context being created before the initial set
192  * of contexts have been retrieved. In such a case, it would possible for
193  * a client to add the context twice (once for the create event, and again
194  * when retrieving the initial list of contexts).
195  * </p>
196  * @function
197  * @param context the {@link BrowserContext} that has been added
198  */
199 Browser.prototype._contextCreated = function(context) {
200 	// if already present, don't add it again
201 	var id = context.getId();
202 	if (this.contexts[id]) {
203 		return;
204 	}
205 	this.contexts[id] = context;
206 	this._dispatch("onContextCreated", [context]);	
207 };
208 
209 /**
210  * Notification the given context has been destroyed.
211  * Removes the context from the list of active contexts and notifies context
212  * listeners.
213  * <p>
214  * Has no effect if the context has already been destroyed or has not yet
215  * been retrieved from the browser. For example, it's possible for a race
216  * condition to occur when a remote browser sends notification of a context
217  * being destroyed before the initial list of contexts is retrieved from the
218  * browser. In this case an implementation could ask to destroy a context that
219  * that has not yet been reported as created.
220  * </p>
221  * 
222  * @function
223  * @param id the identifier of the {@link BrowserContext} that has been destroyed
224  */
225 Browser.prototype._contextDestroyed = function(id) {
226 	var destroyed = this.contexts[id];
227 	if (destroyed) {
228 		destroyed._destroyed();
229 		delete this.contexts[id];
230 		this._dispatch("onContextDestroyed", [destroyed]);
231 	}
232 };
233 
234 /**
235  * Notification the given context has been loaded. Notifies context listeners.
236  * 
237  * @function
238  * @param id the identifier of the {@link BrowserContext} that has been loaded
239  */
240 Browser.prototype._contextLoaded = function(id) {
241 	var loaded = this.contexts[id];
242 	if (loaded) {
243 		loaded._loaded();
244 		this._dispatch("onContextLoaded", [loaded]);
245 	}
246 };
247 
248 /**
249  * Dispatches an event notification to all registered functions for
250  * the specified event type.
251  * 
252  * @param eventType event type
253  * @param arguments arguments to be applied to handler functions
254  */
255 Browser.prototype._dispatch = function(eventType, arguments) {
256 	functions = this.handlers[eventType];
257 	if (functions) {
258 		for ( var i = 0; i < functions.length; i++) {
259 			functions[i].apply(null, arguments);
260 		}
261 	}
262 };
263 
264 /**
265  * Sets the browser context that has focus, possibly <code>null</code>.
266  * 
267  * @function
268  * @param context a {@link BrowserContext} or <code>null</code>
269  */
270 Browser.prototype._setFocusContext = function(context) {
271 	var prev = this.activeContext;
272 	this.activeContext = context;
273 	if (prev != context) {
274 		this._dispatch("onContextChanged", [prev, this.activeContext]);
275 	}
276 };

278 /**
279  * Sets whether this proxy is connected to its underlying browser.
280  * Sends 'onDisconnect' notification when the browser becomes disconnected.
281  * 
282  * @function
283  * @param connected whether this proxy is connected to its underlying browser
284  */
285 Browser.prototype._setConnected = function(connected) {
286 	var wasConnected = this.connected;
287 	this.connected = connected;
288 	if (wasConnected && !connected) {
289 		this._dispatch("onDisconnect", [this]);
290 	}
291 
292 }
