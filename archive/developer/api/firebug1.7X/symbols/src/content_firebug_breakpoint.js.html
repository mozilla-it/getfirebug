315     extractBreakpoints: function(context, breakpoints, errorBreakpoints, monitors)
316     {
317         var breakpoints = [];
318         var errorBreakpoints = [];
319         var monitors = [];
320 
321         var renamer = new SourceFileRenamer(context);
322         var self = this;
323         var Breakpoint = Firebug.Debugger.Breakpoint;
324 
325         for (var url in context.sourceFileMap)
326         {
327             fbs.enumerateBreakpoints(url, {call: function(url, line, props, scripts)
328             {
329                 if (FBTrace.DBG_BP)
330                     FBTrace.sysout("breakpoints.extractBreakpoints type: "+props.type+" in url "+
331                         url+"@"+line+" contxt "+context.getName(), props);
332 
333                 if (renamer.checkForRename(url, line, props)) // some url in this sourceFileMap has changed, we'll be back.
334                     return;
335 
336                 if (scripts)  // then this is a current (not future) breakpoint
337                 {
338                     var script = scripts[0];
339                     var analyzer = Firebug.SourceFile.getScriptAnalyzer(context, script);
340                     if (FBTrace.DBG_BP)
341                         FBTrace.sysout("breakpoints.refresh enumerateBreakpoints for script="+
342                             script.tag+(analyzer?" has analyzer":" no analyzer")+" in context "+
343                             context.getName());
344 
345                     if (analyzer)
346                         var name = analyzer.getFunctionDescription(script, context).name;
347                     else
348                         var name = FBL.guessFunctionName(url, 1, context);
349 
350                     var isFuture = false;
351                 }
352                 else
353                 {
354                     if (FBTrace.DBG_BP)
355                         FBTrace.sysout("breakpoints.refresh enumerateBreakpoints future for url@line="+
356                             url+"@"+line+"\n");
357 
358                     var isFuture = true;
359                 }
360 
361                 var source = context.sourceCache.getLine(url, line);
362                 breakpoints.push(new Breakpoint(name, url, line, !props.disabled, source, isFuture));
363             }});
364 
365             fbs.enumerateErrorBreakpoints(url, {call: function(url, line, props)
366             {
367                 if (renamer.checkForRename(url, line, props)) // some url in this sourceFileMap has changed, we'll be back.
368                     return;
369 
370                 var name = Firebug.SourceFile.guessEnclosingFunctionName(url, line, context);
371                 var source = context.sourceCache.getLine(url, line);
372                 errorBreakpoints.push(new Breakpoint(name, url, line, true, source));
373             }});
374 
375             fbs.enumerateMonitors(url, {call: function(url, line, props)
376             {
377                 if (renamer.checkForRename(url, line, props)) // some url in this sourceFileMap has changed, we'll be back.
378                     return;
379 
380                 var name = Firebug.SourceFile.guessEnclosingFunctionName(url, line, context);
381                 monitors.push(new Breakpoint(name, url, line, true, ""));
382             }});
383         }
384 
385         var result = null;
386 
387         if (renamer.needToRename(context))
388             result = this.extractBreakpoints(context); // since we renamed some sourceFiles we need to refresh the breakpoints again.
389         else
390             result = { breakpoints: breakpoints, errorBreakpoints: errorBreakpoints, monitors: monitors };
391 
392         // even if we did not rename, some bp may be dynamic
393         if (FBTrace.DBG_SOURCEFILES)
394             FBTrace.sysout("breakpoints.extractBreakpoints context.dynamicURLhasBP: "+
395                 context.dynamicURLhasBP, result);
396 
397         return result;
398     },

565 function SourceFileRenamer(context)
566 {
567     this.renamedSourceFiles = [];
568     this.context = context;
569     this.bps = [];
570 }
571 
572 SourceFileRenamer.prototype.checkForRename = function(url, line, props)
573 {
574     var sourceFile = this.context.sourceFileMap[url];
575     if (sourceFile.isEval() || sourceFile.isEvent())
576     {
577         var segs = sourceFile.href.split('/');
578         if (segs.length > 2)
579         {
580             if (segs[segs.length - 2] == "seq")
581             {
582                 this.renamedSourceFiles.push(sourceFile);
583                 this.bps.push(props);
584             }
585         }
586 
587         this.context.dynamicURLhasBP = true;  // whether not we needed to rename, the dynamic sourceFile has a bp.
588 
589         if (FBTrace.DBG_SOURCEFILES)
590             FBTrace.sysout("breakpoints.checkForRename found bp in "+sourceFile+" renamed files:",
591                 this.renamedSourceFiles);
592     }
593     else
594     {
595         if (FBTrace.DBG_SOURCEFILES)
596             FBTrace.sysout("breakpoints.checkForRename found static bp in "+sourceFile+" bp:", props);
597     }
598 
599     return (this.renamedSourceFiles.length > 0);
600 };
601 
602 SourceFileRenamer.prototype.needToRename = function(context)
603 {
604     if (this.renamedSourceFiles.length > 0)
605         this.renameSourceFiles(context);
606 
607     if (FBTrace.DBG_SOURCEFILES)
608         FBTrace.sysout("debugger renamed " + this.renamedSourceFiles.length + " sourceFiles",
609             context.sourceFileMap);
610 
611     return this.renamedSourceFiles.length;
612 }
613 
614 SourceFileRenamer.prototype.renameSourceFiles = function(context)
615 {
616     for (var i = 0; i < this.renamedSourceFiles.length; i++)
617     {
618         var sourceFile = this.renamedSourceFiles[i];
619         var bp = this.bps[i];
620 
621         var oldURL = sourceFile.href;
622         var sameType = bp.type;
623         var sameLineNo = bp.lineNo;
624 
625         var segs = oldURL.split('/');  // last is sequence #, next-last is "seq", next-next-last is kind
626         var kind = segs.splice(segs.length - 3, 3)[0];
627         var callerURL = segs.join('/');
628         if (!sourceFile.source)
629         {
630             FBTrace.sysout("breakpoint.renameSourceFiles no source for "+oldURL+" callerURL "+callerURL, sourceFile)
631             continue;
632         }
633         var newURL = Firebug.Debugger.getURLFromMD5(callerURL, sourceFile.source, kind);
634         sourceFile.href = newURL.href;
635 
636         fbs.removeBreakpoint(bp.type, oldURL, bp.lineNo);
637         delete context.sourceFileMap[oldURL];  // SourceFile delete
638 
639         if (FBTrace.DBG_SOURCEFILES)
640             FBTrace.sysout("breakpoints.renameSourceFiles type: "+bp.type, bp);
641 
642         Firebug.Debugger.watchSourceFile(context, sourceFile);
643         var newBP = fbs.addBreakpoint(sameType, sourceFile, sameLineNo, bp, Firebug.Debugger);
644 
645         var panel = context.getPanel("script", true);
646         if (panel)
647         {
648             panel.context.invalidatePanels("breakpoints");
649             panel.renameSourceBox(oldURL, newURL.href);
650         }
651         if (context.sourceCache.isCached(oldURL))
652         {
653             var lines = context.sourceCache.load(oldURL);
654             context.sourceCache.storeSplitLines(newURL.href, lines);
655             context.sourceCache.invalidate(oldURL);
656         }
657 
658         if (FBTrace.DBG_SOURCEFILES)
659             FBTrace.sysout("SourceFileRenamer renamed "+oldURL +" to "+newURL, { newBP: newBP, oldBP: bp});
660     }
661     return this.renamedSourceFiles.length;
662 }

919     {
920         // We already called hide
921         if (!this.panel)
922             return;
923 
924         // As the box shrinks you don't want text to spill
925         var guts = this.box.getElementsByClassName("balloonContent").item(0);
926         collapse(guts, true);  
927 
928         var msg = this.cause.message;
929         if (msg)
930         {
931             var self = this;
932             var delta = Math.max(20,Math.floor(self.box.clientWidth/20));
933             var interval = setInterval(function slide(event)
934             {

940                     if (self.box.parentNode)

944                         setClass(self.target, "noteInToolTip");

947                     delete self.panel;
948                 }
949                 else
950                     self.box.style.width = (self.box.clientWidth - delta)+"px";
951             }, 15);
952         }
953         else

956             delete this.panel;
957         }
