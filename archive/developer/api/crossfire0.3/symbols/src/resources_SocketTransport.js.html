 76  * @type CrossfireSocketTransport

 79 function getCrossfireServer() {
 80     if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
 81         FBTrace.sysout("getCrossfireServer");
 82 
 83     if (!_instance) {
 84         _instance = new CrossfireSocketTransport(true);
 85     }
 86     return _instance;

 89 /**
 90  * @name CrossfireSocketTransport
 91  * @description Firefox Socket Transport for remote debug protocol.
 92  * Opens a socket connection to a remote host and handles handshaking and
 93  * sending/receiving packets.
 94  * @constructor
 95  * @public
 96  * @param isServer boolean specifying whether to start the transport as client or server
 97  * @type CrossfireSocketTransport
 98  * @returns a new instance of CrossfireSocketTransport
 99  */
100 function CrossfireSocketTransport( isServer) {
101 
102     this.listeners = [];
103     this.connected = false;
104 
105     this.isServer = isServer;
106 
107     if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
108         FBTrace.sysout("Creating new CrossfireSocketTransport");
109 
110     // quit-application observer
111     var transport = this;
112     Cc["@mozilla.org/observer-service;1"].getService(Ci.nsIObserverService).addObserver({
113         observe: function(subject, topic, data)
114         {
115             if (FBTrace.DBG_CROSSFIRE_TRANSPORT) FBTrace.sysout("quit application observed");
116             transport.close();
117         }
118     }, "quit-application", false);
119 
120     return this;
121 
122 }

129     /**
130      * @name addListener
131      * @description Adds a listener to be called when the transport receives requests.
132      * The transport will pass a {@link RequestPacket} as an argument to the listener's <code>handleRequest</code> method.
133      * @function
134      * @public
135      * @memberOf CrossfireSocketTransport
136      * @param listener An object which contains a method named "handleRequest".
137      * @see Packet.js
138      */
139     addListener: function( listener) {
140         // don't push the listener again if it is already there
141         // http://code.google.com/p/fbug/issues/detail?id=3452
142         if(this.listeners.indexOf(listener) < 0) {
143             this.listeners.push(listener);
144         }

147     /**
148      * @name sendResponse
149      * @description Builds and sends a response packet. @see also Packet.js
150      * @function
151      * @public
152      * @memberOf CrossfireSocketTransport
153      * @param command The name of the command for the response.
154      * @param requestSeq Sequence number of the request that initiated the response.
155      * @param body JSON body of the response
156      * @param running boolean indicates if execution is continuing.
157      * @param success boolean indicates whether the command was successful.
158      */
159     sendResponse: function(command, requestSeq, body, running, success) {
160         if (running == null || running == undefined) running = true; // assume we are running unless explicitly told otherwise
161         success = !!(success); // convert to boolean
162         this._defer(function() { this._sendPacket(new ResponsePacket(command, requestSeq, body, running, success)); });
163     },

196     /**
197      * @name close
198      * @description Close a previously opened connection.
199      * <br><br>
200      * Fires a <code>closed</code> event.
201      * @function
202      * @public
203      * @memberOf CrossfireSocketTransport
204      */
205     close: function() {
206         this.sendEvent("closed",{});
207 
208         this._defer(function() {
209             this._notifyConnection(CROSSFIRE_STATUS.STATUS_DISCONNECTED);
210             this.connected = false;
211 
212             if (this._inputStream) {
213                 this._inputStream.close(null);
214             }
215 
216             if (this._outputStream) {
217                 this._outputStream.close();
218             }
219 
220             if (this._transport) {
221                 this._transport.close(null);
222             }
223 
224             this._destroyTransport();
225             this._buffer='';
226             // clean up the listeners
227             // http://code.google.com/p/fbug/issues/detail?id=3452
228             this.listeners = [];
229         });
230     },

233     /**
234      * @name _createTransport
235      * @description Creates a new {@link CrossfireSocketTransport} for the given host and port
236      * @function
237      * @private
238      * @memberOf CrossfireSocketTransport
239      * @param host a {@link String} value for the host
240      * @param port an {@link Integer} value for the port
241      * @type CrossfireSocketTransport
242      * @returns a new {@link CrossfireSocketTransport} for the given host and port
243      * @see https://developer.mozilla.org/en/XPCOM_Interface_Reference/nsIServerSocket
244      * @see https://developer.mozilla.org/en/nsISocketTransportService
245      */
246     _createTransport: function (host, port) {
247 
248         if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
249             FBTrace.sysout("_createTransport");
250 
251         if (this.isServer) {
252 
253             var serverSocket = Cc["@mozilla.org/network/server-socket;1"]
254                                   .createInstance(Ci.nsIServerSocket);
255 
256             serverSocket.init(port, true, -1);
257 
258             var self = this;
259 
260             serverSocket.asyncListen({
261                 QueryInterface: function(iid) {
262                     if(!iid.equals(Ci.nsISupports) && !iid.equals(Ci.nsIServerSocketListener))
263                         throw NS_ERROR_NO_INTERFACE;
264                     return this;
265                 },
266 
267                 onSocketAccepted: function( socket, transport) {
268                     self._transport = transport;
269 
270                     self._createInputStream();
271 
272                     self._createOutputStream();
273 
274                     self._notifyConnection(CROSSFIRE_STATUS.STATUS_CONNECTING);
275                     self._waitHandshake();
276                 }
277             });
278 
279             this._notifyConnection(CROSSFIRE_STATUS.STATUS_WAIT_SERVER);
280         } else {
281             var transportService = Cc["@mozilla.org/network/socket-transport-service;1"]
282                                       .getService(Ci.nsISocketTransportService);
283 
284             this._transport = transportService.createTransport(null,0, host, port, null);
285 
286             this._createInputStream();
287 
288             this._createOutputStream();
289 
290             this._sendHandshake();
291         }
292     },

294     /**
295      * @name _createInputStream
296      * @description Creates a new input stream
297      * @function
298      * @private
299      * @memberOf CrossfireSocketTransport
300      * @see https://developer.mozilla.org/en/nsScriptableInputStream
301      */
302     _createInputStream: function() {
303         if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
304             FBTrace.sysout("_createInputStream");
305 
306         this._inputStream = this._transport.openInputStream(Ci.nsITransport.OPEN_BLOCKING | Ci.nsITransport.OPEN_UNBUFFERED, 0, 0);
307 
308         this._scriptableInputStream = Cc["@mozilla.org/scriptableinputstream;1"]

310 
311         this._scriptableInputStream.init(this._inputStream);
312     },

314     /**
315      * @name _createOutputStream
316      * @description Creates an output stream
317      * @function
318      * @private
319      * @memberOf CrossfireSocketTransport
320      * @see https://developer.mozilla.org/en/nsIOutputStream
321      */
322     _createOutputStream: function() {
323         if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
324             FBTrace.sysout("_createOutputStream");
325 
326         this._outputStream = this._transport.openOutputStream(Ci.nsITransport.OPEN_BLOCKING | Ci.nsITransport.OPEN_UNBUFFERED, 0, 0);
327 
328         this._outputStreamCallback = {
329                 _packets: [],
330 
331                 addPacket: function( packet) {
332                     this._packets.push(packet);
333                 },
334 
335                 QueryInterface: function(iid) {
336                     if(!iid.equals(Ci.nsISupports) && !iid.equals(Ci.nsIOutputStreamCallback))
337                         throw NS_ERROR_NO_INTERFACE;
338                     return this;
339                 },
340 
341                 onOutputStreamReady: function( outputStream) {
342                     try {
343                         var packet = this._packets.pop();
344                         if (packet) {
345                             //while ((packet = this._packets.pop())) {
346 
347                             if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
348                                 FBTrace.sysout("onOutputStreamReady sending packet: " + packet);
349                             outputStream.write(packet.data, packet.length);
350                             outputStream.flush();
351                         }
352                     } catch( ex) {
353                         if (FBTrace.DBG_CROSSFIRE_TRANSPORT) FBTrace.sysout("onOutputStreamReady" + ex);
354                     }
355                 }
356             };
357     },

382     /**
383      * @name _defer
384      * @description Defers a call-back for the given delay time (in milliseconds)
385      * @function
386      * @private
387      * @memberOf CrossfireSocketTransport
388      * @param callback
389      * @param delay the amount of time to wait (in milliseconds)
390      * @see https://developer.mozilla.org/en/nsITimer
391      */
392     _defer: function( callback, delay) {
393         if (!delay) delay = 1;
394         var self = this;
395         var timer = Cc["@mozilla.org/timer;1"].createInstance(Ci.nsITimer);
396         timer.initWithCallback( {
397             QueryInterface: function( iid) {
398                 if(!iid.equals(Ci.nsISupports) && !iid.equals(Ci.nsITimerCallback))
399                     throw NS_ERROR_NO_INTERFACE;
400                 return this;
401             },
402             notify: function( aTimer) {
403                 callback.apply(self);
404             }
405         }, delay, timer.TYPE_ONE_SHOT);
406     },

408     /**
409      * @name _sendHandshake
410      * @description Sends the standard handshake ({@link CROSSFIRE_HANDSHAKE}) over the transport.
411      * @function
412      * @private
413      * @memberOf CrossfireSocketTransport
414      */
415     _sendHandshake: function() {
416         if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
417             FBTrace.sysout("_sendHandshake");
418 
419         var self = this;
420         this._outputStream.asyncWait( {
421             QueryInterface: function( iid) {
422                 if(!iid.equals(Ci.nsISupports) && !iid.equals(Ci.nsIOutputStreamCallback))
423                     throw NS_ERROR_NO_INTERFACE;
424                 return this;
425             },
426             onOutputStreamReady: function( outputStream) {
427                 outputStream.flush();
428                 outputStream.write(CROSSFIRE_HANDSHAKE, CROSSFIRE_HANDSHAKE.length);
429                 outputStream.flush();
430 
431                 if (self.isServer) {
432                     self.connected = true;
433                     self._notifyConnection(CROSSFIRE_STATUS.STATUS_CONNECTED_SERVER);
434                     self._waitOnPacket();
435                 } else {
436                     self._notifyConnection(CROSSFIRE_STATUS.STATUS_CONNECTING);
437                     self._waitHandshake();
438                 }
439 
440             }
441         }, 0, 0, null);
442     },

444     /**
445      * @name _waitHandshake
446      * @description Waits for a handshake acknowledgment for the given timeout interval
447      * @function
448      * @private
449      * @memberOf CrossfireSocketTransport
450      * @param timeout the amount of time to wait for a handshake acknowledgment
451      */
452     _waitHandshake: function( timeout) {
453         if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
454             FBTrace.sysout("_waitHandshake");
455 
456         this._defer(function() {
457             try {
458                 if (this._inputStream.available() == CROSSFIRE_HANDSHAKE.length) {
459                     if (this._scriptableInputStream.read(CROSSFIRE_HANDSHAKE.length) == CROSSFIRE_HANDSHAKE) {
460                         if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
461                             FBTrace.sysout("_waitHandshake read handshake string");
462                         if (this.isServer) {
463                             this._sendHandshake();
464                         } else {
465                             this._waitOnPacket();
466                             this._notifyConnection(CROSSFIRE_STATUS.STATUS_CONNECTED_CLIENT);
467                             this.connected = true;
468                         }
469                         return;
470                     }
471                 }
472                 this._waitHandshake(HANDSHAKE_RETRY);
473             } catch (e) {
474                 //this.close();
475                 if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
476                     FBTrace.sysout("_waitHandshake exception: " + e);
477 
478                 if (this.isServer) {
479                     this._waitHandshake(HANDSHAKE_RETRY);
480                 }
481             }
482         }, timeout);
483     },

485     /** 
486      * @name _sendPacket
487      * @description Sends the given packet over the underlying transport
488      * @function
489      * @private
490      * @memberOf CrossfireSocketTransport
491      * @param packet the packet to send over the underlying transport
492      */
493     _sendPacket: function( packet) {
494         if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
495             FBTrace.sysout("_sendPacket " + packet);
496         if (this._outputStreamCallback) {
497             this._outputStreamCallback.addPacket(packet);
498         }
499         if (this.connected && this._outputStream) {
500             this._outputStream.asyncWait(this._outputStreamCallback,0,0,null);
501         }
502     },

507     /**
508      * @name _waitOnPacket
509      * @description Waits for a packet to be read from the underlying transport
510      * @function
511      * @private
512      * @memberOf CrossfireSocketTransport
513      * @see https://developer.mozilla.org/en/nsIScriptableInputStream/available
514      */
515     _waitOnPacket: function() {
516         var avail, response;
517         if (!this.connected || !this._inputStream)
518             return;
519         try {
520             avail = this._inputStream.available();
521         } catch (e) {
522             if (FBTrace.DBG_CROSSFIRE_TRANSPORT) FBTrace.sysout("_waitOnPacket " + e);
523             this.close();
524             if (this.isServer) {
525                 this.open(this.host, this.port);
526             }
527         }
528         if (avail) {
529             response = this._scriptableInputStream.read(avail);
530 
531             if (FBTrace.DBG_CROSSFIRE_TRANSPORT)
532                 FBTrace.sysout("_waitOnPacket got response => " + response);
533 
534             if (response) {
535                 this._buffer += response;
536 
537                 while(this._parseBuffer()){
538                     // until nothing more is recognized
539                 };
540             }
541         }
542         if (this.connected) {
543             this._defer(function() { this._waitOnPacket();},50);
544         }
545     },

547     /**
548      * @name _parseBuffer
549      * @description Reads packets from the backing buffer
550      * @function
551      * @private
552      * @memberOf CrossfireSocketTransport
553      */
554      _parseBuffer: function(){
555         /*
556          * Buffer always has length header:
557          *   "Content-Length:" + str.length + "\r\n"
558          */
559         var block, packet, length, i, headers = {}, header,
560             headersEnd       = this._buffer.indexOf("\r\n\r\n"),
561             contentBegin     = headersEnd + 4,
562             headersRaw       = this._buffer.substring(0,headersEnd);
563 
564         headersRaw = headersRaw.split("\r\n");
565         for (i=0; i < headersRaw.length; i++)
566         {
567             header = /([\w-]+)\s*:\s*(\S+)/.exec(headersRaw[i]);
568             if (header)
569                 headers[header[1].toLowerCase()] = header[2];
570         }
571 
572         length = parseInt(headers['content-length'],10);
573 
574         if (this._buffer.length >= contentBegin + length) {
575             block = this._buffer.substr(contentBegin, length);
576             this._buffer = this._buffer.slice(contentBegin + length);
577             if (!this.isServer) {
578                 //FIXME: mcollins handle events/requests based on packet type, not server/client mode
579                 packet = new EventPacket(block);
580             } else {
581                 packet = new RequestPacket(block);
582             }
583             this._notifyListeners(packet);
584             return true;
585         }
586         return false;
587     },

589     /**
590      * @name _notifyConnection
591      * @description Notifies <code>onConnectionStatusChanged</code> listeners that the connection state has changed
592      * @function
593      * @private
594      * @memberOf CrossfireSocketTransport
595      * @param status the status of the connection
596      */
597     _notifyConnection: function( status) {
598         for (var i = 0; i < this.listeners.length; ++i) {
599             var listener = this.listeners[i];
600             try {
601                  var handler = listener["onConnectionStatusChanged"];
602                  if (handler)
603                      handler.apply(listener, [status]);
604             } catch (e) {
605                 if (FBTrace.DBG_CROSSFIRE_TRANSPORT) FBTrace.sysout("_notifyConnection " + e);
606             }
607         }
608     },
609

611     /**
612      * @name _notifyListeners
613      * @description Notifies <code>fireEvent</code> or <code>handleRequest</code> listeners based on the kind of the packet
614      * @function
615      * @private
616      * @memberOf CrossfireSocketTransport
617      * @param packet the packet that has been read 
618      */
619     _notifyListeners: function( packet) {
620         var listener, handler;
621         for (var i = 0; i < this.listeners.length; ++i) {
622             listener = this.listeners[i];
623             try {
624 
625                 if (!this.isServer) {
626                     //FIXME: mcollins handle events/requests based on packet type, not server/client mode
627                     handler = listener["fireEvent"];
628                 } else {
629                     handler = listener["handleRequest"];
630                 }
631 
632                 if (handler)
633                     handler.apply(listener, [packet]);
634 
635             } catch (e) {
636                 if (FBTrace.DBG_CROSSFIRE_TRANSPORT) FBTrace.sysout("_notifyListeners " + e);
637             }
638         }
