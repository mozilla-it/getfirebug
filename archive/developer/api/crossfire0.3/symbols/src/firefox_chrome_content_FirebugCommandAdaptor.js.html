  9     /**
 10      * @name FirebugCommandAdaptor
 11      * @constructor FirebugCommandAdaptor An object which accepts protocol
 12      * commands and performs the appropriate Firebug actions, and serializes
 13      * runtime objects from Firebug into expected formats.
 14      *
 15      * Each context in Firebug is associated with a new command adaptor
 16      * object in order to maintain state independently of other contexts.
 17      *
 18      * Arguments to commands are passed as a single JSON object in order to
 19      * allow named parameters. Supported arguments are listed for each command.
 20      */
 21     function FirebugCommandAdaptor( context) {
 22         this.context = context;
 23         this.contextId = context.Crossfire.crossfire_id;
 24         if (FBTrace.DBG_CROSSFIRE)
 25             FBTrace.sysout("CROSSFIRE Creating new FirebugCommandAdaptor for context: " + this.contextId);
 26         this.breakpoints = [];
 27         this.breakpointIds = 1;
 28         this.clearRefs();
 29     }

 38          * @name FirebugCommandAdaptor.continue
 39          * @function
 40          * @description Continue execution of javascript if suspended,
 41          * if no <code>stepaction</code> is passed, simply resumes execution.
 42          * @param {Object} args Arguments object.
 43          * @param args.stepaction <code>stepaction</code>: 'in', 'next', or 'out'.
 44          * @param args.stepcount <code>stepcount</code>: currently ignored.
 45          */
 46         "continue": function( args) {
 47         	if (FBTrace.DBG_CROSSFIRE)
 48         		FBTrace.sysout("CROSSFIRE CommandAdaptor continue");
 49         	var stepAction = null;
 50             if(args) {
 51             	stepAction = args["stepaction"];
 52             }
 53             if (stepAction == "in") {
 54                 Firebug.Debugger.stepInto(this.context);
 55             } else if (stepAction == "next") {
 56                 Firebug.Debugger.stepOver(this.context);
 57             } else if (stepAction == "out") {
 58                 Firebug.Debugger.stepOut(this.context);
 59             } else {
 60                 Firebug.Debugger.resume(this.context);
 61             }
 62             return {};
 63         },

 78          * @name FirebugCommandAdaptor.evaluate
 79          * @function
 80          * @description evaluate a Javascript expression.
 81          * If a frame argument is passed, evaluates the expression in that frame,
 82          * otherwise the expression is evaluated in the context's global scope.
 83          * @param {Object} args Arguments object.
 84          * @param {String} args.expression A string of Javascript to be evaluated.
 85          * @param {Number} args.frame Optional frame to evaluate in.
 86          */
 87         "evaluate": function( args) {
 88              if (FBTrace.DBG_CROSSFIRE)
 89                  FBTrace.sysout("CROSSFIRE CommandAdaptor evaluate");
 90 
 91             var frameNo = args["frame"];
 92             var expression = args["expression"];
 93 
 94             var frame;
 95 
 96             if (this.context.Crossfire.currentFrame) {
 97                 if (frameNo == 0) {
 98                     frame = this.context.Crossfire.currentFrame;
 99                 } else if (frameNo > 0) {
100                     frame = this.context.Crossfire.currentFrame.stack[frameNo];
101                 }
102             }
103 
104             var result = {};
105 
106             if (frame) {
107                 if (FBTrace.DBG_CROSSFIRE)
108                     FBTrace.sysout("CROSSFIRE CommandAdaptor evaluating '" + expression + "' in frame ", frame);
109 
110                 if (frame.eval(expression, "crossfire_eval_" + this.contextId, 1, result)) {
111                     result = unwrapIValue(result.value);
112                 }
113             } else {
114                 if (FBTrace.DBG_CROSSFIRE)
115                     FBTrace.sysout("CROSSFIRE CommandAdaptor evaluating '" + expression + "' in sandbox.");
116                 Firebug.CommandLine.evaluate(expression, this.context,null,null,function(r){
117                     result = r;
118                     if (FBTrace.DBG_CROSSFIRE)
119                         FBTrace.sysout("CROSSFIRE evaluating success",[expression,result]);
120                 },function(){
121                     if (FBTrace.DBG_CROSSFIRE)
122                         FBTrace.sysout("CROSSFIRE evaluating failure",expression,arguments);
123                     throw new Error("Failure to evaluate expression: " + expression);
124                 });
125             }
126 
127             return { "context_id": this.contextId, "result": result };
128         },

294         /**
295          * @name FirebugCommandAdaptor.frame
296          * @function
297          * @description Returns a frame.
298          * @param {Object} args Arguments object.
299          * @param {Number} args.number <code>number</code>: the number (index) of the requested frame.
300          * @param {Boolean} args.includeScopes <code>includeScopes</code> defaults to true.
301          */
302         "frame": function( args) {
303             var number = args["number"];
304 
305             if (FBTrace.DBG_CROSSFIRE)
306                 FBTrace.sysout("CROSSFIRE CommandAdaptor frame: number => ", number);
307 
308             var includeScopes = (typeof(args["includeScopes"]) != "undefined")  ? args["includeScopes"] : true;
309 
310             var frame;
311             if (this.context.Crossfire.currentFrame) {
312                 if (!number) {
313                     number = 0;
314                     frame = this.context.Crossfire.currentFrame;
315                 } else if (this.context.Crossfire.currentFrame.stack) {
316                     frame = this.context.Crossfire.currentFrame.stack[number-1];
317                 }
318 
319                 try {
320 
321                     var locals = {};
322                     for (var l in frame.scope) {
323                         if (l != "parent") { // ignore parent
324                             locals[l] = frame.scope[l];
325                         }
326                     }
327 
328                     if (frame.thisValue) {
329                         locals["this"] = frame.thisValue;
330                     }
331 
332                     if (includeScopes) {
333                         var scopes = (this.scopes({ "frameNumber": number })).scopes;
334                     }
335 
336                     return { "context_id": this.contextId,
337                         "index": frame.frameIndex,
338                         "func": frame.functionName,
339                         "script": frame.script,
340                         "locals": locals,
341                         "line": frame.line,
342                         "scopes": scopes };
343                 } catch (exc) {
344                     if (FBTrace.DBG_CROSSFIRE)
345                         FBTrace.sysout("CROSSFIRE exception returning frame ", exc);
346                 }
347             }
348             return false;
349         },

351         /**
352          * @name FirebugCommandAdaptor.backtrace
353          * @function
354          * @description Returns a backtrace (stacktrace) of frames.
355          * @param {Object} args Arguments object
356          * @param {Number} args.fromFrame <code>fromFrame</code>: the frame to start the trace from.
357          * @param {Number} args.toFrame <code>toFrame</code>: the frame to to stop the trace at.
358          * @param {Boolean} args.includeScopes <code>includeScopes</code> defaults to false.
359          */
360         "backtrace": function( args) {
361             if (FBTrace.DBG_CROSSFIRE)
362                 FBTrace.sysout("CROSSFIRE CommandAdaptor backtrace");
363 
364             args = args || {};
365 
366             if (this.context.Crossfire.currentFrame) {
367                 var fromFrame, toFrame;
368                 var stack = this.context.Crossfire.currentFrame.stack;
369 
370                 if (FBTrace.DBG_CROSSFIRE)
371                     FBTrace.sysout("CROSSFIRE CommandAdaptor backtrace currentFrame.stack => " + stack);
372 
373                 var includeScopes = (typeof(args["includeScopes"]) != "undefined")  ? args["includeScopes"] : true;
374 
375                 if (stack) {
376                     fromFrame = args["fromFrame"] || 0;
377                     // toFrame set to stack.length if not set, or if set to a number higher than the stack sizes
378                     toFrame = (args["toFrame"] && args["toFrame"] <= stack.length) ? args["toFrame"] : stack.length;
379                 } else {
380                     // issue 2559: if there is only one frame, stack is undefined,
381                     // but we still want to return that frame.
382                     fromFrame = 0;
383                     toFrame = 1;
384                 }
385 
386                 var frame;
387                 var frames = [];
388                 for (var i = fromFrame; i <= toFrame; i++) {
389                     frame = this.frame({ "number": i, "includeScopes": includeScopes });
390                     if (frame) {
391                         delete frame.context_id;
392                         frames.push(frame);
393                     }
394                 }
395 
396                 return {
397                     "context_id": this.contextId,
398                     "fromFrame": 0,
399                     "toFrame": frames.length-1,
400                     "totalFrames": frames.length,
401                     "frames": frames
402                 };
403             }
404             return false;
405         },

408          * @name FirebugCommandAdaptor.scope
409          * @function
410          * @description Returns a particular scope for the specified frame.
411          * @param {Object} args Arguments object.

416             if (FBTrace.DBG_CROSSFIRE)

418

420             var scopeNo = args["number"];
421             var frameNo = args["frameNumber"];
422             if (this.context.Crossfire.currentFrame) {
423                 if (scopeNo == 0) {
424                     // only return a reference to the global scope
425                     scope = this.getRef(this.context.window.wrappedJSObject);
426                 } else {
427                     var frame;
428                     if (!frameNo) {
429                         frameNo = 0;
430                         frame = this.context.Crossfire.currentFrame;
431                     } else {
432                         frame = this.context.Crossfire.currentFrame.stack[frameNo-1];
433                     }
434                     scope = frame.scope;
435                     for (var i = 0; i < scopeNo; i++) {
436                         scope = scope.parent;
437                         if (!scope) break;
438                     }
439                 }
440             } else if (scopeNo == 0) {
441               scope = this.context.window.wrappedJSObject;
442               frameNo = -1;
443             }
444 
445             if (scope) {
446                 return {
447                     "context_id": this.contextId,
448                     "index": scopeNo,
449                     "frameIndex": frameNo,
450                     "object": scope
451                 };
452             }
453

649         /**
650          * @function serialize
651          * @description prepare a javascript object to be serialized into JSON.
652          * @param {Object} obj the javascript thing to serialize
653          * @param {Boolean} incContext boolean indicates whether the context id field should be included.
654          */
655         serialize: function( obj, incContext) {
656             try {
657                 //if (FBTrace.DBG_CROSSFIRE)
658                   //  FBTrace.sysout("CROSSFIRE CommandAdaptor serialize => ", obj);
659 
660                 var type = typeof(obj);
661 
662                 var serialized = {
663                         "type": type,
664                         "value": ""
665                 }
666 
667                 if (incContext) {
668                     serialized["context_id"] = this.contextId;
669                 }
670 
671                 if (type == "object") {
672                     if (obj == null) {
673                          serialized["value"] = "null";
674                     } else if (obj.type && obj.type == "ref" && obj.handle) {
675                         // already serialized
676                         serialized = obj;
677                     } else if (obj instanceof Array) {
678                         var arr = "[";
679                         for (var i = 0; i < obj.length; i++) {
680                             arr += JSON.stringify(this.serialize(obj[i]));
681                             if (i < obj.length-1) {
682                                 arr += ',';
683                             }
684                         }
685                         serialized["value"] = arr + "]";
686                     } else {
687                         var ref = this.getRef(obj);
688                         var o = {};
689                         for (var p in obj) {
690                             try {
691                                 if (obj.hasOwnProperty(p) && !(p in ignoreVars)) {
692                                     var prop = obj[p];
693                                     if (typeof(prop) == "object") {
694                                         if (prop == null) {
695                                             o[p] = "null";
696                                         } else if (prop && prop.type && prop.type == "ref" && prop.handle) {
697                                             o[p] = prop;
698                                         } else  {
699                                             o[p] = this.getRef(prop);
700                                         }
701                                     } else if (p === obj) {
702                                         o[p] = ref;
