  4 const Cc = Components.classes;
  5 const Ci = Components.interfaces;
  6 const Cu = Components.utils;

  8 Cu.import("resource://gre/modules/XPCOMUtils.jsm");

 10 /**
 11  * @name CommandLineHandler
 12  * @constructor
 13  * @description Processes command-line arguments to connect to a remote debug host and port.
 14  * <br><br>
 15  * List of accepted command line arguments includes:
 16  * <ul>
 17  * <li><code>crossfire-server-port</code> - the port to start the crossfire server listening on</li>
 18  * <li><code>crossfire-host</code> - the name of the host computer to try and connect to, this is consulted iff <code>crossfire-server-port</code> is not present</li>
 19  * <li><code>crossfire-port</code> - the port of the host computer to try and connect to, this is consulted iff <code>crossfire-server-port</code> is not present</li>
 20  * <li><code>-load-fb-modules</code> - this flag will enable all of the Firebug modules when Firefox starts</li> 
 21  * <li><code>-no-fb-modules</code> - this flag will cause no Firebug modules to be loaded when Firefox starts regardless of any other settings</li>
 22  * </ul>
 23  * @property serverPort the port the server was started on
 24  * @property host the hostname of the computer to connect to
 25  * @property port the port of the computer to connect to
 26  * @property loadFBModules if the Firebug modules should be loaded at startup
 27  * @property noFBModules if no Firebug modules should be loaded at startup
 28  * @type CommandLineHandler
 29  * @returns a new {@link CommandLineHandler}
 30  */
 31 function CommandLineHandler() {
 32     this.wrappedJSObject = this;
 33 }
 34 CommandLineHandler.prototype =
 35 {
 36 
 37       classDescription: "command line handler",
 38       contractID: "@almaden.ibm.com/crossfire/command-line-handler;1",
 39       classID: Components.ID("3AB17C22-D1A6-4FF0-9A66-3DBD42114D61"),
 40       QueryInterface: XPCOMUtils.generateQI([Ci.nsICommandLineHandler]),
 41       _xpcom_categories: [{ category: "command-line-handler", entry: "m-crossfire-clh" }],
 42 
 43     /**
 44      * @name handle
 45      * @description default call-back to look for the crossfire command line arguments.
 46      * @function
 47      * @public
 48      * @memberOf CommandLineHandler
 49      * @param cmdLine an {@link nsICommandLine} object
 50      * @see https://developer.mozilla.org/en/XPCOM_Interface_Reference/nsICommandLine
 51      */
 52     handle: function(cmdLine) {
 53         try {
 54             this.serverPort = cmdLine.handleFlagWithParam("crossfire-server-port", false);
 55             if (!this.serverPort) {
 56                 this.host = cmdLine.handleFlagWithParam("crossfire-host", false);
 57                 this.port = cmdLine.handleFlagWithParam("crossfire-port", false);
 58             }
 59         } catch (e) {
 60             Cu.reportError("Command Line Handler failed: " + e);
 61         }
 62 
 63         try {
 64             this.loadFBModules = cmdLine.handleFlag("-load-fb-modules", false);
 65             this.noFBModules = cmdLine.handleFlag("-no-fb-modules", false);
 66 
 67             if (this.loadFBModules) {
 68                 this._watchAndInitializeFirebug();
 69             }
 70 
 71         } catch (e2) {
 72             Cu.reportError("Command Line Handler failed: " + e2);
 73         }
 74     },
 75 
 76     /**
 77      * @name getServerPort
 78      * @description  returns the port that the server was started on.
 79      * <br><br>
 80      * This value is specified using the <code>crossfire-server-port</code> argument.
 81      * @function
 82      * @public
 83      * @memberOf CommandLineHandler
 84      * @type Integer
 85      * @return the server port that was specified on the command-line.
 86      */
 87     getServerPort: function() {
 88         return this.serverPort;
 89     },
 90 
 91     /**
 92      * @name getPort
 93      * @description  returns the port that Crossfire should try and connect to. 
 94      * <br><br>
 95      * This value is specified using the <code>crossfire-port</code> argument.
 96      * @function
 97      * @public
 98      * @memberOf CommandLineHandler
 99      * @type Integer
100      * @return the port that was specified on the command-line.
101      */
102     getPort: function() {
103         return this.port;
104     },
105 
106     /**
107      * @name getHost
108      * @description returns the host computer name that Crossfire should try and connect to. This value can be a computer name or IP address -
109      * for example <code>localhost</code> or <code>127.0.0.1</code>.
110      * <br><br>
111      * This value is specified using the <code>crossfire-host</code> argument.
112      * @function
113      * @public
114      * @memberOf CommandLineHandler
115      * @type String
116      * @return the host that was specified on the command-line.
117      */
118     getHost: function() {
119         return this.host;
120     },
121 
122     /**
123      * @name shouldLoadFBModules
124      * @description if the Firebug modules should be loaded when Firefox starts.
125      * <br><br>
126      * This value is specified using the <code>-load-fb-modules</code> or <code>-no-fb-modules</code> flag.
127      * <br><br>
128      * If both <code>-no-fb-modules</code> and <code>-load-fb-modules</code> are specified <code>-no-fb-modules</code> will be 
129      * considered to 'win' and no Firebug modules will be loaded.
130      * @function
131      * @public
132      * @memberOf CommandLineHandler
133      * @type Boolean
134      * @return boolean indicating whether <code>-no-fb-modules</code> or <code>-load-fb-modules</code> was specified on the command-line.
135      */
136     shouldLoadFBModules: function() {
137         return (this.loadFBModules && !this.noFBModules);
138     },
139 
140     /** 
141      * @name _watchAndInitializeFirebug
142      * @description registers an observer to an {@link nsIWindowWatcher} 
143      * @function
144      * @private
145      * @memberOf CommandLineHandler
146      * @see https://developer.mozilla.org/en/nsIWindowWatcher
147      */
148     _watchAndInitializeFirebug: function() {
149         // initialize Firebug Modules before panels are loaded...
150         var windowWatcher = Cc["@mozilla.org/embedcomp/window-watcher;1"].getService(Ci.nsIWindowWatcher);
151 
152         var windowObserver = {
153             QueryInterface: function(iid) {
154                     if(!iid.equals(Ci.nsISupports) && !iid.equals(Ci.nsIObserver))
155                         throw NS_ERROR_NO_INTERFACE;
156                     return this;
157             },
158 
159             observe: function( subject, topic, data) {
160                 if (topic == "domwindowopened") {
161                     if (/* nsIDOMWindow */ subject.Firebug) {
162                         subject.Firebug.initialize();
163                         windowWatcher.unregisterNotification(windowObserver);
164                     }
165                 }
166             }
167         };
168 
169         windowWatcher.registerNotification(windowObserver);
170     }
171 
172 };
