  8 /**
  9  * @name Packet
 10  * @description The prototype object for all other packet types.
 11  * Handles parsing packet strings and JSON, and generating sequence numbers.
 12  * @public
 13  * @type Packet
 14  */
 15 var Packet = {
 16     seq: 0,
 17     length: 0,
 18     data: null,
 19 
 20     /**
 21      * @name toJSON
 22      * @description Converts the given object to a JSON {@link String}.
 23      * @function
 24      * @public
 25      * @memberOf Packet
 26      * @param obj the object to convert to a JSON {@link String}
 27      * @type String
 28      * @returns the given object as a JSON {@link String}
 29      */
 30     toJSON: function( obj) {
 31         //FIXME: only works for FF3.5 native JSON
 32         return JSON.stringify(obj);
 33     },
 34 
 35     /**
 36      * @name parseJSON
 37      * @description Parses the {@link Object} from the given JSON {@link String}.
 38      * @function
 39      * @public
 40      * @memberOf Packet
 41      * @param str the JSON {@link String} to parse
 42      * @type Object
 43      * @returns the {@link Object} value from the JSON {@link String}
 44      */
 45     parseJSON: function( str) {
 46         if (str) {
 47             //return eval('(' + str + ')'); //FIXME: dangerous
 48             return JSON.parse(str);
 49         }
 50     },
 51 
 52     /**
 53      * @name toPacketString
 54      * @description Converts the given string to over-the-wire form, with the <code>Content-Length</code> header and lines feed added.
 55      * @function
 56      * @public
 57      * @memberOf Packet
 58      * @param str the {@link String} to add to the {@link Packet} {@link String}
 59      * @type String
 60      * @returns the given {@link String} with the <code>Content-Length</code> header and lines feed added
 61      */
 62     toPacketString: function( str) {
 63         return "Content-Length:" + str.length + "\r\n\r\n" + str; // HTTP-ish style
 64     }
 65 };
 66 
 67 /**
 68  * @name EventPacket
 69  * @description Creates a new {@link EventPacket} object.
 70  * @constructor
 71  * @public
 72  * @param event name of the event.
 73  * @param data JSON object containing additional arguments for the event.
 74  * @type EventPacket
 75  * @returns a new {@link EventPacket}
 76  */
 77 function EventPacket( event, data) {
 78     if (data) {
 79         var sequence = Packet.seq++;
 80         var packet = {
 81                 "seq": sequence,
 82                 "type":	"event",
 83                 "event": event
 84         };
 85         for (var prop in data) {
 86             packet[prop] = data[prop];
 87         }
 88         var json = this.toJSON(packet);
 89         this.data = this.toPacketString(json);
 90         this.length = this.data.length;
 91     } else {
 92         //FIXME: incoming event hack
 93         var json = this.parseJSON(event);
 94         for (var prop in json) {
 95             this[prop] = json[prop];
 96         }
 97         if (json && json.seq)
 98             Packet.seq = json.seq+1;
 99     }
100 };
101 
102 EventPacket.prototype = Packet;
103 
104 /**
105  * @name RequestPacket
106  * @description Creates a {@link RequestPacket} object.
107  * @constructor
108  * @public
109  * @param packetString The unprocessed UTF-8 packet string.
110  * @type RequestPacket
111  * @returns a new {@link RequestPacket}
112  */
113 function RequestPacket( packetString) {
114     var json = this.parseJSON(packetString);
115     for (var prop in json) {
116         this[prop] = json[prop];
117     }
118     if (json && json.seq)
119         Packet.seq = json.seq+1;
120 };
121 
122 RequestPacket.prototype = Packet;

124 /**
125  * @name ResponsePacket
126  * @description Creates a new {@link ResponsePacket} object.
127  * @constructor
128  * @public
129  * @param command The name of the command that requested the response.
130  * @param requestSeq The sequence number of the request that initiated this response.
131  * @param body The JSON body of the response.
132  * @param running boolean indicating whether the context is still running after the command.
133  * @param success boolean indicating whether the command was successful.
134  * @type ResponsePacket
135  * @returns a new {@link ResponsePacket}
136  */
137 function ResponsePacket( command, requestSeq, body, running, success) {
138     var sequence = Packet.seq++;
139     var packet = {
140             "seq": sequence,
141             "type":	"response",
142             "command": command,
143             "request_seq": requestSeq,
144             "body": body,
145             "running": running,
146             "success": success
147     };
148     if (body.context_id) packet.context_id = body.context_id;
149     var json = this.toJSON(packet);
150     this.data = this.toPacketString(json);
151     this.length = this.data.length;
152 };
153 
154 ResponsePacket.prototype = Packet;
