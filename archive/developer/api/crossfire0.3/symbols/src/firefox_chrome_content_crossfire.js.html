188         /**
189          * @name handleRequest
190          * @description Looks up the context by the request object's <code>context_id</code>
191          * property and calls the requested command on that context's command adaptor.
192          * @function
193          * @public
194          * @memberOf CrossfireModule
195          * @param request the original request from {@link SocketTransport}
196          */
197         handleRequest: function( request) {
198             if (FBTrace.DBG_CROSSFIRE)
199                 FBTrace.sysout("CROSSFIRE received request " + request.toSource());
200 
201             var command = request.command;
202 
203             if (FBTrace.DBG_CROSSFIRE)
204                 FBTrace.sysout("CROSSFIRE handling command: " + command + ", with arguments: ",  request.arguments );
205 
206             var response;
207             if (command == "listcontexts") {
208                 response = this.listContexts();
209             } else if (command == "version") {
210                 response =  { "version": CROSSFIRE_VERSION };
211             } else {
212                 var commandAdaptor;
213                 var contextId = request.context_id;
214                 for (var i = 0; i < this.contexts.length; i++) {
215                     var context = this.contexts[i];
216                     if (contextId == context.Crossfire.crossfire_id) {
217                         commandAdaptor = context.Crossfire.commandAdaptor;
218                         break;
219                     }
220                 }
221                 try {
222                     response = commandAdaptor[command].apply(commandAdaptor, [ request.arguments ]);
223                 } catch (e) {
224                     if (FBTrace.DBG_CROSSFIRE)
225                         FBTrace.sysout("CROSSFIRE exception while executing command " + e);
226                 }
227             }
228 
229             if (response) {
230                 if (FBTrace.DBG_CROSSFIRE)
231                     FBTrace.sysout("CROSSFIRE sending response => " + response.toSource());
232                 this.transport.sendResponse(command, request.seq, response, this.running, true);
233             } else {
234                 this.transport.sendResponse(command, request.seq, {}, this.running, false);
235             }
236         },

429         /**
430          * @name listContexts
431          * @description Called in response to a <code>listcontexts</code> command.
432          * This method returns all the context id's that we know about.
433          *
434          * This is the only method that returns a protocol command response
435          * that is not implemented in FirebugCommandAdaptor, because it is
436          * not specific to one context.
437          * @function
438          * @public
439          * @memberOf CrossfireModule
440          * @type Array
441          * @returns an Array of the known list of contexts
442          */
443         listContexts: function() {
444             if (FBTrace.DBG_CROSSFIRE)
445                 FBTrace.sysout("CROSSFIRE listing " + this.contexts.length + " contexts...");
446             var contexts = [];
447             var context, href;
448             for (var i = 0; i < this.contexts.length; i++) {
449                 context = this.contexts[i];
450                 href = "";
451                 if (context.window && !context.window.closed) {
452                     href = context.window.location.href;
453                 }
454                 contexts.push( { "context_id" : context.Crossfire.crossfire_id,
455                                  "href"         : href ,
456                                  "current"      : this.currentContext == context });
457             }
458             return { "contexts": contexts };
459         },

461         /**
462          * @name copyFrame
463          * @description Make a copy of a frame since the jsdIStackFrame's are ephemeral,
464          * but our protocol is asynchronous so the original frame object may
465          * be gone by the time the remote host requests it.
466          * @function
467          * @public
468          * @memberOf CrossfireModule
469          * @param frame the stackframe to copy
470          * @param ctx the current Crossfire context
471          * @param shouldCopyStack is the stack of the frame should also be copied
472          * @type Array
473          * @returns a copy of the given stackframe
474          */
475         copyFrame: function copyFrame( frame, ctx, shouldCopyStack) {
476             if (FBTrace.DBG_CROSSFIRE)
477                 FBTrace.sysout("copy frame => ", frame);
478 
479             if (FBTrace.DBG_CROSSFIRE_FRAMES)
480                 FBTrace.sysout("frame count => " + (++ctx.Crossfire.frameCount));
481 
482             if (ctx) {
483                 var context = ctx;
484             }
485 
486             var frameCopy = {};
487 
488             // recursively copy scope chain
489             function copyScope( aScope) {
490                 if (FBTrace.DBG_CROSSFIRE_FRAMES)
491                     FBTrace.sysout("Copying scope => ", aScope);
492 
493                 var copiedScope = {};
494                 try {
495                     var listValue = {value: null}, lengthValue = {value: 0};
496                     aScope.getProperties(listValue, lengthValue);
497 
498                     for (var i = 0; i < lengthValue.value; ++i) {
499                         var prop = listValue.value[i];
500                         var name = prop.name.getWrappedValue();
501                            copiedScope[name.toString()] = prop.value.getWrappedValue();
502                     }
503 
504                     if (aScope.jsParent) {
505                         //copiedScope.parent = copyScope(aScope.jsParent);
506                     }
507                 } catch (ex) {
508                     if (FBTrace.DBG_CROSSFIRE_FRAMES) FBTrace.sysout("Exception copying scope => " + e);
509                 }
510                 return context.Crossfire.commandAdaptor.serialize(copiedScope);
511                 //return copiedScope;
512             }
513 
514             if (frame && frame.isValid) {
515                 try {
516                     var sourceFile = Firebug.SourceFile.getSourceFileByScript(context, frame.script)
517                     if (sourceFile) {
518                         var analyzer = sourceFile.getScriptAnalyzer(frame.script);
519                         if (analyzer) {
520                             lineno = analyzer.getSourceLineFromFrame(context, frame);
521                             frameCopy["line"] = lineno;
522                             var frameScript = sourceFile.href.toString();
523                             if (FBTrace.DBG_CROSSFIRE_FRAMES)
524                                 FBTrace.sysout("frame.script is " + frameScript);
525 
526                             frameCopy["script"] = frameScript;
527                         }
528                     }
529                 } catch (x) {
530                     if (FBTrace.DBG_CROSSFIRE) FBTrace.sysout("Exception getting script name");
531                     frameCopy["line"] = frame.line;
532                 }
533 
534                 frameCopy["scope"] =  copyScope(frame.scope);
535 
536                 if (frame.thisValue) {
537                     if (FBTrace.DBG_CROSSFIRE_FRAMES)
538                         FBTrace.sysout("copying thisValue from frame...");
539                     try {
540                        var thisVal = frame.thisValue.getWrappedValue();
541                        frameCopy["thisValue"] = context.Crossfire.commandAdaptor.serialize(thisVal);
542                     } catch( e) {
543                         if (FBTrace.DBG_CROSSFIRE) FBTrace.sysout("Exception copying thisValue => " + e);
544                     }
545                 } else if (FBTrace.DBG_CROSSFIRE_FRAMES) {
546                     FBTrace.sysout("no thisValue in frame");
547                 }
548 
549                 /* is 'callee' different from 'callingFrame'?
550                 try {
551                     frameCopy["callee"] = frame.callee.getWrappedValue();
552                 } catch( e) {
553                     frameCopy["callee"] = frame.callee;
554                 }
555                 */
556 
557                 frameCopy["functionName"] = frame.functionName;
558 
559                 // copy eval so we can call it from 'evaluate' command
560                 frameCopy["eval"] = function() { return frame.eval.apply(frame, arguments); };
561 
562                 /**
563                  * @name copyStack
564                  * @description recursively copies all of the stack elements from the given frame
565                  * @function
566                  * @private
567                  * @memberOf CrossfireModule
568                  * @param the current frame
569                  * @type Array
570                  * @returns the Array for the copied stack
571                  */
572                 function copyStack( aFrame) {
573                     if (FBTrace.DBG_CROSSFIRE_FRAMES)
574                         FBTrace.sysout("CROSSFIRE copyStack: calling frame is => ", aFrame.callingFrame);
575                     if (aFrame.callingFrame && aFrame.callingFrame.isValid) {
576                         var stack = copyStack(aFrame.callingFrame);
577                         stack.splice(0,0,copyFrame(aFrame, context, false));
578                         return stack;
579                     } else {
580                         return [ copyFrame(aFrame, context, false) ];
581                     }
582                 }
583 
584                 if (shouldCopyStack) {
585                     if (frame.callingFrame) {
586                         var stack = copyStack(frame.callingFrame);
587                         frameCopy["stack"] = stack;
588                         frameCopy["frameIndex"] = stack.length -1;
589                     } else {
590                         frameCopy["frameIndex"] = 0;
591                     }
592                 }
593             }
594             return frameCopy;
595         },

858         /**
859          * @name _resolveElementPath
860          * @description resolves the path to the given element within the DOM tree.
861          * @function
862          * @private
863          * @memberOf CrossfireModule
864          * @param element the current DOM node context
865          * @param if we should use the tags names when constructing the path. i.e. <code>/html[1]/body[1]/div[4]/span[21]/...</code>
866          * @since 0.3a1
867          */
868 		_resolveElementPath: function(element, useTagNames) {
869 			var nameLookup = [];
870 			nameLookup[Node.COMMENT_NODE] = "comment()";
871 			nameLookup[Node.TEXT_NODE] = "text()";
872 			nameLookup[Node.PROCESSING_INSTRUCTION_NODE] = "processing-instruction()";
873 			var paths = [];
874 			for (; element && element.nodeType != Node.DOCUMENT_NODE; element = element.parentNode) {
875 				var tagName = element.localName || nameLookup[element.nodeType];
876 				var index = 0;
877 				for (var sibling = element.previousSibling; sibling; sibling = sibling.previousSibling) {
878 					var siblingTagName = sibling.localName || nameLookup[sibling.nodeType];
879 					if (!useTagNames || tagName == siblingTagName || !tagName) {
880 						++index;
881 					}
882 				}
883 				var pathIndex = "[" + (index+1) + "]";
884 				paths.splice(0, 0, (useTagNames && tagName ? tagName.toLowerCase() : "node()") + pathIndex);
885 			}
886 			return "/" + paths.join("/");
887 		},
