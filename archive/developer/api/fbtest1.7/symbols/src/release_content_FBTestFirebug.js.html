 17 /**
 18  * Verification method, prints result of a test. If the first <i>pass<i> parameter is true
 19  * the test passes, otherwise fails.
 20  * @param {Boolean} pass Result of a test.
 21  * @param {String} msg A message to be displayed as a test results under the current test
 22  *      within the test console.
 23  */
 24 this.ok = function(pass, msg)
 25 {
 26     if (!pass)
 27         FBTest.sysout("FBTest **** FAILS **** " + msg);
 28     else
 29         FBTest.sysout("FBTest ok " + msg);
 30 
 31     FBTestApp.TestRunner.appendResult(new FBTestApp.TestResult(window, pass, msg));
 32 
 33     if (!pass)
 34         this.onFailure(msg);
 35     else
 36         FBTest.resetTimeout();
 37 
 38     return pass;
 39 };

 41 /**
 42  * Verification method. Compares expected and actuall string (typially from the Firebug UI).
 43  * If <i>actuall</i> and <i>expected<i> parameters are equal the test passes, otherwise fails.
 44  * @param {String} expected Expected value
 45  * @param {String} actual Actual value
 46  * @param {String} msg A message to be displayed as a test result under the current test
 47  *      within the test console.
 48  */
 49 this.compare = function(expected, actual, msg)
 50 {
 51     var result;
 52     if (expected instanceof RegExp)
 53     {
 54         result = actual ? actual.match(expected) : null;
 55         expected = expected ? expected.toString() : null;
 56     }
 57     else
 58     {
 59         // xxxHonza: TODO: lib/textSearch doesn't like '==='
 60         result = (expected == actual);
 61     }
 62 
 63     FBTest.sysout("compare "+(result?"passes":"**** FAILS ****")+" "+msg, {expected: expected, actual: actual});
 64 
 65     FBTestApp.TestRunner.appendResult(new FBTestApp.TestResult(window,
 66         result, msg, expected, actual));
 67 
 68     if (result)
 69         FBTest.resetTimeout();
 70     else
 71         FBTest.onFailure(msg);
 72 
 73     return result;
 74 };

273 /**
274  * Send the char aChar to the node with id aTarget. This method handles casing
275  * of chars (sends the right charcode, and sends a shift key for uppercase chars).
276  * No other modifiers are handled at this point.
277  *
278  * For now this method only works for English letters (lower and upper case)
279  * and the digits 0-9.
280  *
281  * Returns true if the keypress event was accepted (no calls to preventDefault
282  * or anything like that), false otherwise.
283  */
284 this.sendChar = function(aChar, aTarget)
285 {
286     // DOM event charcodes match ASCII (JS charcodes) for a-zA-Z0-9.
287     var hasShift = (aChar == aChar.toUpperCase());
288     var charCode = aChar.charCodeAt(0);
289     var keyCode = charCode;
290     if (!hasShift)
291     {
292         // For lowercase letters, the keyCode is actually 32 less than the charCode
293         keyCode -= 0x20;
294     }
295 
296     return __doEventDispatch(aTarget, charCode, keyCode, hasShift);
297 }

311 /**
312  * Send the non-character key aKey to the node with id aTarget.
313  * The name of the key should be a lowercase
314  * version of the part that comes after "DOM_VK_" in the KeyEvent constant
315  * name for this key.  No modifiers are handled at this point.
316  *
317  * Returns true if the keypress event was accepted (no calls to preventDefault
318  * or anything like that), false otherwise.
319  */
320 this.sendKey = function(aKey, aTarget)
321 {
322     keyName = "DOM_VK_" + aKey.toUpperCase();
323 
324     if (!KeyEvent[keyName]) {
325         throw "Unknown key: " + keyName;
326     }
327 
328     return __doEventDispatch(aTarget, 0, KeyEvent[keyName], false);
329 }

331 /**
332  * Actually perform event dispatch given a charCode, keyCode, and boolean for
333  * whether "shift" was pressed.  Send the event to the node with id aTarget.  If
334  * aTarget is not provided, use "target".
335  *
336  * Returns true if the keypress event was accepted (no calls to preventDefault
337  * or anything like that), false otherwise.
338  */
339 function __doEventDispatch(aTarget, aCharCode, aKeyCode, aHasShift)
340 {
341     if (aTarget && aTarget instanceof Node)
342         aTarget = aTarget;
343     else if (aTarget)
344         aTarget = FBTest.FirebugWindow.document.getElementById(aTarget);
345     else
346         aTarget= FBTest.FirebugWindow.document.documentElement;
347 
348     var doc = aTarget.ownerDocument
349 
350     var event = doc.createEvent("KeyEvents");
351     event.initKeyEvent("keydown", true, true, doc.defaultView,
352         false, false, aHasShift, false, aKeyCode, 0);
353 
354     var accepted = aTarget.dispatchEvent(event);
355 
356     // Preventing the default keydown action also prevents the default
357     // keypress action.
358     event = doc.createEvent("KeyEvents");
359     if (aCharCode)
360     {
361         event.initKeyEvent("keypress", true, true, doc.defaultView,
362             false, false, aHasShift, false, 0, aCharCode);
363     }
364     else
365     {
366         event.initKeyEvent("keypress", true, true, doc.defaultView,
367             false, false, aHasShift, false, aKeyCode, 0);
368     }
369 
370     if (!accepted)
371         event.preventDefault();
372 
373     accepted = aTarget.dispatchEvent(event);
374 
375     // Always send keyup
376     var event = doc.createEvent("KeyEvents");
377     event.initKeyEvent("keyup", true, true, doc.defaultView,
378         false, false, aHasShift, false, aKeyCode, 0);
379     aTarget.dispatchEvent(event);
380 
381     return accepted;
382 }

384 /**
385  * Synthesize a key event. It is targeted at whatever would be targeted by an
386  * actual keypress by the user, typically the focused element.
387  *
388  * aKey should be either a character or a keycode starting with VK_ such as
389  * VK_ENTER. See list of all possible key-codes here:
390  * http://www.w3.org/TR/2000/WD-DOM-Level-3-Events-20000901/events.html
391  *
392  * aEvent is an object which may contain the properties:
393  *   shiftKey, ctrlKey, altKey, metaKey, accessKey, type
394  *
395  * If the type is specified, a key event of that type is fired. Otherwise,
396  * a keydown, a keypress and then a keyup event are fired in sequence.
397  *
398  * aWindow is optional, and defaults to the current window object.
399  */
400 this.synthesizeKey = function(aKey, aWindow)
401 {
402     if (!aWindow)
403         aWindow = window;
404 
405     var utils = aWindow.QueryInterface(Components.interfaces.nsIInterfaceRequestor).
406         getInterface(Components.interfaces.nsIDOMWindowUtils);
407     if (!utils)
408         return;
409 
410     var keyCode = 0, charCode = 0;
411     if (aKey.indexOf("VK_") == 0)
412         keyCode = KeyEvent["DOM_" + aKey];
413     else
414         charCode = aKey.charCodeAt(0);
415 
416     var keyDownDefaultHappened = utils.sendKeyEvent("keydown", keyCode, charCode, 0);
417     utils.sendKeyEvent("keypress", keyCode, charCode, 0, !keyDownDefaultHappened);
418     utils.sendKeyEvent("keyup", keyCode, charCode, 0);
419 }

568 /**
569  * Helper method for wait till a window is *really* loaded.
570  * @param {Object} browser Window's parent browser.
571  * @param {Window} callback Executed when the window is loaded. The window is passed in
572  *      as the parameter.
573  */
574 function waitForWindowLoad(browser, callback)
575 {
576     // If the callback isn't specified don't watch the window load at all.
577     if (!callback)
578         return;
579 
580     var loaded = false;
581     var painted = false;
582 
583     // Wait for all event that must be fired before the window is loaded.
584     // Any event is missing?
585     // xxxHonza: In case of Firefox 3.7 the new 'content-document-global-created'
586     // (bug549539) could be utilized.
587     var waitForEvents = function(event)
588     {
589         if (event.type == "load")
590         {
591             browser.removeEventListener("load", waitForEvents, true);
592             loaded = true;
593         }
594         else if (event.type == "MozAfterPaint")
595         {
596             browser.removeEventListener("MozAfterPaint", waitForEvents, true);
597             painted = true;
598         }
599 
600         // Execute callback after 100ms timout (the inspector tests need it for now),
601         // but this shoud be set to 0.
602         if (loaded && painted)
603             setTimeout(executeCallback, 100);
604     }
605 
606     // All expected events have been fired, execute the callback.
607     var executeCallback = function()
608     {
609         try
610         {
611             var win = browser.contentWindow;
612 
613             // This is a workaround for missing wrappedJSObject property,
614             // if the test case comes from http (and not from chrome)
615             // xxxHonza: this is rather a hack, it should be removed if possible.
616             //if (!win.wrappedJSObject)
617             //    win.wrappedJSObject = win;
618             FBTest.sysout("callback <<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< "+win.location);
619             // The window is loaded, execute the callback now.
620             callback(win);
621         }
622         catch (exc)
623         {
624             FBTest.sysout("runTest FAILS " + exc, exc);
625             FBTest.ok(false, "runTest FAILS " + exc);
626         }
627     }
628 FBTest.sysout("addinge event listeenr<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<");
629     browser.addEventListener("load", waitForEvents, true);
630     browser.addEventListener("MozAfterPaint", waitForEvents, true);
631 }

1686 /**
1687  * Support for set of asynchronouse actions within a FBTest.
1688  * @param {Array} tests List of asynchronous functions to be executed in order.
1689  * @param {Function} callback A callback that is executed as soon as all fucntions
1690  * in the list are finished.<br/><br/>
1691  * Example:<br/>
1692  *  <pre>// A suite of asynchronous tests.
1693  *  var testSuite = [];
1694  *  testSuite.push(function(callback) {
1695  *      // TODO: test implementation
1696  *      // Continue with other tests.
1697  *      callback();
1698  *  });
1699  *  testSuite.push(function(callback) {
1700  *      // TODO: test implementation
1701  *      // Continue with other tests.
1702  *      callback();
1703  *  });
1704  *  // Run entire suite.
1705  *  runTestSuite(testSuite, function() {
1706  *      FBTestFirebug.testDone("DONE");
1707  *  });</pre>
1708  */
1709 this.runTestSuite = function(tests, callback)
1710 {
1711     setTimeout(function()
1712     {
1713         var test = tests.shift();
1714         if (!test)
1715         {
1716             callback();
1717             return;
1718         }
1719 
1720         function runNext()
1721         {
1722             if (tests.length > 0)
1723                 FBTestFirebug.runTestSuite(tests, callback);
1724             else
1725                 callback();
1726         }
1727 
1728         try
1729         {
1730             test.call(this, runNext);
1731         }
1732         catch (err)
1733         {
1734             FBTest.progress("EXCEPTION " + err);
1735         }
1736     }, 200);
1737 }
