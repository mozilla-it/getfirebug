  4 // Text selection conroller.
  5 
  6 var SelectionController =
  7 {
  8     listeners: [],
  9 
 10     addListener: function(listener)
 11     {
 12         FBTestApp.SelectionController.addListener(listener);
 13         this.listeners.push(listener);
 14     },
 15 
 16     removeListener: function(listener)
 17     {
 18         FBTestApp.SelectionController.removeListener(listener);
 19         FW.FBL.remove(this.listeners, listener);
 20     },
 21 
 22     cleanUp: function()
 23     {
 24         // Remove all listeners registered by the current test.
 25         while (this.listeners.length)
 26             this.removeListener(this.listeners[0]);
 27     }
 28 }
 29 
 30 // ************************************************************************************************
 31 // Clean up
 32 
 33 window.addEventListener("unload", function testSelectionUnload() {
 34     SelectionController.cleanUp();
 35 }, true);
