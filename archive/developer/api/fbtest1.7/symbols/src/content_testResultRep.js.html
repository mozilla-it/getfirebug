468             if (this.isImage(result.expected))
469                 this.insertImage(result.expected, expectedNode);
470             else if (this.isXml(result.expected))
471                 this.insertXml(result.expected, expectedNode);
472             else
473                 this.insertText(result.expected, expectedNode);
474 
475             if (this.isImage(result.result))
476                 this.insertImage(result.result, actualNode);
477             else if (this.isXml(result.expected))
478                 this.insertXml(result.result, actualNode);
479             else
480                 this.insertText(result.result, actualNode);
481 
482             // The diff is generated only if there are any differences.
483             if (result.expected != result.result) {

493         {
494             tabExceptionBody.updated = true;

564         // Create helper root element (for the case where there is no signle root).
565         var tempXml = "<wrapper>" + xml + "</wrapper>";

568 
569         // Error handling

580             return;
581         }
582 
583         // Generate UI. Get appropriate domplate tag for every element that is found
584         // within the helper <wrapper> and append it into the parent container.
585         for (var i=0; i<docElem.childNodes.length; i++)
586             Firebug.HTMLPanel.CompleteElement.getNodeTag(docElem.childNodes[i]).

604 /**
605  * This template displays a parse-erros that can occurs when parsing
606  * expected and acuall results (see compare method).
607  */
608 var ParseErrorRep = domplate(
609 {
610     tag:
611         DIV({"class": "xmlInfoError"},
612             DIV({"class": "xmlInfoErrorMsg"}, "$error.message"),
613             PRE({"class": "xmlInfoErrorSource"}, "$error|getSource"),
614             BR(),
615             PRE({"class": "xmlInfoSource"})
616         ),
617 
618     getSource: function(error)
619     {
620         var parts = error.source.split("\n");
621         if (parts.length != 2)
622             return error.source;
623 
624         var limit = 50;
625         var column = parts[1].length;
626         if (column >= limit) {
627             parts[0] = "..." + parts[0].substr(column - limit);
628             parts[1] = "..." + parts[1].substr(column - limit);
629         }
630 
631         if (parts[0].length > 80)
632             parts[0] = parts[0].substr(0, 80) + "...";
633 
634         return parts.join("\n");
635     }

641 /**
642  * This object represents a test-result.
643  */
644 FBTestApp.TestResult = function(win, pass, msg, expected, result)
645 {
646     var location = win.location.href;
647     this.fileName = location.substr(location.lastIndexOf("/") + 1);
648 
649     this.pass = pass ? true : false;
650     this.msg = clean(msg);
651 
652     // Make sure the following values are strings.
653     this.expected = expected + "";
654     this.result = result + "";
655 
656     // xxxHonza: there should be perhaps simple API in lib.js to get the stack trace.
657     this.stack = [];
658     for (var frame = Components.stack, i=0; frame; frame = frame.caller, i++)
659     {
660         var fileName = unescape(frame.filename ? frame.filename : "");
661         //if (fileName.indexOf("chrome://fbtest/content") == 0)
662         //    continue;
663 
664         var lineNumber = frame.lineNumber ? frame.lineNumber : "";
665         this.stack.push({fileName:fileName, lineNumber:lineNumber});
666     }
667 }
668 
669 FBTestApp.TestException = function(win, msg, err)
670 {
671     FBTestApp.TestResult.call(this, win, false, msg + err);
672 
673     this.err = err;
674     this.expected = null;
675     this.result = null;
676 }

680 function clean(str)
681 {
682     try
683     {
684         return str ? str.replace(/&/g, "&").replace(/</g, "<").replace(/>/g, ">") : "";
685     }
686     catch (err)
687     {
688         FBTrace.sysout("fbtest.clean; EXCEPTION " + str, err);
689     }
690 
691     return str;
692 }
