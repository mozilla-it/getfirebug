 31 
 32 function History() {}
 33 History.prototype =
 34 {
 35     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
 36     // XPCOM
 37 
 38     QueryInterface: XPCOMUtils.generateQI([
 39         Ci.nsISupports,
 40         Ci.nsIAutoCompleteSearch
 41     ]),
 42 
 43     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
 44     // nsIAutoCompleteSearch
 45 
 46     startSearch: function(searchString, searchParam, previousResult, listener)
 47     {
 48         // Get all test-lists from preferences.
 49         var history = prefs.getCharPref(this.pref);
 50         var arr = history.split(",");
 51 
 52         var map = {};
 53         var results = [];
 54         for (var i=0; i<arr.length; i++)
 55         {
 56             var url = trimSpaces(arr[i]);
 57             if (url && !map[url] && (!searchString || url.indexOf(searchString) > 0))
 58             {
 59                 map[url] = true;
 60                 results.push(url);
 61             }
 62         }
 63 
 64         listener.onSearchResult(this, new SearchResult(searchString,
 65             Ci.nsIAutoCompleteResult.RESULT_SUCCESS,
 66             0, results));
 67     },
 68 
 69     stopSearch: function()
 70     {
 71     }
 72 };
 73 
 74 function trimSpaces(text)
 75 {
 76     return text.replace(/^\s*|\s*$/g,"");
 77 }

 80 // Implements nsIAutoCompleteResult
 81 
 82 function SearchResult(searchString, searchResult, defaultIndex, results)
 83 {
 84     this.searchString = searchString;
 85     this.searchResult = searchResult;
 86     this.defaultIndex = defaultIndex;
 87     this.results = results;
 88 }
 89 
 90 SearchResult.prototype =
 91 {
 92     searchString: "",
 93     searchResult: 0,
 94     defaultIndex: 0,
 95     results: [],
 96     errorDescription: "",
 97 
 98     get matchCount() {
 99         return this.results.length;
100     },
101 
102     getValueAt: function(index) {
103         return this.results[index];
104     },
105 
106     getCommentAt: function(index) {
107         return "";
108     },
109 
110     getStyleAt: function(index) {
111         return null;
112     },
113 
114     getImageAt: function (index) {
115         return "";
116     },
117 
118     removeValueAt: function(index, removeFromDb) {
119         this._results.splice(index, 1);
120     },
121 
122     QueryInterface: function(aIID)
123     {
124         if (!aIID.equals(Ci.nsIAutoCompleteResult) &&
125             !aIID.equals(Ci.nsISupports))
126             throw Components.results.NS_ERROR_NO_INTERFACE;
127 
128         return this;
129     }
130 };

135 function extend(l, r)
136 {
137     var newOb = {};
138     for (var n in l)
139         newOb[n] = l[n];
140     for (var n in r)
141         newOb[n] = r[n];
142     return newOb;
143 };
