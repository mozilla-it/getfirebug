 17 /**
 18  * @constructor This object is intended for handling HTML changes that can occur on a page.
 19  * This is useful e.g. in cases when a test expects specific element to be created and
 20  * wants to asynchronously wait for it.
 21  * @param {Window} win Parent window.
 22  * @param {String} tagName Name of the element.
 23  * @param {Object} attributes List of attributes that identifies the element.
 24  * @param {String} text Specific text that should be created. The tagName must be set to
 25  * <i>Text</i> in this case.
 26  */
 27 var MutationRecognizer = function(win, tagName, attributes, text)
 28 {
 29    this.win = win;
 30    this.tagName = tagName;
 31    this.attributes = attributes;
 32    this.characterData = text;
 33 };
 34 
 35 MutationRecognizer.prototype.getDescription = function()
 36 {
 37     var obj = {
 38         tagName: this.tagName,
 39         attributes: this.attributes,
 40         characterData: this.characterData
 41     };
 42 
 43     return JSON.stringify(obj);
 44 };
 45 
 46 /**
 47  * Passes a callback handler that is called when specific HTML change
 48  * occurs on the page.
 49  * @param {Function} handler Callback handler gets one parameter specifing the founded element.
 50  */
 51 MutationRecognizer.prototype.onRecognize = function(handler)
 52 {
 53     return new MutationEventFilter(this, handler);
 54 }
 55 
 56 /**
 57  * Passes a callback handler that is called when specific HTML change
 58  * occurs on the page. After the change is caught, the handler is executed yet
 59  * asynchronously.
 60  * @param {Function} handler Callback handler gets one parameter specifing the founded element.
 61  * @delay {Number} delay Number of milliseconds delay (10ms by default).
 62  */
 63 MutationRecognizer.prototype.onRecognizeAsync = function(handler, delay)
 64 {
 65     if (!delay)
 66         delay = 10;
 67 
 68     return new MutationEventFilter(this, function(element) {
 69         setTimeout(function delayMutationEventFilter() {
 70             FBTest.sysout("testFirebug.MutationEventFilter.onRecognizeAsync:", element);
 71             handler(element);
 72         }, delay);
 73     });
 74 }

 76 MutationRecognizer.prototype.getWindow = function()
 77 {
 78     return this.win;

 81 MutationRecognizer.prototype.matches = function(elt)
 82 {
 83     // Note Text nodes have no tagName
 84     if (this.tagName == "Text")
 85     {
 86         if (elt.data && elt.data.indexOf(this.characterData) != -1)
 87         {
 88             if (FBTrace.DBG_TESTCASE_MUTATION)
 89                 FBTrace.sysout("MutationRecognizer matches Text character data "+this.characterData);
 90             return elt.parentNode;
 91         }
 92         else
 93         {
 94             if (FBTrace.DBG_TESTCASE_MUTATION)
 95                 FBTrace.sysout("MutationRecognizer no match in Text character data "+this.characterData+" vs "+elt.data,{element: elt, recogizer: this});
 96             return null;
 97         }
 98     }
 99 
100     if (!(elt instanceof Element))
101     {
102         if (FBTrace.DBG_TESTCASE_MUTATION)
103             FBTrace.sysout("MutationRecognizer Node not an Element ", elt);
104         return null;
105     }
106 
107     if (elt.tagName && (elt.tagName.toLowerCase() != this.tagName) )
108     {
109         if (FBTrace.DBG_TESTCASE_MUTATION)
110             FBTrace.sysout("MutationRecognizer no match on tagName "+this.tagName+
111                 " vs "+elt.tagName.toLowerCase()+" "+FW.FBL.getElementCSSSelector(elt), {element: elt, recogizer: this});
112         return null;
113     }
114 
115     for (var p in this.attributes)
116     {
117         if (this.attributes.hasOwnProperty(p))
118         {
119             var eltP = elt.getAttribute(p);
120             if (!eltP)
121             {
122                 if (FBTrace.DBG_TESTCASE_MUTATION)
123                     FBTrace.sysout("MutationRecognizer no attribute "+p+" in "+
124                         FW.FBL.getElementHTML(elt), {element: elt, recogizer: this});
125                 return null;
126             }
127             if (this.attributes[p] != null)
128             {
129                 if (p == 'class')
130                 {
131                     if (!FW.FBL.hasClass.apply(FW.FBL, [elt, this.attributes[p]]))
132                     {
133                         if (FBTrace.DBG_TESTCASE_MUTATION)
134                             FBTrace.sysout("MutationRecognizer no match for class " +
135                                 this.attributes[p]+" vs "+eltP+" p==class: "+(p=='class') +
136                                 " indexOf: "+eltP.indexOf(this.attributes[p]));
137                         return null;
138                     }
139                 }
140                 else if (eltP != this.attributes[p])
141                 {
142                     if (FBTrace.DBG_TESTCASE_MUTATION)
143                         FBTrace.sysout("MutationRecognizer no match for attribute "+p+": "+
144                             this.attributes[p]+" vs "+eltP,{element: elt, recogizer: this});
145                     return null;
146                 }
147             }
148         }
149     }
150 
151     if (this.characterData)
152     {
153         if (elt.textContent.indexOf(this.characterData) < 0)
154         {
155             if (FBTrace.DBG_TESTCASE_MUTATION)
156                 FBTrace.sysout("MutationRecognizer no match for characterData "+this.characterData+
157                     " vs "+elt.textContent, {element: elt, recogizer: this});
158             return null;
159         }
160     }
161 
162     // tagName and all attributes match
163     FBTest.sysout("MutationRecognizer tagName and all attributes match "+elt, elt);
164     return elt;
165 }

167 function MutationEventFilter(recognizer, handler)
168 {
169     this.recognizer = recognizer;
170 
171     this.winName = new String(window.location.toString());
172     var filter = this;
173     this.onMutateAttr = function handleAttrMatches(event)
174     {
175         if (window.closed)
176             throw "WINDOW CLOSED watching:: "+(filter.recognizer.win.closed?"closed":filter.recognizer.win.location)+" closed window: "+filter.winName;
177 
178         if (!recognizer.attributes)
179             return; // we don't care about attribute mutation
180 
181         if (FBTrace.DBG_TESTCASE_MUTATION)
182             FBTrace.sysout("onMutateAttr "+event.attrName+"=>"+event.newValue+" on "+event.target+
183                 " in "+event.target.ownerDocument.location, event.target);
184 
185         // We care about some attribute mutation.
186         if (!recognizer.attributes.hasOwnProperty(event.attrName))
187         {
188             if (FBTrace.DBG_TESTCASE_MUTATION)
189                 FBTrace.sysout("onMutateAttr not interested in "+event.attrName+"=>"+event.newValue+
190                     " on "+event.target+" in "+event.target.ownerDocument.location, event.target);
191             return;  // but not the one that changed.
192         }
193 
194         try
195         {
196             if (filter.checkElement(event.target))
197                 handler(event.target);
198         }
199         catch(exc)
200         {
201             if (FBTrace.DBG_TESTCASE_MUTATION)
202                 FBTrace.sysout("onMutateNode FAILS "+exc, exc);
203         }
204     }
205 
206     // the matches() function could be tuned to each kind of mutation for improved efficiency
207     this.onMutateNode = function handleNodeMatches(event)
208     {
209         if (window.closed)
210             throw "WINDOW CLOSED watching:: "+(filter.recognizer.win.closed?"closed":filter.recognizer.win.location)+" closed window: "+filter.winName;
211 
212         if (FBTrace.DBG_TESTCASE_MUTATION)
213             FBTrace.sysout("onMutateNode "+event.target+" in "+event.target.ownerDocument.location, event.target);
214 
215         try
216         {
217             var child = filter.checkElementDeep(event.target)
218             if (child)
219                 handler(child);
220         }
221         catch(exc)
222         {
223             if (FBTrace.DBG_TESTCASE_MUTATION)
224                 FBTrace.sysout("onMutateNode FAILS "+exc, exc);
225         }
226     }
227 
228     this.onMutateText = function handleTextMatches(event)
229     {
230         if (window.closed)
231             throw "WINDOW CLOSED watching:: "+(filter.recognizer.win.closed?"closed":filter.recognizer.win.location)+" closed window: "+filter.winName;
232 
233         if (!recognizer.characterData)
234             return; // we don't care about text
235 
236         // We care about text and the text for this element mutated.  If it matches we must have hit.
237         if (FBTrace.DBG_TESTCASE_MUTATION)
238             FBTrace.sysout("onMutateText =>"+event.newValue+" on "+event.target.ownerDocument.location, event.target);
239 
240         try
241         {
242             if (filter.checkElement(event.target))  // target is CharacterData node
243                 handler(event.target);
244         }
245         catch(exc)
246         {
247             if (FBTrace.DBG_TESTCASE_MUTATION)
248                 FBTrace.sysout("onMutateNode FAILS "+exc, exc);
249         }
250     }
251 
252     filter.checkElement = function(elt)
253     {
254         var element = recognizer.matches(elt);
255         if (element)
256         {
257             filter.unwatchWindow(recognizer.getWindow())
258             return element;
259         }
260         return null;
261     }
262 
263     filter.checkElementDeep = function(elt)
264     {
265         var element = filter.checkElement(elt);
266         if (element)
267         {
268             return element;
269         }
270         else
271         {
272             var child = elt.firstChild;
273             for (; child; child = child.nextSibling)
274             {
275                 var element = this.checkElementDeep(child)
276                 if (element)
277                     return element;
278             }
279         }
280 
281         return null;
282     }
283 
284     filter.watchWindow(recognizer.win);
285 }

290 var filterInstance = 1;
291 var activeFilters = {};
292 MutationEventFilter.prototype.watchWindow = function(win)
293 {
294     var doc = win.document;
295     doc.addEventListener("DOMAttrModified", this.onMutateAttr, false);
296     doc.addEventListener("DOMCharacterDataModified", this.onMutateText, false);
297     doc.addEventListener("DOMNodeInserted", this.onMutateNode, false);
298     // doc.addEventListener("DOMNodeRemoved", this.onMutateNode, false);
299 
300     var filter = this;
301     filterInstance++;
302     activeFilters[filterInstance] = filter;
303     this.filterInstance = filterInstance;
304 
305     filter.cleanUp = function(event)
306     {
307         try
308         {
309             if (window.closed)
310             {
311                 throw new Error("Filter cleanup in window.closed event.target:"+event.target);
312             }
313             FBTest.sysout("Filter.cleanup "+filter.filterInstance);
314             filter.unwatchWindow(win);
315             document.removeEventListener("FBTestCleanup", filter.cleanUp, true);
316         }
317         catch (e)
318         {
319           FBTest.sysout("Filter.cleanup FAILS "+e, e);
320         }
321     }
322     win.addEventListener("unload", filter.cleanUp, true);
323     window.addEventListener("unload", filter.cleanUp, true);
324     document.addEventListener("FBTestCleanup", filter.cleanUp, true);
325     //window.FBTest.progress("added MutationWatcher to "+doc.location+" and FBTestCleanup to "+document.location);
326     //window.FBTest.progress("added FBTestCleanup "+filterInstance+" to "+document.location);
327 }
328 
329 MutationEventFilter.prototype.unwatchWindow = function(win)
330 {
331     var doc = win.document;
332 
333     doc.removeEventListener("DOMAttrModified", this.onMutateAttr, false);
334     doc.removeEventListener("DOMCharacterDataModified", this.onMutateText, false);
335     doc.removeEventListener("DOMNodeInserted", this.onMutateNode, false);
336     win.removeEventListener("unload", this.cleanUp, true);
337     window.removeEventListener("unload", this.cleanUp, true);
338     window.FBTest.sysout("unwatchWindow removed MutationWatcher "+this.filterInstance+" from "+doc.location);
339     delete activeFilters[this.filterInstance];
340 }

342 // ************************************************************************************************

345 window.addEventListener('unload', function sayUnload()
346 {
347     FBTest.sysout(" UNLOAD "+window.location);
348     for (var p in activeFilters)
349     {
350         FBTest.sysout(p+" still active filter ");
351         activeFilters[p].cleanUp();
352     }
353 
354 }, true);
