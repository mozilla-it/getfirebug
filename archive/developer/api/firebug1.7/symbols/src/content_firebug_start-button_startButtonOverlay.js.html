  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ********************************************************************************************* //
  6 // Constants
  7 
  8 const Cc = Components.classes;
  9 const Ci = Components.interfaces;
 10 
 11 var appInfo = Cc["@mozilla.org/xre/app-info;1"].getService(Ci.nsIXULAppInfo);
 12 var versionChecker = Cc["@mozilla.org/xpcom/version-comparator;1"].getService(Ci.nsIVersionComparator);
 13 
 14 var popup = $("fbStatusContextMenu");
 15 var statusBar = $("fbStatusBar");
 16 var statusText = $("fbStatusText");
 17 
 18 // ********************************************************************************************* //
 19 // Module Implementation
 20 
 21 /**
 22  * @module StartButton module represents the UI entry point to Firebug. This "start buttton"
 23  * formerly known as "the status bar icon" is automatically appended into Firefox toolbar
 24  * (since Firefox 4).
 25  *
 26  * Start button is associated with a menu (fbStatusContextMenu) that contains basic actions
 27  * such as panel activation and also indicates whether Firebug is activated/deactivated for
 28  * the current page (by changing its color).
 29  */
 30 Firebug.StartButton = extend(Firebug.Module,
 31 /** @lends Firebug.StartButton */
 32 {
 33     initializeUI: function()
 34     {
 35         Firebug.Module.initializeUI.apply(this, arguments);
 36 
 37         // Associate a popup-menu with the start button (the same as it's
 38         // used for the obsolete status bar icon.
 39         var startButton = $("firebug-button");
 40         if (startButton)
 41         {
 42             startButton.appendChild(popup.cloneNode(true));
 43 
 44             // Append the button into Firefox toolbar automatically.
 45             this.appendToToolbar();
 46 
 47             // In case of Firefox 4+ the button is a bit different.
 48             if (versionChecker.compare(appInfo.version, "4.0*") >= 0)
 49                 startButton.setAttribute("firefox", "4");
 50         }
 51 
 52         if (FBTrace.DBG_INITIALIZE)
 53             FBTrace.sysout("Startbutton initializeUI "+startButton);
 54     },
 55 
 56     shutdown: function()
 57     {
 58     },
 59 
 60     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * //
 61 
 62     /**
 63      * Appends Firebug start button into Firefox toolbar automatically after installation.
 64      * The button is appended only once so, if the user removes it, it isn't appended again.
 65      */
 66     appendToToolbar: function()
 67     {
 68         if (Firebug.getPref(Firebug.prefDomain, "toolbarCustomizationDone"))
 69             return;
 70 
 71         Firebug.setPref(Firebug.prefDomain, "toolbarCustomizationDone", true);
 72 
 73         // Get the current navigation bar button set (a string of button IDs) and append
 74         // ID of the Firebug start button into it.
 75         var startButtonId = "firebug-button";
 76         var navBarId = "nav-bar";
 77         var navBar = $(navBarId);
 78 
 79         // In SeaMonkey we need to read the attribute (see issue 4086)
 80         // In Firefox the currentSet property must be used.
 81         var currentSet = navBar.getAttribute("currentset");
 82         if (!currentSet)
 83             currentSet = navBar.currentSet;
 84 
 85         if (FBTrace.DBG_INITIALIZE)
 86             FBTrace.sysout("Startbutton; curSet: " + currentSet);
 87 
 88         // Append only if the button is not already there.
 89         var curSet = currentSet.split(",");
 90         if (curSet.indexOf(startButtonId) == -1)
 91         {
 92             var set = curSet.concat(startButtonId);
 93             navBar.setAttribute("currentset", set.join(","));
 94             document.persist(navBarId, "currentset");
 95 
 96             if (FBTrace.DBG_INITIALIZE)
 97                 FBTrace.sysout("Startbutton; curSet (after modification): " + set.join(","));
 98 
 99             try
100             {
101                 BrowserToolboxCustomizeDone(true);
102             }
103             catch (e)
104             {
105                 if (FBTrace.DBG_ERRORS)
106                     FBTrace.sysout("startButton; appendToToolbar EXCEPTION " + e, e);
107             }
108         }
109 
110         // Don't forget to show the navigation bar - just in case it's hidden.
111         collapse(navBar, false);
112         document.persist(navBarId, "collapsed");
113     },
114 
115     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * //
116     // Support for the status bar (OBSOLETE in Fx4)
117 
118     onClickStatusText: function(context, event)
119     {
120         if (event.button != 0)
121             return;
122 
123         if (!context || !context.errorCount)
124             return;
125 
126         var panel = Firebug.chrome.getSelectedPanel();
127         if (panel && panel.name != "console")
128         {
129             Firebug.chrome.selectPanel("console");
130             cancelEvent(event);
131         }
132     },
133 
134     onClickStatusIcon: function(context, event)
135     {
136         if (event.button != 0)
137             return;
138         else if (isControl(event))
139             Firebug.toggleDetachBar(true);
140         else if (context && context.errorCount)
141             Firebug.toggleBar(undefined, "console");
142         else
143             Firebug.toggleBar();
144     },
145 
146     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * //
147     // Error count
148 
149     showCount: function(errorCount)
150     {
151         if (!statusBar)
152             return;
153 
154         var firebugButton = $("firebug-button");
155         if (errorCount && Firebug.showErrorCount)
156         {
157             statusBar.setAttribute("showErrors", "true")
158             statusText.setAttribute("value", $STRP("plural.Error_Count2", [errorCount]));
159 
160             if (firebugButton)
161             {
162                 firebugButton.setAttribute("showErrors", "true");
163                 firebugButton.setAttribute("errorCount", errorCount);
164             }
165         }
166         else
167         {
168             statusBar.removeAttribute("showErrors");
169             statusText.setAttribute("value", "");
170 
171             if (firebugButton)
172             {
173                 firebugButton.removeAttribute("showErrors");
174 
175                 // Use '0' so, the horizontal space for the number is still allocated.
176                 // The button will cause re-layout if there is more than 9 errors.
177                 firebugButton.setAttribute("errorCount", "0");
178             }
179         }
180     },
181 
182     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * //
183     // Tooltip
184 
185     resetTooltip: function()
186     {
187         if (FBTrace.DBG_TOOLTIP)
188             FBTrace.sysout("resetTooltip called");
189 
190         var tooltip = "Firebug " + Firebug.getVersion();
191 
192         tooltip += "\n" + this.getEnablementStatus();
193 
194         if (Firebug.getSuspended())
195             tooltip += "\n" + Firebug.getSuspended();
196         else
197             tooltip += "\n" + $STRP("plural.Total_Firebugs2", [Firebug.TabWatcher.contexts.length]);
198 
199         if (Firebug.allPagesActivation == "on")
200         {
201             var label = $STR("enablement.on");
202             tooltip += "\n"+label+" "+$STR("enablement.for all pages");
203         }
204         // else allPagesActivation == "none" we don't show it.
205 
206         tooltip += "\n" + $STR(Firebug.getPlacement());
207 
208         var firebugStatus = $("firebugStatus");
209         firebugStatus.setAttribute("tooltiptext", tooltip);
210     },
211 
212     getEnablementStatus: function()
213     {
214         var strOn = $STR("enablement.on");
215         var strOff = $STR("enablement.off");
216 
217         var status = "";
218         var firebugStatus = $("firebugStatus");
219 
220         if (firebugStatus.getAttribute("console") == "on")
221             status += "Console: " + strOn + ",";
222         else
223             status += "Console: " + strOff + ",";
224 
225         if (firebugStatus.getAttribute("net") == "on")
226             status += " Net: " + strOn + ",";
227         else
228             status += " Net: " + strOff + ",";
229 
230         if (firebugStatus.getAttribute("script") == "on")
231             status += " Script: " + strOn;
232         else
233             status += " Script: " + strOff + "";
234 
235         return status;
236     },
237 });
238 
239 // ********************************************************************************************* //
240 // Registration
241 
242 Firebug.registerModule(Firebug.StartButton);
243 
244 // ********************************************************************************************* //
245 }});
