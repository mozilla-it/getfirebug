239     // moz, called by back end
240     stop: function(context, frame, type, rv)
241     {
242         if (context.stopped)
243             return RETURN_CONTINUE;
244 
245         if (!this.isAlwaysEnabled())
246             return RETURN_CONTINUE;
247 
248         if (FBTrace.DBG_UI_LOOP)
249             FBTrace.sysout("debugger.stop "+context.getName()+" frame",frame);
250 
251         context.stoppedFrame = frame;  // the frame we stopped in, don't change this elsewhere.
252         context.currentFrame = frame;  // the frame we show to user, depends on selection
253         context.stopped = true;
254 
255         var hookReturn = dispatch2(this.fbListeners,"onStop",[context,frame, type,rv]);
256         if ( hookReturn && hookReturn >= 0 )
257         {
258             delete context.stopped;
259             delete context.stoppedFrame;
260             delete context.currentFrame;
261             if (FBTrace.DBG_UI_LOOP)
262                 FBTrace.sysout("debugger.stop extension vetoed stop with hookReturn "+hookReturn);
263 
264             return hookReturn;
265         }
266 
267         try
268         {
269             this.freeze(context);
270 
271             // We will pause here until resume is called
272             var depth = fbs.enterNestedEventLoop({onNest: bindFixed(this.startDebugging, this, context)});
273             // For some reason we don't always end up here

276                 FBTrace.sysout("debugger.stop, nesting depth:"+depth+" jsd.pauseDepth: "+
277                     jsd.pauseDepth+" context:"+context.getName());
278         }
279         catch (exc)
280         {
281             // Just ignore exceptions that happened while in the nested loop
282             if (FBTrace.DBG_ERRORS)
283                 FBTrace.sysout("debugger exception in nested event loop: "+exc, exc);
284             else
285                 ERROR("debugger exception in nested event loop: "+exc+"\n");
286         }
287         finally
288         {
289             this.thaw(context);
290         }
291 
292         this.stopDebugging(context);
293 
294         dispatch(this.fbListeners,"onResume",[context]);
295 
296         if (context.aborted)
297         {
298             delete context.aborted;
299             return RETURN_ABORT;
300         }
301         else if (Firebug.rerun)
302         {
303             setTimeout(function reExecute()
304             {
305                 var rerun = context.savedRerun = Firebug.rerun;
306                 delete Firebug.rerun;
307 
308                 if (FBTrace.DBG_UI_LOOP)
309                     FBTrace.sysout("Firebug.debugger.reExecute ", {rerun: rerun});
310 
311                 // fire the prestored script

313                 {
314                     if (FBTrace.DBG_UI_LOOP)
315                         FBTrace.sysout("Firebug.debugger.reExecute success", result);
316                     dispatch(Firebug.Debugger.fbListeners, "onRerunComplete", [true, result]);
317                 }
318 
319                 function exceptionFunction(result, context)
320                 {
321                     if (FBTrace.DBG_ERRORS)
322                         FBTrace.sysout("Firebug.debugger.reExecute FAILED "+result, result);
323                     dispatch(Firebug.Debugger.fbListeners, "onRerunComplete", [false, result]);
324                 }
325 
326                 Firebug.CommandLine.evaluate("window._firebug.rerunFunction()", context, null,
327                     context.window, successConsoleFunction, exceptionFunction);
328             });
329 
330             if (FBTrace.DBG_UI_LOOP)
331                 FBTrace.sysout("Firebug.debugger.reExecute return "+RETURN_HOOK_ERROR);
332 
333             return RETURN_HOOK_ERROR;
334         }
335         else
336             return RETURN_CONTINUE;

361     getRerun: function(context)
362     {
363         if (FBTrace.DBG_UI_LOOP)
364                 FBTrace.sysout("debugger.rerun for "+context.getName());
365         try
366         {
367             // walk back to the oldest frame, but not top level
368             var frame = context.stoppedFrame;
369             while (frame.callingFrame && frame.callingFrame.script.functionName)
370             {
371                 frame = frame.callingFrame;
372 
373                 if (frame.script.functionName == "_firebugRerun") // re-reRun
374                 {
375                     if (FBTrace.DBG_UI_LOOP)
376                         FBTrace.sysout("getRerun re-rerun ", context.savedRerun);
377                     return context.savedRerun;
378                 }
379             }
380 
381             // In this oldest frame we have element.onclick(event) or window.foo()
382             // We want to cause the page to run this again after we abort this call stack.
383             //
384             function getStoreRerunInfoScript(fnName)
385             {
386                 var str = "if (!window._firebug)window._firebug={};\n";
387                 str += "window._firebug.rerunThis = this;\n";
388                 str += "window._firebug.rerunArgs = [];\n"
389                 str += "if (arguments && arguments.length) for (var i = 0; i < arguments.length; i++) window._firebug.rerunArgs.push(arguments[i]);\n"
390                 str += "window._firebug.rerunFunctionName = "+fnName+";\n"
391                 str +="window._firebug.rerunFunction = function _firebugRerun() { "+fnName+".apply(window._firebug.rerunThis, window._firebug.rerunArgs); }"
392                 return str;
393             }
394 
395             var rerun = {};
396 
397             var fnName = getFunctionName(frame.script, context, frame, true);
398             rerun.script = getStoreRerunInfoScript(fnName);
399             var jsdFunctionName = frame.script.functionName;
400 
401             // now run the script that stores the rerun info in the page
402             var result = {};
403             var ok = frame.eval(rerun.script, context.window.location + "/RerunScript", 1, result);
404 
405             // If the eval goes off somewhere wacky, the frame may be invalid by this point.
406             if (FBTrace.DBG_UI_LOOP)
407                 FBTrace.sysout("debugger.rerun "+ok+" and result: "+result+" for "+context.getName(),
408                     {result: result, rerun: rerun, functionName: jsdFunctionName});
409         }
410         catch(exc)
411         {
412             if (FBTrace.DBG_ERRORS)
413                 FBTrace.sysout("debugger.rerun FAILS for "+context.getName()+" because "+exc,
414                     {exc:exc, rerun: rerun});
415         }
416 
417         return rerun;
418     },

502     freeze: function(context)
503     {
504         var executionContext = context.stoppedFrame.executionContext;
505         try {
506             executionContext.scriptsEnabled = false;
507             this.suppressEventHandling(context);
508             context.isFrozen = true;
509 
510             // https://developer.mozilla.org/en/XUL_Tutorial/Focus_and_Selection#Getting_the_currently_focused_element
511             if (context.window && context.window.document.commandDispatcher)
512             {
513                 context.saveFocus = context.window.document.commandDispatcher.focusedElement;
514                 if (context.saveFocus && !context.discardBlurEvents)
515                 {
516                     context.discardBlurEvents = function blurDiscarder(event)
517                     {
518                         if (!context.saveFocus)
519                         {
520                             context.window.removeEventListener('blur', context.discardBlurEvents, true);
521                             delete context.discardBlurEvents;
522                         }
523 
524                         if (FBTrace.DBG_UI_LOOP)
525                             FBTrace.sysout("debugger.freeze discard blur event "+context.saveFocus+
526                                 " while focus is "+context.window.document.commandDispatcher.focusedElement,
527                                 event);
528 
529                         event.preventDefault();
530                         event.stopPropagation();
531                     },
532 
533                     context.window.addEventListener('blur', context.discardBlurEvents, true);
534                 }
535             }
536 
537             if (FBTrace.DBG_UI_LOOP)
538             {
539                 FBTrace.sysout("debugger.freeze context.saveFocus "+context.saveFocus,
540                     context.saveFocus);
541 
542                 FBTrace.sysout("debugger.freeze try to disable scripts "+
543                     (context.eventSuppressor?"and events":"but not events")+" in "+
544                     context.getName()+" executionContext.tag "+executionContext.tag+
545                     ".scriptsEnabled: "+executionContext.scriptsEnabled);
546             }
547         }
548         catch (exc)
549         {
550             // This attribute is only valid for contexts which implement nsIScriptContext.
551             if (FBTrace.DBG_UI_LOOP)
552                 FBTrace.sysout("debugger.freeze, freeze exception "+exc+" in "+context.getName(), exc);
553         }
554     },

566     thaw: function(context)
567     {
568         try {
569             if (context.isFrozen)
570                 delete context.isFrozen;
571             else
572                 return; // bail, we did not freeze this context
573 
574                 var executionContext = context.stoppedFrame.executionContext;
575             if (executionContext.isValid)
576             {
577                 this.unsuppressEventHandling(context);
578 
579                 // Before we release JS, put the focus back
580                 if (context.saveFocus)
581                 {
582                     context.window.focus();
583                     context.saveFocus.focus();
584                     delete context.saveFocus;
585 
586                     if (FBTrace.DBG_UI_LOOP)
587                     {
588                         var nowFocused = context.window.document.commandDispatcher ?
589                             context.window.document.commandDispatcher.focusedElement : null;
590                         FBTrace.sysout("debugger.thaw context.saveFocus "+context.saveFocus+
591                             " vs "+nowFocused, context.saveFocus);
592                     }
593                 }
594 
595                 executionContext.scriptsEnabled = true;
596             }
597             else
598             {
599                 if (FBTrace.DBG_UI_LOOP)
600                     FBTrace.sysout("debugger.thaw "+executionContext.tag+" executionContext is not valid");
601             }
602 
603             if (FBTrace.DBG_UI_LOOP)
604                 FBTrace.sysout("debugger.thaw try to enable scripts " +
605                     (context.eventSuppressor?"with events suppressed":"events enabled")+
606                     " in "+context.getName()+" executionContext.tag "+executionContext.tag+
607                     ".scriptsEnabled: "+executionContext.scriptsEnabled);
608         }
609         catch (exc)
610         {
611             if (FBTrace.DBG_UI_LOOP)

2791 // Recursively look for obj in container using array of visited objects
2792 function findObjectPropertyPath(containerName, container, obj, visited)
2793 {
2794     if (!container || !obj || !visited)
2795         return false;
2796 
2797     var referents = [];
2798     visited.push(container);
2799     for (var p in container)
2800     {
2801         if (container.hasOwnProperty(p))
2802         {
2803             var candidate = null;
2804             try
2805             {
2806                 candidate = container[p];
2807             }
2808             catch(exc)
2809             {
2810                 // eg sessionStorage
2811             }
2812 
2813             if (candidate === obj) // then we found a property pointing to our obj
2814             {
2815                 referents.push(new Referent(containerName, container, p, obj));
2816             }
2817             else // recurse
2818             {
2819                 var candidateType = typeof (candidate);
2820                 if (candidateType === 'object' || candidateType === 'function')
2821                 {
2822                     if (visited.indexOf(candidate) === -1)
2823                     {
2824                         var refsInChildren = findObjectPropertyPath(p, candidate, obj, visited);
2825                         if (refsInChildren.length)
2826                         {
2827                             // As we unwind the recursion we tack on layers of the path.
2828                             for (var i = 0; i < refsInChildren.length; i++)
2829                             {
2830                                 var refInChildren = refsInChildren[i];
2831                                 refInChildren.prependPath(containerName, container);
2832                                 referents.push(refInChildren);
2833 
2834                                 FBTrace.sysout(" Did prependPath with p "+p+" gave "+
2835                                     referents[referents.length - 1].getObjectPathExpression(),
2836                                     referents[referents.length - 1]);
2837                             }
2838                         }
2839                     }
2840                     //else we already looked at that object.
2841                 } // else the object has no properties
2842             }
2843         }
2844     }
2845 
2846     FBTrace.sysout(" Returning "+referents.length+ " referents", referents);
2847 
2848     return referents;
2849 }
