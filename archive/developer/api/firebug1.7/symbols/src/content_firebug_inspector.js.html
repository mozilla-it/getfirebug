 34     /**
 35      * Main highlighter method. Can be used to highlight elements using the box model, frame or image map highlighters. Can highlight single or multiple elements.
 36      *
 37      * Examples:
 38      * Firebug.Inspector.highlightObject([window.content.document.getElementById('gbar'), window.content.document.getElementById('logo')], window.content, "frame", null, ["#ff0000",{background:"#0000ff", border:"#ff0000"}])
 39      * or
 40      * Firebug.Inspector.highlightObject([window.content.document.getElementById('gbar'), window.content.document.getElementById('logo')], window.content, "boxModel", null, [{content: "#ff0000", padding: "#eeeeee", border: "#00ff00", margin: "#0000ff"},{content: "#00ff00", padding: "#eeeeee", border: "#00ff00", margin: "#0000ff"}])
 41      *
 42      * @param {Array} elementArr Elements to highlight
 43      * @param {Window} context Context of the elements to be highlighted
 44      * @param {String} [highlightType] Either "frame" or "boxModel". Default is configurable.
 45      * @param {String} [boxFrame] Displays the line guides for the box model layout view. Valid values are
 46      *        "content", "padding", "border" or "margin"
 47      * @param {String | Array} [colorObj] Any valid html color e.g. red, #f00, #ff0000, etc., a valid color object
 48      *        or any valid highlighter color array.
 49      */
 50     highlightObject: function(elementArr, context, highlightType, boxFrame, colorObj)
 51     {
 52         var i, elt, elementLen, oldContext, usingColorArray;
 53         var highlighter = highlightType ? getHighlighter(highlightType) : this.defaultHighlighter;
 54 
 55         if (!elementArr || !FirebugReps.Arr.isArray(elementArr))
 56         {
 57             // highlight a single element
 58             if (!elementArr || !isElement(elementArr) || !isVisible(unwrapObject(elementArr)))
 59             {
 60                 if(elementArr && elementArr.nodeType == 3)
 61                     elementArr = elementArr.parentNode;
 62                 else
 63                     elementArr = null;
 64             }
 65 
 66             if (elementArr && context && context.highlightTimeout)
 67             {
 68                 context.clearTimeout(context.highlightTimeout);
 69                 delete context.highlightTimeout;
 70             }
 71 
 72             oldContext = this.highlightedContext;
 73             if (oldContext && oldContext.window)
 74                 this.clearAllHighlights();
 75 
 76             this.highlighter = highlighter;
 77             this.highlightedContext = context;
 78 
 79             if (elementArr)
 80             {
 81                 if(!isVisibleElement(elementArr))
 82                     highlighter.unhighlight(context);
 83                 else if (context && context.window && context.window.document)
 84                     highlighter.highlight(context, elementArr, boxFrame, colorObj, false);
 85             }
 86             else if (oldContext)
 87             {
 88                 oldContext.highlightTimeout = oldContext.setTimeout(function()
 89                 {
 90                     if (FBTrace.DBG_INSPECT)
 91                         FBTrace.sysout("Removing inspector highlighter due to setTimeout loop");
 92 
 93                     delete oldContext.highlightTimeout;
 94 
 95                     if (oldContext.window && oldContext.window.document)
 96                     {
 97                         highlighter.unhighlight(oldContext);
 98 
 99                         if (oldContext.inspectorMouseMove)
100                             oldContext.window.document.removeEventListener("mousemove",
101                                 oldContext.inspectorMouseMove, true);
102                     }
103                 }, inspectDelay);
104             }
105         }
106         else
107         {
108             // Highlight multiple elements
109             if (context && context.highlightTimeout)
110             {
111                 context.clearTimeout(context.highlightTimeout);
112                 delete context.highlightTimeout;
113             }
114             this.clearAllHighlights();
115             usingColorArray = FirebugReps.Arr.isArray(colorObj);
116 
117             if (context && context.window && context.window.document)
118             {
119                 for (i = 0, elementLen = elementArr.length; i < elementLen; i++)
120                 {
121                     elt = elementArr[i];
122 
123                     if (elt)
124                     {
125                         if(elt.nodeType === 3)
126                             elt = elt.parentNode;
127 
128                         if (usingColorArray)
129                             highlighter.highlight(context, elt, null, colorObj[i], true);
130                         else
131                             highlighter.highlight(context, elt, null, colorObj, true);
132                     }
133                 }
134             }
135 
136             storeHighlighterParams(null, context, elementArr, null, colorObj, highlightType, true);
137         }
138     },

214     /**
215      * Highlight a node using the frame highlighter. Can only be used after inspecting has already started.
216      * @param {Element} node The element to inspect
217      */
218     inspectNode: function(node)
219     {
220         if (node && node.nodeType != 1)
221             node = node.parentNode;
222 
223         if (node && unwrapObject(node).firebugIgnore && !node.fbProxyFor)
224             return;
225 
226         var context = this.inspectingContext;
227 
228         if (this.inspectTimeout)
229         {
230             context.clearTimeout(this.inspectTimeout);
231             delete this.inspectTimeout;
232         }
233 
234         if (node && node.fbProxyFor)
235             node = node.fbProxyFor;
236 
237         var inspectingPanel = this.inspectingPanel;
238 
239         // Some panels may want to only allow inspection of panel-supported objects
240         node = inspectingPanel ? inspectingPanel.getInspectNode(node) : node;
241 
242         var highlightColor = inspectingPanel ? inspectingPanel.inspectHighlightColor : "";
243         this.highlightObject(node, context, "frame", undefined, highlightColor);
244 
245         this.inspectingNode = node;
246 
247         if (node)
248         {
249             var _this = this;
250 
251             this.inspectTimeout = context.setTimeout(function()
252             {
253                 var selection = inspectingPanel.inspectNode(node);
254                 dispatch(_this.fbListeners, "onInspectNode", [context, node]);
255                 if (selection)
256                     inspectingPanel.select(node);
257             }, inspectDelay);
258         }
259     },

264      * @param {Boolean} [waitForClick] Indicates whether the next click will still forward you to the clicked element in the HTML panel

361     /**
362      * Repaint the highlighter. Called from the window scroll and resize handlers.
363      */
364     repaint: function()
365     {
366         var rp = this.repaint;
367         var highlighter = rp.highlighter;
368         var context = rp.context;
369         var element = rp.element;
370         var boxFrame = rp.boxFrame;
371         var colorObj = rp.colorObj;
372         var highlightType = rp.highlightType;
373         var isMulti = rp.isMulti;
374 
375         if(!context || (!highlighter && !isMulti))
376             return;
377 
378         if(isMulti && element)
379             this.highlightObject(element, context, highlightType, null, colorObj);
380         else
381         {
382             var highlighterNode = highlighterCache.get(highlighter.ident);
383 
384             if(highlighterNode && highlighter.ident === ident.boxModel)
385                 highlighterNode = highlighterNode.offset;
386 
387             if(highlighterNode && highlighterNode.parentNode)
388             {
389                 this.clearAllHighlights();
390                 highlighter.highlight(context, element, boxFrame, colorObj, isMulti);
391             }
392         }

397     /**
398      * Attach the scroll and resize handlers to elts window. Called from every highlight call.
399      * @param {Element} elt Passed in order to reliably obtain context
400      */
401     attachRepaintInspectListeners: function(elt)
402     {
403         if (!elt || !elt.ownerDocument || !elt.ownerDocument.defaultView)
404             return;
405 
406         var win = elt.ownerDocument.defaultView;
407 
408         if (FBTrace.DBG_INSPECT)
409             FBTrace.sysout("inspector.attachRepaintInspectListeners to " + win.location.href, elt);
410 
411         // there is no way to check if the listeners have already been added and we should avoid adding properties
412         // to the users page. Adding them again will do no harm so lets just do that.
413         win.document.addEventListener("resize", this.onInspectingResizeWindow, true);
414         win.document.addEventListener("scroll", this.onInspectingScroll, true);
415     },

417     /**
418      * Attach key and mouse events to windows recursively.
419      * @param {Window} context Context of the main browser window
420      */
421     attachInspectListeners: function(context)
422     {
423         var win = context.window;
424         if (!win || !win.document)
425             return;
426 
427         if (FBTrace.DBG_INSPECT)
428             FBTrace.sysout("inspector.attachInspectListeners to all subWindows of " + win.location);
429 
430         var chrome = Firebug.chrome;
431 
432         this.keyListeners =
433         [
434             chrome.keyCodeListen("RETURN", null, bindFixed(this.stopInspecting, this)),
435             chrome.keyCodeListen("ESCAPE", null, bindFixed(this.stopInspecting, this, true)),
436             chrome.keyCodeListen("UP", isControl, bindFixed(this.inspectNodeBy, this, "up"), true),
437             chrome.keyCodeListen("DOWN", isControl, bindFixed(this.inspectNodeBy, this, "down"), true),
438         ];
439 
440         iterateWindows(win, bind(function(subWin)
441         {
442             if (FBTrace.DBG_INSPECT)
443                 FBTrace.sysout("inspector.attacheInspectListeners to " + subWin.location +
444                     " subWindow of " + win.location);
445 
446             subWin.document.addEventListener("mouseover", this.onInspectingMouseOver, true);
447             subWin.document.addEventListener("mousedown", this.onInspectingMouseDown, true);
448             subWin.document.addEventListener("mouseup", this.onInspectingMouseUp, true);
449             subWin.document.addEventListener("click", this.onInspectingClick, true);
450         }, this));
451     },

453     /**
454      * Remove all event listeners except click listener from windows recursively.
455      * @param {Window} context Context of the main browser window
456      */
457     detachInspectListeners: function(context)
458     {
459         var i, keyListenersLen;
460         var win = context.window;
461 
462         if (!win || !win.document)
463             return;
464 
465         var chrome = Firebug.chrome;
466 
467         if (this.keyListeners)  // XXXjjb for some reason this is null some times.
468         {
469             keyListenersLen = this.keyListeners.length;
470             for (i = 0; i < keyListenersLen; ++i)
471                 chrome.keyIgnore(this.keyListeners[i]);
472             delete this.keyListeners;
473         }
474 
475         iterateWindows(win, bind(function(subWin)
476         {
477             subWin.document.removeEventListener("mouseover", this.onInspectingMouseOver, true);
478             subWin.document.removeEventListener("mousedown", this.onInspectingMouseDown, true);
479             subWin.document.removeEventListener("mouseup", this.onInspectingMouseUp, true);
480         }, this));
481     },
