  4 // Constants
  5 
  6 const prefs = fbXPCOMUtils.CCSV("@mozilla.org/preferences-service;1", "nsIPrefBranch");
  7 const names = ["label", "executable", "cmdline", "image"];
  8 
  9 // ************************************************************************************************
 10 // Globals
 11 
 12 var gEditorManager =
 13 {
 14     _tree : null,
 15     _data : [],
 16     _removeButton : null,
 17     _changeButton : null,
 18     _moveUpButton : null,
 19 
 20     init: function()
 21     {
 22         var args = window.arguments[0];
 23         this._FBL = args.FBL;
 24         this._prefName = args.prefName;
 25 
 26         (this._removeButton = document.getElementById("removeEditor")).disabled = true;
 27         (this._changeButton = document.getElementById("changeEditor")).disabled = true;
 28         (this._moveUpButton = document.getElementById("moveUpEditor")).disabled = true;
 29 
 30         this._tree = document.getElementById("editorsList");
 31 
 32         this._treeView =
 33         {
 34             data: this._data,
 35             selection: null,
 36 
 37             get rowCount() { return this.data.length; },
 38             getCellText: function(row, column)
 39             {
 40                 switch(column.id)
 41                 {
 42                 case "editorName":
 43                     return " "+this.data[row].label;
 44                 case "editorExecutable":
 45                     return this.data[row].executable;
 46                 case "editorParams":
 47                     return this.data[row].cmdline;
 48                 }
 49                 return "";
 50             },
 51             setTree: function(treebox){ this.treebox = treebox; },
 52             isContainer: function(row) { return false; },
 53             isContainerOpen: function(row) { return false; },
 54             isContainerEmpty: function(row) { return false; },
 55             isSeparator: function(row) { return false; },
 56             isSorted: function() { return false; },
 57             getLevel: function(row) { return 0; },
 58             getImageSrc: function(row,column) { return column.id=="editorName" ? this.data[row].image : null; },
 59             getRowProperties: function(row,props) {},
 60             getCellProperties: function(row,column,props) {},
 61             getColumnProperties: function(colid,column,props) {}
 62         };
 63 
 64         this._load();
 65         this._tree.view = this._treeView;
 66 
 67         this.internationalizeUI(document);
 68     },
 69 
 70     uninit: function()
 71     {
 72     },
 73 
 74     internationalizeUI: function(doc)
 75     {
 76         var elements = doc.getElementsByClassName("fbInternational");
 77         var attributes = ["title", "label", "value"];
 78         for (var i=0; i<elements.length; i++)
 79         {
 80             for(var j=0; j<attributes.length; j++)
 81             {
 82                 if (elements[i].hasAttribute(attributes[j]))
 83                     this._FBL.internationalize(elements[i], attributes[j]);
 84             }
 85         }
 86     },
 87 
 88     onSelectionChanged: function()
 89     {
 90         var selection = this._tree.view.selection;
 91         this._removeButton.disabled = (selection.count != 1);
 92         this._changeButton.disabled = (selection.count != 1);
 93         this._moveUpButton.disabled = (selection.count != 1) || (selection.currentIndex == 0);
 94     },
 95 
 96     addEditorHandler: function()
 97     {
 98         var item = { label: "", executable: null, cmdline: "" };
 99         var result = {};
100         var args = {
101             item: item,
102             FBL: this._FBL
103         };
104         openDialog("chrome://firebug/content/external/changeeditor.xul",  "_blank", "modal,centerscreen,resizable", args, result);
105         if (result.saveChanges)
106         {
107             item.id = item.label.replace(/\W/g, "_");
108             this._saveItem(item);
109 
110             this._loadItem(item);
111             this._data.push(item);
112             this._tree.view = this._treeView;
113 
114             var editors = [];
115             try {
116                 editors = prefs.getCharPref(this._prefName).split(",");
117                 for( var i = 0; i < editors.length; ++i )
118                 {
119                     if ( editors[i].replace(/^\s+|\s+$/,"") == "" )
120                         editors.splice(i, 1);
121                 }
122             }
123             catch(exc)
124             {
125                 this._FBL.ERROR(exc);
126             }
127             editors.push(item.id);
128             prefs.setCharPref(this._prefName, editors.join(","));
129         }
130     },
131 
132     removeEditorHandler: function()
133     {
134         var selection = this._tree.view.selection;
135         if (selection.count < 1)
136             return;
137         var item = this._data[selection.currentIndex];
138         this._data.splice(selection.currentIndex, 1);
139         this._tree.view = this._treeView;
140 
141         try {
142             var editors = prefs.getCharPref(this._prefName).split(",");
143             this._FBL.remove(editors, item.id);
144             prefs.setCharPref(this._prefName, editors.join(","));
145             prefs.deleteBranch(this._prefName+"."+item.id);
146         }
147         catch(exc)
148         {
149             this._FBL.ERROR(exc);
150         }
151     },
152 
153     changeEditorHandler: function()
154     {
155         var selection = this._tree.view.selection;
156         if (selection.count != 1)
157             return;
158         var item = this._data[selection.currentIndex];
159         var args = {
160             item: item,
161             FBL: this._FBL
162         };
163         var result = {};
164         openDialog("chrome://firebug/content/external/changeeditor.xul",  "_blank", "modal,centerscreen", args, result);
165         if (result.saveChanges)
166         {
167             this._saveItem(item);
168         }
169         this._loadItem(item);
170         this._tree.view = this._treeView;
171     },
172 
173     moveUpEditorHandler: function()
174     {
175         var selection = this._tree.view.selection;
176         if (selection.count < 1)
177             return;
178         var item = this._data[selection.currentIndex];
179         this._data.splice(selection.currentIndex, 1);
180         this._data.unshift(item);
181         this._tree.view = this._treeView;
182         try {
183             var editors = this._data.map(function(x) x.id);
184             prefs.setCharPref(this._prefName, editors.join(","));
185         }
186         catch(exc)
187         {
188             this._FBL.ERROR(exc);
189         }
190     },
191 
192 
193     _loadItem: function(item)
194     {
195         const prefName = this._prefName;
196         for( var i = 0; i < names.length; ++i )
197         {
198             try {
199                 item[names[i]] = prefs.getCharPref(prefName+"."+item.id+"."+names[i]);
200             }
201             catch(exc)
202             {}
203         }
204         if (!item.image)
205             item.image = this._FBL.getIconURLForFile(item.executable);
206     },
207 
208     _saveItem: function(item)
209     {
210         if ( item.image && item.image == this._FBL.getIconURLForFile(item.executable) )
211             item.image = null;
212 
213         const prefName = this._prefName;
214         for( var i = 0; i < names.length; ++i )
215         {
216             try {
217                 var value = item[names[i]];
218                 if ( value )
219                     prefs.setCharPref(prefName+"."+item.id+"."+names[i], value);
220                 else
221                     prefs.clearUserPref(prefName+"."+item.id+"."+names[i]);
222             }
223             catch(exc)
224             {}
225         }
226     },
227 
228     _load: function()
229     {
230         try {
231             var list = prefs.getCharPref(this._prefName).split(",");
232             for (var i = 0; i < list.length; ++i)
233             {
234                 var editorId = list[i].replace(/\s/g, "_");
235                 if ( !editorId )
236                     continue;
237                 var item = { id: editorId };
238                 this._data.push(item);
239                 this._loadItem(item);
240             }
241         }
242         catch(exc)
243         {
244             this._FBL.ERROR(exc);
245         }
246     }
247 
248 };
