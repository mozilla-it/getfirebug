  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 // Constants
  7 
  8 // ************************************************************************************************
  9 // Implementation
 10 
 11 /**
 12  * @module Command Line availability in other panels.
 13  */
 14 Firebug.CommandLine.Popup = extend(Firebug.Module,
 15 {
 16     lastFocused : null,
 17 
 18     initializeUI: function()
 19     {
 20         Firebug.Module.initializeUI.apply(this, arguments);
 21 
 22         this.setPopupBrowserStyle(Firebug.chrome);
 23 
 24         this.onKeyPress = bind(this.onKeyPress, this);
 25 
 26         this.attachListeners();
 27     },
 28 
 29     shutdown: function()
 30     {
 31         Firebug.chrome.$("fbContentBox").removeEventListener("keypress", this.onKeyPress, false);
 32     },
 33 
 34     reattachContext: function(browser, context)
 35     {
 36         this.setPopupBrowserStyle(Firebug.chrome);
 37         this.attachListeners();
 38     },
 39 
 40     showPanel: function(browser, panel)
 41     {
 42         if (FBTrace.DBG_COMMANDLINE)
 43             FBTrace.sysout("commandLine.Popup.showPanel; " + (panel?panel.name:"null panel"));
 44 
 45         var chrome = Firebug.chrome;
 46         var visible = this.isVisible();
 47         var isConsole = (panel && panel.name == "console");
 48         var largeCmd = Firebug.largeCommandLine;
 49 
 50         // Disable the console popup button (Firebug toolbar) if the Console panel
 51         // is disabled or selected.
 52         var consolePanelType = Firebug.getPanelType("console");
 53         var disabled = consolePanelType.prototype.isEnabled() ? "false" : "true";
 54         if (isConsole || !panel)
 55             disabled = "true";
 56 
 57         chrome.$("fbCommandPopupButton").setAttribute("disabled", disabled);
 58 
 59         if ((largeCmd && isConsole) || !panel)
 60         {
 61             collapse(chrome.$("fbPanelSplitter"), panel ? false : true);
 62             collapse(chrome.$("fbSidePanelDeck"), panel ? false : true);
 63             collapse(chrome.$("fbCommandBox"), true);
 64             chrome.$("fbSidePanelDeck").selectedPanel = chrome.$("fbLargeCommandBox");
 65         }
 66 
 67         // The console can't be multiline on other panels so, hide the toggle-to-multiline
 68         // button (displayed at the end of the one line command line)
 69         collapse(chrome.$("fbCommandToggleSmall"), !isConsole);
 70 
 71         // Update visibility of the console-popup (hidden if the Console panel is selected).
 72         this.updateVisibility(visible && !isConsole && panel && disabled != "true");
 73 
 74         // Make sure the console panel is attached to the proper document
 75         // (the one used by all panels, or the one used by console popup and available
 76         // for all the panels).
 77         if (panel)
 78             this.reattach(panel.context);
 79 
 80         // If the the console panel is opened on another panel, simulate show event for it.
 81         if (panel && !isConsole && visible)
 82             this.showPopupPanel(panel.context);
 83     },
 84 
 85     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
 86 
 87     setPopupBrowserStyle: function(chrome)
 88     {
 89         // Set additional style so we can make the panelNode-console node
 90         // always visible regardless of the currently selected panel.
 91         var doc = chrome.$("fbCommandPopupBrowser").contentDocument;
 92         var body = getBody(doc);
 93         setClass(body, "commandPopup");
 94     },
 95 
 96     attachListeners: function()
 97     {
 98         // Register event listeners.
 99         Firebug.chrome.$("fbContentBox").addEventListener("keypress", this.onKeyPress, false);
100     },
101 
102     toggle: function(context)
103     {
104         var panel = Firebug.chrome.getSelectedPanel();
105         if (panel && panel.name == "console")
106             return;
107 
108         if (FBTrace.DBG_COMMANDLINE)
109             FBTrace.sysout("commandLine.Popup.toggle;");
110 
111         var newState = !this.isVisible();
112         Firebug.chrome.setGlobalAttribute("cmd_toggleCommandPopup", "checked", newState);
113         this.updateVisibility(newState);
114 
115         this.reattach(context);
116         this.showPopupPanel(context);
117     },
118 
119     showPopupPanel: function(context)
120     {
121         // If the the console panel is opened on another panel, simulate show event for it.
122         if (this.isVisible())
123         {
124             var panel = context.getPanel("console", true);
125             if (panel)
126             {
127                 var state = Firebug.getPanelState(panel);
128                 panel.showPanel(state);
129             }
130         }
131     },
132 
133     updateVisibility: function(visible)
134     {
135         var chrome = Firebug.chrome;
136         var popup = chrome.$("fbCommandPopup");
137         var splitter = chrome.$("fbCommandPopupSplitter")
138         var cmdbox = chrome.$("fbCommandBox");
139         var toggle = chrome.$("fbCommandToggleSmall");
140 
141         // If all the visual parts are already visible then bail out.
142         if (visible && !isCollapsed(popup) && !isCollapsed(splitter) &&
143             !isCollapsed(cmdbox) && !isCollapsed(toggle))
144             return;
145 
146         collapse(popup, !visible);
147         collapse(splitter, !visible);
148         collapse(cmdbox, !visible);
149 
150         // The command line can't be multiline in other panels.
151         collapse(toggle, visible);
152 
153         var commandLineSmall = Firebug.CommandLine.getCommandLineLarge();
154         var commandLineLarge = Firebug.CommandLine.getCommandLineSmall();
155 
156         // Focus the command line if it has been just displayed.
157         if (visible)
158         {
159             this.lastFocused = document.commandDispatcher.focusedElement;
160             commandLineSmall.focus();
161         }
162         else if (this.lastFocused && isVisible(this.lastFocused) &&
163             typeof this.lastFocused.focus == "function")
164         {
165             this.lastFocused.focus();
166             this.lastFocused = null;
167         }
168 
169         if (Firebug.largeCommandLine)
170         {
171             if (visible)
172                 commandLineSmall.value = stripNewLines(commandLineLarge.value);
173             else
174                 commandLineLarge.value = cleanIndentation(commandLineSmall.value);
175         }
176     },
177 
178     isVisible: function()
179     {
180         var checked = Firebug.chrome.getGlobalAttribute("cmd_toggleCommandPopup", "checked");
181         return (checked == "true") ? true : false;
182     },
183 
184     reattach: function(context)
185     {
186         var consolePanelType = Firebug.getPanelType("console");
187         var doc = Firebug.chrome.getPanelDocument(consolePanelType);
188 
189         // Console doesn't have to be available (e.g. disabled)
190         var panel = context.getPanel("console", true);
191         if (panel)
192             panel.reattach(doc);
193     },
194 
195     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
196     // Event Listeners
197 
198     onKeyPress: function(event)
199     {
200         if (event.altKey || event.ctrlKey || event.metaKey || event.shiftKey)
201             return false;
202 
203         // ESC
204         var target = event.target;
205         // prevent conflict with inline editors being closed
206         if (target && event.keyCode == 27 && !hasClass(event.target, "textEditorInner"))
207             this.toggle(Firebug.currentContext);
208     }
209 });
210 
211 // ************************************************************************************************
212 // Registration
213 
214 Firebug.registerModule(Firebug.CommandLine.Popup);
215 
216 // ************************************************************************************************
217 return Firebug.CommandLine.Popup;
218 }});
