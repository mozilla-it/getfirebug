 25 /*
 26  * @param global: the global object to use for the execution context associated with the module loader.
 27  * use null for system modules, then {context:<string>} to config (required)
 28  * use |window| for window modules, config optional
 29  * @param requirejsConfig: object matching http://requirejs.org/docs/api.html#config
 30  * @param securityOrigin: a window, URI, or nsIPrincipal. Overrides global as origin URI for scripts in sandbox
 31  * Operations on the global object in the evaled code must be carefully studied unless
 32  * the securityOrigin is more restricting.

 35 function ModuleLoader(global, requirejsConfig, securityOrigin) {
 36     this.config = ModuleLoader.copyProperties({}, requirejsConfig);
 37     this.global = global;
 38     this.securityOrigin;
 39 
 40     this.registry = {};
 41     this.totalEvals = 0;
 42     this.totalEntries = 0;
 43 
 44     ModuleLoader.instanceCount += 1;
 45     this.instanceCount = ModuleLoader.instanceCount;
 46 
 47     var self = this;
 48     this.loadDepsThenCallback  = function() { // during the ctor call, bind a ref to the loader
 49         return self.prefixWithConfig.apply(self, arguments);  // use the bound ref to call apply with proper |this|
 50     }
 51     this.define = this.loadDepsThenCallback;
 52 
 53     ModuleLoader.currentModuleLoader = this;
 54 
 55     if (!ModuleLoader.loaders) {
 56         ModuleLoader.loaders = [];
 57     }
 58     ModuleLoader.loaders.push(this);
 59 }

110 /*
111  * @return reference to lhs, overridden by properties of rhs
112  * @param lhs left hand side object, properties may be overwritten by rhs
113  * @param rhs (optional) right hand side object, properties copied over lhs
114  */
115 ModuleLoader.copyProperties = function(lhs, rhs) {
116     for (var p in rhs) {
117         try {
118             lhs[p] = rhs[p];
119         } catch (exc) {
120             // On the Mozila platform we get "Operation is not supported"
121         }
122     }
123     return lhs;
124 }

129 ModuleLoader.bootStrap = function(requirejsPath) {
130     var primordialLoader = new ModuleLoader(null, {context: "_Primordial"});
131     try {
132         if (ModuleLoader.debug) consoleService.logStringMessage("ModuleLoader bootstrap from "+requirejsPath);
133         ModuleLoader.bootstrapUnit = primordialLoader.loadModule(requirejsPath);
134     } catch (exc) {
135         consoleService.logStringMessage("ModuleLoader bootstrap ERROR "+exc);
136     }
137 
138     // require.js does not export so we need to fix that
139     ModuleLoader.bootstrapUnit.exports = {
140         require: ModuleLoader.bootstrapUnit.sandbox.require,
141         define: ModuleLoader.bootstrapUnit.sandbox.define
142     };
143     return ModuleLoader.bootstrapUnit.exports;
144 }

151     /*
152      *  @return produces the global object for the execution context associated with moduleLoader.

154     globalObject: function () {

157     /*
158      * @return registers a frozen object as a top-level module in the module loader's registry. The own-properties of the object are treated as the exports of the module instance.
159      */
160     attachModule: function(name, module) {
161         this.registry[name] = module;  // its a lie, we register compilation units
162         this.totalEntries++;
163     },
164     /*
165      * @return the module instance object registered at name, or null if there is no such module in the registry.
166      */
167     getModule: function(name) {
168         var entry = this.registry[name];
169         if (entry) return entry.exports;
170     },
171     /*
172      * @param unit compilation unit: {
173      * 	source: a string of JavaScript source,
174      *  url: identifier,
175      *  jsVersion: JavaScript version to compile under
176      *  staringLineNumber: offset for source line numbers.
177      * @return completion value
178      */
179     evalScript: function(unit) {
180         try {
181             unit.jsVersion = unit.jsVersion || "1.8";
182             unit.url = unit.url || (this.getModuleLoaderName() + this.totalEvals)
183             unit.startingLineNumber = unit.startingLineNumber || 1;
184             // beforeCompilationUnit
185             var evalResult = Cu.evalInSandbox(unit.source, unit.sandbox,  unit.jsVersion, unit.url, unit.startingLineNumber);
186             // afterCompilationUnit
187             this.totalEvals += 1;
188             return evalResult;
189         } catch (exc) {
190             var msg = exc.toString() +" "+(exc.fileName || exc.sourceName) + "@" + exc.lineNumber;
191             return ModuleLoader.onError(msg, {exception: exc, compilationUnit: unit, moduleLoader: this});
192         }
193     },

195     loadModule: function(mrl, callback) {
196         try {
197             var mozURI = ModuleLoader.mozIOService.newURI(mrl, null, (this.baseURI ? this.baseURI : null));
198             var url = mozURI.spec;
199 
200             if (!this.baseURI) {  // then we did not have one configured before, use the first one we see
201                 var baseURL = url.split('/').slice(0,-1).join('/');
202                 this.baseURI =  ModuleLoader.mozIOService.newURI(mrl, null, null);
203             }
204 
205         } catch (exc) {
206             return ModuleLoader.onError(new Error("ModuleLoader could not convert "+mrl+" to absolute URL using baseURI "+this.baseURI.spec), {exception: exc, moduleLoader: this});
207         }
208         if (ModuleLoader.debug) ModuleLoader.onDebug("ModuleLoader loadModule reading "+url);
209 
210         var unit = {
211             source: this.mozReadTextFromFile(url),
212             url: url,
213             mrl: mrl, // relative
214         }
215         var thatGlobal = unit.sandbox = this.getSandbox(unit);
216 
217         // **** For security analysis we need to recognize that these added properties are visible to evaled code. ****
218 
219         // Any properties of this.global that are functions compiled in chrome scope become exposed to evaled code.
220         if (this.global) {
221             ModuleLoader.copyProperties(thatGlobal, this.global);
222         }
223 
224         this.loadModuleLoading(thatGlobal);  // only for system sandboxes.
225 
226         // *** end of added properties ****
227 
228         thatGlobal.exports = {}; // create the container for the module to fill with exported properties
229         unit.exports = thatGlobal.exports; // point to the container before the source can muck with it.
230         unit.evalResult = this.evalScript(unit);
231         for (var p in unit.exports) {
232             if (unit.exports.hasOwnProperty(p)) { // then we had at least on export
233                 if (callback) {
234                     callback(unit.exports);  // this call throws we do not register the module?
235                 }
236             }
237         }
238         this.attachModule(url, unit);  // even if we don't have any valid exports, so we can try to finish dependencies
239 
240         if (ModuleLoader.debug) ModuleLoader.onDebug("ModuleLoader loaded "+url, {compilationUnit: unit, moduleLoader: this});
241         return unit;
242     },

253     prefixWithConfig: function (deps, callback) {
254         var firstArg = arguments[0];
255 
256         if (firstArg) {
257             if (coreRequire.isArray(firstArg) || typeof( firstArg ) === "string") {  // then deps is first arg
258                 var args = [{}];  // start with our our new first arg
259                 for (var i = 0; i < arguments.length; ++i) {
260                        args.push(arguments[i]);
261                 }
262             } else { // then caller wants requirejs config api
263                 var args = arguments;
264             }
265             args[0] = this.remapConfig(ModuleLoader.copyProperties(args[0], this.config));
266 
267             if (args[0].onError) {
268                 if (ModuleLoader.debug) ModuleLoader.onDebug("ModuleLoader overriding onError ");
269 
270                 this.saveOnError = coreRequire.onError;
271                 coreRequire.onError = args[0].onError;
272             }
273 
274             coreRequire.apply(null, args);
275 
276             if (this.saveOnError) {
277                 coreRequire.onError = this.saveOnError;
278                 delete this.saveOnError;
279             }
280         } else {
281             if (ModuleLoader.debug) ModuleLoader.onDebug("ModuleLoader.loadDepsThenCallback(deps, callback), deps string or array of strings, callback called after strings resolved and loaded", this);
282         }
283     },

285     remapConfig: function(cfg) {
286         if (!cfg.context) {
287             // The require.js config object uses 'context' property name to mean 'contextName'.
288             cfg.context = this.getModuleLoaderName();
289         } // else caller better know what they are doing...
290 
291         if (cfg.baseUrl) {
292             try {
293                 this.baseURI = ModuleLoader.mozIOService.newURI(cfg.baseUrl, null, null);
294             } catch (exc) {
295                 throw new Error("ModuleLoader ERROR failed to create baseURI from baseUrl =\'"+cfg.baseUrl+"\'");
296             }
297         }
298         else if (this.baseURI) {
299             cfg.baseUrl = this.baseURI.spec;
300         }
301 
302         if (cfg.debug)
303             ModuleLoader.debug = !!cfg.debug;
304         if (cfg.onDebug)
305             ModuleLoader.onDebug = cfg.onDebug;
306 
307         if (cfg.onError)
308             ModuleLoader.onError = cfg.onError;
309 
310         return cfg;
311     },

319     getPrincipal: function() {
320         if (!this.principal) {
321             if (this.securityOrigin) {
322                 this.principal = this.securityOrigin;
323             } else if (this.global && (this.global instanceof Ci.nsIDOMWindow)) {
324                 this.principal = this.global;
325             } else {
326                 this.principal = ModuleLoader.systemPrincipal;
327             }
328         }
329         return this.principal;
330     },

332     getModuleLoaderName: function()	{
333         if (!this.name)	{
334             if (this.config.context) {
335                 this.name = this.config.context
336             } else if (this.global) {
337                 if (this.global instanceof Ci.nsIDOMWindow) {
338                     this.name = this.safeGetWindowLocation(this.global);
339                 } else {
340                     this.name = (this.global + "" + this.instanceCount).replace(/\s/,'_');
341                 }
342             }
343             else {
344                 this.name = "ModuleLoader_"+this.instanceCount;
345             }
346         }
347         return this.name;
348     },

354     safeGetWindowLocation: function(window)	{
355         try {
356             if (window) {
357                 if (window.closed) {
358                     return "(window.closed)";
359                 }
360                 else if ("location" in window) {
361                     return window.location+"";
362                 }
363                 else {
364                     return "(no window.location)";
365                 }
366             }
367             else {
368                 return "(no context.window)";
369             }
370         } catch(exc) {
371             return "(getWindowLocation: "+exc+")";
372         }

375     mozReadTextFromFile: function(pathToFile) {
376         try {
377             var channel = ModuleLoader.mozIOService.newChannel(pathToFile, null, null);
378             var inputStream = channel.open();
379 
380             var ciStream = Cc["@mozilla.org/intl/converter-input-stream;1"]
381                 .createInstance(Ci.nsIConverterInputStream);
382 
383             var bufLen = 0x8000;
384             ciStream.init(inputStream, "UTF-8", bufLen,
385                           Ci.nsIConverterInputStream.DEFAULT_REPLACEMENT_CHARACTER);
386             var chunk = {};
387             var data = "";
388 
389             while (ciStream.readString(bufLen, chunk) > 0) {
390                 data += chunk.value;
391             }
392 
393             ciStream.close();
394             inputStream.close();
395 
396             return data;
397         } catch (err) {
398             if (err.name === "NS_ERROR_FILE_NOT_FOUND") {
399                 // TODO: The call stack needs to point to the caller here
400                 var callsite = "";
401                 var caller = err.location ? err.location.caller : null;
402                 while (caller) {
403                     if (caller.filename === "resource://firebug/moduleLoader.js" && caller.lineNumber === 49) {
404                         if (caller.caller) {
405                             callsite = caller.caller.filename +"@"+caller.caller.lineNumber;
406                         }
407                         break;
408                     }
409                     caller = caller.caller;
410                 }
411                 return ModuleLoader.onError(new Error("ModuleLoader file not found "+pathToFile+" "+callsite), {err:err, pathToFile: pathToFile, moduleLoader: this});
412             }
413             return ModuleLoader.onError(new Error("mozReadTextFromFile; EXCEPTION "+err), {err:err, pathToFile: pathToFile, moduleLoader: this});
414         }
415     },

458 function loadCompilationUnit(moduleLoader, context, url, moduleName) {
459     try {
460         var unit = moduleLoader.loadModule(url);
461         context.completeLoad(moduleName);             // round up all the dependencies
462         return unit;
463     } catch (exc) {
464         var errorURL = exc.filename || exc.sourceName || exc.fileName;
465         var errorLineNumber = exc.lineNumber;
466         ModuleLoader.onError("loadCompilationUnit got exception "+exc+" on "+errorURL+"@"+errorLineNumber+", trying "+moduleLoader.config.edit);
467         if (moduleLoader.config.edit) {
468             return moduleLoader.config.edit(exc, errorURL, errorLineNumber);
469         }
470         throw exc;
471     }
472 }

474 // Override to connect require.js to our loader
475 coreRequire.load = function (context, moduleName, url) {
476 
477     this.s.isDone = false; // signal for require.ready()
478 
479     context.loaded[moduleName] = false; //in process of loading.
480     context.scriptCount += 1;
481 
482     var moduleLoader = ModuleLoader.get(context.contextName); // set in config for each subsystem
483 
484     if (moduleLoader) {
485         var unit = null;
486         while (!unit) {
487             ModuleLoader.onDebug("ModuleLoader coreRequire.load loadCompilationUnit "+url);
488             unit = loadCompilationUnit(moduleLoader, context, url, moduleName);
489         }
490         unit.exports = context.defined[moduleName];   // remember what we exported.
491         if (ModuleLoader.debug) ModuleLoader.onDebug(context.contextName+" loaded "+url);
492     } else {
493         return ModuleLoader.onError( new Error("require.attach called with unknown moduleLoaderName "+context.contextName+" for url "+url), ModuleLoader );
494     }
495 };

497 coreRequire.analyzeFailure = function(context, managers, specified, loaded) {
498     for (var i = 0; i < managers.length; i++) {
499         var manager = managers[i];
500         context.config.onDebug("require.js ERROR failed to complete "+manager.fullName+" isDone:"+manager.isDone+" #defined: "+manager.depCount+" #required: "+manager.depMax);
501         var theNulls = [];
502         var theUndefineds = [];
503         var theUnstrucks = manager.strikeList;
504         var depsTotal = 0;
505         for( var depName in manager.deps) {
506             if (typeof (manager.deps[depName]) === "undefined") theUndefineds.push(depName);
507             if (manager.deps[depName] === null) theNulls.push(depName);
508             var strikeIndex = manager.strikeList.indexOf(depName);
509             manager.strikeList.splice(strikeIndex, 1);
510         }
511         context.config.onDebug("require.js: "+theNulls.length+" null dependencies "+theNulls.join(',')+" << check module ids.", theNulls);
512         context.config.onDebug("require.js: "+theUndefineds.length+" undefined dependencies: "+theUndefineds.join(',')+" << check module return values.", theUndefineds);
513         context.config.onDebug("require.js: "+theUnstrucks.length+" unstruck dependencies "+theUnstrucks.join(',')+" << check duplicate requires", theUnstrucks);
514 
515         for (var j = 0; j < manager.depArray.length; j++) {
516             var id = manager.depArray[j];
517             var module = manager.deps[id];
518             context.config.onDebug("require.js: "+j+" specified: "+specified[id]+" loaded: "+loaded[id]+" "+id+" "+module);
519         }
520     }
