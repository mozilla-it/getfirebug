 45 // JSD Breakpoints
 46 
 47 // jsd breakpoints are on a PC in a jsdIScript
 48 // Users breakpoint on a line of source
 49 // Because test.js can be included multiple times, the URL+line number from the UI is not unique.
 50 // sourcefile.href != script.fileName, generally script.fileName cannot be used.
 51 // If the source is compiled, then we have zero, one, or more jsdIScripts on a line.
 52 //    If zero, cannot break at that line
 53 //    If one, set a jsd breakpoint
 54 //    If more than one, set jsd breakpoint on each script
 55 // Else we know that the source will be compiled before it is run.
 56 //    Save the sourceFile.href+line and set the jsd breakpoint when we compile
 57 //    Venkman called these "future" breakpoints
 58 //    We cannot prevent future breakpoints on lines that have no script.
 59 //    Break onCreate with error?
 60 var Breakpoints =
 61 {
 62     onBreakpoint: function(frame, type, val)
 63     {
 64         if (fbs.isTopLevelScript(frame, type, val))
 65         {
 66             if (FBTrace.DBG_FBS_BP)
 67                 FBTrace.sysout("onBreakpoint isTopLevel returning "+RETURN_CONTINUE);
 68 
 69             return RETURN_CONTINUE;
 70         }
 71 
 72         var bp = this.findBreakpointByScript(frame.script, frame.pc);
 73         if (bp)
 74         {
 75             var theDebugger = fbs.getDebuggerByName(bp.debuggerName);
 76             if (!theDebugger)
 77                 theDebugger = this.findDebugger(frame);  // sets debuggr.breakContext
 78 
 79             // See issue 1179, should not break if we resumed from a single step and have not advanced.
 80             if (disabledCount || monitorCount || conditionCount || runningUntil)
 81             {
 82                 if (FBTrace.DBG_FBS_BP)
 83                 {
 84                     FBTrace.sysout("onBreakpoint("+getExecutionStopNameFromType(type)+") disabledCount:"+disabledCount
 85                               +" monitorCount:"+monitorCount+" conditionCount:"+conditionCount+" runningUntil:"+runningUntil, bp);
 86                 }
 87 
 88                 if (bp.type & BP_MONITOR && !(bp.disabled & BP_MONITOR))
 89                 {
 90                     if (bp.type & BP_TRACE && !(bp.disabled & BP_TRACE) )
 91                         this.hookCalls(theDebugger.onFunctionCall, true);  // TODO
 92                      else
 93                         theDebugger.onMonitorScript(frame);
 94                 }
 95 
 96                 if (bp.type & BP_UNTIL)  // then we hit the runningUntil breakpoint
 97                 {
 98                     if (runningUntil)
 99                     {
100                         this.removeBreakpoint(BP_UNTIL, runningUntil.href, runningUntil.lineNo);
101                         runningUntil = null;
102                     }
103                     else
104                     {
105                         if (FBTrace.DBG_FBS_ERRORS)
106                             FBTrace.sysout("BP_UNTIL but not runningUntil!", bp);
107                     }
108 
109                     if (theDebugger)
110                         return this.breakIntoDebugger(theDebugger, frame, type);
111                 }
112                 else if (!(bp.type & BP_NORMAL) || bp.disabled & BP_NORMAL)
113                 {
114                     return  RETURN_CONTINUE;
115                 }
116                 else if (bp.type & BP_NORMAL)
117                 {
118                     var passed = testBreakpoint(frame, bp);
119                     if (!passed)
120                         return RETURN_CONTINUE;
121                 }
122                 // type was normal, but passed test
123             }
124             else  // not special, just break for sure
125                 return this.breakIntoDebugger(theDebugger, frame, type);
126         }
127         else
128         {
129             if (FBTrace.DBG_FBS_BP)
130                 FBTrace.sysout("onBreakpoint("+getExecutionStopNameFromType(type)+") NO bp match with frame.script.tag="+frame.script.tag+" clearing and continuing");
131             // We did not find a logical breakpoint to match the one set into JSD, so stop trying.
132             frame.script.clearBreakpoint(frame.pc);
133             return RETURN_CONTINUE;
134         }
135 
136         if (runningUntil)
137             return RETURN_CONTINUE;
138         else
139             return fbs.routeBreakToDebuggr(frame, type, val);
140     },

144     setBreakpoint: function(sourceFile, lineNo, props, debuggr)
145     {
146         var bp = this.addBreakpoint(BP_NORMAL, sourceFile, lineNo, props, debuggr);
147         if (bp)
148         {
149             dispatch(debuggers, "onToggleBreakpoint", [sourceFile.href, lineNo, true, bp]);
150             this.saveBreakpoints(sourceFile.href);  // after every call to onToggleBreakpoint
151             return true;
152         }
153         return false;
154     },
155 
156     clearBreakpoint: function(url, lineNo)
157     {
158         var bp = this.removeBreakpoint(BP_NORMAL, url, lineNo);
159         if (bp)
160         {
161             dispatch(debuggers, "onToggleBreakpoint", [url, lineNo, false, bp]);
162             this.saveBreakpoints(url);
163         }
164         return bp;
165     },

167     enableBreakpoint: function(url, lineNo)
168     {
169         var bp = this.findBreakpoint(url, lineNo);
170         if (bp && bp.type & BP_NORMAL)
171         {
172             bp.disabled &= ~BP_NORMAL;
173             dispatch(debuggers, "onToggleBreakpoint", [url, lineNo, true, bp]);
174             this.saveBreakpoints(url);
175             --disabledCount;
176         }
177         else {
178             if (FBTrace.DBG_FBS_BP)
179                 FBTrace.sysout("fbs.enableBreakpoint no find for "+lineNo+"@"+url);
180         }
181     },
182 
183     disableBreakpoint: function(url, lineNo)
184     {
185         var bp = this.findBreakpoint(url, lineNo);
186         if (bp && bp.type & BP_NORMAL)
187         {
188             bp.disabled |= BP_NORMAL;
189             ++disabledCount;
190             dispatch(debuggers, "onToggleBreakpoint", [url, lineNo, true, bp]);
191             this.saveBreakpoints(url);
192         }
193         else
194         {
195             if (FBTrace.DBG_FBS_BP)
196                 FBTrace.sysout("fbs.disableBreakpoint no find for "+lineNo+"@"+url);
197         }
198     },

209     setBreakpointCondition: function(sourceFile, lineNo, condition, debuggr)
210     {
211         var bp = this.findBreakpoint(sourceFile.href, lineNo);
212         if (!bp)
213         {
214             bp = this.addBreakpoint(BP_NORMAL, sourceFile, lineNo, null, debuggr);
215         }
216 
217         if (!bp)
218             return;
219 
220         if (bp.hitCount <= 0 )
221         {
222             if (bp.condition && !condition)
223             {
224                 --conditionCount;
225             }
226             else if (condition && !bp.condition)
227             {
228                 ++conditionCount;
229             }
230         }
231         bp.condition = condition;
232 
233         dispatch(debuggers, "onToggleBreakpoint", [sourceFile.href, lineNo, true, bp]);
234         this.saveBreakpoints(sourceFile.href);
235         return bp;
236     },

244     clearAllBreakpoints: function(sourceFiles)
245     {
246         for (var i = 0; i < sourceFiles.length; ++i)
247         {
248             var url = sourceFiles[i].href;
249             if (!url)
250                 continue;
251 
252             var urlBreakpoints = this.getBreakpoints(url);
253 
254             if (FBTrace.DBG_FBS_BP)
255                 FBTrace.sysout("clearAllBreakpoints "+url+" urlBreakpoints: "+
256                     (urlBreakpoints?urlBreakpoints.length:"null"));
257 
258             if (!urlBreakpoints)
259                 return false;
260 
261             for(var ibp = 0; ibp < urlBreakpoints.length; ibp++)
262             {
263                 var bp = urlBreakpoints[ibp];
264                 this.clearBreakpoint(url, bp.lineNo);
265             }
266          }

269     enumerateBreakpoints: function(url, cb)  // url is sourceFile.href, not jsd script.fileName
270     {
271         if (url)
272         {
273             var urlBreakpoints = this.getBreakpoints(url);
274             if (urlBreakpoints)
275             {
276                 for (var i = 0; i < urlBreakpoints.length; ++i)
277                 {
278                     var bp = urlBreakpoints[i];
279                     if (bp.type & BP_NORMAL && !(bp.type & BP_ERROR) )
280                     {
281                         if (bp.scriptsWithBreakpoint && bp.scriptsWithBreakpoint.length > 0)
282                         {
283                             var rc = cb.call.apply(bp, [url, bp.lineNo, bp, bp.scriptsWithBreakpoint]);
284                             if (rc)
285                                 return [bp];
286                         }
287                         else
288                         {
289                             var rc = cb.call.apply(bp, [url, bp.lineNo, bp]);
290                             if (rc)
291                                 return [bp];
292                         }
293                     }
294                 }
295             }
296         }
297         else
298         {
299             var bps = [];
300             var urls = fbs.getBreakpointURLs();
301             for (var i = 0; i < urls.length; i++)
302                 bps.push(this.enumerateBreakpoints(urls[i], cb));
303             return bps;
304         }
305     },

322                     dispatch(debuggers, "onToggleErrorBreakpoint", [url, lineNo, true, debuggr]);

324                 }
325 
326             }
327             catch(exc)

329                 FBTrace.sysout("fbs.setErrorBreakpoint ERROR "+exc, exc);
330             }
331         }

337         if (index != -1)

341             errorBreakpoints.splice(index, 1);
342             dispatch(debuggers, "onToggleErrorBreakpoint", [url, lineNo, false, debuggr]);
343             this.saveBreakpoints(url);  // after every call to onToggleBreakpoint
344         }

374     {

387     addBreakpoint: function(type, sourceFile, lineNo, props, debuggr)
388     {
389         var url = sourceFile.href;
390         var bp = this.findBreakpoint(url, lineNo);
391         if (bp && bp.type & type)
392             return null;
393 
394         if (bp)
395         {
396             bp.type |= type;
397 
398             if (debuggr)
399             {
400                 bp.debuggerName = debuggr.debuggerName;
401             }
402             else
403             {
404                 if (FBTrace.DBG_FBS_BP)
405                     FBTrace.sysout("fbs.addBreakpoint with no debuggr:\n");
406             }
407         }
408         else
409         {
410             bp = this.recordBreakpoint(type, url, lineNo, debuggr, props, sourceFile);
411         }
412 
413         if (FBTrace.DBG_FBS_BP)
414             FBTrace.sysout("addBreakpoint for "+url, [bp, sourceFile]);
415 
416         return bp;
417     },
418 
419     recordBreakpoint: function(type, url, lineNo, debuggr, props, sourceFile)
420     {
421         var urlBreakpoints = this.getBreakpoints(url);
422         if (!urlBreakpoints)
423             urlBreakpoints = [];
424 
425         if (typeof(lineNo) !== 'number')
426             throw new Error("firebug-service line numbers must be numbers "+lineNo+"@"+url);
427 
428         var bp = {type: type, href: url, lineNo: lineNo, disabled: 0,
429             debuggerName: debuggr.debuggerName,
430             condition: "", onTrue: true, hitCount: -1, hit: 0};
431 
432         if (props)
433         {
434             bp.condition = props.condition;
435             bp.onTrue = props.onTrue;
436             bp.hitCount = props.hitCount;
437             if (bp.condition || bp.hitCount > 0)
438                 ++conditionCount;
439             if (props.disabled)
440             {
441                 bp.disabled |= BP_NORMAL;
442                 ++disabledCount;
443             }
444         }
445 
446         urlBreakpoints.push(bp);
447         fbs.setJSDBreakpoint(sourceFile, bp);
448         fbs.setBreakpoints(url, urlBreakpoints);
449         ++breakpointCount;
450         return bp;
451     },
452 
453     removeBreakpoint: function(type, url, lineNo)
454     {
455         var urlBreakpoints = this.getBreakpoints(url);
456 
457         if (FBTrace.DBG_FBS_BP)
458             FBTrace.sysout("removeBreakpoint for "+url+", need to check bps="+
459                 (urlBreakpoints?urlBreakpoints.length:"none"));
460 
461         if (!urlBreakpoints)
462             return false;
463 
464         for (var i = 0; i < urlBreakpoints.length; ++i)
465         {
466             var bp = urlBreakpoints[i];
467             if (FBTrace.DBG_FBS_BP)
468                 FBTrace.sysout("removeBreakpoint checking bp.lineNo vs lineNo="+bp.lineNo+" vs "+lineNo);
469 
470             if (bp.lineNo === lineNo)
471             {
472                 bp.type &= ~type;
473                 if (!bp.type)
474                 {
475                     if (bp.scriptsWithBreakpoint)
476                     {
477                         for (var j = 0; j < bp.scriptsWithBreakpoint.length; j++)
478                         {
479                             var script = bp.scriptsWithBreakpoint[j];
480                             if (script && script.isValid)
481                             {
482                                 try
483                                 {
484                                     script.clearBreakpoint(bp.pc[j]);
485                                     if (FBTrace.DBG_FBS_BP)
486                                         FBTrace.sysout("removeBreakpoint in tag="+script.tag+" at "+
487                                             lineNo+"@"+url);
488                                 }
489                                 catch (exc)
490                                 {
491                                     FBTrace.sysout("Firebug service failed to remove breakpoint in "+
492                                         script.tag+" at lineNo="+lineNo+" pcmap:"+bp.pcmap);
493                                 }
494                             }
495                         }
496                     }
497                     // else this was a future breakpoint that never hit or a script that was GCed
498 
499                     urlBreakpoints.splice(i, 1);
500                     --breakpointCount;
501 
502                     if (bp.disabled)
503                         --disabledCount;
504 
505                     if (bp.condition || bp.hitCount > 0)
506                         --conditionCount;
507 
508                     fbs.setBreakpoints(url, urlBreakpoints);
509                 }
510                 return bp;
511             }
512         }
513 
514         if (FBTrace.DBG_FBS_BP)
515             FBTrace.sysout("fbs.removeBreakpoint no find for "+lineNo+"@"+url+" in "+urlBreakpoints.length, urlBreakpoints);
516 
517         return false;
518     },

520     findBreakpoint: function(url, lineNo)
521     {
522         var urlBreakpoints = this.getBreakpoints(url);
523         if (urlBreakpoints)
524         {
525             for (var i = 0; i < urlBreakpoints.length; ++i)
526             {
527                 var bp = urlBreakpoints[i];
528                 if (bp.lineNo === lineNo)
529                     return bp;
530             }
531         }
532         if (FBTrace.DBG_FBS_BP)
533             FBTrace.sysout("findBreakpoint no find for "+lineNo+"@"+url, urlBreakpoints);
534         return null;
535     },

537     // When we are called, scripts have been compiled so all relevant breakpoints are not "future"
538     findBreakpointByScript: function(script, pc)
539     {
540         var urlsWithBreakpoints = fbs.getBreakpointURLs();
541         for (var iURL = 0; iURL < urlsWithBreakpoints.length; iURL++)
542         {
543             var url = urlsWithBreakpoints[iURL];
544             var urlBreakpoints = this.getBreakpoints(url);
545             if (urlBreakpoints)
546             {
547                 for (var iBreakpoint = 0; iBreakpoint < urlBreakpoints.length; ++iBreakpoint)
548                 {
549                     var bp = urlBreakpoints[iBreakpoint];
550                     if (bp.scriptsWithBreakpoint)
551                     {
552                         for (var iScript = 0; iScript < bp.scriptsWithBreakpoint.length; iScript++)
553                         {
554                             if (FBTrace.DBG_FBS_BP)
555                             {
556                                 var vs = (bp.scriptsWithBreakpoint[iScript] ? bp.scriptsWithBreakpoint[iScript].tag+"@"+bp.pc[iScript]:"future")+" on "+url;
557                                 FBTrace.sysout("findBreakpointByScript["+iURL+","+iBreakpoint+","+iScript+"]"+" looking for "+script.tag+"@"+pc+" vs "+vs);
558                             }
559                             if ( bp.scriptsWithBreakpoint[iScript] && (bp.scriptsWithBreakpoint[iScript].tag == script.tag) && (bp.pc[iScript] == pc) )
560                                 return bp;
561                         }
562                     }
563                 }
564             }
565         }
566 
567         return null;
568     },

570     resetBreakpoints: function(sourceFile, debuggr) // the sourcefile has just been created after compile
571     {
572         // If the new script is replacing an old script with a breakpoint still
573         var url = sourceFile.href;
574         var urlBreakpoints = this.getBreakpoints(url);
575         if (FBTrace.DBG_FBS_BP)
576         {
577             try
578             {
579                 var msg = "resetBreakpoints: breakpoints["+sourceFile.href;
580                 msg += "]="+(urlBreakpoints?urlBreakpoints.length:"NONE")+"\n";
581                 FBTrace.sysout(msg);
582             }
583             catch (exc)
584             {
585                 FBTrace.sysout("Failed to give resetBreakpoints trace in url: "+url+" because "+exc+" for urlBreakpoints=", urlBreakpoints);
586             }
587         }
588 
589         if (urlBreakpoints)
590         {
591             if (FBTrace.DBG_FBS_BP)
592                 FBTrace.sysout("resetBreakpoints total bp="+urlBreakpoints.length+" for url="+url);
593 
594             fbs.deleteBreakpoints(url);
595 
596             for (var i = 0; i < urlBreakpoints.length; ++i)
597             {
598                 var bp = urlBreakpoints[i];
599                 fbs.recordBreakpoint(bp.type, url, bp.lineNo, debuggr, bp, sourceFile);
600                 if (bp.disabled & BP_NORMAL)
601                 {
602                     if (FBTrace.DBG_FBS_BP)
603                         FBTrace.sysout("resetBreakpoints:  mark breakpoint disabled: "+
604                             bp.lineNo+"@"+sourceFile);
605 
606                     fbs.disableBreakpoint(url, bp.lineNo);
607                 }
608                 else
609                 {
610                     if (FBTrace.DBG_FBS_BP)
611                         FBTrace.sysout("resetBreakpoints: "+bp.lineNo+"@"+sourceFile);
612                 }
613             }
614         }
615         else
616         {
617             if (FBTrace.DBG_FBS_BP)
618                 FBTrace.sysout("resetBreakpoints no breakpoints for "+url);
619         }

622     setJSDBreakpoint: function(sourceFile, bp)
623     {
624         var scripts = sourceFile.getScriptsAtLineNumber(bp.lineNo);
625         if (!scripts)
626         {
627              if (FBTrace.DBG_FBS_BP)
628                 FBTrace.sysout("setJSDBreakpoint:  NO inner scripts: "+bp.lineNo+"@"+sourceFile);
629              if (!sourceFile.outerScript || !sourceFile.outerScript.isValid)
630              {
631                 if (FBTrace.DBG_FBS_BP)
632                     FBTrace.sysout("setJSDBreakpoint:  NO valid outerScript\n");
633                 return;
634              }
635              scripts = [sourceFile.outerScript];
636         }
637 
638         if (!bp.scriptsWithBreakpoint)
639         {
640             bp.scriptsWithBreakpoint = [];
641             bp.pc = [];
642         }
643 
644         for (var i = 0; i < scripts.length; i++)
645         {
646             var script = scripts[i];
647             if (!script.isValid)
648             {
649                 if (FBTrace.DBG_FBS_BP)
650                     FBTrace.sysout("setJSDBreakpoint:  tag "+script.tag+", "+i+"/"+scripts.length+
651                         " is invalid\n");
652                 continue;
653             }
654 
655             var haveScript = false;
656             for (var j = 0; j < bp.scriptsWithBreakpoint.length; j++)
657             {
658                 if (bp.scriptsWithBreakpoint[j].tag === script.tag)
659                    {
660                     haveScript = true;
661                     break;
662                    }
663             }
664             if (haveScript)
665                 continue;
666 
667             var pcmap = sourceFile.pcmap_type;
668             if (!pcmap)
669             {
670                 if (FBTrace.DBG_FBS_ERRORS)
671                     FBTrace.sysout("fbs.setJSDBreakpoint pcmap undefined "+sourceFile, sourceFile);
672                 pcmap = PCMAP_SOURCETEXT;
673             }
674             // we subtraced this offset when we showed the user so lineNo is a user line number; now we need to talk
675             // to jsd its line world
676             var jsdLine = bp.lineNo + sourceFile.getBaseLineOffset();
677             // test script.isLineExecutable(jsdLineNo, pcmap) ??
678 
679             var isExecutable = false;
680             try {
681                  isExecutable = script.isLineExecutable(jsdLine, pcmap);
682             } catch(e) {
683                 // guess not then...
684             }
685 
686             if (isExecutable)
687             {
688                 var pc = script.lineToPc(jsdLine, pcmap);
689                 var pcToLine = script.pcToLine(pc, pcmap);  // avoid calling this unless we have to...
690 
691                 if (pcToLine == jsdLine)
692                 {
693                     script.setBreakpoint(pc);
694 
695                     bp.scriptsWithBreakpoint.push(script);
696                     bp.pc.push(pc);
697                     bp.pcmap = pcmap;
698                     bp.jsdLine = jsdLine;
699 
700                     if (pc == 0)  // signal the breakpoint handler to break for user
701                         sourceFile.breakOnZero = script.tag;
702 
703                     if (FBTrace.DBG_FBS_BP)
704                         FBTrace.sysout("setJSDBreakpoint tag: "+script.tag+" line.pc@url="+bp.lineNo +
705                             "."+pc+"@"+sourceFile.href+" using offset:"+sourceFile.getBaseLineOffset()+
706                             " jsdLine: "+jsdLine+" pcToLine: "+pcToLine+
707                             (isExecutable?" isExecuable":" notExecutable"), {sourceFile: sourceFile, script: script});
708                 }
709                 else
710                 {
711                     if (FBTrace.DBG_FBS_BP)
712                         FBTrace.sysout("setJSDBreakpoint LINE MISMATCH for tag: "+script.tag+
713                             " line.pc@url="+bp.lineNo +"."+pc+"@"+sourceFile.href+" using offset:"+
714                             sourceFile.getBaseLineOffset()+" jsdLine: "+jsdLine+" pcToLine: "+
715                             pcToLine+(isExecutable?" isExecuable":" notExecutable"), sourceFile);
716                 }
717             }
718             else
719             {
720                 if (FBTrace.DBG_FBS_BP)
721                     FBTrace.sysout("setJSDBreakpoint NOT isExecutable tag: "+script.tag+
722                         " jsdLine@url="+jsdLine +"@"+sourceFile.href+" pcmap:"+pcmap+
723                         " baselineOffset:"+sourceFile.getBaseLineOffset(), script);
724             }
725          }
726     },

730     saveBreakpoints: function(url)
731     {
732         // Do not call fbs.setBreakpoints() it calls us.
733         try
734         {
735             var urlBreakpoints = this.getBreakpoints(url);
736 
737             if (!urlBreakpoints || !urlBreakpoints.length)
738             {
739                 fbs.breakpointStore.removeItem(url);
740                 fbs.deleteBreakpoints(url);
741                 return;
742             }
743 
744             var cleanBPs = [];
745             for(var i = 0; i < urlBreakpoints.length; i++)
746             {
747                 var bp = urlBreakpoints[i];
748                 var cleanBP = {};
749                 for (var p in bp)
750                     cleanBP[p] = bp[p];
751                 delete cleanBP.scriptsWithBreakpoint; // not JSON-able
752                 delete cleanBP.pc; // co-indexed with scriptsWithBreakpoint
753                 delete cleanBP.debuggerName;
754                 cleanBPs.push(cleanBP);
755             }
756             fbs.breakpointStore.setItem(url, cleanBPs);
757         }
758         catch (exc)
759         {
760             FBTrace.sysout("firebug-service.saveBreakpoints ERROR "+exc, exc);
761         }
762     },
763 
764     setBreakpoints: function(url, urlBreakpoints)
765     {
766         fbs.breakpoints[url] = urlBreakpoints;
767         this.saveBreakpoints(url);
768     },

771     {
772         return fbs.breakpoints[url];

776     {
777         delete fbs.breakpoints[url];

782          var urls = this.getBreakpointStore().getKeys();

786     getBreakpointStore: function()
787     {
788         if (this.breakpointStore)
789             return this.breakpointStore;
790 
791         try
792         {
793             Components.utils.import("resource://firebug/storageService.js");
794 
795             if (typeof(StorageService) != "undefined")
796             {
797                 this.breakpointStore = StorageService.getStorage("breakpoints.json");
798             }
799             else
800             {
801                 ERROR("firebug-service breakpoint StorageService ERROR");
802                 this.breakpointStore =
803                 {
804                     setItem: function(){},
805                     removeItem: function(){},
806                     getKeys: function(){return [];},
807                 };
808             }
809             return this.breakpointStore;
810         }
811         catch(exc)
812         {
813             ERROR("firebug-service restoreBreakpoints ERROR "+exc);
814         }
815 
816     },
817 
818     restoreBreakpoints: function()
819     {
820         this.breakpoints = {};
821         var breakpointStore = this.getBreakpointstore();
822         var urls =  fbs.getBreakpointURLs();
823         for (var i = 0; i < urls.length; i++)
824         {
825             var url = urls[i];
826             var bps = breakpointStore.getItem(url);
827             this.breakpoints[url] = bps;
828             for (var j = 0; j < bps.length; j++)
829             {
830                 var bp = bps[j];
831                 if (bp.condition)
832                     ++conditionCount;
833                 if (bp.disabled)
834                     ++disabledCount;
835                 if (bp.type & BP_MONITOR)
836                     ++monitorCount;
837                 if (bp.type & BP_ERROR)
838                     errorBreakpoints.push({href: url, lineNo: bp.lineNo, type: BP_ERROR });
839             }
840         }
841 
842         if (FBTrace.DBG_FBS_BP)
843         {
844             FBTrace.sysout("restoreBreakpoints "+urls.length+", disabledCount:"+disabledCount+
845                 " monitorCount:"+monitorCount+" conditionCount:"+conditionCount+", restored ",
846                 this.breakpoints);
847 
848             for (var p in this.breakpoints)
849                 FBTrace.sysout("restoreBreakpoints restored "+p+" condition "+
850                     this.breakpoints[p].condition);
851         }
852     },

854     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * //
855     // Monitoring
856 
857     monitor: function(sourceFile, lineNo, debuggr)
858     {
859         if (lineNo == -1)
860             return null;
861 
862         var bp = this.addBreakpoint(BP_MONITOR, sourceFile, lineNo, null, debuggr);
863         if (bp)
864         {
865             ++monitorCount;
866             dispatch(debuggers, "onToggleMonitor", [sourceFile.href, lineNo, true]);
867         }
868         return bp;
869     },
870 
871     unmonitor: function(href, lineNo)
872     {
873         if (lineNo != -1 && this.removeBreakpoint(BP_MONITOR, href, lineNo))
874         {
875             --monitorCount;
876             dispatch(debuggers, "onToggleMonitor", [ href, lineNo, false]);
877         }
878     },
879 
880     isMonitored: function(url, lineNo)
881     {
882         var bp = lineNo != -1 ? this.findBreakpoint(url, lineNo) : null;
883         return bp && bp.type & BP_MONITOR;
884     },
885 
886     enumerateMonitors: function(url, cb)
887     {
888         if (url)
889         {
890             var urlBreakpoints = this.getBreakpoints(url);
891             if (urlBreakpoints)
892             {
893                 for (var i = 0; i < urlBreakpoints.length; ++i)
894                 {
895                     var bp = urlBreakpoints[i];
896                     if (bp.type & BP_MONITOR)
897                         cb.call(url, bp.lineNo, bp);
898                 }
899             }
900         }
901         else
902         {
903             for (var url in breakpoints)
904                 this.enumerateBreakpoints(url, cb);
905         }
906     },
