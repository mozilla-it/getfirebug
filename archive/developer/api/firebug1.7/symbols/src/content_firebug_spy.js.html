 17 /**
 18  * @module Represents a XHR Spy module. The main purpose of the XHR Spy feature is to monitor
 19  * XHR activity of the current page and create appropriate log into the Console panel.
 20  * This feature can be controlled by an option <i>Show XMLHttpRequests</i> (from within the
 21  * console panel).
 22  *
 23  * The module is responsible for attaching/detaching a HTTP Observers when Firebug is
 24  * activated/deactivated for a site.
 25  */
 26 Firebug.Spy = extend(Firebug.Module,
 27 /** @lends Firebug.Spy */
 28 {
 29     dispatchName: "spy",
 30 
 31     initialize: function()
 32     {
 33         if (Firebug.TraceModule)
 34             Firebug.TraceModule.addListener(this.TraceListener);
 35 
 36         Firebug.Module.initialize.apply(this, arguments);
 37     },
 38 
 39     shutdown: function()
 40     {
 41         Firebug.Module.shutdown.apply(this, arguments);
 42 
 43         if (Firebug.TraceModule)
 44             Firebug.TraceModule.removeListener(this.TraceListener);
 45     },
 46 
 47     initContext: function(context)
 48     {
 49         context.spies = [];
 50 
 51         if (Firebug.showXMLHttpRequests && Firebug.Console.isAlwaysEnabled())
 52             this.attachObserver(context, context.window);
 53 
 54         if (FBTrace.DBG_SPY)
 55             FBTrace.sysout("spy.initContext " + contexts.length + " ", context.getName());
 56     },
 57 
 58     destroyContext: function(context)
 59     {
 60         // For any spies that are in progress, remove our listeners so that they don't leak
 61         this.detachObserver(context, null);
 62 
 63         if (FBTrace.DBG_SPY && context.spies.length)
 64             FBTrace.sysout("spy.destroyContext; ERROR There are leaking Spies ("
 65                 + context.spies.length + ") " + context.getName());
 66 
 67         delete context.spies;
 68 
 69         if (FBTrace.DBG_SPY)
 70             FBTrace.sysout("spy.destroyContext " + contexts.length + " ", context.getName());
 71     },
 72 
 73     watchWindow: function(context, win)
 74     {
 75         if (Firebug.showXMLHttpRequests && Firebug.Console.isAlwaysEnabled())
 76             this.attachObserver(context, win);
 77     },
 78 
 79     unwatchWindow: function(context, win)
 80     {
 81         try
 82         {
 83             // This make sure that the existing context is properly removed from "contexts" array.
 84             this.detachObserver(context, win);
 85         }
 86         catch (ex)
 87         {
 88             // Get exceptions here sometimes, so let's just ignore them
 89             // since the window is going away anyhow
 90             ERROR(ex);
 91         }
 92     },

 95     {

 97         // but somehow seems not correct
 98         if (name == "showXMLHttpRequests")
 99         {
100             var tach = value ? this.attachObserver : this.detachObserver;
101             for (var i = 0; i < Firebug.TabWatcher.contexts.length; ++i)
102             {
103                 var context = Firebug.TabWatcher.contexts[i];
104                 iterateWindows(context.window, function(win)
105                 {
106                     tach.apply(this, [context, win]);
107                 });
108             }
109         }

113     // Attaching Spy to XHR requests.
114 
115     /**
116      * Returns false if Spy should not be attached to XHRs executed by the specified window.
117      */
118     skipSpy: function(win)
119     {
120         if (!win)
121             return true;
122 
123         // Don't attach spy to chrome.
124         var uri = safeGetWindowLocation(win);
125         if (uri && (uri.indexOf("about:") == 0 || uri.indexOf("chrome:") == 0))
126             return true;
127     },
128 
129     attachObserver: function(context, win)
130     {
131         if (Firebug.Spy.skipSpy(win))
132             return;
133 
134         for (var i=0; i<contexts.length; ++i)
135         {
136             if ((contexts[i].context == context) && (contexts[i].win == win))
137                 return;
138         }
139 
140         // Register HTTP observers only once.
141         if (contexts.length == 0)
142         {
143             httpObserver.addObserver(SpyHttpObserver, "firebug-http-event", false);
144             SpyHttpActivityObserver.registerObserver();
145         }
146 
147         contexts.push({context: context, win: win});
148 
149         if (FBTrace.DBG_SPY)
150             FBTrace.sysout("spy.attachObserver (HTTP) " + contexts.length + " ", context.getName());
151     },
152 
153     detachObserver: function(context, win)
154     {
155         for (var i=0; i<contexts.length; ++i)
156         {
157             if (contexts[i].context == context)
158             {
159                 if (win && (contexts[i].win != win))
160                     continue;
161 
162                 contexts.splice(i, 1);
163 
164                 // If no context is using spy, remvove the (only one) HTTP observer.
165                 if (contexts.length == 0)
166                 {
167                     httpObserver.removeObserver(SpyHttpObserver, "firebug-http-event");
168                     SpyHttpActivityObserver.unregisterObserver();
169                 }
170 
171                 if (FBTrace.DBG_SPY)
172                     FBTrace.sysout("spy.detachObserver (HTTP) " + contexts.length + " ",
173                         context.getName());
174                 return;
175             }
176         }
177     },

179     /**
180      * Return XHR object that is associated with specified request <i>nsIHttpChannel</i>.
181      * Returns null if the request doesn't represent XHR.
182      */
183     getXHR: function(request)
184     {
185         // Does also query-interface for nsIHttpChannel.
186         if (!(request instanceof Ci.nsIHttpChannel))
187             return null;
188 
189         try
190         {
191             var callbacks = request.notificationCallbacks;
192             if (callbacks)
193             {
194                 suspendShowStackTrace();
195                 return callbacks.getInterface(Ci.nsIXMLHttpRequest);
196             }
197         }
198         catch (exc)
199         {
200             if (exc.name == "NS_NOINTERFACE")
201             {
202                 if (FBTrace.DBG_SPY)
203                     FBTrace.sysout("spy.getXHR; Request is not nsIXMLHttpRequest: " +
204                         safeGetRequestName(request));
205             }
206         }
207         finally
208         {
209             resumeShowStackTrace();
210         }
211 
212        return null;
213     },

218 /**
219  * @class This observer uses {@link HttpRequestObserver} to monitor start and end of all XHRs.
220  * using <code>http-on-modify-request</code>, <code>http-on-examine-response</code> and
221  * <code>http-on-examine-cached-response</code> events. For every monitored XHR a new
222  * instance of {@link Firebug.Spy.XMLHttpRequestSpy} object is created. This instance is removed
223  * when the XHR is finished.
224  */
225 var SpyHttpObserver =
226 /** @lends SpyHttpObserver */
227 {
228     observe: function(request, topic, data)
229     {
230         try
231         {
232             if (topic != "http-on-modify-request" &&
233                 topic != "http-on-examine-response" &&
234                 topic != "http-on-examine-cached-response")
235             {
236                 if (FBTrace.DBG_ERRORS || FBTrace.DBG_SPY)
237                     FBTrace.sysout("spy.SpyHttpObserver.observe; ERROR Unknown topic: " + topic);
238                 return;
239             }
240 
241             this.observeRequest(request, topic);
242         }
243         catch (exc)
244         {
245             if (FBTrace.DBG_ERRORS || FBTrace.DBG_SPY)
246                 FBTrace.sysout("spy.SpyHttpObserver EXCEPTION", exc);
247         }
248     },
249 
250     observeRequest: function(request, topic)
251     {
252         var win = getWindowForRequest(request);
253         var xhr = Firebug.Spy.getXHR(request);
254 
255         // The request must be associated with window (i.e. tab) and it also must be
256         // real XHR request.
257         if (!win || !xhr)
258             return;
259 
260         for (var i=0; i<contexts.length; ++i)
261         {
262             var context = contexts[i];
263             if (context.win == win)
264             {
265                 var spyContext = context.context;
266                 var requestName = request.URI.asciiSpec;
267                 var requestMethod = request.requestMethod;
268 
269                 if (topic == "http-on-modify-request")
270                     this.requestStarted(request, xhr, spyContext, requestMethod, requestName);
271                 else if (topic == "http-on-examine-response")
272                     this.requestStopped(request, xhr, spyContext, requestMethod, requestName);
273                 else if (topic == "http-on-examine-cached-response")
274                     this.requestStopped(request, xhr, spyContext, requestMethod, requestName);
275 
276                 return;
277             }
278         }
279     },
280 
281     requestStarted: function(request, xhr, context, method, url)
282     {
283         var spy = getSpyForXHR(request, xhr, context);
284         spy.method = method;
285         spy.href = url;
286 
287         if (FBTrace.DBG_SPY)
288             FBTrace.sysout("spy.requestStarted; " + spy.href, spy);
289 
290         // Get "body" for POST and PUT requests. It will be displayed in
291         // appropriate tab of the XHR.
292         if (method == "POST" || method == "PUT")
293             spy.postText = readPostTextFromRequest(request, context);
294 
295         spy.urlParams = parseURLParams(spy.href);
296 
297         // In case of redirects there is no stack and the source link is null.
298         spy.sourceLink = getStackSourceLink();
299 
300         if (!spy.requestHeaders)
301             spy.requestHeaders = getRequestHeaders(spy);
302 
303         // If it's enabled log the request into the console tab.
304         if (Firebug.showXMLHttpRequests && Firebug.Console.isAlwaysEnabled())
305         {
306             spy.logRow = Firebug.Console.log(spy, spy.context, "spy", null, true);
307             setClass(spy.logRow, "loading");
308         }
309 
310         // Notify registered listeners. The onStart event is fired once for entire XHR
311         // (even if there is more redirects within the process).
312         var name = request.URI.asciiSpec;
313         var origName = request.originalURI.asciiSpec;
314         if (name == origName)
315             dispatch(Firebug.Spy.fbListeners, "onStart", [context, spy]);
316 
317         // Remember the start time et the end, so it's most accurate.
318         spy.sendTime = new Date().getTime();
319     },
320 
321     requestStopped: function(request, xhr, context, method, url)
322     {
323         var spy = getSpyForXHR(request, xhr, context);
324         if (!spy)
325             return;
326 
327         spy.endTime = new Date().getTime();
328         spy.responseTime = spy.endTime - spy.sendTime;
329         spy.mimeType = Firebug.NetMonitor.Utils.getMimeType(request.contentType, request.name);
330 
331         if (!spy.responseHeaders)
332             spy.responseHeaders = getResponseHeaders(spy);
333 
334         if (!spy.statusText)
335         {
336             try
337             {
338                 spy.statusCode = request.responseStatus;
339                 spy.statusText = request.responseStatusText;
340             }
341             catch (exc)
342             {
343                 if (FBTrace.DBG_SPY)
344                     FBTrace.sysout("spy.requestStopped " + spy.href + ", status access FAILED", exc);
345             }
346         }
347 
348         if (spy.logRow)
349         {
350             updateLogRow(spy);
351             updateHttpSpyInfo(spy);
352         }
353 
354         // Remove only the Spy object that has been created for an intermediate rediret
355         // request. These exist only to be also displayed in the console and they
356         // don't attach any listeners to the original XHR object (which is always created
357         // only once even in case of redirects).
358         // xxxHonza: These requests are not observer by the activityObserver now
359         // (if they should be observed we have to remove them in the activityObserver)
360         if (!spy.onLoad && spy.context.spies)
361             remove(spy.context.spies, spy);
362 
363         if (FBTrace.DBG_SPY)
364             FBTrace.sysout("spy.requestStopped: " + spy.href + ", responseTime: " +
365                 spy.responseTime + "ms, spy.responseText: " +
366                 (spy.reponseText ? spy.responseText.length : 0) + " bytes");
367     }
368 };

373 /**
374  * @class This observer is used to properly monitor even mulipart XHRs. It's based on
375  * an activity-observer component that has been introduced in Firefox 3.6.
376  */
377 var SpyHttpActivityObserver = extend(Firebug.NetMonitor.NetHttpActivityObserver,
378 /** @lends SpyHttpActivityObserver */
379 {
380     activeRequests: [],
381 
382     observeRequest: function(request, activityType, activitySubtype, timestamp,
383         extraSizeData, extraStringData)
384     {
385         if (activityType != Ci.nsIHttpActivityObserver.ACTIVITY_TYPE_HTTP_TRANSACTION &&
386            (activityType == Ci.nsIHttpActivityObserver.ACTIVITY_TYPE_SOCKET_TRANSPORT &&
387             activitySubtype != Ci.nsISocketTransport.STATUS_RECEIVING_FROM))
388             return;
389 
390         // xxxHonza: this code is duplicated in net.js, it should be refactored.
391         var win = getWindowForRequest(request);
392         if (!win)
393         {
394             var index = this.activeRequests.indexOf(request);
395             if (index == -1)
396                 return;
397 
398             if (!(win = this.activeRequests[index+1]))
399                 return;
400         }
401 
402         for (var i=0; i<contexts.length; ++i)
403         {
404             var context = contexts[i];
405             if (context.win == win)
406             {
407                 var spyContext = context.context;
408                 var spy = getSpyForXHR(request, null, spyContext, true);
409                 if (spy)
410                     this.observeXHRActivity(win, spy, request, activitySubtype, timestamp);
411                 return;
412             }
413         }
414     },
415 
416     observeXHRActivity: function(win, spy, request, activitySubtype, timestamp)
417     {
418         // Activity observer has precise time info so, use it.
419         var time = new Date();
420         time.setTime(timestamp/1000);
421 
422         if (activitySubtype == Ci.nsIHttpActivityObserver.ACTIVITY_SUBTYPE_REQUEST_HEADER)
423         {
424             if (FBTrace.DBG_SPY)
425                 FBTrace.sysout("spy.observeXHRActivity REQUEST_HEADER " + safeGetRequestName(request));
426 
427             this.activeRequests.push(request);
428             this.activeRequests.push(win);
429 
430             spy.sendTime = time;
431             spy.transactionStarted = true;
432         }
433         else if (activitySubtype == Ci.nsIHttpActivityObserver.ACTIVITY_SUBTYPE_TRANSACTION_CLOSE)
434         {
435             if (FBTrace.DBG_SPY)
436                 FBTrace.sysout("spy.observeXHRActivity TRANSACTION_CLOSE " + safeGetRequestName(request));
437 
438             var index = this.activeRequests.indexOf(request);
439             this.activeRequests.splice(index, 2);
440 
441             spy.endTime = time;
442             spy.transactionClosed = true;
443 
444             // This should be the proper time to detach the Spy object, but only
445             // in the case when the XHR is already loaded. If the XHR is made as part of the
446             // page load, it may happen that the event (readyState == 4) comes later
447             // than actual TRANSACTION_CLOSE.
448             if (spy.loaded)
449                 spy.detach();
450         }
451         else if (activitySubtype == Ci.nsISocketTransport.STATUS_RECEIVING_FROM)
452         {
453             spy.endTime = time;
454         }
455     }
456 });

460 function getSpyForXHR(request, xhrRequest, context, noCreate)
461 {
462     var spy = null;
463 
464     // Iterate all existing spy objects in this context and look for one that is
465     // already created for this request.
466     var length = context.spies.length;
467     for (var i=0; i<length; i++)
468     {
469         spy = context.spies[i];
470         if (spy.request == request)
471             return spy;
472     }
473 
474     if (noCreate)
475         return null;
476 
477     spy = new Firebug.Spy.XMLHttpRequestSpy(request, xhrRequest, context);
478     context.spies.push(spy);
479 
480     var name = request.URI.asciiSpec;
481     var origName = request.originalURI.asciiSpec;
482 
483     // Attach spy only to the original request. Notice that there can be more network requests
484     // made by the same XHR if redirects are involved.
485     if (name == origName)
486         spy.attach();
487 
488     if (FBTrace.DBG_SPY)
489         FBTrace.sysout("spy.getSpyForXHR; New spy object created (" +
490             (name == origName ? "new XHR" : "redirected XHR") + ") for: " + name, spy);
491 
492     return spy;
493 }

497 /**
498  * @class This class represents a Spy object that is attached to XHR. This object
499  * registers various listeners into the XHR in order to monitor various events fired
500  * during the request process (onLoad, onAbort, etc.)
501  */
502 Firebug.Spy.XMLHttpRequestSpy = function(request, xhrRequest, context)
503 {
504     this.request = request;
505     this.xhrRequest = xhrRequest;
506     this.context = context;
507     this.responseText = "";
508 
509     // For compatibility with the Net templates.
510     this.isXHR = true;
511 
512     // Support for activity-observer
513     this.transactionStarted = false;
514     this.transactionClosed = false;
515 };
516 
517 Firebug.Spy.XMLHttpRequestSpy.prototype =
518 /** @lends Firebug.Spy.XMLHttpRequestSpy */
519 {
520     attach: function()
521     {
522         var spy = this;
523         this.onReadyStateChange = function(event) { onHTTPSpyReadyStateChange(spy, event); };
524         this.onLoad = function() { onHTTPSpyLoad(spy); };
525         this.onError = function() { onHTTPSpyError(spy); };
526         this.onAbort = function() { onHTTPSpyAbort(spy); };
527 
528         // xxxHonza: #502959 is still failing on Fx 3.5
529         // Use activity distributor to identify 3.6
530         if (SpyHttpActivityObserver.getActivityDistributor())
531         {
532             this.onreadystatechange = this.xhrRequest.onreadystatechange;
533             this.xhrRequest.onreadystatechange = this.onReadyStateChange;
534         }
535 
536         this.xhrRequest.addEventListener("load", this.onLoad, false);
537         this.xhrRequest.addEventListener("error", this.onError, false);
538         this.xhrRequest.addEventListener("abort", this.onAbort, false);
539 
540         // xxxHonza: should be removed from FB 3.6
541         if (!SpyHttpActivityObserver.getActivityDistributor())
542             this.context.sourceCache.addListener(this);
543     },
544 
545     detach: function()
546     {
547         // Bubble out if already detached.
548         if (!this.onLoad)
549             return;
550 
551         // If the activity distributor is available, let's detach it when the XHR
552         // transaction is closed. Since, in case of multipart XHRs the onLoad method
553         // (readyState == 4) can be called mutliple times.
554         // Keep in mind:
555         // 1) It can happen that that the TRANSACTION_CLOSE event comes before
556         // the onLoad (if the XHR is made as part of the page load) so, detach if
557         // it's already closed.
558         // 2) In case of immediate cache responses, the transaction doesn't have to
559         // be started at all (or the activity observer is no available in Firefox 3.5).
560         // So, also detach in this case.
561         if (this.transactionStarted && !this.transactionClosed)
562             return;
563 
564         if (FBTrace.DBG_SPY)
565             FBTrace.sysout("spy.detach; " + this.href);
566 
567         // Remove itself from the list of active spies.
568         remove(this.context.spies, this);
569 
570         if (this.onreadystatechange)
571             this.xhrRequest.onreadystatechange = this.onreadystatechange;
572 
573         try { this.xhrRequest.removeEventListener("load", this.onLoad, false); } catch (e) {}
574         try { this.xhrRequest.removeEventListener("error", this.onError, false); } catch (e) {}
575         try { this.xhrRequest.removeEventListener("abort", this.onAbort, false); } catch (e) {}
576 
577         this.onreadystatechange = null;
578         this.onLoad = null;
579         this.onError = null;
580         this.onAbort = null;
581 
582         // xxxHonza: shouuld be removed from FB 1.6
583         if (!SpyHttpActivityObserver.getActivityDistributor())
584             this.context.sourceCache.removeListener(this);
585     },
586 
587     getURL: function()
588     {
589         return this.xhrRequest.channel ? this.xhrRequest.channel.name : this.href;
590     },
591 
592     // Cache listener
593     onStopRequest: function(context, request, responseText)
594     {
595         if (FBTrace.DBG_SPY)
596             FBTrace.sysout("spy.onStopRequest: " + safeGetRequestName(request), responseText);
597 
598         if (!responseText)
599             return;
600 
601         if (request == this.request)
602             this.responseText = responseText;
603     },
604 };

608 function onHTTPSpyReadyStateChange(spy, event)
609 {
610     if (FBTrace.DBG_SPY)
611         FBTrace.sysout("spy.onHTTPSpyReadyStateChange " + spy.xhrRequest.readyState +
612             " (multipart: " + spy.xhrRequest.multipart + ")");
613 
614     // Remember just in case spy is detached (readyState == 4).
615     var originalHandler = spy.onreadystatechange;
616 
617     // Force response text to be updated in the UI (in case the console entry
618     // has been already expanded and the response tab selected).
619     if (spy.logRow && spy.xhrRequest.readyState >= 3)
620     {
621         var netInfoBox = getChildByClass(spy.logRow, "spyHead", "netInfoBody");
622         if (netInfoBox)
623         {
624             netInfoBox.htmlPresented = false;
625             netInfoBox.responsePresented = false;
626         }
627     }
628 
629     // If the request is loading update the end time.
630     if (spy.logRow && spy.xhrRequest.readyState == 3 && spy.sendTime && spy.endTime)
631     {
632         spy.responseTime = spy.endTime - spy.sendTime;
633         updateTime(spy);
634     }
635 
636     // Request loaded. Get all the info from the request now, just in case the
637     // XHR would be aborted in the original onReadyStateChange handler.
638     if (spy.xhrRequest.readyState == 4)
639     {
640         // Cumulate response so, multipart response content is properly displayed.
641         if (SpyHttpActivityObserver.getActivityDistributor())
642             spy.responseText += spy.xhrRequest.responseText;
643         else
644         {
645             // xxxHonza: remove from FB 1.6
646             if (!spy.responseText)
647                 spy.responseText = spy.xhrRequest.responseText;
648         }
649 
650         // The XHR is loaded now (used also by the activity observer).
651         spy.loaded = true;
652 
653         // Update UI.
654         updateHttpSpyInfo(spy);
655 
656         // Notify Net pane about a request beeing loaded.
657         // xxxHonza: I don't think this is necessary.
658         var netProgress = spy.context.netProgress;
659         if (netProgress)
660             netProgress.post(netProgress.stopFile, [spy.request, spy.endTime, spy.postText, spy.responseText]);
661 
662         // Notify registered listeners about finish of the XHR.
663         dispatch(Firebug.Spy.fbListeners, "onLoad", [spy.context, spy]);
664     }
665 
666     // Pass the event to the original page handler.
667     callPageHandler(spy, event, originalHandler);
668 }

670 function onHTTPSpyLoad(spy)
671 {
672     if (FBTrace.DBG_SPY)
673         FBTrace.sysout("spy.onHTTPSpyLoad: " + spy.href, spy);
674 
675     // Detach must be done in onLoad (not in onreadystatechange) otherwise
676     // onAbort would not be handled.
677     spy.detach();
678 
679     // xxxHonza: Still needed for Fx 3.5 (#502959)
680     if (!SpyHttpActivityObserver.getActivityDistributor())
681         onHTTPSpyReadyStateChange(spy, null);
682 
683     // If the spy is not loaded yet (and so, the response was not cached), do it now.
684     // This can happen since synchronous XHRs don't fire onReadyStateChange event (issue 2868).
685     if (!spy.loaded)
686     {
687         spy.loaded = true;
688         spy.responseText = spy.xhrRequest.responseText;
689     }
690 }

692 function onHTTPSpyError(spy)
693 {
694     if (FBTrace.DBG_SPY)
695         FBTrace.sysout("spy.onHTTPSpyError; " + spy.href, spy);
696 
697     spy.detach();
698     spy.loaded = true;
699 
700     if (spy.logRow)
701     {
702         removeClass(spy.logRow, "loading");

707 function onHTTPSpyAbort(spy)
708 {
709     if (FBTrace.DBG_SPY)
710         FBTrace.sysout("spy.onHTTPSpyAbort: " + spy.href, spy);
711 
712     spy.detach();
713     spy.loaded = true;
714 
715     // Ignore aborts if the request already has a response status.
716     if (spy.xhrRequest.status)
717         return;
718 
719     if (spy.logRow)
720     {
721         removeClass(spy.logRow, "loading");
722         setClass(spy.logRow, "error");
723     }
724 
725     spy.statusText = "Aborted";
726     updateLogRow(spy);
727 
728     // Notify Net pane about a request beeing aborted.
729     // xxxHonza: the net panel shoud find out this itself.
730     var netProgress = spy.context.netProgress;
731     if (netProgress)
732         netProgress.post(netProgress.abortFile, [spy.request, spy.endTime, spy.postText, spy.responseText]);
733 }

738 {
739     try
740     {
741         // Calling the page handler throwed an exception (see #502959)
742         // This should be fixed in Firefox 3.5
743         if (originalHandler && event)
744             originalHandler.handleEvent(event);
745     }
746     catch (exc)
747     {
748         if (FBTrace.DBG_ERRORS)
749             FBTrace.sysout("spy.onHTTPSpyReadyStateChange: EXCEPTION "+exc, [exc, event]);
750 
751         var xpcError = Firebug.Errors.reparseXPC(exc, spy.context);
752         if (xpcError) //
753         {
754             // TODO attach trace
755             if (FBTrace.DBG_ERRORS)
756                 FBTrace.sysout("spy.onHTTPSpyReadyStateChange: reparseXPC", xpcError);
757 
758             // Make sure the exception is displayed in both Firefox & Firebug console.
759             throw new Error(xpcError.message, xpcError.href, xpcError.lineNo);
760         }
761         else
762         {
763             throw exc;
764         }
765     }

970 function updateHttpSpyInfo(spy)
971 {
972     if (!spy.logRow || !hasClass(spy.logRow, "opened"))
973         return;
974 
975     if (!spy.params)
976         spy.params = parseURLParams(spy.href+"");
977 
978     if (!spy.requestHeaders)
979         spy.requestHeaders = getRequestHeaders(spy);
980 
981     if (!spy.responseHeaders && spy.loaded)
982         spy.responseHeaders = getResponseHeaders(spy);
983 
984     var template = Firebug.NetMonitor.NetInfoBody;
985     var netInfoBox = getChildByClass(spy.logRow, "spyHead", "netInfoBody");
986     if (!netInfoBox)
987     {
988         var head = getChildByClass(spy.logRow, "spyHead");
989         netInfoBox = template.tag.append({"file": spy}, head);
990         dispatch(template.fbListeners, "initTabBody", [netInfoBox, spy]);
991         template.selectTabByName(netInfoBox, "Response");
992     }
993     else
994     {
995         template.updateInfo(netInfoBox, spy, spy.context);
996     }
997 }

1001 function getRequestHeaders(spy)
1002 {
1003     var headers = [];
1004 
1005     var channel = spy.xhrRequest.channel;
1006     if (channel instanceof Ci.nsIHttpChannel)
1007     {
1008         channel.visitRequestHeaders({
1009             visitHeader: function(name, value)
1010             {
1011                 headers.push({name: name, value: value});
1012             }
1013         });
1014     }
1015 
1016     return headers;
1017 }
1018 
1019 function getResponseHeaders(spy)
1020 {
1021     var headers = [];
1022 
1023     try
1024     {
1025         var channel = spy.xhrRequest.channel;
1026         if (channel instanceof Ci.nsIHttpChannel)
1027         {
1028             channel.visitResponseHeaders({
1029                 visitHeader: function(name, value)
1030                 {
1031                     headers.push({name: name, value: value});
1032                 }
1033             });
1034         }
1035     }
1036     catch (exc)
1037     {
1038         if (FBTrace.DBG_SPY || FBTrace.DBG_ERRORS)
1039             FBTrace.sysout("spy.getResponseHeaders; EXCEPTION " +
1040                 safeGetRequestName(spy.request), exc);
1041     }
1042 
1043     return headers;
1044 }
