 15 /**
 16  * Describes a compilation unit in a browser context. A compilation unit
 17  * may originate from a JavaScript source file or a script element in HTML.
 18  *
 19  * @constructor
 20  * @param url compilation unit URL - a {@link String} or <code>null</code> if none
 21  * @param context the {@link BrowserContext} this compilation unit is contained in
 22  * @type CompilationUnit
 23  * @return a new CompilationUnit
 24  * @version 1.0
 25  */
 26 function CompilationUnit(url, context)
 27 {
 28     this.url = url;
 29     this.context = context;
 30     this.breakpoints = [];
 31     this.numberOfLines = 0;
 32     this.kind = CompilationUnit.SCRIPT_TAG;
 33 }
 34 
 35 /*
 36  * Kinds of Compilation Units
 37  */
 38 CompilationUnit.SCRIPT_TAG = "script_tag";
 39 CompilationUnit.EVAL = "eval";
 40 CompilationUnit.BROWSER_GENERATED = "event";

 92 /**
 93  * Returns the breakpoints that have been created in this compilation unit and
 94  * have not been cleared.
 95  * <p>
 96  * This function does not require communication with
 97  * the browser.
 98  * </p>
 99  * @function
100  * @returns an array of {@link Breakpoint}'s
101  */
102 CompilationUnit.prototype.getBreakpoints = function()
103 {
104     // return a copy of scripts so the master copy is not corrupted
105     var bps = [];
106     for ( var i = 0; i < this.breakpoints.length; i++)
107         bps.push(this.breakpoints[i]);
108     return bps;
109 };

111 /**
112  * Requests the source of this compilation unit asynchronously. Source will be
113  * retrieved from the browser and reported back to the listener function when available.
114  * The handler may be called before or after this function returns.
115  * <p>
116  * TODO: what if the compilation unit no longer exists in the browser
117  * </p>
118  * @function
119  * @param firstLineNumber requested line number starting point; < 1 means from lowest line number
120  * @param lastLineNumber request last line number; < 1 means up to maximum line
121  * @param listener a listener (function) that accepts (compilationUnit, firstLineNumber, lastLineNumber, array of source code lines)
122  */
123 CompilationUnit.prototype.getSourceLines = function(firstLine, lastLine, listener)
124 {
125     this.lines = this.sourceFile.loadScriptLines(this.context);
126     this.numberOfLines = (this.lines ? this.lines.length : 0);
127     listener(this, 1, this.numberOfLines, this.lines);
128 };

163 /**
164  * Adds the specified breakpoint to this compilation unit's collection of breakpoints.
165  * Implementation should call this method when a breakpoint is created in a compilation
166  * unit.
167  *
168  * @param breakpoint the breakpoint that was created
169  * @function
170  */
171 CompilationUnit.prototype._addBreakpoint = function(breakpoint)
172 {
173     this.breakpoints.push(breakpoint);
174 };
175 
176 /**
177  * Removes the specified breakpoint from this compilation unit's collection of breakpoints.
178  * Implementation should call this method when a breakpoint is cleared from a compilation
179  * unit.
180  *
181  * @param breakpoint the breakpoint that was removed
182  * @function
183  */
184 CompilationUnit.prototype._removeBreakpoint = function(breakpoint)
185 {
186     for ( var i = 0; i < this.breakpoints.length; i++)
187     {
188         if (this.breakpoints[i] === breakpoint)
189         {
190             this.breakpoints.splice(i, 1);
191             return;
192         }
193     }
194 };
