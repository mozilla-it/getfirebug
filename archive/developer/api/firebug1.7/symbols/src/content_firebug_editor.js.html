 39     startEditing: function(target, value, editor, selectionData)
 40     {
 41         this.stopEditing();
 42 
 43         if (hasClass(target, "insertBefore") || hasClass(target, "insertAfter"))
 44             return;
 45 
 46         var panel = Firebug.getElementPanel(target);
 47         if (!panel.editable)
 48             return;
 49 
 50         if (FBTrace.DBG_EDITOR)
 51             FBTrace.sysout("editor.startEditing " + value, target);
 52 
 53         defaultValue = target.getAttribute("defaultValue");
 54         if (value == undefined)
 55         {
 56             value = target.textContent;
 57             if (value == defaultValue)
 58                 value = "";
 59         }
 60 
 61         originalValue = previousValue = value;
 62 
 63         invalidEditor = false;
 64         currentTarget = target;
 65         currentPanel = panel;
 66         currentGroup = getAncestorByClass(target, "editGroup");
 67 
 68         currentPanel.editing = true;
 69 
 70         var panelEditor = currentPanel.getEditor(target, value);
 71         currentEditor = editor ? editor : panelEditor;
 72         if (!currentEditor)
 73             currentEditor = getDefaultEditor(currentPanel);
 74 
 75         var inlineParent = getInlineParent(target);
 76         var targetSize = getOffsetSize(inlineParent);
 77 
 78         setClass(panel.panelNode, "editing");
 79         setClass(target, "editing");
 80         if (currentGroup)
 81             setClass(currentGroup, "editing");
 82

 84         dispatch(this.fbListeners, "onBeginEditing", [currentPanel, currentEditor, target, value]);
 85         currentEditor.beginEditing(target, value);
 86         if (FBTrace.DBG_EDITOR)

100     stopEditing: function(cancel)
101     {
102         if (!currentTarget)
103             return;
104 
105         if (FBTrace.DBG_EDITOR)
106             FBTrace.sysout("editor.stopEditing cancel:" + cancel+" saveTimeout: "+this.saveTimeout);
107 
108         clearTimeout(this.saveTimeout);
109         delete this.saveTimeout;
110 
111         this.detachListeners(currentEditor, currentPanel.context);
112 
113         removeClass(currentPanel.panelNode, "editing");
114         removeClass(currentTarget, "editing");
115         if (currentGroup)
116             removeClass(currentGroup, "editing");
117 
118         var value = currentEditor.getValue();
119         if (value == defaultValue)
120             value = "";
121 
122         // Reset the editor's value so it isn't accidentaly reused the next time
123         // the editor instance is reused (see also 3280, 3332).
124         currentEditor.setValue("");
125 
126         var removeGroup = currentEditor.endEditing(currentTarget, value, cancel);
127 
128         try
129         {
130             if (cancel)
131             {
132                 dispatch(currentPanel.fbListeners, 'onInlineEditorClose', [currentPanel, currentTarget, removeGroup && !originalValue]);
133                 if (value != originalValue)
134                     this.saveEditAndNotifyListeners(currentTarget, originalValue, previousValue);
135 
136                 if (removeGroup && !originalValue && currentGroup)
137                     currentGroup.parentNode.removeChild(currentGroup);
138             }
139             else if (!value)
140             {
141                 this.saveEditAndNotifyListeners(currentTarget, "", previousValue);
142 
143                 if (removeGroup && currentGroup)
144                     currentGroup.parentNode.removeChild(currentGroup);
145             }
146             else
147                 this.save(value);
148         }
149         catch (exc)
150         {
151             ERROR(exc);
152         }
153 
154         currentEditor.hide();
155         currentPanel.editing = false;
156 
157         dispatch(this.fbListeners, "onStopEdit", [currentPanel, currentEditor, currentTarget]);
158         if (FBTrace.DBG_EDITOR)
159             FBTrace.sysout("Editor stop panel "+currentPanel.name);
160         currentTarget = null;
161         currentGroup = null;
162         currentPanel = null;
163         currentEditor = null;
164         originalValue = null;
165         invalidEditor = false;
166 
167         return value;
168     },

349         }
350         else

352             this.listeners.push(
353                 chrome.keyCodeListen("RETURN", null, bindFixed(this.stopEditing, this))
354             );
355 
356             if (currentEditor.tabCompletion)
357             {
358                 this.listeners.push(
359                     chrome.keyCodeListen("TAB", null, bind(editor.completeValue, editor, 1)),
360                     chrome.keyCodeListen("TAB", isShift, bind(editor.completeValue, editor, -1)),
361                     chrome.keyCodeListen("UP", null, bindFixed(editor.completeValue, editor, -1, true)),

697     getAutoCompleter: function()
698     {
699         if (!this.autoCompleter)
700         {
701             this.autoCompleter = new Firebug.AutoCompleter(null,
702                 bind(this.getAutoCompleteRange, this), bind(this.getAutoCompleteList, this),

709     completeValue: function(amt)
710     {
711         if (this.getAutoCompleter().complete(currentPanel.context, this.input, null, true, amt < 0, true))
712             Firebug.Editor.update(true);
713         else
714             this.incrementValue(amt);
715     },
716 
717     incrementValue: function(amt)
718     {
719         var value = this.input.value;
720         var start = this.input.selectionStart, end = this.input.selectionEnd;
721 
722         var range = this.getAutoCompleteRange(value, start);
723         if (!range || range.type != "int")
724             range = {start: 0, end: value.length-1};
725 
726         var expr = value.substr(range.start, range.end-range.start+1);
727         preExpr = value.substr(0, range.start);
728         postExpr = value.substr(range.end+1);
729 
730         // See if the value is an integer, and if so increment it
731         var intValue = parseInt(expr);
732         if (!!intValue || intValue == 0)
733         {
734             var m = /\d+/.exec(expr);
735             var digitPost = expr.substr(m.index+m[0].length);
736 
737             var completion = intValue-amt;
738             this.input.value = preExpr + completion + digitPost + postExpr;
739             this.input.setSelectionRange(start, end);
740 
741             Firebug.Editor.update(true);
742 
743             return true;
744         }
745         else
746             return false;
747     },

751     onKeyPress: function(event)
752     {
753         if (event.keyCode == 27 && !this.completeAsYouType)
754         {
755             var reverted = this.getAutoCompleter().revert(this.input);
756             if (reverted)
757                 cancelEvent(event);
758         }
759         else if (event.charCode && this.advanceToNext(this.target, event.charCode))
760         {
761             Firebug.Editor.tabNextEditor();
762             cancelEvent(event);
763         }
764         else if (this.numeric && event.charCode && (event.charCode < 48 || event.charCode > 57) && event.charCode != 45 && event.charCode != 46)
765         {
766             FBL.cancelEvent(event);
767         }
768         else
769         {
770             // If the user backspaces, don't autocomplete after the upcoming input event
771             this.ignoreNextInput = event.keyCode == 8;
772         }

893 Firebug.AutoCompleter = function(getExprOffset, getRange, evaluator, selectMode, caseSensitive,
894         noCompleteOnBlank, noShowGlobal, showCompletionPopup, isValidProperty)
895 {
896     var candidates = null;
897     var originalValue = null;
898     var originalOffset = -1;
899     var lastExpr = null;
900     var lastOffset = -1;
901     var exprOffset = 0;
902     var lastIndex = -2;  // adding 1 will still be less then zero
903     var preParsed = null;
904     var preExpr = null;
905     var postExpr = null;
906     var completionPopup = Firebug.chrome.$("fbCommandLineCompletionList");
907     var commandCompletionLineLimit = 40;
908     var reJavascriptChar = /[a-zA-Z0-9$_]/;
909     var reJavaScriptGroup = /([\{\"\/\(\'])/;
910     // current completion state values
911     var completionEnd = 0;
912     var value = "";
913     var preCompletion = "";
914     var completionStart = -1;
915     var completionEnd = -1;
916     var accepted = false;
917 
918     this.revert = function(textBox)
919     {
920         if (originalOffset != -1)
921         {
922             textBox.value = originalValue;
923             textBox.setSelectionRange(originalOffset, originalOffset);
924 
925             this.reset();
926             return true;
927         }
928         else
929         {
930             this.reset();
931             return false;
932         }
933     };
934 
935     this.reset = function()
936     {
937         candidates = null;
938         originalValue = null;
939         originalOffset = -1;
940         lastExpr = null;
941         lastOffset = 0;
942         exprOffset = 0;
943         lastIndex = -2;
944         accepted = false;
945     };
946 
947     this.complete = function(context, textBox, completionBox, cycle, reverse, showGlobals)
948     {
949         this.clearCandidates(textBox, completionBox);
950 
951         if (!this.getVerifiedText(textBox) && !showGlobals) // then no completion is desired
952             return false;
953 
954         var offset = textBox.selectionStart; // defines the cursor position
955 
956         var found =  this.pickCandidates(textBox, offset, context, cycle, reverse, showGlobals);
957 
958         if (completionBox && found)
959                 this.showCandidates(textBox, completionBox);
960 
961         return found;
962     };
963 
964     /*
965      * returns true if candidate list was created
966      */
967     this.pickCandidates = function(textBox, offset, context, cycle, reverse, showGlobals)
968     {
969         var value = textBox.value;
970 
971         if (!candidates || !cycle || offset != lastOffset)
972         {
973             originalOffset = offset;
974             originalValue = value;
975 
976             // Find the part of the string that will be parsed
977             var parseStart = getExprOffset ? getExprOffset(value, offset, context) : 0;
978             preParsed = value.substr(0, parseStart);
979             var parsed = value.substr(parseStart);
980 
981             // Find the part of the string that is being completed
982             var range = getRange ? getRange(parsed, offset-parseStart, context) : null;
983             if (!range)
984                     range = {start: 0, end: parsed.length-1 };
985 
986             var expr = parsed.substr(range.start, range.end-range.start+1);
987             preExpr = parsed.substr(0, range.start);
988             postExpr = parsed.substr(range.end+1);
989             exprOffset = parseStart + range.start;
990 
991             if (FBTrace.DBG_EDITOR)
992             {
993                 var sep = (parsed.indexOf('|') > -1) ? '^' : '|';
994                 FBTrace.sysout(preExpr+sep+expr+sep+postExpr+" offset: "+offset+" parseStart:"+parseStart);
995             }
996 
997             if (!cycle)
998             {
999                 if (!expr)
1000                 {
1001                     this.hide();
1002                     return false;
1003                 }
1004                 else if (lastExpr && lastExpr.indexOf(expr) != 0)
1005                 {
1006                     candidates = null;
1007                 }
1008                 else if (lastExpr && lastExpr.length >= expr.length)
1009                 {
1010                     candidates = null;
1011                     lastExpr = expr;
1012                     this.hide();
1013                     return false;
1014                 }
1015             }
1016 
1017             lastExpr = expr;
1018             lastOffset = offset;
1019 
1020             var searchExpr;
1021 
1022             // Check if the cursor is at the very right edge of the expression, or
1023             // somewhere in the middle of it
1024             if (expr && offset != parseStart+range.end+1)
1025             {
1026                 if (cycle)
1027                 {
1028                     // We are in the middle of the expression, but we can
1029                     // complete by cycling to the next item in the values
1030                     // list after the expression
1031                     offset = range.start;
1032                     searchExpr = expr;
1033                     expr = "";
1034                 }
1035                 else
1036                 {
1037                     // We can't complete unless we are at the ridge edge
1038                     this.hide();
1039                     return false;
1040                 }
1041             }
1042 
1043             if (!showGlobals && !preExpr && !expr && !postExpr)
1044             {
1045                 // Don't complete globals unless we are forced to do so.
1046                 this.hide();
1047                 return false;
1048             }
1049 
1050             var m = reJavaScriptGroup.exec(value);
1051             if (m) // then we have group operator
1052             {
1053                 return false; // give up, we need at least to balance
1054             }
1055             else
1056             {
1057                 var values = evaluator(preExpr, expr, postExpr, context);
1058                 if (!values)
1059                 {
1060                     this.hide();
1061                     return false;
1062                 }
1063 
1064                 if (expr)
1065                 {
1066                     this.setCandidatesByExpr(expr, values, reverse);
1067                 }
1068                 else if (searchExpr)
1069                 {
1070                     if (!this.setCandidatesBySearchExpr(searchExpr, expr, values))
1071                         return false;
1072                     expr = searchExpr;
1073                 }
1074                 else
1075                 {
1076                     this.setCandidatesByValues(values);
1077                 }
1078             }
1079         }
1080 
1081         if (cycle)
1082             expr = lastExpr;
1083 
1084         if (!candidates.length)
1085         {
1086             this.hide();
1087             return false;
1088         }
1089 
1090         this.adjustLastIndex(cycle, reverse);
1091 
1092         var completion = candidates[lastIndex];
1093         preCompletion = expr.substr(0, offset-exprOffset);
1094         var postCompletion = completion.substr(offset-exprOffset);
1095 
1096         var line = preParsed + preExpr + preCompletion + postCompletion + postExpr;
1097         var offsetEnd = preParsed.length + preExpr.length + completion.length;
1098 
1099 
1100         if (selectMode) // inline completion  uses this
1101         {
1102             textBox.value = line;
1103             textBox.setSelectionRange(offset, offsetEnd);
1104         }
1105         else
1106         {
1107             textBox.setSelectionRange(offsetEnd, offsetEnd);
1108         }
1109 
1110         // store current state of completion
1111         currentLine = line;
1112         completionStart = offset;
1113         completionEnd = offsetEnd;
1114 
1115         return true;
1116     };
1117 
1118     this.setCandidatesByExpr = function(expr, values, reverse)
1119     {
1120         // Filter the list of values to those which begin with expr. We
1121         // will then go on to complete the first value in the resulting list
1122         candidates = [];
1123 
1124         if (caseSensitive)
1125         {
1126             for (var i = 0; i < values.length; ++i)
1127             {
1128                 var name = values[i];
1129                 if (name.indexOf && name.indexOf(expr) == 0)
1130                     candidates.push(name);
1131             }
1132         }
1133         else
1134         {
1135             var lowerExpr = caseSensitive ? expr : expr.toLowerCase();
1136             for (var i = 0; i < values.length; ++i)
1137             {
1138                 var name = values[i];
1139                 if (name.indexOf && name.toLowerCase().indexOf(lowerExpr) == 0)
1140                     candidates.push(name);
1141             }
1142         }
1143 
1144         lastIndex = -2;
1145     };
1146 
1147     this.setCandidatesBySearchExpr = function(searchExpr, expr, values)
1148     {
1149         var searchIndex = -1;
1150 
1151         // Find the first instance of searchExpr in the values list. We
1152         // will then complete the string that is found
1153         if (caseSensitive)
1154         {
1155             searchIndex = values.indexOf(expr);
1156         }
1157         else
1158         {
1159             var lowerExpr = searchExpr.toLowerCase();
1160             for (var i = 0; i < values.length; ++i)
1161             {
1162                 var name = values[i];
1163                 if (name && name.toLowerCase().indexOf(lowerExpr) == 0)
1164                 {
1165                     searchIndex = i;
1166                     break;
1167                 }
1168             }
1169         }
1170 
1171         // Nothing found, so there's nothing to complete to
1172         if (searchIndex == -1)
1173         {
1174             this.reset();
1175             return false;
1176         }
1177 
1178         candidates = cloneArray(values);
1179         lastIndex = searchIndex;
1180         return true;
1181     };
1182 
1183     this.setCandidatesByValues = function(values)
1184     {
1185         expr = "";
1186         candidates = [];
1187         for (var i = 0; i < values.length; ++i)
1188         {
1189             var value = values[i];
1190             if (isValidProperty(value))
1191                 candidates.push(value);
1192         }
1193         lastIndex = -2;
1194     }
1195

1197     this.adjustLastIndex = function(cycle, reverse)
1198     {
1199         if (!cycle) // we have a valid lastIndex but we are not cycling, so reset it
1200             lastIndex = this.pickDefaultCandidate();
1201         else if (candidates.length === 1)
1202             lastIndex = 0;
1203         else if (lastIndex >= candidates.length)  // use default on first completion, else cycle
1204             lastIndex = (lastIndex === -2) ? this.pickDefaultCandidate() : 0;
1205         else if (lastIndex < 0)
1206             lastIndex = (lastIndex === -2) ? this.pickDefaultCandidate() : (candidates.length - 1);
1207         else // we have cycle == true
1208         {
1209             lastIndex += reverse ? -1 : 1;
1210             if (lastIndex >= candidates.length)
1211                 lastIndex = 0;
1212             else if (lastIndex < 0)
1213                 lastIndex = candidates.length - 1;
1214         }
1215     };
1216 
1217     this.cycle = function(reverse)
1218     {
1219         if (lastIndex < 0)
1220             return false;
1221 
1222         this.adjustLastIndex(true, reverse);
1223 
1224         var completion = candidates[lastIndex];
1225         var postCompletion = completion.substr(preCompletion.length);
1226         var line = currentLine.substr(0, completionStart);
1227         line += postCompletion;
1228         var end = line.length;
1229         line += currentLine.substr(completionEnd);
1230 
1231         // preCompletion and completionStart do not change
1232         currentLine = line;
1233         completionEnd = end;
1234         return true;
1235     },

1237     this.pickDefaultCandidate = function()
1238     {
1239         // The shortest candidate is default value
1240         var pick = 0;
1241         for (var i = 1; i < candidates.length; i++)
1242         {
1243             if (candidates[i].length < candidates[pick].length)
1244                 pick = i;
1245         }
1246         return pick;
1247     };

1388     this.handledKeyDown = function(event, context, textBox, completionBox)
1389     {
1390         var clearedTabWarning = this.clearTabWarning(completionBox);
1391 
1392         if (event.altKey || event.metaKey)
1393             return false;
1394 
1395         if (event.ctrlKey && event.keyCode === 32) // Control space
1396         {
1397             this.complete(context, textBox, completionBox, false, false, true); // force completion incl globals
1398             return true;
1399         }
1400         else if (event.keyCode === 9 || // TAB
1401             (event.keyCode === 39 && completionBox.value.length && textBox.selectionStart === textBox.value.length)) // right arrow
1402         {
1403             if (!completionBox.value.length)  // then no completion text,
1404             {
1405                 if (clearedTabWarning) // then you were warned,
1406                     return false; //  pass TAB along
1407 
1408                 this.setTabWarning(textBox, completionBox);
1409                 cancelEvent(event);
1410                 return true;
1411             }
1412             else  // complete
1413             {
1414                 this.acceptCompletionInTextBox(textBox, completionBox);
1415                 cancelEvent(event);
1416                 return true;
1417             }
1418         }
1419         else if (event.keyCode === 27) // ESC, close the completer
1420         {
1421             if (this.hide(completionBox))  // then we closed the popup
1422             {
1423                 cancelEvent(event); // Stop event bubbling if it was used to close the popup.
1424                 return true;
1425             }
1426         }
1427         else if (event.keyCode === 38 || event.keyCode === 40) // UP of DOWN arrow
1428         {
1429             if (this.getCompletionText(completionBox))
1430             {
1431                 if (this.cycle(event.keyCode === 38))
1432                     this.showCandidates(textBox, completionBox);
1433                 cancelEvent(event);
1434                 return true;
1435             }
1436             // else the arrow will fall through to command history
1437         }
1438     };

1522 /**
1523  * An outsider is the first element matching the stepper element that
1524  * is not an child of group. Elements tagged with insertBefore or insertAfter
1525  * classes are also excluded from these results unless they are the sibling
1526  * of group, relative to group's parent editGroup. This allows for the proper insertion
1527  * rows when groups are nested.
1528  */
1529 function getOutsider(element, group, stepper)
1530 {
1531     var parentGroup = getAncestorByClass(group.parentNode, "editGroup");
1532     var next;
1533     do
1534     {
1535         next = stepper(next || element);
1536     }
