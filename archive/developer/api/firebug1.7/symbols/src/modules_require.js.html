 64     function mixin(target, source, force) {
 65         for (var prop in source) {
 66             if (!(prop in empty) && (!(prop in target) || force)) {
 67                 target[prop] = source[prop];
 68             }

 74      * Used to set up package paths from a packagePaths or packages config object.
 75      * @param {Object} packages the object to store the new package config
 76      * @param {Array} currentPackages an array of packages to configure
 77      * @param {String} [dir] a prefix dir to use.

 79     function configurePackageDir(packages, currentPackages, dir) {
 80         var i, location, pkgObj;
 81         for (i = 0; (pkgObj = currentPackages[i]); i++) {
 82             pkgObj = typeof pkgObj === "string" ? { name: pkgObj } : pkgObj;
 83             location = pkgObj.location;
 84 
 85             //Add dir to the path, but avoid paths that start with a slash
 86             //or have a colon (indicates a protocol)
 87             if (dir && (!location || (location.indexOf("/") !== 0 && location.indexOf(":") === -1))) {
 88                 pkgObj.location = dir + "/" + (pkgObj.location || pkgObj.name);
 89             }
 90 
 91             //Normalize package paths.
 92             pkgObj.location = pkgObj.location || pkgObj.name;
 93             pkgObj.lib = pkgObj.lib || "lib";
 94             //Remove leading dot in main, so main paths are normalized.
 95             pkgObj.main = (pkgObj.main || "lib/main").replace(currDirRegExp, '');
 96 
 97             packages[pkgObj.name] = pkgObj;
 98         }

148         /**
149          * Trims the . and .. from an array of path segments.
150          * It will keep a leading path segment if a .. will become
151          * the first path segment, to help with module name lookups,
152          * which act like paths, but can be remapped. But the end result,
153          * all paths that use this function should look normalized.
154          * NOTE: this method MODIFIES the input array.
155          * @param {Array} ary the array of path segments.
156          */
157         function trimDots(ary) {
158             var i, part;
159             for (i = 0; (part = ary[i]); i++) {
160                 if (part === ".") {
161                     ary.splice(i, 1);
162                     i -= 1;
163                 } else if (part === "..") {
164                     if (i === 1 && (ary[2] === '..' || ary[0] === '..')) {
165                         //End of the line. Keep at least one non-dot
166                         //path segment at the front so it can be mapped
167                         //correctly to disk. Otherwise, there is likely
168                         //no path mapping for a path starting with '..'.
169                         //This can still fail, but catches the most reasonable
170                         //uses of ..
171                         break;
172                     } else if (i > 0) {
173                         ary.splice(i - 1, 2);
174                         i -= 2;
175                     }
176                 }
177             }
178         }

180         /**
181          * Given a relative module name, like ./something, normalize it to
182          * a real name that can be mapped to a path.
183          * @param {String} name the relative name
184          * @param {String} baseName a real name that the name arg is relative
185          * to.
186          * @returns {String} normalized name
187          */
188         function normalizeName(name, baseName) {
189             var pkgName, pkgConfig;
190 
191             //Adjust any relative paths.
192             if (name.charAt(0) === ".") {
193                 //If have a base name, try to normalize against it,
194                 //otherwise, assume it is a top-level require that will
195                 //be relative to baseUrl in the end.
196                 if (baseName) {
197                     if (config.packages[baseName]) {
198                         //If the baseName is a package name, then just treat it as one
199                         //name to concat the name with.
200                         baseName = [baseName];
201                     } else {
202                         //Convert baseName to array, and lop off the last part,
203                         //so that . matches that "directory" and not name of the baseName's
204                         //module. For instance, baseName of "one/two/three", maps to
205                         //"one/two/three.js", but we want the directory, "one/two" for
206                         //this normalization.
207                         baseName = baseName.split("/");
208                         baseName = baseName.slice(0, baseName.length - 1);
209                     }
210 
211                     name = baseName.concat(name.split("/"));
212                     trimDots(name);
213 
214                     //Some use of packages may use a . path to reference the
215                     //"main" module name, so normalize for that.
216                     pkgConfig = config.packages[(pkgName = name[0])];
217                     name = name.join("/");
218                     if (pkgConfig && name === pkgName + '/' + pkgConfig.main) {
219                         name = pkgName;
220                     }
221                 }
222             }
223             return name;
224         }

226         /**
227          * Creates a module mapping that includes plugin prefix, module
228          * name, and path. If parentModuleMap is provided it will
229          * also normalize the name via require.normalizeName()
230          *
231          * @param {String} name the module name
232          * @param {String} [parentModuleMap] parent module map
233          * for the module name, used to resolve relative names.
234          *
235          * @returns {Object}
236          */
237         function makeModuleMap(name, parentModuleMap) {
238             var index = name ? name.indexOf("!") : -1,
239                 prefix = null,
240                 parentName = parentModuleMap ? parentModuleMap.name : null,
241                 normalizedName, url;
242 
243             if (index !== -1) {
244                 prefix = name.substring(0, index);
245                 name = name.substring(index + 1, name.length);
246             }
247 
248             //Account for relative paths if there is a base name.
249             if (name) {
250                 normalizedName = normalizeName(name, parentName);
251 
252                 url = urlMap[normalizedName];
253                 if (!url) {
254                     //Calculate url for the module, if it has a name.
255                     if (req.toModuleUrl) {
256                         //Special logic required for a particular engine,
257                         //like Node.
258                         url = req.toModuleUrl(context, name, parentModuleMap);
259                     } else {
260                         url = context.nameToUrl(name, null, parentModuleMap);
261                     }
262 
263                     //Store the URL mapping for later.
264                     urlMap[normalizedName] = url;
265                 }
266             }
267 
268             if (prefix) {
269                 prefix = normalizeName(prefix, parentName);
270                 //Allow simpler mappings for some plugins
271                 prefix = requirePlugins[prefix] || prefix;
272             }
273 
274             return {
275                 prefix: prefix,
276                 name: normalizedName,
277                 parentMap: parentModuleMap,
278                 url: url,
279                 fullName: prefix ? prefix + "!" + normalizedName : normalizedName
280             };
281         }

287             var priorityDone = true,

289                 priorityName, i;

301             return priorityDone;

327          * Helper function that creates a require function object to give to
328          * modules that ask for it as a dependency. It needs to be specific
329          * per module because of the implication of path mappings that may
330          * need to be relative to the module name.
331          */
332         function makeRequire(relModuleMap) {
333             var modRequire = makeContextModuleFunc(context.require, relModuleMap);
334 
335             mixin(modRequire, {
336                 nameToUrl: makeContextModuleFunc(context.nameToUrl, relModuleMap),
337                 toUrl: makeContextModuleFunc(context.toUrl, relModuleMap),
338                 isDefined: makeContextModuleFunc(context.isDefined, relModuleMap),
339                 ready: req.ready,
340                 isBrowser: req.isBrowser
341             });
342             //Something used by node.
343             if (req.paths) {
344                 modRequire.paths = req.paths;
345             }
346             return modRequire;
347         }

349         /*
350          * Queues a dependency for checking after the loader is out of a
351          * "paused" state, for example while a script file is being loaded
352          * in the browser, where it may have many modules defined in it.
353          *
354          * depName will be fully qualified, no relative . or .. path.
355          */
356         function queueDependency(dep) {
357             //Make sure to load any plugin and associate the dependency
358             //with that plugin.
359             var prefix = dep.prefix,
360                 fullName = dep.fullName;
361 
362             //Do not bother if the depName is already in transit
363             if (specified[fullName] || fullName in defined) {
364                 return;
365             }
366 
367             if (prefix) {
368                 //Queue up loading of the dependency, track it
369                 //via context.plugins
370                 context.plugins[prefix] = undefined;
371                 queueDependency(makeModuleMap(prefix));
372             }
373 
374             context.paused.push(dep);
375         }

377         function execManager(manager) {
378             var i, ret, waitingCallbacks,
379                 cb = manager.callback,
380                 fullName = manager.fullName,
381                 args = [],
382                 ary = manager.depArray;
383 
384             //Call the callback to define the module, if necessary.
385             if (cb && isFunction(cb)) {
386                 //Pull out the defined dependencies and pass the ordered
387                 //values to the callback.
388                 if (ary) {
389                     for (i = 0; i < ary.length; i++) {
390                         args.push(manager.deps[ary[i]]);
391                     }
392                 }
393 
394                 if (context.config.onDebug) {
395                     context.config.onDebug("require.js: defining "+fullName+" with "+args.length+" dependents", {defineFunction: manager.callback, dependents: args});
396                 }
397 
398                 ret = req.execCb(fullName, manager.callback, args);
399 
400                 if (fullName) {
401                     //If using exports and the function did not return a value,
402                     //and the "module" object for this definition function did not
403                     //define an exported value, then use the exports object.
404                     if (manager.usingExports && ret === undefined && (!manager.cjsModule || !("exports" in manager.cjsModule))) {
405                         ret = defined[fullName];
406                     } else {
407                         if (manager.cjsModule && "exports" in manager.cjsModule) {
408                             ret = defined[fullName] = manager.cjsModule.exports;
409                         } else {
410                             if (fullName in defined && !manager.usingExports) {
411                                 return req.onError(new Error(fullName + " has already been defined"));
412                             }
413                             defined[fullName] = ret;
414                         }
415                     }
416                 }
417             } else if (fullName) {
418                 //May just be an object definition for the module. Only
419                 //worry about defining if have a module name.
420                 ret = defined[fullName] = cb;
421             }
422 
423             if (fullName) {
424                 //If anything was waiting for this module to be defined,
425                 //notify them now.
426                 waitingCallbacks = managerCallbacks[fullName];
427                 if (waitingCallbacks) {
428                     for (i = 0; i < waitingCallbacks.length; i++) {
429                         waitingCallbacks[i].onDep(fullName, ret);
430                     }
431                     delete managerCallbacks[fullName];
432                 }
433             }
434 
435             //Clean up waiting.
436             if (waiting[manager.waitId]) {
437                 delete waiting[manager.waitId];
438                 manager.isDone = true;
439                 context.waitCount -= 1;
440                 if (context.waitCount === 0) {
441                     //Clear the wait array used for cycles.
442                     waitAry = [];
443                 }
444             }
445 
446             return undefined;
447         }

449         function main(inName, depArray, callback, relModuleMap) {
450             var moduleMap = makeModuleMap(inName, relModuleMap),
451                 name = moduleMap.name,
452                 fullName = moduleMap.fullName,
453                 manager = {
454                     //Use a wait ID because some entries are anon
455                     //async require calls.
456                     waitId: name || reqWaitIdPrefix + (waitIdCounter++),
457                     depCount: 0,
458                     depMax: 0,
459                     prefix: moduleMap.prefix,
460                     name: name,
461                     fullName: fullName,
462                     deps: {},
463                     depArray: depArray,
464                     strikeList: [], // length depMax, value depName,
465                     callback: callback,
466                     onDep: function (depName, value) {
467                         if (!(depName in manager.deps)) {
468                             manager.deps[depName] = value;
469                             manager.depCount += 1;
470                             if (manager.depCount === manager.depMax) {
471                                 //All done, execute!
472                                 execManager(manager);
473                             }
474                         }
475                     }
476                 },
477                 i, depArg, depName, cjsMod;
478 
479             if (fullName) {
480                 //If module already defined for context, or already loaded,
481                 //then leave.
482                 if (fullName in defined || loaded[fullName] === true) {
483                     return;
484                 }
485 
486                 //Set specified/loaded here for modules that are also loaded
487                 //as part of a layer, where onScriptLoad is not fired
488                 //for those cases. Do this after the inline define and
489                 //dependency tracing is done.
490                 specified[fullName] = true;
491                 loaded[fullName] = true;
492 
493                 //Also check if auto-registry of jQuery needs to be skipped.
494                 context.jQueryDef = (fullName === "jquery");
495             }
496 
497             //Add the dependencies to the deps field, and register for callbacks
498             //on the dependencies.
499             for (i = 0; i < depArray.length; i++) {
500                 depArg = depArray[i];
501                 //There could be cases like in IE, where a trailing comma will
502                 //introduce a null dependency, so only treat a real dependency
503                 //value as a dependency.
504                 if (depArg) {
505                     //Split the dependency name into plugin and name parts
506                     depArg = makeModuleMap(depArg, (name ? moduleMap : relModuleMap));
507                     depName = depArg.fullName;
508 
509                     //Fix the name in depArray to be just the name, since
510                     //that is how it will be called back later.
511                     depArray[i] = depName;
512 
513                     //Fast path CommonJS standard dependencies.
514                     if (depName === "require") {
515                         manager.deps[depName] = makeRequire(moduleMap);
516                     } else if (depName === "exports") {
517                         //CommonJS module spec 1.1
518                         manager.deps[depName] = defined[fullName] = {};
519                         manager.usingExports = true;
520                     } else if (depName === "module") {
521                         //CommonJS module spec 1.1
522                         manager.cjsModule = cjsMod = manager.deps[depName] = {
523                             id: name,
524                             uri: name ? context.nameToUrl(name, null, relModuleMap) : undefined
525                         };
526                         cjsMod.setExports = makeSetExports(cjsMod);
527                     } else if (depName in defined && !(depName in waiting)) {
528                         //Module already defined, no need to wait for it.
529                         manager.deps[depName] = defined[depName];
530                     } else {
531                         //A dynamic dependency.
532                         manager.depMax += 1;
533                         manager.strikeList.push(depName);
534 
535                         queueDependency(depArg);
536 
537                         //Register to get notification when dependency loads.
538                         (managerCallbacks[depName] ||
539                         (managerCallbacks[depName] = [])).push(manager);
540                     }
541                 }
542             }
543 
544             //Do not bother tracking the manager if it is all done.
545             if (manager.depCount === manager.depMax) {
546                 //All done, execute!
547                 execManager(manager);
548             } else {
549                 waiting[manager.waitId] = manager;
550                 waitAry.push(manager);
551                 context.waitCount += 1;
552             }
553         }

567         /**
568          * As of jQuery 1.4.3, it supports a readyWait property that will hold off
569          * calling jQuery ready callbacks until all scripts are loaded. Be sure
570          * to track it if readyWait is available. Also, since jQuery 1.4.3 does
571          * not register as a module, need to do some global inference checking.
572          * Even if it does register as a module, not guaranteed to be the precise
573          * name of the global. If a jQuery is tracked for this context, then go
574          * ahead and register it as a module too, if not already in process.

576         function jQueryCheck(jqCandidate) {
577             if (!context.jQuery) {
578                 var $ = jqCandidate || (typeof jQuery !== "undefined" ? jQuery : null);
579                 if ($ && "readyWait" in $) {
580                     context.jQuery = $;
581 
582                     //Manually create a "jquery" module entry if not one already
583                     //or in process.
584                     callDefMain(["jquery", [], function () {
585                         return jQuery;
586                     }]);
587 
588                     //Increment jQuery readyWait if ncecessary.
589                     if (context.scriptCount) {
590                         $.readyWait += 1;
591                         context.jQueryIncremented = true;
592                     }
593                 }
594             }
595         }

597         function forceExec(manager, traced) {
598             if (manager.isDone) {
599                 return undefined;
600             }
601 
602             var fullName = manager.fullName,
603                 depArray = manager.depArray,
604                 depName, i;
605             if (fullName) {
606                 if (traced[fullName]) {
607                     return defined[fullName];
608                 }
609 
610                 traced[fullName] = true;
611             }
612 
613             //forceExec all of its dependencies.
614             for (i = 0; i < depArray.length; i++) {
615                 //Some array members may be null, like if a trailing comma
616                 //IE, so do the explicit [i] access and check if it has a value.
617                 depName = depArray[i];
618                 if (depName) {
619                     if (!manager.deps[depName] && waiting[depName]) {
620                         manager.onDep(depName, forceExec(waiting[depName], traced));
621                     }
622                 }
623             }
624 
625             return fullName ? defined[fullName] : undefined;
626         }

628         /**
629          * Checks if all modules for a context are loaded, and if so, evaluates the
630          * new ones in right dependency order.
631          *
632          * @private
633          */
634         function checkLoaded() {
635             var waitInterval = config.waitSeconds * 1000,
636                 //It is possible to disable the wait interval by using waitSeconds of 0.
637                 expired = waitInterval && (context.startTime + waitInterval) < new Date().getTime(),
638                 noLoads = "", hasLoadedProp = false, stillLoading = false, prop,
639                 err, manager;
640 
641             //Determine if priority loading is done. If so clear the priority. If
642             //not, then do not check
643             if (config.priorityWait) {
644                 if (isPriorityDone()) {
645                     //Call resume, since it could have
646                     //some waiting dependencies to trace.
647                     resume();
648                 } else {
649                     return undefined;
650                 }
651             }
652 
653             //See if anything is still in flight.
654             for (prop in loaded) {
655                 if (!(prop in empty)) {
656                     hasLoadedProp = true;
657                     if (!loaded[prop]) {
658                         if (expired) {
659                             noLoads += prop + " ";
660                         } else {
661                             stillLoading = true;
662                             break;
663                         }
664                     }
665                 }
666             }
667 
668             //Check for exit conditions.
669             if (!hasLoadedProp && !context.waitCount) {
670                 //If the loaded object had no items, then the rest of
671                 //the work below does not need to be done.
672                 return undefined;
673             }
674             if (expired && noLoads) {
675                 //If wait time expired, throw error of unloaded modules.
676                 err = new Error("require.js load timeout for modules: " + noLoads);
677                 err.requireType = "timeout";
678                 err.requireModules = noLoads;
679                 return req.onError(err);
680             }
681             if (stillLoading || context.scriptCount) {
682                 //Something is still waiting to load. Wait for it.
683                 if (isBrowser || isWebWorker) {
684                     setTimeout(checkLoaded, 50);
685                 }
686                 return undefined;
687             }
688 
689             //If still have items in the waiting cue, but all modules have
690             //been loaded, then it means there are some circular dependencies
691             //that need to be broken.
692             //However, as a waiting thing is fired, then it can add items to
693             //the waiting cue, and those items should not be fired yet, so
694             //make sure to redo the checkLoaded call after breaking a single
695             //cycle, if nothing else loaded then this logic will pick it up
696             //again.
697             if (context.waitCount) {
698                 //Cycle through the waitAry, and call items in sequence.
699                 for (i = 0; (manager = waitAry[i]); i++) {
700                     forceExec(manager, {});
701                 }
702 
703                 if (!context.checkLoadedDepth) {
704                     context.checkLoadedDepth = 0;
705                 }

1094             /**
1095              * Converts a module name to a file path. Supports cases where
1096              * moduleName may actually be just an URL.
1097              */
1098             nameToUrl: function (moduleName, ext, relModuleMap) {
1099 
1100                 var paths, packages, pkg, pkgPath, syms, i, parentModule, url,
1101                     config = context.config;
1102 
1103                 if (moduleName.indexOf("./") === 0 || moduleName.indexOf("../") === 0) {
1104                     //A relative ID, just map it relative to relModuleMap's url
1105                     syms = relModuleMap && relModuleMap.url ? relModuleMap.url.split('/') : [];
1106                     //Pop off the file name.
1107                     if (syms.length) {
1108                         syms.pop();
1109                     }
1110                     syms = syms.concat(moduleName.split('/'));
1111                     trimDots(syms);
1112                     url = syms.join('/') +
1113                           (ext ? ext :
1114                           (req.jsExtRegExp.test(moduleName) ? "" : ".js"));
1115                 } else {
1116 
1117                     //Normalize module name if have a base relative module name to work from.
1118                     moduleName = normalizeName(moduleName, relModuleMap);
1119 
1120                     //If a colon is in the URL, it indicates a protocol is used and it is just
1121                     //an URL to a file, or if it starts with a slash or ends with .js, it is just a plain file.
1122                     //The slash is important for protocol-less URLs as well as full paths.
1123                     if (req.jsExtRegExp.test(moduleName)) {
1124                         //Just a plain path, not module name lookup, so just return it.
1125                         //Add extension if it is included. This is a bit wonky, only non-.js things pass
1126                         //an extension, this method probably needs to be reworked.
1127                         url = moduleName + (ext ? ext : "");
1128                     } else {
1129                         //A module that needs to be converted to a path.
1130                         paths = config.paths;
1131                         packages = config.packages;
1132 
1133                         syms = moduleName.split("/");
1134                         //For each module name segment, see if there is a path
1135                         //registered for it. Start with most specific name
1136                         //and work up from it.
1137                         for (i = syms.length; i > 0; i--) {
1138                             parentModule = syms.slice(0, i).join("/");
1139                             if (paths[parentModule]) {
1140                                 syms.splice(0, i, paths[parentModule]);
1141                                 break;
1142                             } else if ((pkg = packages[parentModule])) {
1143                                 //If module name is just the package name, then looking
1144                                 //for the main module.
1145                                 if (moduleName === pkg.name) {
1146                                     pkgPath = pkg.location + '/' + pkg.main;
1147                                 } else {
1148                                     pkgPath = pkg.location + '/' + pkg.lib;
1149                                 }
1150                                 syms.splice(0, i, pkgPath);
1151                                 break;
1152                             }
1153                         }
1154 
1155                         //Join the path parts together, then figure out if baseUrl is needed.
1156                         url = syms.join("/") + (ext || ".js");
1157                         url = (url.charAt(0) === '/' || url.match(/^\w+:/) ? "" : config.baseUrl) + url;
1158                     }
1159                 }
1160 
1161                 return config.urlArgs ? url +
1162                                         ((url.indexOf('?') === -1 ? '?' : '&') +
1163                                          config.urlArgs) : url;
1164             }

1311     /**
1312      * The function that handles definitions of modules. Differs from
1313      * require() in that a string for the module should be the first argument,
1314      * and the function to execute after dependencies are loaded should
1315      * return a value to define the module corresponding to the first argument's
1316      * name.
1317      */
1318     define = req.def = function (name, deps, callback) {
1319         var node, context;
1320 
1321         //Allow for anonymous functions
1322         if (typeof name !== 'string') {
1323             //Adjust args appropriately
1324             callback = deps;
1325             deps = name;
1326             name = null;
1327         }
1328 
1329         //This module may not have dependencies
1330         if (!req.isArray(deps)) {
1331             callback = deps;
1332             deps = [];
1333         }
1334 
1335         //If no name, and callback is a function, then figure out if it a
1336         //CommonJS thing with dependencies.
1337         if (!name && !deps.length && req.isFunction(callback)) {
1338             //Remove comments from the callback string,
1339             //look for require calls, and pull them into the dependencies,
1340             //but only if there are function args.
1341             if (callback.length) {

1343                     .toString()
1344                     .replace(commentRegExp, "")
1345                     .replace(cjsRequireRegExp, function (match, dep) {
1346                         deps.push(dep);
1347                     });

1350                 //could use exports, and such, so always add those as dependencies.
1351                 //This is a bit wasteful for RequireJS modules that do not need
1352                 //an exports or module object, but erring on side of safety.
1353                 //REQUIRES the function to expect the CommonJS variables in the
1354                 //order listed below.
1355                 deps = ["require", "exports", "module"].concat(deps);
1356             }

1364                 return req.onError(new Error("ERROR: No matching script interactive for " + callback));
1365             }
1366             if (!name) {
1367                 name = node.getAttribute("data-requiremodule");
1368             }
1369             context = contexts[node.getAttribute("data-requirecontext")];
1370         }

1373         //This allows multiple modules to be in a file without prematurely
1374         //tracing dependencies, and allows for anonymous module support,
1375         //where the module name is not known until the script onload event
1376         //occurs. If no context, use the global queue, and get it processed
1377         //in the onscript load callback.

1432     /**
1433      * Attaches the script represented by the URL to the current
1434      * environment. Right now only supports browser loading,
1435      * but can be redefined in other environments to do the right thing.
1436      * @param {String} url the url of the script to attach.
1437      * @param {String} contextName the name of the context that wants the script.
1438      * @param {moduleName} the name of the module that is associated with the script.
1439      * @param {Function} [callback] optional callback, defaults to require.onScriptLoad
1440      * @param {String} [type] optional type, defaults to text/javascript
1441      */
1442     req.attach = function (url, contextName, moduleName, callback, type) {
1443         var node, loaded, context;
1444         if (isBrowser) {
1445             //In the browser so use a script tag
1446             callback = callback || req.onScriptLoad;
1447             node = document.createElement("script");
1448             node.type = type || "text/javascript";
1449             node.charset = "utf-8";
1450             //Use async so Gecko does not block on executing the script if something
1451             //like a long-polling comet tag is being run first. Gecko likes
1452             //to evaluate scripts in DOM order, even for dynamic scripts.
1453             //It will fetch them async, but only evaluate the contents in DOM
1454             //order, so a long-polling script tag can delay execution of scripts
1455             //after it. But telling Gecko we expect async gets us the behavior
1456             //we want -- execute it whenever it is finished downloading. Only
1457             //Helps Firefox 3.6+
1458             //Allow some URLs to not be fetched async. Mostly helps the order!
1459             //plugin
1460             node.async = !s.skipAsync[url];
1461 
1462             node.setAttribute("data-requirecontext", contextName);
1463             node.setAttribute("data-requiremodule", moduleName);
1464 
1465             //Set up load listener.
1466             if (node.addEventListener) {
1467                 node.addEventListener("load", callback, false);
1468             } else {
1469                 //Probably IE. If not it will throw an error, which will be
1470                 //useful to know. IE (at least 6-8) do not fire
1471                 //script onload right after executing the script, so
1472                 //we cannot tie the anonymous require.def call to a name.
1473                 //However, IE reports the script as being in "interactive"
1474                 //readyState at the time of the require.def call.
1475                 useInteractive = true;
1476                 node.attachEvent("onreadystatechange", callback);
1477             }
1478             node.src = url;
1479 
1480             //For some cache cases in IE 6-8, the script executes before the end
1481             //of the appendChild execution, so to tie an anonymous require.def
1482             //call to the module name (which is stored on the node), hold on
1483             //to a reference to this node, but clear after the DOM insertion.
1484             currentlyAddingScript = node;
1485             if (baseElement) {
1486                 head.insertBefore(node, baseElement);
1487             } else {
1488                 head.appendChild(node);
1489             }
1490             currentlyAddingScript = null;
1491             return node;
1492         } else if (isWebWorker) {
1493             //In a web worker, use importScripts. This is not a very
1494             //efficient use of importScripts, importScripts will block until
1495             //its script is downloaded and evaluated. However, if web workers
1496             //are in play, the expectation that a build has been done so that
1497             //only one script needs to be loaded anyway. This may need to be
1498             //reevaluated if other use cases become common.
1499             context = contexts[contextName];
1500             loaded = context.loaded;
1501             loaded[moduleName] = false;
1502 
1503             importScripts(url);
1504 
1505             //Account for anonymous modules
1506             context.completeLoad(moduleName);
1507         }
1508         return null;
1509     };
