 11 /**
 12  * Describes the execution of JavaScript within a browser context. A Stack
 13  * can be suspended, then provide stack frames, and then be resumed.

107 /**
108  * Sets this execution context as currently suspended. Fires notification
109  * of the suspend to registered listeners. Subclasses may call this method
110  * when a suspend occurs.
111  * <p>
112  * Has no effect if this execution context is already suspended.
113  * </p>
114  *
115  * @function
116  * @param compilationUnit the compilation unit where the suspend occurred
117  * @param lineNumber the line number the suspend occurred at
118  */
119 JavaScriptStack.prototype._suspended = function(compilationUnit, lineNumber)
120 {
121     if (!this.is_suspended)
122     {
123         this.is_suspended = true;
124         this.getBrowserContext().getBrowser()._dispatch("onBreak", [compilationUnit, lineNumber]);
125     }
126 };
127 
128 /**
129  * Sets this execution context as currently running. Fires notification
130  * of the resume to registered listeners. Subclasses may call this method
131  * when a resume occurs.
132  * <p>
133  * Has no effect if this execution context is already running.
134  * </p>
135  *
136  * @function
137  */
138 JavaScriptStack.prototype._resumed = function()
139 {
140     if (this.is_suspended)
141     {
142         this.is_suspended = false;
143         this.getBrowserContext().getBrowser()._dispatch("onResume", [this]);
144     }
145 };
