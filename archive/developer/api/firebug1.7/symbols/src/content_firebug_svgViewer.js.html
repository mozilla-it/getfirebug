  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 // Constants
  7 
  8 // List of SVG related content types.
  9 var contentTypes =
 10 [
 11     "image/svg+xml",
 12 ];
 13 
 14 // ************************************************************************************************
 15 // Model implementation
 16 
 17 /**
 18  * @module Implements viewer for SVG based network responses. In order to create a new
 19  * tab wihin network request detail, a listener is registered into
 20  * <code>Firebug.NetMonitor.NetInfoBody</code> object.
 21  */
 22 Firebug.SVGViewerModel = extend(Firebug.Module,
 23 {
 24     dispatchName: "svgViewer",
 25 
 26     initialize: function()
 27     {
 28         Firebug.ActivableModule.initialize.apply(this, arguments);
 29         Firebug.NetMonitor.NetInfoBody.addListener(this);
 30     },
 31 
 32     shutdown: function()
 33     {
 34         Firebug.ActivableModule.shutdown.apply(this, arguments);
 35         Firebug.NetMonitor.NetInfoBody.removeListener(this);
 36     },
 37 
 38     /**
 39      * Check response's content-type and if it's a SVG, create a new tab with SVG preview.
 40      */
 41     initTabBody: function(infoBox, file)
 42     {
 43         if (FBTrace.DBG_SVGVIEWER)
 44             FBTrace.sysout("svgviewer.initTabBody", infoBox);
 45 
 46         // If the response is SVG let's display a pretty preview.
 47         if (this.isSVG(safeGetContentType(file.request)))
 48         {
 49             Firebug.NetMonitor.NetInfoBody.appendTab(infoBox, "SVG",
 50                 $STR("svgviewer.tab.SVG"));
 51 
 52             if (FBTrace.DBG_SVGVIEWER)
 53                 FBTrace.sysout("svgviewer.initTabBody; SVG response available");
 54         }
 55     },
 56 
 57     isSVG: function(contentType)
 58     {
 59         if (!contentType)
 60             return false;
 61 
 62         // Look if the response is SVG based.
 63         for (var i=0; i<contentTypes.length; i++)
 64         {
 65             if (contentType.indexOf(contentTypes[i]) == 0)
 66                 return true;
 67         }
 68 
 69         return false;
 70     },
 71 
 72     /**
 73      * Parse SVG response and render pretty printed preview.
 74      */
 75     updateTabBody: function(infoBox, file, context)
 76     {
 77         var tab = infoBox.selectedTab;
 78         var tabBody = infoBox.getElementsByClassName("netInfoSVGText").item(0);
 79         if (!hasClass(tab, "netInfoSVGTab") || tabBody.updated)
 80             return;
 81 
 82         tabBody.updated = true;
 83 
 84         this.insertSVG(tabBody, file.responseText);
 85     },
 86 
 87     insertSVG: function(parentNode, text)
 88     {
 89         var parser = CCIN("@mozilla.org/xmlextras/domparser;1", "nsIDOMParser");
 90         var doc = parser.parseFromString(text, "text/xml");
 91         var root = doc.documentElement;
 92 
 93         // Error handling
 94         var nsURI = "http://www.mozilla.org/newlayout/xml/parsererror.xml";
 95         if (root.namespaceURI == nsURI && root.nodeName == "parsererror")
 96         {
 97             this.ParseError.tag.replace({error: {
 98                 message: root.firstChild.nodeValue,
 99                 source: root.lastChild.textContent
100             }}, parentNode);
101             return;
102         }
103 
104         if (FBTrace.DBG_SVGVIEWER)
105             FBTrace.sysout("svgviewer.updateTabBody; SVG response parsed", doc);
106 
107         // Override getHidden in these templates. The parsed SVG document is
108         // hidden, but we want to display it using 'visible' styling.
109         var templates = [
110             Firebug.HTMLPanel.CompleteElement,
111             Firebug.HTMLPanel.Element,
112             Firebug.HTMLPanel.TextElement,
113             Firebug.HTMLPanel.EmptyElement,
114             Firebug.HTMLPanel.XEmptyElement,
115         ];
116 
117         var originals = [];
118         for (var i=0; i<templates.length; i++)
119         {
120             originals[i] = templates[i].getHidden;
121             templates[i].getHidden = function() {
122                 return "";
123             }
124         }
125 
126         // Generate SVG preview.
127         Firebug.HTMLPanel.CompleteElement.tag.replace({object: doc.documentElement}, parentNode);
128 
129         for (var i=0; i<originals.length; i++)
130             templates[i].getHidden = originals[i];
131     }
132 });
133 
134 // ************************************************************************************************
135 // Domplate
136 
137 /**
138  * @domplate Represents a template for displaying SVG parser errors. Used by
139  * <code>Firebug.SVGViewerModel</code>.
140  */
141 Firebug.SVGViewerModel.ParseError = domplate(Firebug.Rep,
142 {
143     tag:
144         DIV({"class": "svgInfoError"},
145             DIV({"class": "svgInfoErrorMsg"}, "$error.message"),
146             PRE({"class": "svgInfoErrorSource"}, "$error|getSource")
147         ),
148 
149     getSource: function(error)
150     {
151         var parts = error.source.split("\n");
152         if (parts.length != 2)
153             return error.source;
154 
155         var limit = 50;
156         var column = parts[1].length;
157         if (column >= limit) {
158             parts[0] = "..." + parts[0].substr(column - limit);
159             parts[1] = "..." + parts[1].substr(column - limit);
160         }
161 
162         if (parts[0].length > 80)
163             parts[0] = parts[0].substr(0, 80) + "...";
164 
165         return parts.join("\n");
166     }
167 });
168 
169 // ************************************************************************************************
170 // Registration
171 
172 Firebug.registerModule(Firebug.SVGViewerModel);
173 
174 return Firebug.SVGViewerModel;
175 }});
