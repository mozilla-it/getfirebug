 21 /**
 22  * Proxy to a debuggable web browser. A browser may be remote and contain one or more
 23  * JavaScript execution contexts. Each JavaScript execution context may contain one or
 24  * more compilation units. A browser provides notification to registered listeners describing
 25  * events that occur in the browser.
 26  *
 27  * @constructor
 28  * @type Browser
 29  * @return a new Browser
 30  * @version 1.0
 31  */
 32 function Browser()
 33 {
 34     this.contexts = {}; // map of contexts, indexed by context ID
 35     this.activeContext = null;
 36     this.handlers = {}; // map of event types to array of handler functions
 37     this.EVENT_TYPES = ["onBreak", "onConsoleDebug", "onConsoleError", "onConsoleInfo",
 38         "onConsoleLog", "onConsoleWarn", "onContextCreated", "onContextDestroyed",
 39         "onContextChanged", "onContextLoaded", "onInspectNode", "onResume", "onScript",
 40         "onSuspend", "onToggleBreakpoint", "onBreakpointError", "onDisconnect"];
 41     this.connected = false;
 42 }

 76 /**
 77  * Returns the root contexts being browsed. A {@link BrowserContext} represents the
 78  * content that has been served up and is being rendered for a location (URL) that
 79  * has been navigated to.
 80  * <p>
 81  * This function does not require communication with the remote browser.
 82  * </p>
 83  * @function
 84  * @returns an array of {@link BrowserContext}'s
 85  */
 86 Browser.prototype.getBrowserContexts = function()
 87 {
 88     var knownContexts = [];
 89     for (var id in this.contexts)
 90         knownContexts.push(this.contexts[id]);
 91     return knownContexts;
 92 };

118 /**
119  * Registers a listener (function) for a specific type of event. Listener
120  * call back functions are specified in {@link BrowserEventListener}.
121  * <p>
122  * The supported event types are:
123  * <ul>
124  *   <li>onBreak</li>
125  *   <li>onConsoleDebug</li>
126  *   <li>onConsoleError</li>
127  *   <li>onConsoleInfo</li>
128  *   <li>onConsoleLog</li>
129  *   <li>onConsoleWarn</li>
130  *   <li>onContextCreated</li>
131  *   <li>onContextChanged</li>
132  *   <li>onContextDestroyed</li>
133  *   <li>onDisconnect</li>
134  *   <li>onInspectNode</li>
135  *   <li>onResume</li>
136  *   <li>onScript</li>
137  *   <li>onToggleBreakpoint</li>
138  * </ul>
139  * <ul>
140  * <li>TODO: how can clients remove (deregister) listeners?</li>
141  * </ul>
142  * </p>
143  * @function
144  * @param eventType an event type ({@link String}) listed above
145  * @param listener a listener (function) that handles the event as specified
146  *   by {@link BrowserEventListener}
147  * @exception Error if an unsupported event type is specified
148  */
149 Browser.prototype.addEventListener = function(eventType, listener)
150 {
151     var i = this.EVENT_TYPES.indexOf(eventType);
152     if (i < 0)
153     {
154         // unsupported event type
155         throw new Error("eventType '" + eventType + "' is not supported");
156     }
157     var list = this.handlers[eventType];
158     if (!list)
159     {
160         list = [];
161         this.handlers[eventType] = list;
162     }
163     list.push(listener);
164 };

180 /**
181  * Notification the given context has been added to this browser.
182  * Adds the context to the list of active contexts and notifies context
183  * listeners.
184  * <p>
185  * Has no effect if the context has already been created. For example,
186  * it's possible for a race condition to occur when a remote browser
187  * sends notification of a context being created before the initial set
188  * of contexts have been retrieved. In such a case, it would possible for
189  * a client to add the context twice (once for the create event, and again
190  * when retrieving the initial list of contexts).
191  * </p>
192  * @function
193  * @param context the {@link BrowserContext} that has been added
194  */
195 Browser.prototype._contextCreated = function(context)
196 {
197     // if already present, don't add it again
198     var id = context.getId();
199     if (this.contexts[id])
200         return;
201 
202     this.contexts[id] = context;
203     this._dispatch("onContextCreated", [context]);
204 };
205 
206 /**
207  * Notification the given context has been destroyed.
208  * Removes the context from the list of active contexts and notifies context
209  * listeners.
210  * <p>
211  * Has no effect if the context has already been destroyed or has not yet
212  * been retrieved from the browser. For example, it's possible for a race
213  * condition to occur when a remote browser sends notification of a context
214  * being destroyed before the initial list of contexts is retrieved from the
215  * browser. In this case an implementation could ask to destroy a context that
216  * that has not yet been reported as created.
217  * </p>
218  *
219  * @function
220  * @param id the identifier of the {@link BrowserContext} that has been destroyed
221  */
222 Browser.prototype._contextDestroyed = function(id)
223 {
224     var destroyed = this.contexts[id];
225     if (destroyed)
226     {
227         destroyed._destroyed();
228         delete this.contexts[id];
229         this._dispatch("onContextDestroyed", [destroyed]);
230     }
231 };
232 
233 /**
234  * Notification the given context has been loaded. Notifies context listeners.
235  *
236  * @function
237  * @param id the identifier of the {@link BrowserContext} that has been loaded
238  */
239 Browser.prototype._contextLoaded = function(id)
240 {
241     var loaded = this.contexts[id];
242     if (loaded)
243     {
244         loaded._loaded();
245         this._dispatch("onContextLoaded", [loaded]);
246     }
247 };
248 
249 /**
250  * Dispatches an event notification to all registered functions for
251  * the specified event type.
252  *
253  * @param eventType event type
254  * @param arguments arguments to be applied to handler functions
255  */
256 Browser.prototype._dispatch = function(eventType, args)
257 {
258     var functions = this.handlers[eventType];
259     if (functions)
260     {
261         for ( var i = 0; i < functions.length; i++)
262             functions[i].apply(null, args);
263     }
264 };
265 
266 /**
267  * Sets the browser context that has focus, possibly <code>null</code>.
268  *
269  * @function
270  * @param context a {@link BrowserContext} or <code>null</code>
271  */
272 Browser.prototype._setFocusContext = function(context)
273 {
274     var prev = this.activeContext;
275     this.activeContext = context;
276     if (prev !== context)
277         this._dispatch("onContextChanged", [prev, this.activeContext]);
278 };
279 
280 /**
281  * Sets whether this proxy is connected to its underlying browser.
282  * Sends 'onDisconnect' notification when the browser becomes disconnected.
283  *
284  * @function
285  * @param connected whether this proxy is connected to its underlying browser
286  */
287 Browser.prototype._setConnected = function(connected)
288 {
289     var wasConnected = this.connected;
290     this.connected = connected;
291     if (wasConnected && !connected)
292         this._dispatch("onDisconnect", [this]);
293 };
