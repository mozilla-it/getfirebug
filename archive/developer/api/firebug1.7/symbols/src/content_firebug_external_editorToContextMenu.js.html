  5 // Runs during overlay processing
  6 function OpenEditorShowHide(event)
  7 {
  8     var item = document.getElementById("menu_firebugOpenWithEditor");
  9 
 10     var popupNode = document.popupNode;
 11     var hidden = (popupNode instanceof HTMLInputElement
 12         || popupNode instanceof HTMLIFrameElement
 13         || popupNode instanceof HTMLTextAreaElement)
 14     if(hidden)
 15     {
 16         item.hidden = true;
 17         return;
 18     }
 19     var editor=Firebug.ExternalEditors.getDefaultEditor();
 20     if(!editor)
 21     {
 22         item.hidden = true;
 23         return;
 24     }
 25     item.hidden = false;
 26     item.setAttribute('image', editor.image);
 27     item.setAttribute('label', editor.label);
 28     item.value = editor.id;
 29 }
 30 
 31 function addOpenEditorShowHide(event)
 32 {
 33     window.removeEventListener("load", addOpenEditorShowHide, false);
 34 
 35     var contextMenu = document.getElementById("contentAreaContextMenu");
 36     if (contextMenu)
 37     {
 38         addContextToForms();
 39         contextMenu.addEventListener("popupshowing", OpenEditorShowHide, false);
 40     }
 41 };
 42 
 43 function addContextToForms(contextMenu)
 44 {
 45     // https://bugzilla.mozilla.org/show_bug.cgi?id=433168
 46     var setTargetOriginal = nsContextMenu.prototype.setTarget;
 47     nsContextMenu.prototype.setTarget = function(aNode, aRangeParent, aRangeOffset)
 48     {
 49         setTargetOriginal.apply(this, arguments);
 50         if (this.isTargetAFormControl(aNode))
 51             this.shouldDisplay = true;
 52     };
 53 }
