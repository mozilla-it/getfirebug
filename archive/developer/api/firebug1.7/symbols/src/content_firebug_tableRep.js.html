  9 FirebugReps.Table = domplate(Firebug.Rep,
 10 {
 11     className: "table",
 12 
 13     tag:
 14         DIV({"class": "dataTableSizer", "tabindex": "-1" },
 15             TABLE({"class": "dataTable", cellspacing: 0, cellpadding: 0, width: "100%",
 16                 "role": "grid"},
 17                 THEAD({"class": "dataTableThead", "role": "presentation"},
 18                     TR({"class": "headerRow focusRow dataTableRow subFocusRow", "role": "row",
 19                         onclick: "$onClickHeader"},
 20                         FOR("column", "$object.columns",
 21                             TH({"class": "headerCell a11yFocus", "role": "columnheader",
 22                                 $alphaValue: "$column.alphaValue"},
 23                                 DIV({"class": "headerCellBox"},
 24                                     "$column.label"
 25                                 )
 26                             )
 27                         )
 28                     )
 29                 ),
 30                 TBODY({"class": "dataTableTbody", "role": "presentation"},
 31                     FOR("row", "$object.data|getRows",
 32                         TR({"class": "focusRow dataTableRow subFocusRow", "role": "row"},
 33                             FOR("column", "$row|getColumns",
 34                                 TD({"class": "a11yFocus dataTableCell", "role": "gridcell"},
 35                                     TAG("$column|getValueTag", {object: "$column"})
 36                                 )
 37                             )
 38                         )
 39                     )
 40                 )
 41             )
 42         ),
 43 
 44     getValueTag: function(object)
 45     {
 46         var rep = Firebug.getRep(object);
 47         return rep.shortTag || rep.tag;
 48     },
 49 
 50     getRows: function(data)
 51     {
 52         var props = this.getProps(data);
 53         if (!props.length)
 54             return [];
 55         return props;
 56     },
 57 
 58     getColumns: function(row)
 59     {
 60         if (typeof(row) != "object")
 61             return [row];
 62 
 63         var cols = [];
 64         for (var i=0; i<this.columns.length; i++)
 65         {
 66             var prop = this.columns[i].property;
 67             
 68             if (typeof row[prop] === "undefined")
 69             {
 70                 var props = (typeof(prop) == "string") ? prop.split(".") : [prop];
 71 
 72                 var value = row;
 73                 for (var p in props)
 74                     value = (value && value[props[p]]) || undefined;
 75             }
 76             else
 77             {
 78                 value = row[prop];
 79             }
 80 
 81             cols.push(value);
 82         }
 83         return cols;
 84     },
 85 
 86     getProps: function(obj)
 87     {
 88         if (typeof(obj) != "object")
 89             return [obj];
 90 
 91         if (obj.length)
 92             return cloneArray(obj);
 93 
 94         var arr = [];
 95         for (var p in obj)
 96         {
 97             var value = obj[p];
 98             if (this.domFilter(value, p))
 99                 arr.push(value);
100         }
101         return arr;
102     },
103 
104     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
105     // Sorting
106 
107     onClickHeader: function(event)
108     {
109         var table = getAncestorByClass(event.target, "dataTable");
110         var header = getAncestorByClass(event.target, "headerCell");
111         if (!header)
112             return;
113 
114         var numerical = !hasClass(header, "alphaValue");
115 
116         var colIndex = 0;
117         for (header = header.previousSibling; header; header = header.previousSibling)
118             ++colIndex;
119 
120         this.sort(table, colIndex, numerical);
121     },
122 
123     sort: function(table, colIndex, numerical)
124     {
125         var tbody = getChildByClass(table, "dataTableTbody");
126         var thead = getChildByClass(table, "dataTableThead");
127 
128         var values = [];
129         for (var row = tbody.childNodes[0]; row; row = row.nextSibling)
130         {
131             var cell = row.childNodes[colIndex];
132             var value = numerical ? parseFloat(cell.textContent) : cell.textContent;
133             values.push({row: row, value: value});
134         }
135 
136         values.sort(function(a, b) { return a.value < b.value ? -1 : 1; });
137 
138         var headerRow = thead.firstChild;
139         var headerSorted = getChildByClass(headerRow, "headerSorted");
140         removeClass(headerSorted, "headerSorted");
141         if (headerSorted)
142             headerSorted.removeAttribute('aria-sort');
143 
144         var header = headerRow.childNodes[colIndex];
145         setClass(header, "headerSorted");
146 
147         if (!header.sorted || header.sorted == 1)
148         {
149             removeClass(header, "sortedDescending");
150             setClass(header, "sortedAscending");
151             header.setAttribute("aria-sort", "ascending");
152 
153             header.sorted = -1;
154 
155             for (var i = 0; i < values.length; ++i)
156                 tbody.appendChild(values[i].row);
157         }
158         else
159         {
160             removeClass(header, "sortedAscending");
161             setClass(header, "sortedDescending");
162             header.setAttribute("aria-sort", "descending")
163 
164             header.sorted = 1;
165 
166             for (var i = values.length-1; i >= 0; --i)
167                 tbody.appendChild(values[i].row);
168         }
169     },
170 
171     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
172     // Console logging
173 
174     log: function(data, cols, context)
175     {
176         // No arguments passed into console.table method, bail out for now,
177         // but some error message could be displayed in the future.
178         if (!data)
179             return;
180 
181         // Get header info from passed argument (can be null).
182         var columns = [];
183         for (var i=0; cols && i<cols.length; i++)
184         {
185             var col = cols[i];
186             var prop = (typeof(col.property) != "undefined") ? col.property : col;
187             var label = (typeof(col.label) != "undefined") ? col.label : prop;
188 
189             columns.push({
190                 property: prop,
191                 label: label,
192                 alphaValue: true
193             });
194         }
195 
196         // Generate header info from the data dynamically.
197         if (!columns.length)
198             columns = this.getHeaderColumns(data);
199 
200         // Don't limit strings in the table. It should be mostly ok. In case of
201         // complaints we need an option.
202         var prevValue = Firebug.stringCropLength;
203         Firebug.stringCropLength = -1;
204 
205         try
206         {
207             this.columns = columns;
208             var row = Firebug.Console.log({data: data, columns: columns}, context, "table", this, true);
209 
210             // Set vertical height for scroll bar.
211             var tBody = row.querySelector(".dataTableTbody");
212             var maxHeight = Firebug.tabularLogMaxHeight;
213             if (maxHeight > 0 && tBody.clientHeight > maxHeight)
214                 tBody.style.height = maxHeight + "px";
215         }
216         catch (err)
217         {
218             if (FBTrace.DBG_CONSOLE)
219                 FBTrace.sysout("consoleInjector.table; EXCEPTION " + err, err);
220         }
221         finally
222         {
223             Firebug.stringCropLength = prevValue;
224             delete this.columns;
225         }
226     },
227 
228     /**
229      * Analyse data and return dynamically created list of columns.
230      * @param {Object} data
231      */
232     getHeaderColumns: function(data)
233     {
234         // Get the first row in the object.
235         var firstRow;
236         for (var p in data)
237         {
238             firstRow = data[p];
239             break;
240         }
241 
242         if (typeof(firstRow) != "object")
243             return [{label: $STR("firebug.reps.table.ObjectProperties")}];
244 
245         // Put together a column property, label and type (type for default sorting logic).
246         var header = [];
247         for (var p in firstRow)
248         {
249             var value = firstRow[p];
250             if (!this.domFilter(value, p))
251                 continue;
252 
253             header.push({
254                 property: p,
255                 label: p,
256                 alphaValue: (typeof(value) != "number")
257             });
258         }
259 
260         return header;
261     },
262 
263     /**
264      * Filtering based on options set in the DOM panel.
265      * @param {Object} value - a property value under inspection.
266      * @param {String} name - name of the property.
267      * @returns true if the value should be displayed, otherwise false.
268      */
269     domFilter: function(object, name)
270     {
271         var domMembers = getDOMMembers(object, name);
272 
273         if (typeof(object) == "function")
274         {
275             if (isDOMMember(object, name) && !Firebug.showDOMFuncs)
276                 return false;
277             else if (!Firebug.showUserFuncs)
278                 return false;
279         }
280         else
281         {
282             if (isDOMMember(object, name) && !Firebug.showDOMProps)
283                 return false;
284             else if (isDOMConstant(object, name) && !Firebug.showDOMConstants)
285                 return false;
286             else if (!Firebug.showUserProps)
287                 return false;
288         }
289 
290         return true;
291     }
292 });
