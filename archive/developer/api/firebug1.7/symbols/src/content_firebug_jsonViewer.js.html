  7 // List of JSON content types.
  8 var contentTypes =
  9 {
 10     "text/plain": 1,
 11     "text/javascript": 1,
 12     "text/x-javascript": 1,
 13     "text/json": 1,
 14     "text/x-json": 1,
 15     "application/json": 1,
 16     "application/x-json": 1,
 17     "application/javascript": 1,
 18     "application/x-javascript": 1,
 19     "application/json-rpc": 1

 22 // ************************************************************************************************
 23 // Model implementation
 24 
 25 Firebug.JSONViewerModel = extend(Firebug.Module,
 26 {
 27     dispatchName: "jsonViewer",
 28 
 29     initialize: function()
 30     {
 31         Firebug.NetMonitor.NetInfoBody.addListener(this);
 32     },
 33 
 34     shutdown: function()
 35     {
 36         Firebug.NetMonitor.NetInfoBody.removeListener(this);
 37     },
 38 
 39     initTabBody: function(infoBox, file)
 40     {
 41         if (FBTrace.DBG_JSONVIEWER)
 42             FBTrace.sysout("jsonviewer.initTabBody", infoBox);
 43 
 44         // Let listeners to parse the JSON.
 45         dispatch(this.fbListeners, "onParseJSON", [file]);
 46 
 47         // The JSON is still no there, try to parse most common cases.
 48         if (!file.jsonObject)
 49         {
 50             if (this.isJSON(safeGetContentType(file.request), file.responseText))
 51                 file.jsonObject = this.parseJSON(file);
 52         }
 53 
 54         // The jsonObject is created so, the JSON tab can be displayed.
 55         if (file.jsonObject && hasProperties(file.jsonObject))
 56         {
 57             Firebug.NetMonitor.NetInfoBody.appendTab(infoBox, "JSON",
 58                 $STR("jsonviewer.tab.JSON"));
 59 
 60             if (FBTrace.DBG_JSONVIEWER)
 61                 FBTrace.sysout("jsonviewer.initTabBody; JSON object available " +
 62                     (typeof(file.jsonObject) != "undefined"), file.jsonObject);
 63         }
 64     },
 65 
 66     isJSON: function(contentType, data)
 67     {
 68         // Workaround for JSON responses without proper content type
 69         // Let's consider all responses starting with "{" as JSON. In the worst
 70         // case there will be an exception when parsing. This means that no-JSON
 71         // responses (and post data) (with "{") can be parsed unnecessarily,
 72         // which represents a little overhead, but this happens only if the request
 73         // is actually expanded by the user in the UI (Net & Console panels).
 74         var responseText = data ? trimLeft(data) : null;
 75         if (responseText && responseText.indexOf("{") == 0)
 76             return true;
 77 
 78         if (!contentType)
 79             return false;
 80 
 81         contentType = contentType.split(";")[0];
 82         contentType = trim(contentType);
 83         return contentTypes[contentType];
 84     },
 85 
 86     // Update listener for TabView
 87     updateTabBody: function(infoBox, file, context)
 88     {
 89         var tab = infoBox.selectedTab;
 90         var tabBody = infoBox.getElementsByClassName("netInfoJSONText").item(0);
 91         if (!hasClass(tab, "netInfoJSONTab") || tabBody.updated)
 92             return;
 93 
 94         tabBody.updated = true;
 95         tabBody.context = context;
 96 
 97         this.Preview.render(tabBody, file, context);
 98     },
 99 
100     parseJSON: function(file)
101     {
102         var jsonString = new String(file.responseText);
103         return parseJSONString(jsonString, "http://" + file.request.originalURI.host);
104     },
105 });

109 Firebug.JSONViewerModel.Preview = domplate(
110 {
111     bodyTag:
112         DIV({"class": "jsonPreview", _repObject: "$file"},
113             DIV({"class": "title"},
114                 DIV({"class": "sortLink", onclick: "$onSort", $sorted: "$sorted"},
115                     SPAN({"class": "doSort"}, $STR("jsonviewer.sort")),
116                     SPAN({"class": "doNotSort"}, $STR("jsonviewer.do not sort"))
117                 )
118             ),
119             DIV({"class": "jsonPreviewBody"})
120         ),
121 
122     onSort: function(event)
123     {
124         var target = event.target;
125         var sortLink = FBL.getAncestorByClass(target, "sortLink");
126         if (!sortLink)
127             return;
128 
129         FBL.cancelEvent(event);
130 
131         FBL.toggleClass(sortLink, "sorted");
132         Firebug.setPref(Firebug.prefDomain, "sortJsonPreview", !Firebug.sortJsonPreview);
133 
134         var preview = FBL.getAncestorByClass(sortLink, "jsonPreview");
135         var body = FBL.getAncestorByClass(sortLink, "netInfoJSONText");
136         this.render(body, preview.repObject, body.context);
137     },
138 
139     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * //
140 
141     render: function(body, file, context)
142     {
143         if (!file.jsonObject)
144             return;
145 
146         if (!body.jsonTree)
147             body.jsonTree = new JSONTreePlate();
148 
149         var input = {file: file, sorted: Firebug.sortJsonPreview};
150         parentNode = this.bodyTag.replace(input, body, this);
151         parentNode = parentNode.getElementsByClassName("jsonPreviewBody").item(0);
152 
153         body.jsonTree.render(file.jsonObject, parentNode, context);
154     }
155 });

159 function JSONTreePlate()
160 {
161     // Used by Firebug.DOMPanel.DirTable domplate.
162     this.toggles = new ToggleBranch();
163 }
164 
165 // xxxHonza: this object is *not* a panel (using Firebug terminology), but
166 // there is no other way how to subclass the DOM Tree than to derive from the DOMBasePanel.
167 // Better solution would be to have a middle object between DirTablePlate and DOMBasePanel.
168 JSONTreePlate.prototype = extend(Firebug.DOMBasePanel.prototype,
169 {
170     dispatchName: "JSONTreePlate",
171 
172     render: function(jsonObject, parentNode, context)
173     {
174         try
175         {
176             this.panelNode = parentNode;
177             this.context = context;
178 
179             var members = this.getMembers(jsonObject, 0);
180             this.expandMembers(members, this.toggles, 0, 0, context);
181             this.showMembers(members, false, false);
182         }
183         catch (err)
184         {
185             if (FBTrace.DBG_JSONVIEWER)
186                 FBTrace.sysout("jsonviewer.render; EXCEPTION", err);
187         }
188     },
189 
190     getMembers: function(object, level, context)
191     {
192         if (!level)
193             level = 0;
194 
195         var members = [];
196 
197         for (var name in object)
198         {
199             var val = object[name];
200             this.addMember(object, "user", members, name, val, level, 0);
201         }
202 
203         function sortName(a, b) { return a.name > b.name ? 1 : -1; }
204         if (Firebug.sortJsonPreview)
205             members.sort(sortName);
206 
207         return members;
208     }
209 });
