 10 // Breakpoint

 12 /**
 13  * Describes a breakpoint in a compilation unit. A breakpoint is specific to a compilation unit and
 14  * execution context.
 15  * <p>
 16  * A breakpoint proceeds through states in its lifecycle.
 17  * <ul>
 18  * <li>{@link Breakpoint.PENDING_INSTALL} - created and pending installation in the runtime</li>
 19  * <li>{@link Breakpoint.FAILED_INSTALL} - failed to install in the runtime</li>
 20  * <li>{@link Breakpoint.INSTALLED} - installed in the runtime</li>
 21  * <li>{@link Breakpoint.PENDING_CLEAR} - pending removal from the runtime</li>
 22  * <li>{@link Breakpoint.FAILED_CLEAR} - failed to clear from the runtime</li>
 23  * <li>{@link Breakpoint.CLEARED} - cleared from the runtime</li>
 24  * </ul>
 25  * </p>
 26  * 
 27  * @constructor
 28  * @param compilationUnit the {@link CompilationUnit} unit that contains this breakpoint
 29  * @param lineNumber the source code line number the breakpoint is set on
 30  * @type Breakpoint
 31  * @return a new Breakpoint
 32  * @version 1.0
 33  */
 34 function Breakpoint(compilationUnit, lineNumber)
 35 {
 36     this.compilationUnit = compilationUnit;
 37     this.lineNumber = lineNumber;
 38     this.state = this.PENDING_INSTALL;
 39 }
 40 
 41 // ************************************************************************************************
 42 // API
 43 
 44 /**
 45  * Breakpoint state indicating a breakpoint has been created and is pending installation.
 46  * @constant 
 47  */
 48 Breakpoint.prototype.PENDING_INSTALL = 1;
 49 /**
 50  * Breakpoint state indicating a breakpoint failed to install.
 51  * @constant 
 52  */
 53 Breakpoint.prototype.FAILED_INSTALL = 2;
 54 /**
 55  * Breakpoint state indicating a breakpoint has been installed.
 56  * @constant 
 57  */
 58 Breakpoint.prototype.INSTALLED = 3;
 59 /**
 60  * Breakpoint state indicating a breakpoint pending a clear. 
 61  * @constant 
 62  */
 63 Breakpoint.prototype.PENDING_CLEAR = 4;
 64 /**
 65  * Breakpoint state indicating a clear request failed.
 66  * @constant 
 67  */
 68 Breakpoint.prototype.FAILED_CLEAR = 5;
 69 /**
 70  * Breakpoint state indicating a breakpoint has been cleared.
 71  * @constant 
 72  */
 73 Breakpoint.prototype.CLEARED = 6;

138 // Private

140 /**
141  * Implementations must call this method when a breakpoint gets installed in the browser.
142  * Notification will be sent to registered listeners that the breakpoint has been installed.
143  * Updates the installed property of this breakpoint. This method should only be called once when
144  * a breakpoint becomes installed.
145  * 
146  * @function
147  */
148 Breakpoint.prototype._installed = function()
149 {
150     if (this.state === this.PENDING_INSTALL)
151     {
152     	this.state = this.INSTALLED;
153         this.getCompilationUnit().getBrowserContext().getBrowser()._dispatch("onToggleBreakpoint", [this]);	
154     }
155 };
156 
157 /**
158  * Implementations must call this method when a breakpoint is cleared from the browser.
159  * Notification will be sent to registered listeners that the breakpoint has been cleared.
160  * This method should only be called once when a breakpoint is cleared.
161  * 
162  * @function
163  */
164 Breakpoint.prototype._cleared = function()
165 {
166     if (this.state != this.CLEARED)
167     {
168         this.state = this.CLEARED;
169         this.getCompilationUnit().getBrowserContext().getBrowser()._dispatch("onToggleBreakpoint", [this]);
170     }
171 };
172 
173 /**
174  * Implementations must call this method when a breakpoint fails to install in the browser.
175  * Notification will be sent to registered listeners that breakpoint installation has failed.
176  * Updates breakpoint state. This method should only be called once when a breakpoint fails to
177  * install.
178  * 
179  * @function
180  */
181 Breakpoint.prototype._failedInstall = function()
182 {
183     if (this.state === this.PENDING_INSTALL)
184     {
185         this.state = this.FAILED_INSTALL;
186         this.getCompilationUnit().getBrowserContext().getBrowser()._dispatch("onBreakpointError", [this]);
187     }
188 };
189 
190 /**
191  * Implementations must call this method when a breakpoint fails to clear from the browser.
192  * Notification will be sent to registered listeners that breakpoint clear has failed.
193  * Updates breakpoint state. This method should only be called once when a breakpoint fails to
194  * clear.
195  * 
196  * @function
197  */
198 Breakpoint.prototype._failedClear = function()
199 {
200     if (this.state === this.PENDING_CLEAR)
201     {
202         this.state = this.FAILED_CLEAR;
203         this.getCompilationUnit().getBrowserContext().getBrowser()._dispatch("onBreakpointError", [this]);
204     }
205 };
