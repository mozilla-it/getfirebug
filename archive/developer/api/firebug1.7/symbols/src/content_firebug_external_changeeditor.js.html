 11 function onLoad()
 12 {
 13     var args = window.arguments[0];
 14     item = args.item;
 15     FBL = args.FBL;
 16 
 17     browseButton = document.getElementById("browse-button");
 18 
 19     document.getElementById("name").value = item.label;
 20     if (item.executable)
 21     {
 22         try
 23         {
 24             var file = fbXPCOMUtils.CCIN("@mozilla.org/file/local;1", "nsILocalFile");
 25             file.initWithPath(item.executable);
 26             document.getElementById("executable").file = file;
 27         }
 28         catch(exc) {}
 29     }
 30 
 31     if (item.cmdline)
 32         document.getElementById("cmdline").value = item.cmdline;
 33 
 34     onChange();
 35 
 36     // Localization
 37     internationalizeUI(document);
 38 
 39     window.sizeToContent();
 40 
 41     if (document.getAnonymousElementByAttribute && !document.getElementById("executable").file)
 42     {
 43         setTimeout(function()
 44         {
 45             internalFilefieldTextbox = document.getAnonymousElementByAttribute(
 46                 document.getElementById("executable"), "class", "fileFieldLabel");
 47 
 48             if (internalFilefieldTextbox)
 49             {
 50                 internalFilefieldTextbox.readOnly = false;
 51                 internalFilefieldTextbox.addEventListener("input", function(e) {
 52                     browseButton.disabled = (this.value != "");
 53                     onChange();
 54                 }, false);
 55             }
 56         }, 100);
 57     }
 58 }

 60 function internationalizeUI(doc)
 61 {
 62     var elements = doc.getElementsByClassName("fbInternational");
 63     var attributes = ["title", "label", "value"];
 64     for (var i=0; i<elements.length; i++)
 65     {
 66         if (elements[i].nodeName == "description")
 67         {
 68             var localized = FBL.$STR(elements[i].textContent);
 69             var parser = fbXPCOMUtils.CCIN("@mozilla.org/xmlextras/domparser;1", "nsIDOMParser");
 70             var doc = parser.parseFromString("<vbox>" + localized + "</vbox>", "text/xml");
 71             var root = doc.documentElement;
 72 
 73             while(elements[i].firstChild)
 74                 elements[i].removeChild(elements[i].firstChild);
 75 
 76             for(var j=0; j<root.childNodes.length; j++)
 77             {
 78                 // ToDo: Show labels correctly
 79                 // Namespaces are not inherited from doc, so labels 
 80                 // are not shown as links
 81                 node = doc.importNode(root.childNodes[j], true);
 82                 elements[i].appendChild(node);
 83             }
 84         }
 85         else
 86         {
 87             for(var j=0; j<attributes.length; j++)
 88             {
 89                 if (elements[i].hasAttribute(attributes[j]))
 90                     FBL.internationalize(elements[i], attributes[j]);
 91             }
 92         }
 93     }
 94 }

 96 function onAccept()
 97 {
 98     item.label = document.getElementById("name").value;
 99     if (!browseButton.disabled)
100     {
101         var file = document.getElementById("executable").file;
102         item.executable = "";
103         if (file)
104             item.executable = file.path;
105     }
106     else
107     {
108         item.executable = internalFilefieldTextbox.value.replace(/^\s+|\s+$/g, '');
109     }
110 
111     item.cmdline = document.getElementById("cmdline").value;
112 
113     try
114     {
115         var file = fbXPCOMUtils.CCIN("@mozilla.org/file/local;1", "nsILocalFile");
116         file.initWithPath(item.executable);
117         if (!file.isExecutable())
118            throw "NotAnExecutable";
119 
120         window.arguments[1].saveChanges = true;
121         return true;
122     }
123     catch (exc)
124     {
125         const Ci = Components.interfaces;
126         const nsIPromptService = nsIPromptService;
127         var promptService = fbXPCOMUtils.CCIN("@mozilla.org/embedcomp/prompt-service;1",
128             "nsIPromptService");
129 
130         if (exc == "NotAnExecutable")
131         {
132             promptService.alert(null, FBL.$STR("changeEditor.Invalid_Application_Path"),
133                 FBL.$STR("changeEditor.Path_is_not_an_executable"));
134         }
135         else
136         {
137             promptService.alert(null, FBL.$STR("changeEditor.Invalid_Application_Path"),
138                 FBL.$STR("changeEditor.Application_does_not_exist"));
139         }
140 
141         return false;
142     }
143 }
144 
145 function onChange()
146 {
147     document.documentElement.getButton("accept").disabled = !(
148         document.getElementById("name").value && (
149             (browseButton.disabled && internalFilefieldTextbox &&
150                 internalFilefieldTextbox.value &&
151                 internalFilefieldTextbox.value.replace(/^\s+|\s+$/g, '')) ||
152             (!browseButton.disabled && document.getElementById("executable").file)
153         )
154     );
155 }
156 
157 function onBrowse()
158 {
159     const Ci = Components.interfaces;
160     const nsIFilePicker = Ci.nsIFilePicker;
161     var picker = fbXPCOMUtils.CCIN("@mozilla.org/filepicker;1", "nsIFilePicker");
162     picker.init(window, "", nsIFilePicker.modeOpen);
163     picker.appendFilters(nsIFilePicker.filterApps);
164 
165     if (picker.show() == nsIFilePicker.returnOK && picker.file)
166     {
167         var nameField = document.getElementById("name");
168         var execField = document.getElementById("executable");
169         execField.file = picker.file;
170 
171         if (internalFilefieldTextbox)
172             internalFilefieldTextbox.readOnly = true;
173 
174         if (nameField.value == "")
175             nameField.value = execField.file.leafName.replace(".exe","");
176 
177         onChange();
178         return true;
179     }
180 
181     return false;
182 }

184 function insertText(text, whole)
185 {
186     var textbox = document.getElementById("cmdline")
187     if(whole)
188         textbox.select();
189 
190     textbox.editor.QueryInterface(Components.interfaces.nsIPlaintextEditor).insertText(text);
191     textbox.focus()
192 }

196 // would be good to have autosuggest for popular editors
197 var defaultCommandLines =
198 {
199     "sublimetext": "%file:%line",
200     "notepad++":   "-n%line %file",
201     "emeditor":    "/l %line %file",
202     "browser":     "%url",
203     "explorer":    "/select,%file"
204 }
205 
206 function suggestionPopupShowing(popup)
207 {
208     FBL.eraseNode(popup);
209 
210     for (var i in defaultCommandLines)
211     {
212         var box = document.createElement('hbox');
213         var label = document.createElement('label');
214         label.setAttribute('value', i + ': ');
215         box.appendChild(label);
216 
217         var spacer = document.createElement('spacer');
218         spacer.setAttribute('flex', 1);
219         box.appendChild(spacer);
220 
221         label = document.createElement('label');
222         label.setAttribute('value', defaultCommandLines[i]);
223         label.className = 'text-link'
224         box.appendChild(label);
225 
226         popup.appendChild(box)
227     }
228 }
