 18 Firebug.ScriptPanel.reLineNumber = /^[^\\]?#(\d*)$/;
 19 /*
 20  * object used to markup Javascript source lines.
 21  * In the namespace Firebug.ScriptPanel.
 22  */
 23 Firebug.ScriptPanel.decorator = extend(new Firebug.SourceBoxDecorator,
 24 {
 25     decorate: function(sourceBox, unused)
 26     {
 27         this.markExecutableLines(sourceBox);
 28         this.setLineBreakpoints(sourceBox.repObject, sourceBox)
 29     },
 30 
 31     markExecutableLines: function(sourceBox)
 32     {
 33         var compilationUnit = sourceBox.repObject;
 34         if (FBTrace.DBG_BP || FBTrace.DBG_LINETABLE)
 35             FBTrace.sysout("script.markExecutableLines START: "+compilationUnit.toString());
 36 
 37         var lineNo = sourceBox.firstViewableLine;
 38         while( lineNode = sourceBox.getLineNode(lineNo) )
 39         {
 40             if (lineNode.alreadyMarked)
 41             {
 42                 lineNo++;
 43                 continue;
 44             }
 45 
 46             var script = compilationUnit.isExecutableLine(lineNo);
 47 
 48             if (FBTrace.DBG_LINETABLE)
 49                 FBTrace.sysout("script.markExecutableLines ["+lineNo+"]="+script);
 50 
 51             if (script)
 52                 lineNode.setAttribute("executable", "true");
 53             else
 54                 lineNode.removeAttribute("executable");
 55 
 56             lineNode.alreadyMarked = true;
 57 
 58             if (lineNo > sourceBox.lastViewableLine)
 59                 break;
 60 
 61             lineNo++;
 62         }
 63 
 64         if (FBTrace.DBG_BP || FBTrace.DBG_LINETABLE)
 65             FBTrace.sysout("script.markExecutableLines DONE: "+compilationUnit.toString()+"\n");
 66     },
 67 
 68     setLineBreakpoints: function(compilationUnit, sourceBox)
 69     {
 70         fbs.enumerateBreakpoints(compilationUnit.getURL(), {call: function(url, line, props, scripts)
 71         {
 72             var scriptRow = sourceBox.getLineNode(line);
 73             if (scriptRow)
 74             {
 75                 scriptRow.setAttribute("breakpoint", "true");
 76                 if (props.disabled)
 77                     scriptRow.setAttribute("disabledBreakpoint", "true");
 78                 if (props.condition)
 79                     scriptRow.setAttribute("condition", "true");
 80             }
 81 
 82             if (FBTrace.DBG_LINETABLE)
 83                 FBTrace.sysout("script.setLineBreakpoints found "+scriptRow+" for "+line+"@"+
 84                     compilationUnit.getURL()+"\n");
 85         }});
 86     },
 87 });

334     populateInfoTip: function(infoTip, expr)
335     {
336         if (!expr || isJavaScriptKeyword(expr))
337             return false;
338 
339         var self = this;
340         // If the evaluate fails, then we report an error and don't show the infoTip
341         Firebug.CommandLine.evaluate(expr, this.context, null, this.context.getGlobalScope(),
342             function success(result, context)
343             {
344                 var rep = Firebug.getRep(result, context);
345                 var tag = rep.shortTag ? rep.shortTag : rep.tag;
346 
347                 if (FBTrace.DBG_STACK)
348                     FBTrace.sysout("populateInfoTip result is "+result, result);
349 
350                 tag.replace({object: result}, infoTip);
351 
352                 Firebug.chrome.contextMenuObject = result;  // for context menu select()
353 
354                 self.infoTipExpr = expr;
355             },
356             function failed(result, context)
357             {
358                 self.infoTipExpr = "";
359             }
360         );
361         return (self.infoTipExpr == expr);
362     },

671     search: function(text, reverse)
672     {
673         var sourceBox = this.selectedSourceBox;
674         if (!text || !sourceBox)
675         {
676             delete this.currentSearch;
677             return false;
678         }
679 
680         // Check if the search is for a line number
681         var m = Firebug.ScriptPanel.reLineNumber.exec(text);
682         if (m)
683         {
684             if (!m[1])
685                 return true; // Don't beep if only a # has been typed
686 
687             var lineNo = parseInt(m[1]);
688             if (!isNaN(lineNo) && (lineNo > 0) && (lineNo < sourceBox.lines.length) )
689             {
690                 this.scrollToLine(sourceBox.repObject.getURL(), lineNo,  this.jumpHighlightFactory(lineNo, this.context))
691                 return true;
692             }
693         }
694 
695         var curDoc = this.searchCurrentDoc(!Firebug.searchGlobal, text, reverse);
696         if (!curDoc && Firebug.searchGlobal)
697         {
698             return this.searchOtherDocs(text, reverse);
699         }
700         return curDoc;
701     },
702 
703     searchOtherDocs: function(text, reverse)
704     {
705         var scanRE = Firebug.Search.getTestingRegex(text);
706 
707         var self = this;
708 
709         function scanDoc(compilationUnit)
710         {
711             var lines = null;
712 
713             // TODO the source lines arrive async in general
714             compilationUnit.getSourceLines(-1, -1, function loadSource(unit, firstLineNumber, lastLineNumber, linesRead)
715             {
716                 lines = linesRead;
717             });
718 
719             if (!lines)
720                 return;
721             // we don't care about reverse here as we are just looking for existence,
722             // if we do have a result we will handle the reverse logic on display
723             for (var i = 0; i < lines.length; i++) {
724                 if (scanRE.test(lines[i]))
725                 {
726                     return true;
727                 }
728             }
729         }
730 
731         if (this.navigateToNextDocument(scanDoc, reverse))
732         {
733             return this.searchCurrentDoc(true, text, reverse);
734         }
735     },
