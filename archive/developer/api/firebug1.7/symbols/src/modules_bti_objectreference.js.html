 11 /**
 12  * Describes an instance of an object in a JavaScript program - for example a number,
 13  * string or object.
 14  * 
 15  * @constructor
 16  * @param type type of object
 17  * @param id unique object identifier (a number)
 18  * @type ObjectReference
 19  * @return a new {@link ObjectReference}
 20  * @version 1.0
 21  */
 22 function ObjectReference(type, id)
 23 {
 24     this.type = type;
 25     this.id = id;
 26 }
 27 
 28 /**
 29  * Returns the unique identifier of this object as a number.
 30  * <p>
 31  * This function does not require communication with
 32  * the browser.
 33  * </p>
 34  * @function
 35  * @returns unique object identifier (number)
 36  */
 37 ObjectReference.prototype.getId = function()
 38 {
 39     return this.id;
 40 };
 41 
 42 /**
 43  * Returns the type of data this object contains as a {@link String}.
 44  * One of the following is returned:
 45  * <ul>
 46  * <li><code>object</code> - implies this object is an instance of {@link ObjectReference}</li>
 47  * <li><code>function</code> - implies this object is an instance of {@link FunctionReference}</li>
 48  * <li><code>boolean</code> - implies this object is an instance of {@link Primitive}</li>
 49  * <li><code>number</code> - implies this object is an instance of {@link Primitive}</li>
 50  * <li><code>string</code> - implies this object is an instance of {@link Primitive}</li>
 51  * <li><code>array</code> - implies this object is an instance of {@link ArrayReference}</li>
 52  * </ul>
 53  * <p>
 54  * This function does not require communication with
 55  * the browser.
 56  * </p>
 57  * @function
 58  * @returns the type of data this object contains
 59  */
 60 ObjectReference.prototype.getType = function()
 61 {
 62     return this.type;
 63 };
