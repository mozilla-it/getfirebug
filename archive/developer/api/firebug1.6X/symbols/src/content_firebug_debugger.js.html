204     stop: function(context, frame, type, rv)
205     {
206         if (context.stopped)
207             return RETURN_CONTINUE;
208 
209         if (!this.isAlwaysEnabled())
210             return RETURN_CONTINUE;
211 
212         if (FBTrace.DBG_UI_LOOP)
213             FBTrace.sysout("debugger.stop "+context.getName()+" frame",frame);
214 
215         context.stoppedFrame = frame;  // the frame we stopped in, don't change this elsewhere.
216         context.currentFrame = frame;  // the frame we show to user, depends on selection
217         context.stopped = true;
218 
219         var hookReturn = dispatch2(this.fbListeners,"onStop",[context,frame, type,rv]);
220         if ( hookReturn && hookReturn >= 0 )
221         {
222             delete context.stopped;
223             delete context.stoppedFrame;
224             delete context.currentFrame;
225             if (FBTrace.DBG_UI_LOOP)
226                 FBTrace.sysout("debugger.stop extension vetoed stop with hookReturn "+hookReturn);
227 
228             return hookReturn;
229         }
230 
231         try
232         {
233              this.freeze(context);
234             // We will pause here until resume is called
235             var depth = fbs.enterNestedEventLoop({onNest: bindFixed(this.startDebugging, this, context)});
236             // For some reason we don't always end up here
237             if (FBTrace.DBG_UI_LOOP) FBTrace.sysout("debugger.stop, depth:"+depth+" context:"+context.getName());
238         }
239         catch (exc)
240         {
241             // Just ignore exceptions that happened while in the nested loop
242             if (FBTrace.DBG_ERRORS)
243                 FBTrace.sysout("debugger exception in nested event loop: "+exc, exc);
244             else
245                 ERROR("debugger exception in nested event loop: "+exc+"\n");
246         }
247         finally
248         {
249             this.thaw(context);
250         }
251 
252         this.stopDebugging(context);
253 
254         dispatch(this.fbListeners,"onResume",[context]);
255 
256         if (context.aborted)
257         {
258             delete context.aborted;
259             return RETURN_ABORT;
260         }
261         else
262             return RETURN_CONTINUE;

337     // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
338     freeze: function(context)
339     {
340         var executionContext = context.stoppedFrame.executionContext;
341         try {
342             executionContext.scriptsEnabled = false;
343             this.suppressEventHandling(context);
344             context.isFrozen = true;
345 
346             // https://developer.mozilla.org/en/XUL_Tutorial/Focus_and_Selection#Getting_the_currently_focused_element
347             if (context.window.document.commandDispatcher)
348             {
349                 context.saveFocus = context.window.document.commandDispatcher.focusedElement;
350                 if (context.saveFocus && !context.discardBlurEvents)
351                 {
352                     context.discardBlurEvents = function blurDiscarder(event)
353                     {
354                         if (!context.saveFocus)
355                         {
356                             context.window.removeEventListener('blur', context.discardBlurEvents, true);
357                             delete context.discardBlurEvents;
358                         }
359 
360                         if (FBTrace.DBG_UI_LOOP)
361                             FBTrace.sysout("debugger.freeze discard blur event "+context.saveFocus+" while focus is "+context.window.document.commandDispatcher.focusedElement, event);
362                         event.preventDefault();
363                         event.stopPropagation();
364                     },
365 
366                     context.window.addEventListener('blur', context.discardBlurEvents, true);
367                 }
368             }
369 
370             if (FBTrace.DBG_UI_LOOP)
371             {
372                 FBTrace.sysout("debugger.freeze context.saveFocus "+context.saveFocus, context.saveFocus);
373                 FBTrace.sysout("debugger.freeze try to disable scripts "+(context.eventSuppressor?"and events":"but not events")+" in "+context.getName()+" executionContext.tag "+executionContext.tag+".scriptsEnabled: "+executionContext.scriptsEnabled);
374             }
375         }
376         catch (exc)
377         {
378             // This attribute is only valid for contexts which implement nsIScriptContext.
379             if (FBTrace.DBG_UI_LOOP) FBTrace.sysout("debugger.freeze, freeze exception "+exc+" in "+context.getName(), exc);
380         }

412                 }
413 
414                 if (FBTrace.DBG_UI_LOOP)
415                 {
416                     var nowFocused = context.window.document.commandDispatcher ? context.window.document.commandDispatcher.focusedElement : null;
417                     FBTrace.sysout("debugger.thaw context.saveFocus "+context.saveFocus+" vs "+nowFocused, context.saveFocus);
418             }
419 
420                 executionContext.scriptsEnabled = true;
421             }
422             else
423             {
424                 if (FBTrace.DBG_UI_LOOP)
425                     FBTrace.sysout("debugger.thaw "+executionContext.tag+" executionContext is not valid");
426             }
427             if (FBTrace.DBG_UI_LOOP)
428                 FBTrace.sysout("debugger.thaw try to enable scripts "+(context.eventSuppressor?"with events suppressed":"events enabled")+" in "+context.getName()+" executionContext.tag "+executionContext.tag+".scriptsEnabled: "+executionContext.scriptsEnabled);

713     startDebugging: function(context)
714     {
715         if (FBTrace.DBG_UI_LOOP) FBTrace.sysout("startDebugging enter context.stopped:"+context.stopped+" for context: "+context.getName()+"\n");
716         try {
717 
718             fbs.lockDebugger();
719 
720             context.executingSourceFile = Firebug.SourceFile.getSourceFileByScript(context, context.stoppedFrame.script);
721 
722             if (!context.executingSourceFile)  // bail out, we don't want the user stuck in debug with out source.
723             {
724                 if (FBTrace.DBG_UI_LOOP)
725                     FBTrace.sysout("startDebugging resuming, no sourceFile for "+context.stoppedFrame.script.fileName, context.stoppedFrame.script.functionSource);
726                 this.resume(context);
727                 return;
728             }
729 
730             var currentBreakable = Firebug.chrome.getGlobalAttribute("cmd_breakOnNext", "breakable");
731 
732             if (FBTrace.DBG_BP)
733                 FBTrace.sysout("debugger.startDebugging; currentBreakable "+currentBreakable+" in " + context.getName());
734 
735             if (currentBreakable == "false") // then we are armed but we broke
736                 Firebug.chrome.setGlobalAttribute("cmd_breakOnNext", "breakable", "true");
737 
738             if (context != Firebug.currentContext || Firebug.isDetached())
739                 Firebug.selectContext(context);  // Make Firebug.currentContext = context and sync the UI
740 
741             if (Firebug.isMinimized()) // then open the UI to show we are stopped
742                 Firebug.unMinimize();
743 
744             this.syncCommands(context);
745             this.syncListeners(context);
746 
747             // Update Break on Next lightning.
748             var panel = context.getPanel("script", true);
749             Firebug.Breakpoint.updatePanelTab(panel, false);
750             Firebug.chrome.syncPanel("script");  // issue 3463
751             Firebug.chrome.select(context.stoppedFrame, "script", null, true);
752             Firebug.chrome.focus();
753         }
754         catch(exc)
755         {
756             if (FBTrace.DBG_ERRORS)
757                 FBTrace.sysout("Resuming debugger: error during debugging loop: "+exc, exc);
758             Firebug.Console.log("Resuming debugger: error during debugging loop: "+exc);
759             this.resume(context);
760         }
761 
762         dispatch(this.fbListeners, "onStartDebugging", [context]);
763 
764         if (FBTrace.DBG_UI_LOOP) FBTrace.sysout("startDebugging exit context.stopped:"+context.stopped+" for context: "+context.getName()+"\n");
765     },

3836 // Recursively look for obj in container using array of visited objects
3837 function findObjectPropertyPath(containerName, container, obj, visited)
3838 {
3839     if (!container || !obj || !visited)
3840         return false;
3841 
3842     var referents = [];
3843     visited.push(container);
3844     for (var p in container)
3845     {
3846         if (container.hasOwnProperty(p))
3847         {
3848             var candidate = null;
3849             try
3850             {
3851                 candidate = container[p];
3852             }
3853             catch(exc)
3854             {
3855                 // eg sessionStorage
3856             }
3857 
3858             if (candidate === obj) // then we found a property pointing to our obj
3859             {
3860                 referents.push(new Referent(containerName, container, p, obj));
3861             }
3862             else // recurse
3863             {
3864                 var candidateType = typeof (candidate);
3865                 if (candidateType === 'object' || candidateType === 'function')
3866                 {
3867                     if (visited.indexOf(candidate) === -1)
3868                     {
3869                         var refsInChildren = findObjectPropertyPath(p, candidate, obj, visited);
3870                         if (refsInChildren.length)
3871                         {
3872                             // As we unwind the recursion we tack on layers of the path.
3873                             for (var i = 0; i < refsInChildren.length; i++)
3874                             {
3875                                 var refInChildren = refsInChildren[i];
3876                                 refInChildren.prependPath(containerName, container);
3877                                 referents.push(refInChildren);
3878                                 FBTrace.sysout(" Did prependPath with p "+p+" gave "+referents[referents.length - 1].getObjectPathExpression(), referents[referents.length - 1]);
3879 
3880                             }
3881                         }
3882                     }
3883                     //else we already looked at that object.
3884                 } // else the object has no properties
3885             }
3886         }
3887     }
3888     FBTrace.sysout(" Returning "+referents.length+ " referents", referents);
3889 
3890     return referents;
3891 }
