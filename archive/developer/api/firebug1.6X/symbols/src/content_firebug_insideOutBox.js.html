 32 /**
 33  * Creates a tree based on objects provided by a separate "view" object.
 34  *
 35  * Construction uses an "inside-out" algorithm, meaning that the view's job is first
 36  * to tell us the ancestry of each object, and secondarily its descendants.
 37  */
 38 top.InsideOutBox = function(view, box)
 39 {
 40     this.view = view;
 41     this.box = box;
 42 
 43     this.rootObject = null;
 44 
 45     this.rootObjectBox = null;
 46     this.selectedObjectBox = null;
 47     this.highlightedObjectBox = null;
 48 
 49     this.onMouseDown = bind(this.onMouseDown, this);
 50     box.addEventListener("mousedown", this.onMouseDown, false);
 51 };

151     selectObjectBox: function(objectBox, forceOpen)
152     {
153         var panel = Firebug.getElementPanel(objectBox);
154 
155         if (!panel)
156         {
157             if (FBTrace.DBG_ERRORS || FBTrace.DBG_HTML)
158                 FBTrace.sysout("selectObjectBox no panel for "+objectBox, objectBox);
159             return;
160         }
161

164         if (!isSelected)
165         {
166             removeClass(this.selectedObjectBox, "selected");
167             dispatch(panel.fbListeners, 'onObjectBoxUnselected', [this.selectedObjectBox]);
168             this.selectedObjectBox = objectBox;

170             if (objectBox)
171             {
172                 setClass(objectBox, "selected");
173 
174                 // Force it open the first time it is selected
175                 if (forceOpen)
176                     this.toggleObjectBox(objectBox, true);
177             }
178         }
179         dispatch(panel.fbListeners, 'onObjectBoxSelected', [objectBox]);

187             var parentBox = this.getParentObjectBox(objectBox);
188             var labelBox;

193                 if (labelBox)

199     expandObjectBox: function(objectBox, expandAll, event)
200     {
201         var nodeChildBox = this.getChildObjectBox(objectBox);
202         if (!nodeChildBox)
203             return;
204 
205         if (!objectBox.populated)
206         {
207             var firstChild = this.view.getChildObject(objectBox.repObject, 0);
208             this.populateChildBox(firstChild, nodeChildBox);
209         }
210 
211         var labelBox = objectBox.getElementsByClassName('nodeLabelBox').item(0);
212         if (labelBox)
213             labelBox.setAttribute('aria-expanded', 'true');
214         setClass(objectBox, "open");
215 
216         // Recursively expand all child boxes.
217         if (expandAll)
218         {
219             for (var child = nodeChildBox.firstChild; child; child = child.nextSibling)
220             {
221                 if (event)
222                 {
223                     var localName = child.repObject.localName;
224                     var localName = localName ? localName.toLowerCase() : "";
225 
226                     // Avoid expanding SCRIPT and LINK tags. Their content is not
227                     // event part of the DOM and it isn't usually what the user
228                     // wants to see when exploring HTML in the HTML panel.
229                     // The user can force expanding by pressing SHIFT key.
230                     // xxxHonza: I believe this entire logic belongs int html.js
231                     //    Refactor when implementing the breakpoint column.
232                     if ((localName == "script" || localName == "link") && !isShift(event))
233                         continue;
234                 }
235 
236                 if (hasClass(child, "containerNodeBox"))
237                     this.expandObjectBox(child, expandAll);
238             }
239         }
240     },

247         var labelBox = nodeLabel.getElementsByClassName('nodeLabelBox').item(0);
248         if (labelBox)
249             labelBox.setAttribute('aria-expanded', 'false');
250 
251         // Recursively (one level) contract all child boxes.
252         if (contractAll)
253         {
254             for (var child = nodeChildBox.firstChild; child; child = child.nextSibling)
255             {
256                 if (hasClass(child, "containerNodeBox"))
257                     this.contractObjectBox(childBoxes[i], contractAll);
258             }
259         }

286     /**
287      * Creates all of the boxes for an object, its ancestors, and siblings.
288      */
289     createObjectBox: function(object)
290     {
291         if (!object)
292             return null;
293 
294         this.rootObject = this.getRootNode(object);
295 
296         // Get or create all of the boxes for the target and its ancestors
297         var objectBox = this.createObjectBoxes(object, this.rootObject);
298 
299         if (FBTrace.DBG_HTML)
300             FBTrace.sysout("----insideOutBox.createObjectBox: createObjectBoxes(object="+formatNode(object)+", rootObject="+formatNode(this.rootObject)+") ="+formatNode(objectBox), objectBox);
301 
302         if (!objectBox)  // we found an object outside of the navigatible tree
303             return;
304         else if (object == this.rootObject)
305             return objectBox;
306         else
307             return this.populateChildBox(object, objectBox.parentNode);
308     },
309 
310     /**
311      * Creates all of the boxes for an object, its ancestors, and siblings up to a root.
312      */
313     createObjectBoxes: function(object, rootObject)
314     {
315         if (!object)
316             return null;
317 
318         if (object == rootObject)
319         {
320             if (!this.rootObjectBox || this.rootObjectBox.repObject != rootObject)
321             {
322                 if (this.rootObjectBox)
323                 {
324                     try
325                     {
326                         this.box.removeChild(this.rootObjectBox);
327                     }
328                     catch (exc)
329                     {
330                         if (FBTrace.DBG_HTML)
331                             FBTrace.sysout(" this.box.removeChild(this.rootObjectBox) FAILS "+
332                                 this.box+" must not contain "+this.rootObjectBox);
333                     }
334                 }
335 
336                 this.highlightedObjectBox = null;
337                 this.selectedObjectBox = null;
338                 this.rootObjectBox = this.view.createObjectBox(object, true);
339                 this.box.appendChild(this.rootObjectBox);
340             }
341 
342             if (FBTrace.DBG_HTML)
343             {
344                 FBTrace.sysout("insideOutBox.createObjectBoxes("+formatNode(object)+","+
345                     formatNode(rootObject)+") rootObjectBox: "+this.rootObjectBox, object);
346             }
347 
348             if ((FBTrace.DBG_HTML || FBTrace.DBG_ERRORS) && !this.rootObjectBox.parentNode)
349                 FBTrace.sysout("insideOutBox.createObjectBoxes; ERROR - null parent node. "+
350                     "object: " + formatNode(object)+", rootObjectBox: "+
351                         formatObjectBox(this.rootObjectBox), object);
352 
353             return this.rootObjectBox;
354         }
355         else
356         {
357             var parentNode = this.view.getParentObject(object);
358 
359             // recurse towards parent, eventually returning rootObjectBox
360             var parentObjectBox = this.createObjectBoxes(parentNode, rootObject);
361 
362             if (FBTrace.DBG_HTML)
363                 FBTrace.sysout("insideOutBox.createObjectBoxes createObjectBoxes("+
364                     formatNode(parentNode)+","+formatNode(rootObject)+"):parentObjectBox: "+
365                         formatObjectBox(parentObjectBox), parentObjectBox);
366 
367             if (!parentObjectBox)
368                 return null;
369 
370             // Returns an inner box (nodeChildBox) that contains list of child boxes (nodeBox).
371             var childrenBox = this.getChildObjectBox(parentObjectBox);
372 
373             if (FBTrace.DBG_HTML)
374                 FBTrace.sysout("insideOutBox.createObjectBoxes getChildObjectBox("+
375                     formatObjectBox(parentObjectBox)+")= childrenBox: "+formatObjectBox(childrenBox));
376 
377             if (!childrenBox)
378             {
379                 if (FBTrace.DBG_ERRORS)
380                     FBTrace.sysout("insideOutBox.createObjectBoxes FAILS for "+formatNode(object)+" getChildObjectBox("+
381                         formatObjectBox(parentObjectBox)+")= childrenBox: "+formatObjectBox(childrenBox));
382                 // This is where we could try to create a box for objects we cannot get to by navigation via walker or DOM nodes (native anonymous)
383                 return null;
384             }
385 
386             var childObjectBox = this.findChildObjectBox(childrenBox, object);
387 
388             if (FBTrace.DBG_HTML)
389                 FBTrace.sysout("insideOutBox.createObjectBoxes findChildObjectBox("+
390                     formatNode(childrenBox)+","+formatNode(object)+"): childObjectBox: "+
391                         formatObjectBox(childObjectBox), childObjectBox);
392 
393             return childObjectBox ? childObjectBox : this.populateChildBox(object, childrenBox);
394         }
395     },

397     findObjectBox: function(object)
398     {
399         if (!object)
400             return null;
401 
402         if (object == this.rootObject)
403             return this.rootObjectBox;
404         else
405         {
406             var parentNode = this.view.getParentObject(object);
407             var parentObjectBox = this.findObjectBox(parentNode);
408             if (!parentObjectBox)
409                 return null;
410 
411             var childrenBox = this.getChildObjectBox(parentObjectBox);
412             if (!childrenBox)
413                 return null;
414 
415             return this.findChildObjectBox(childrenBox, object);
416         }
417     },
418 
419     appendChildBox: function(parentNodeBox, repObject)
420     {
421         var childBox = this.getChildObjectBox(parentNodeBox);
422         var objectBox = this.findChildObjectBox(childBox, repObject);
423         if (objectBox)
424             return objectBox;
425 
426         objectBox = this.view.createObjectBox(repObject);
427         if (objectBox)
428         {
429             var childBox = this.getChildObjectBox(parentNodeBox);
430             childBox.appendChild(objectBox);
431         }
432         return objectBox;
433     },

435     insertChildBoxBefore: function(parentNodeBox, repObject, nextSibling)
436     {
437         var childBox = this.getChildObjectBox(parentNodeBox);
438         var objectBox = this.findChildObjectBox(childBox, repObject);
439         if (objectBox)
440             return objectBox;
441 
442         objectBox = this.view.createObjectBox(repObject);
443         if (objectBox)
444         {
445             var siblingBox = this.findChildObjectBox(childBox, nextSibling);
446             childBox.insertBefore(objectBox, siblingBox);
447         }
448         return objectBox;

451     removeChildBox: function(parentNodeBox, repObject)
452     {
453         var childBox = this.getChildObjectBox(parentNodeBox);
454         var objectBox = this.findChildObjectBox(childBox, repObject);
455         if (objectBox)
456             childBox.removeChild(objectBox);

459     populateChildBox: function(repObject, nodeChildBox)  // We want all children of the parent of repObject.
460     {
461         if (!repObject)
462             return null;
463 
464         var parentObjectBox = getAncestorByClass(nodeChildBox, "nodeBox");
465         if (FBTrace.DBG_HTML)
466             FBTrace.sysout("+++insideOutBox.populateChildBox("+getElementCSSSelector(repObject)+") parentObjectBox.populated "+parentObjectBox.populated+"\n");
467         if (parentObjectBox.populated)
468             return this.findChildObjectBox(nodeChildBox, repObject);
469 
470         var lastSiblingBox = this.getChildObjectBox(nodeChildBox);
471         var siblingBox = nodeChildBox.firstChild;
472         var targetBox = null;
473 
474         var view = this.view;
475 
476         var targetSibling = null;
477         var parentNode = view.getParentObject(repObject);
478         for (var i = 0; 1; ++i)
479         {
480             targetSibling = view.getChildObject(parentNode, i, targetSibling);
481             if (!targetSibling)
482                 break;
483 
484             // Check if we need to start appending, or continue to insert before
485             if (lastSiblingBox && lastSiblingBox.repObject == targetSibling)
486                 lastSiblingBox = null;
487 
488             if (!siblingBox || siblingBox.repObject != targetSibling)
489             {
490                 var newBox = view.createObjectBox(targetSibling);
491                 if (newBox)
492                 {
493                     if (lastSiblingBox)
494                         nodeChildBox.insertBefore(newBox, lastSiblingBox);
495                     else
496                         nodeChildBox.appendChild(newBox);
497                 }
498 
499                 siblingBox = newBox;
500             }
501 
502             if (targetSibling == repObject)
503                 targetBox = siblingBox;
504 
505             if (siblingBox && siblingBox.repObject == targetSibling)
506                 siblingBox = siblingBox.nextSibling;
507         }
508 
509         if (targetBox)
510             parentObjectBox.populated = true;
511 
512         if (FBTrace.DBG_HTML)
513             FBTrace.sysout("---insideOutBox.populateChildBox("+(repObject.localName?repObject.localName:repObject)+") targetBox "+targetBox+"\n");
514 
515         return targetBox;
516     },
517 
518     getParentObjectBox: function(objectBox)
519     {
520         var parent = objectBox.parentNode ? objectBox.parentNode.parentNode : null;
521         return parent && parent.repObject ? parent : null;
522     },
523 
524     getChildObjectBox: function(objectBox)
525     {
526         return objectBox.getElementsByClassName("nodeChildBox").item(0);
527     },
528 
529     findChildObjectBox: function(parentNodeBox, repObject)
530     {
531         for (var childBox = parentNodeBox.firstChild; childBox; childBox = childBox.nextSibling)
532         {
533             if (childBox.repObject == repObject)
534                 return childBox;
535         }
536         if (FBTrace.DBG_HTML)
537             FBTrace.sysout("insideOutBox.findChildObjectBox no match for repObject: " +(repObject && (repObject.localName || repObject))+" in "+parentNodeBox.localName);
538     },
539 
540     /**
541      * Determines if the given node is an ancestor of the current root.
542      */
543     isInExistingRoot: function(node)
544     {
545         if (FBTrace.DBG_HTML)
546             var dbg_isInExistingRoot = "";
547 
548         var parentNode = node;
549         while (parentNode && parentNode != this.rootObject)
550         {
551             if (FBTrace.DBG_HTML)
552                 dbg_isInExistingRoot = dbg_isInExistingRoot + parentNode.localName+" < ";
553 
554             parentNode = this.view.getParentObject(parentNode);
555         }
556         if (FBTrace.DBG_HTML)
557             FBTrace.sysout("insideOutBox.isInExistingRoot  "+dbg_isInExistingRoot+": (parentNode == this.rootObject)="+(parentNode == this.rootObject));
558         return parentNode == this.rootObject;
559     },
560 
561     getRootNode: function(node)
562     {
563         if (FBTrace.DBG_HTML)
564             var dbg_getRootNode ="";
565 
566         while (1)
567         {
568             if (FBTrace.DBG_HTML)
569                 dbg_getRootNode += node.localName+" < ";
570 
571             var parentNode = this.view.getParentObject(node);
572 
573             if (!parentNode)
574                 break;
575             else
576                 node = parentNode;
577         }
578         if (FBTrace.DBG_HTML)
579             FBTrace.sysout("insideOutBox.getRootNode "+dbg_getRootNode);
580 
581         return node;
582     },

606 function isVisibleTarget(node)
607 {
608     if (node.repObject && node.repObject.nodeType == Node.ELEMENT_NODE)
609     {
610         for (var parent = node.parentNode; parent; parent = parent.parentNode)
611         {
612             if (hasClass(parent, "nodeChildBox")
613                 && !hasClass(parent.parentNode, "open")
614                 && !hasClass(parent.parentNode, "highlightOpen"))
615                 return false;
616         }
617         return true;
618     }

621 function formatNode(object)
622 {
623     if (object)
624     {
625         if (!object.localName)
626         {
627             var str = object.toString();
628             if (str)
629                 return str;
630             else
631                 return "(an object with no localName or toString result)";
632         }
633         else  return getElementCSSSelector(object);
634     }
635     else
636         return "(null object)";
637 }

639 function formatObjectBox(object)
640 {
641     if (object)
642     {
643         if (object.localName)
644             return getElementCSSSelector(object);
645         return object.textContent;
646     }
647     else
648         return "(null object)";
649 }
