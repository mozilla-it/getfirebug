  6 var _FirebugCommandLine =
  7 {
  8     initFirebugCommandLine: function()
  9     {
 10         // Define console functions.
 11         var commands = ["$", "$$", "$x", "$n", "cd", "clear", "inspect", "keys",
 12             "values", "debug", "undebug", "monitor", "unmonitor", "traceCalls", "untraceCalls",
 13             "traceAll", "untraceAll", "monitorEvents", "unmonitorEvents", "profile", "profileEnd", "copy"];
 14         for (var i=0; i<commands.length; i++)
 15         {
 16             var command = commands[i];
 17 
 18             // If the method is already defined, don't override it.
 19             if (window[command])
 20                 continue;
 21 
 22             this[command] = new Function(
 23                 "return window.console.notifyFirebug(arguments, '" + command + "', 'firebugExecuteCommand');");
 24         }
 25 
 26         // Define console shortcuts
 27         var consoleShortcuts = ["dir", "dirxml"];
 28         for (var i=0; i<consoleShortcuts.length; i++)
 29         {
 30             var command = consoleShortcuts[i];
 31             this[command] = new Function("return window.console." + command + ".apply(window.console, arguments)");
 32         }
 33 
 34         // Define console variables.
 35         var props = ["$0", "$1"];
 36         for (var j=0; j<props.length; j++)
 37         {
 38             var prop = props[j];
 39             if (window[prop])
 40                 continue;
 41 
 42             this.__defineGetter__(prop, new Function(
 43                 "return window.console.notifyFirebug(arguments, '" + prop + "', 'firebugExecuteCommand');"));
 44         }
 45 
 46         this.attachCommandLine();
 47     },
 48 
 49     attachCommandLine: function()
 50     {
 51         // DBG window.dump("attachCommandLine "+window.location+"\n");
 52         if (!window.console)
 53         {
 54             // DBG     debugger;
 55             window.loadFirebugConsole();
 56         }
 57         var self = this;
 58 
 59         this._firebugEvalEvent = function _firebugEvalEvent(event)
 60         {
 61             // DBG window.dump("attachCommandLine firebugCommandLine "+window.location+"\n");
 62             var expr = document.getUserData("firebug-expr"); // see commandLine.js
 63             self.evaluate(expr);
 64             // DBG window.dump("attachCommandLine did evaluate on "+expr+"\n");
 65         }
 66 
 67         document.addEventListener("firebugCommandLine",this._firebugEvalEvent, true);
 68         document.setUserData("firebug-CommandLineAttached", "true", null);
 69         // DBG window.dump("Added listener for firebugCommandLine event "+window.location+"\n");
 70     },
 71 
 72     detachCommandLine: function()
 73     {
 74          document.removeEventListener("firebugCommandLine", this._firebugEvalEvent, true);
 75          delete window._FirebugCommandLine; // suicide!
 76          // DBG window.dump("detachCommmandLine<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<\n")
 77     },
 78 
 79     evaluate: function _firebugInjectedEvaluate(expr)
 80     {
 81         try
 82         {
 83             var result = window.eval(expr);
 84             window.console.notifyFirebug([result], "evaluated", "firebugAppendConsole");
 85         }
 86         catch(exc)
 87         {
 88             var result = exc;
 89             result.source = expr;
 90             window.console.notifyFirebug([result], "evaluateError", "firebugAppendConsole");
 91         }
 92     },
 93 };
 94 
 95 (function()
 96 {
 97     try
 98     {
 99         // DBG window.dump("_FirebugCommandLine init console is "+window.console+" in "+window.location+"\n");
100         _FirebugCommandLine.initFirebugCommandLine();
101     }
102     catch(exc)
103     {
104         var wrappedException = {
105             cause: exc,
106             message: "_FirebugCommandLine init failed in "+window.location+" because "+exc,
107             toString: function() { return this.message; }
108         };
109         throw wrappedException;
110     }
111 })();
