 16 // HTTP Request Observer implementation
 17 
 18 var FBTrace = null;
 19 
 20 /**
 21  * @service This service is intended as the only HTTP observer registered by Firebug.
 22  * All FB extensions and Firebug itself should register a listener within this
 23  * service in order to listen for http-on-modify-request, http-on-examine-response and
 24  * http-on-examine-cached-response events.
 25  *
 26  * See also: <a href="http://developer.mozilla.org/en/Setting_HTTP_request_headers">
 27  * Setting_HTTP_request_headers</a>
 28  */
 29 var httpRequestObserver =
 30 /** lends HttpRequestObserver */
 31 {
 32     preInitialize: function()
 33     {
 34         this.observers = [];
 35         this.observerCount = 0;
 36 
 37         // Get firebug-trace service for logging (the service should be already
 38         // registered at this moment).
 39         Components.utils["import"]("resource://firebug/firebug-trace-service.js");
 40         FBTrace = traceConsoleService.getTracer("extensions.firebug");
 41 
 42         // Get firebug-service to listen for suspendFirebug and resumeFirebug events.
 43         // TODO is this really the way we want to do suspendFirebug?
 44         Components.utils["import"]("resource://firebug/firebug-service.js");
 45 
 46         this.initialize(fbs);
 47     },
 48 
 49     initialize: function(fbs)
 50     {
 51         this.firebugService = fbs;
 52 
 53         this.firebugService.registerClient(FirebugClient);
 54 
 55         observerService.addObserver(this, "quit-application", false);
 56 
 57         if (FBTrace.DBG_HTTPOBSERVER)
 58             FBTrace.sysout("httpObserver.initialize OK");
 59     },
 60 
 61     shutdown: function()
 62     {
 63         this.firebugService.unregisterClient(FirebugClient);
 64 
 65         observerService.removeObserver(this, "quit-application");
 66 
 67         if (FBTrace.DBG_HTTPOBSERVER)
 68             FBTrace.sysout("httpObserver.shutdown OK");
 69     },
 70 
 71     registerObservers: function()
 72     {
 73         if (FBTrace.DBG_HTTPOBSERVER)
 74             FBTrace.sysout("httpObserver.registerObservers; wasObserving: " +
 75                 this.observerCount + " with observers "+this.observers.length, this.observers);
 76 
 77         if (this.observerCount == 0)
 78         {
 79             observerService.addObserver(this, "http-on-modify-request", false);
 80             observerService.addObserver(this, "http-on-examine-response", false);
 81             observerService.addObserver(this, "http-on-examine-cached-response", false);
 82         }
 83 
 84         this.observerCount++;
 85     },
 86 
 87     unregisterObservers: function()
 88     {
 89         if (FBTrace.DBG_HTTPOBSERVER)
 90             FBTrace.sysout("httpObserver.unregisterObservers; wasObserving: " +
 91                 this.observerCount + " with observers "+this.observers.length, this.observers);
 92 
 93         if (this.observerCount == 1)
 94         {
 95             observerService.removeObserver(this, "http-on-modify-request");
 96             observerService.removeObserver(this, "http-on-examine-response");
 97             observerService.removeObserver(this, "http-on-examine-cached-response");
 98         }
 99 
100         this.observerCount--;
101     },
102 
103     /* nsIObserve */
104     observe: function(subject, topic, data)
105     {
106         if (topic == "quit-application")
107         {
108             this.shutdown();
109             return;
110         }
111 
112         try
113         {
114             if (!(subject instanceof Ci.nsIHttpChannel))
115                 return;
116 
117             if (FBTrace.DBG_HTTPOBSERVER)
118                 FBTrace.sysout("httpObserver.observe " + (topic ? topic.toUpperCase() : topic) +
119                     ", " + safeGetName(subject));
120 
121             // Notify all registered observers.
122             if (topic == "http-on-modify-request" ||
123                 topic == "http-on-examine-response" ||
124                 topic == "http-on-examine-cached-response")
125             {
126                 this.notifyObservers(subject, topic, data);
127             }
128         }
129         catch (err)
130         {
131             if (FBTrace.DBG_ERRORS)
132                 FBTrace.sysout("httpObserver.observe EXCEPTION", err);
133         }
134     },
135 
136     /* nsIObserverService */
137     addObserver: function(observer, topic, weak)
138     {
139         if (topic != "firebug-http-event")
140             throw Cr.NS_ERROR_INVALID_ARG;
141 
142         this.observers.push(observer);
143     },
144 
145     removeObserver: function(observer, topic)
146     {
147         if (topic != "firebug-http-event")
148             throw Cr.NS_ERROR_INVALID_ARG;
149 
150         for (var i=0; i<this.observers.length; i++)
151         {
152             if (this.observers[i] == observer)
153             {
154                 this.observers.splice(i, 1);
155                 return;
156             }
157         }
158 
159         if (FBTrace.DBG_HTTPOBSERVER)
160             FBTrace.sysout("httpObserver.removeObserver FAILED (no such observer)");
161     },
162 
163     notifyObservers: function(subject, topic, data)
164     {
165         if (FBTrace.DBG_HTTPOBSERVER)
166             FBTrace.sysout("httpObserver.notifyObservers (" + this.observers.length + ") " + topic);
167 
168         for (var i=0; i<this.observers.length; i++)
169             this.observers[i].observe(subject, topic, data);
170     }
171 }

174 // Request helpers
175 
176 function safeGetName(request)
177 {
178     try
179     {
180         return request.name;
181     }
182     catch (exc)
183     {
184     }
185 
186     return null;
187 }

191 // Debugging helper.
192 function dumpStack(message)
193 {
194     dump(message + "\n");
195 
196     for (var frame = Components.stack, i = 0; frame; frame = frame.caller, i++)
197     {
198         if (i < 1)
199             continue;
200 
201         var fileName = unescape(frame.filename ? frame.filename : "");
202         var lineNumber = frame.lineNumber ? frame.lineNumber : "";
203 
204         dump(fileName + ":" + lineNumber + "\n");
205     }
206 }
