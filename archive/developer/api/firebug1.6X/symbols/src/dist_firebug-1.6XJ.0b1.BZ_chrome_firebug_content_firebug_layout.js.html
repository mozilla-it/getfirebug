180     updateSelection: function(element)
181     {
182         var view = element ? element.ownerDocument.defaultView : null;
183         if (!view)
184             return this.panelNode.innerHTML = "";
185 
186         var prev = getPreviousElement(element.previousSibling);
187         var next = getNextElement(element.nextSibling);
188 
189         var style = view.getComputedStyle(element, "");
190         var prevStyle = prev ? view.getComputedStyle(prev, "") : null;
191         var nextStyle = next ? view.getComputedStyle(next, "") : null;
192 
193         var args = getBoxFromStyles(style, element);
194 
195         args.outerLeft = args.outerRight = args.outerTop = args.outerBottom = '';
196         args.outerLeftMode = args.outerRightMode = args.outerTopMode = args.outerBottomMode = "";
197         args.zIndex = args.zIndex ? args.zIndex : "auto";
198 
199         var position = style.getPropertyCSSValue("position").cssText;
200         args.position = position;
201         args.outerLabel = '';
202         
203         if (isElementSVG(element) || isElementMathML(element) || isElementXUL(element))
204         {
205             var rect = element.getBoundingClientRect();
206             if (rect.wrappedJSObject)
207                 rect = rect.wrappedJSObject;
208 
209             args.width = Math.round(rect.width);
210             args.height = Math.round(rect.height);
211         }
212         
213         // these Modes are classes on the domplate
214         args.outerLeftMode = args.outerRightMode = args.outerTopMode
215         = args.outerBottomMode = "blankEdge";
216         
217         if (position == "absolute" || position == "fixed" || position == "relative")
218         {
219             function getStyle(style, name) { var v = style.getPropertyCSSValue(name); return (v && v.cssText) ? parseInt(v.cssText) : ' '; }
220 
221             args.outerLabel = $STR("LayoutPosition");
222             
223             args.outerLeft = getStyle(style,'left');
224             args.outerTop = getStyle(style,'top');
225             args.outerRight = getStyle(style,'right');
226             args.outerBottom = getStyle(style,'bottom');
227             
228             args.outerLeftMode = args.outerRightMode = args.outerTopMode
229                 = args.outerBottomMode = "absoluteEdge";
230         }
231         
232         var node = this.template.tag.replace(args, this.panelNode);
233         this.adjustCharWidth(this.getMaxCharWidth(args, node), this.panelNode);
234 
235         dispatch(this.fbListeners, 'onLayoutBoxCreated', [this, node, args]);
236     },

249         this.adjustBoxWidth(node, "marginLayoutBox", maxWidth);

256         this.adjustLabelWidth(node, "layoutLabelLeft", maxWidth);

330     saveEdit: function(target, value, previousValue)
331     {
332         if (!this.panel.selection.style)
333             return;
334 
335         var labelBox = getAncestorByClass(target, "layoutLabel");
336         var layoutBox = getLayoutBox(labelBox);
337 
338         var boxFrame = getBoxFrame(layoutBox);
339         var boxEdge = getBoxEdge(labelBox);
340 
341         var styleName;
342         if (boxFrame == "content" || boxFrame == "position")
343             styleName = boxEdge.toLowerCase();
344         else if (boxFrame == "border")
345             styleName = boxFrame+boxEdge+"Width";
346         else
347             styleName = boxFrame+boxEdge;
348 
349         var intValue = value ? value : 0;
350         this.panel.selection.style[styleName] = intValue + "px";
351 
352         if (Firebug.Inspector.highlightedElement == this.panel.selection)
353         {
354             var boxFrame = this.highlightedBox ? getBoxFrame(this.highlightedBox) : null;
355             Firebug.Inspector.highlightObject(this.panel.selection, this.panel.context, "boxModel", boxFrame);
356         }
357 
358         if (hasClass(target, "layoutVerticalText"))
359             target.innerHTML = getVerticalText(intValue);
360         else
361             target.innerHTML = intValue;
362 
363         if (previousValue == "0" && !!value)
364             removeClass(target.parentNode, "v0");
365         else if (!value)
366             setClass(target.parentNode, "v0");
367     },
368

379 function getLayoutBox(element)
380 {
381     var re = /([^\s]+)LayoutBox/;
382     for (var box = element; box; box = box.parentNode)
383     {
384         if (re.exec(box.className))
385             return box;
386     }
387 }
388 
389 function getBoxFrame(element)
390 {
391     var re = /([^\s]+)LayoutBox/;
392     var m = re.exec(element.className);
393     return m ? m[1] : "";
394 }
395 
396 function getBoxEdge(element)
397 {
398     var re = /layoutLabel([^\s]+)/;
399     var m = re.exec(element.className);
400     return m ? m[1] : "";
401 }
402 
403 function getVerticalText(n)
404 {
405     n = n+"";
406     var text = [];
407     for (var i = 0; i < n.length; ++i)
408         text.push(n[i]);
409     return text.join("<br>");
410 }
