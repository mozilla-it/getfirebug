379     handlePanelBarKeyPress : function (event)
380     {
381         var target = event.originalTarget;
382         var isTab = target.nodeName.toLowerCase() == "paneltab";
383         var isButton = target.nodeName.search(/(xul:)?((toolbar)?button)|(checkbox)/) != -1;
384         var isDropDownMenu = isButton && (target.getAttribute('type') == "menu" || target.id == "fbLocationList") ;
385         var siblingTab, forward, toolbar, buttons;
386         var keyCode = event.keyCode || (event.type=='keypress' ? event.charCode : null);
387         if (keyCode == KeyEvent.DOM_VK_TAB)
388             this.ensurePanelTabStops(); //TODO: need a better solution to prevent loss of panel tabstop
389         if (isTab || isButton )
390         {
391             switch (keyCode)
392             {
393                 case KeyEvent.DOM_VK_LEFT:
394                 case KeyEvent.DOM_VK_RIGHT:
395                 case KeyEvent.DOM_VK_UP:
396                 case KeyEvent.DOM_VK_DOWN:
397                     forward = event.keyCode == KeyEvent.DOM_VK_RIGHT || event.keyCode == KeyEvent.DOM_VK_DOWN;
398                     if (isTab)
399                     {
400                         //will only work as long as long as siblings only consist of paneltab elements
401                         siblingTab = target[forward ? 'nextSibling' : 'previousSibling'];
402                         if (!siblingTab)
403                             siblingTab = target.parentNode[forward ? 'firstChild' : 'lastChild'];
404                         if (siblingTab)
405                         {
406                             var panelBar = getAncestorByClass(target, 'panelBar')
407                             setTimeout(bindFixed(function()
408                             {
409                                 panelBar.selectTab(siblingTab);
410                                 this.focus(siblingTab);
411                             }, this));
412                         }
413                    }
414                    else if (isButton)
415                    {
416                        if (target.id=="fbFirebugMenu" && !forward)
417                        {
418                             cancelEvent(event);
419                             return;
420                        }
421                        toolbar = getAncestorByClass(target, 'innerToolbar');
422                        if (toolbar)
423                        {
424                            var doc = target.ownerDocument;
425                            //temporarily make all buttons in the toolbar part of the tab order,
426                            //to allow smooth, native focus advancement
427                            setClass(toolbar, 'hasTabOrder');
428                            setTimeout(bindFixed(function() // time out needed to fix this behavior in 3.6
429                            {
430                                doc.commandDispatcher[forward ? 'advanceFocus' : 'rewindFocus']();
431                                //remove the buttons from the tab order again, so that it will remain uncluttered
432                                //Very ugly hack, but it works well. This prevents focus to 'spill out' of a
433                                //toolbar when using the left and right arrow keys
434                                if (!isAncestor(doc.commandDispatcher.focusedElement, toolbar))
435                                {
436                                    //we moved focus to somewhere out of the toolbar: not good. Move it back to where it was.
437                                    doc.commandDispatcher[!forward ? 'advanceFocus' : 'rewindFocus']();
438                                }
439                                removeClass(toolbar, 'hasTabOrder');
440                            }, this));
441 
442                        }
443                         cancelEvent(event);
444                         return;
445                    }
446                 break;
447                 case KeyEvent.DOM_VK_RETURN:
448                 case KeyEvent.DOM_VK_SPACE:
449                     if (isTab && target.tabMenu)
450                         target.tabMenu.popup.showPopup(target.tabMenu, -1, -1, "popup", "bottomleft", "topleft");
451                     else if (isButton)
452                     {
453                         if (isDropDownMenu)
454                         {
455                             if (target.id == "fbLocationList")
456                                 target.showPopup();
457                             else
458                                 target.open = true;
459                             cancelEvent(event);
460                             return false;
461                         }
462                     }
463                 break;
464                 case KeyEvent.DOM_VK_F4:
465                     if (isTab && target.tabMenu)
466                         target.tabMenu.popup.showPopup(target.tabMenu, -1, -1, "popup", "bottomleft", "topleft");
467                 break;
468             }
469         }
470     },

506     ensurePanelTabStop: function(panel)
507     {
508         var panelA11y = this.getPanelA11y(panel);
509         if (!panelA11y)
510             return;
511         if (panelA11y.manageFocus)
512         {
513             var tabStop = this.getPanelTabStop(panel);
514             if (!tabStop || !this.isVisibleByStyle(tabStop) || !isVisible(tabStop))
515             {
516                 this.tabStop = null;
517                 this.findPanelTabStop(panel, 'focusRow', panelA11y.lastIsDefault);
518             }
519             else if (tabStop.getAttribute('tabindex') !== "0")
520                 tabStop.setAttribute('tabindex', "0");
521         if (tabStop)
522             this.checkModifiedState(panel, tabStop, true);
523         }
