 12 const Cc = Components.classes;

 16 const dirService = Cc["@mozilla.org/file/directory_service;1"].getService(Ci.nsIProperties);

 18 // ************************************************************************************************
 19 // Annotation service implementation
 20 
 21 var FBTrace = null;
 22 
 23 /**
 24  * @class Represents an internal Firebug annotation service. This service is used to
 25  * annotate sites with an info whether Firebug should be activated for them or not.
 26  */
 27 
 28 var annotationService =
 29 {
 30     annotations: [],
 31 
 32     setPageAnnotation: function(uri, value)
 33     {
 34         if (FBTrace.DBG_ANNOTATIONS)
 35             FBTrace.sysout("AnnotationService.setPageAnnotation; " + value + ", " + uri.spec);
 36 
 37         this.annotations[uri.spec] = value;
 38     },
 39 
 40     getPageAnnotation: function(uri)
 41     {
 42         return this.annotations[uri.spec];
 43     },
 44 
 45     pageHasAnnotation: function(uri)
 46     {
 47         return this.annotations[uri.spec] ? true : false;
 48     },
 49 
 50     removePageAnnotation: function(uri)
 51     {
 52         if (FBTrace.DBG_ANNOTATIONS)
 53             FBTrace.sysout("AnnotationService.removePageAnnotation; " + uri.spec);
 54 
 55         delete this.annotations[uri.spec];
 56     },
 57 
 58     getAnnotations: function()
 59     {
 60         return this.annotations;
 61     },
 62 
 63     clear: function()
 64     {
 65         this.annotations = [];
 66     },
 67 
 68     initialize: function()
 69     {
 70         Components.utils.import("resource://firebug/firebug-trace-service.js");
 71         FBTrace = traceConsoleService.getTracer("extensions.firebug");
 72 
 73         // Get annotation file stored within the profile directory.
 74         this.file = dirService.get("ProfD", Ci.nsIFile);
 75         this.file.append("firebug");
 76         this.file.append("annotations.json");
 77 
 78         // Load annotations.
 79         this.loadAnnotations();
 80     },
 81 
 82     // Persistence
 83     loadAnnotations: function()
 84     {
 85         try
 86         {
 87             this.clear();
 88 
 89             if (!this.file.exists())
 90             {
 91                 this.file.create(Ci.nsIFile.NORMAL_FILE_TYPE, 0666);
 92                 if (FBTrace.DBG_ANNOTATIONS)
 93                     FBTrace.sysout("AnnotationService.loadAnnotations; Annotations file created " +
 94                         this.file.path);
 95                 return;
 96             }
 97 
 98             var inputStream = Cc["@mozilla.org/network/file-input-stream;1"]
 99                 .createInstance(Ci.nsIFileInputStream);
100             var cstream = Cc["@mozilla.org/intl/converter-input-stream;1"]
101                 .createInstance(Ci.nsIConverterInputStream);
102 
103             // loadAnnotations input stream.
104             inputStream.init(this.file, 0x01 | 0x08, 0666, 0); // read, create
105             cstream.init(inputStream, "UTF-8", 0, 0);
106 
107             // Load annotations.
108             var json = "";
109             var data = {};
110             while (cstream.readString(-1, data) != 0)
111                 json += data.value;
112 
113             if (!json.length)
114                 return;
115 
116             var nativeJSON = Cc["@mozilla.org/dom/json;1"].createInstance(Ci.nsIJSON);
117             var arr = nativeJSON.decode(json);
118             if (!arr)
119                 return;
120 
121             // Convert to map for faster lookup.
122             for (var i=0; i<arr.length; i++)
123                 this.annotations[arr[i].uri] = arr[i].value;
124 
125             if (FBTrace.DBG_ANNOTATIONS)
126                 FBTrace.sysout("AnnotationService.loadAnnotations; Annotations loaded from " +
127                     this.file.path, arr);
128         }
129         catch (err)
130         {
131             if (FBTrace.DBG_ERRORS || FBTrace.DBG_ANNOTATIONS)
132                 FBTrace.sysout("AnnotationService.loadAnnotations; EXCEPTION", err);
133         }
134     },
135 
136     flush: function()
137     {
138         try
139         {
140             // Initialize output stream.
141             var outputStream = Cc["@mozilla.org/network/file-output-stream;1"]
142                 .createInstance(Ci.nsIFileOutputStream);
143             outputStream.init(this.file, 0x02 | 0x08 | 0x20, 0666, 0); // write, create, truncate
144 
145             // Convert data to JSON.
146             var arr = [];
147             for (var uri in this.annotations)
148                 arr.push({
149                     uri: uri,
150                     value: this.annotations[uri]
151                 });
152 
153             var nativeJSON = Cc["@mozilla.org/dom/json;1"].createInstance(Ci.nsIJSON);
154             var jsonString = nativeJSON.encode(arr);
155 
156             // Store annotations.
157             outputStream.write(jsonString, jsonString.length);
158             outputStream.close();
159 
160             if (FBTrace.DBG_ANNOTATIONS)
161                 FBTrace.sysout("AnnotationService.loadAnnotations; Annotations stored to " +
162                     this.file.path, jsonString);
163         }
164         catch (err)
165         {
166             if (FBTrace.DBG_ERRORS || FBTrace.DBG_ANNOTATIONS)
167                 FBTrace.sysout("AnnotationService.flush; EXCEPTION", err);
168         }
169     },
170 
171     /* nsISupports */
172     QueryInterface: function(iid)
173     {
174         if (iid.equals(Ci.nsISupports))
175             return this;
176 
177         throw Cr.NS_ERROR_NO_INTERFACE;
178     }
179 };
