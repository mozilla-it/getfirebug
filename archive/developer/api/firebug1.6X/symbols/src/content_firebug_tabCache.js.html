 67 /**
 68  * Implementation of cache model. The only purpose of this object is to register an HTTP
 69  * observer so, HTTP communication can be interecepted and all incoming data stored within
 70  * a cache.
 71  */
 72 Firebug.TabCacheModel = extend(Firebug.Module,
 73 {
 74     dispatchName: "tabCache",
 75     contentTypes: contentTypes,
 76     fbListeners: [],
 77 
 78     initializeUI: function(owner)
 79     {
 80         if (FBTrace.DBG_CACHE)
 81             FBTrace.sysout("tabCache.initializeUI; Cache model initialized.");
 82 
 83         // Read maximum size limit for cached response from preferences.
 84         responseSizeLimit = Firebug.getPref(Firebug.prefDomain, "cache.responseLimit");
 85 
 86         // Read additional text mime-types from preferences.
 87         var mimeTypes = Firebug.getPref(Firebug.prefDomain, "cache.mimeTypes");
 88         if (mimeTypes)
 89         {
 90             var list = mimeTypes.split(" ");
 91             for (var i=0; i<list.length; i++)
 92                 contentTypes[list[i]] = 1;
 93         }
 94 
 95         // Register for HTTP events.
 96         if (Ci.nsITraceableChannel)
 97             httpObserver.addObserver(this, "firebug-http-event", false);
 98     },
 99 
100     shutdown: function()
101     {
102         if (FBTrace.DBG_CACHE)
103             FBTrace.sysout("tabCache.shutdown; Cache model destroyed.");
104 
105         if (Ci.nsITraceableChannel)
106             httpObserver.removeObserver(this, "firebug-http-event");
107     },
108 
109     initContext: function(context)
110     {
111         if (FBTrace.DBG_CACHE)
112             FBTrace.sysout("tabCache.initContext for: " + context.getName());
113     },
114 
115     /* nsIObserver */
116     observe: function(subject, topic, data)
117     {
118         try
119         {
120             if (!(subject instanceof Ci.nsIHttpChannel))
121                 return;
122 // XXXjjb this same code is in net.js, better to have it only once
123             var win = getWindowForRequest(subject);
124             if (win)
125                 var tabId = Firebug.getTabIdForWindow(win); // TODO remove, the tabId is not used after all
126             if (!tabId)
127                 return;
128 
129             if (topic == "http-on-modify-request")
130                 this.onModifyRequest(subject, win, tabId);
131             else if (topic == "http-on-examine-response")
132                 this.onExamineResponse(subject, win, tabId);
133             else if (topic == "http-on-examine-cached-response")
134                 this.onCachedResponse(subject, win, tabId);
135         }
136         catch (err)
137         {
138             if (FBTrace.DBG_ERRORS)
139                 FBTrace.sysout("tabCache.observe EXCEPTION", err);
140         }
141     },
142 
143     onModifyRequest: function(request, win, tabId)
144     {
145     },
146 
147     onExamineResponse: function(request, win, tabId)
148     {
149         this.registerStreamListener(request, win);
150     },
151 
152     onCachedResponse: function(request, win, tabId)
153     {
154         this.registerStreamListener(request, win);
155     },

157     registerStreamListener: function(request, win)
158     {
159         try
160         {
161             if (FBTrace.DBG_CACHE)
162                 FBTrace.sysout("tabCache.registerStreamListener; " + safeGetRequestName(request));
163 
164             // Due to #489317, the content type must be checked in onStartRequest
165             //if (!this.shouldCacheRequest(request))
166             //    return;
167 
168             request.QueryInterface(Ci.nsITraceableChannel);
169 
170             // Create Firebug's stream listener that is tracing HTTP responses.
171             Components.utils.import("resource://firebug/firebug-channel-listener.js");
172             var newListener = new ChannelListener();  // TODO pass args rather than set properties
173             newListener.window = win;
174 
175             // Set proxy listener for back notification from XPCOM to chrome (using real interface
176             // so nsIRequest object is properly passed from XPCOM scope).
177             newListener.QueryInterface(Ci.nsITraceableChannel);
178             newListener.setNewListener(new ChannelListenerProxy(win));
179 
180             // Add tee listener into the chain of request stream listeners so, the chain
181             // doesn't include a JS code. This way all exceptions are propertly distributed
182             // (#515051).
183             // The nsIStreamListenerTee differes in following branches:
184             // 1.9.1: not possible to registere nsIRequestObserver with tee
185             // 1.9.2: implements nsIStreamListenerTee_1_9_2 with initWithObserver methods
186             // 1.9.3: adds third parameter to the existing init method.
187             if (versionChecker.compare(appInfo.version, "3.6*") >= 0)
188             {
189                 var tee = CCIN("@mozilla.org/network/stream-listener-tee;1", "nsIStreamListenerTee");
190                 tee = tee.QueryInterface(Ci.nsIStreamListenerTee);
191 
192                 if (Ci.nsIStreamListenerTee_1_9_2)
193                     tee = tee.QueryInterface(Ci.nsIStreamListenerTee_1_9_2);
194 
195                 // The response will be written into the outputStream of this pipe.
196                 // Both ends of the pipe must be blocking.
197                 var sink = CCIN("@mozilla.org/pipe;1", "nsIPipe");
198                 sink.init(false, false, 0x20000, 0x4000, null);
199 
200                 // Remember the input stream, so it isn't released by GC.
201                 // See issue 2788 for more details.
202                 newListener.inputStream = sink.inputStream;
203 
204                 var originalListener = request.setNewListener(tee);
205                 newListener.sink = sink;
206 
207                 if (tee.initWithObserver)
208                     tee.initWithObserver(originalListener, sink.outputStream, newListener);
209                 else
210                     tee.init(originalListener, sink.outputStream, newListener);
211             }
212             else
213             {
214                 newListener.listener = request.setNewListener(newListener);
215             }
216 
217             // xxxHonza: this is a workaround for #489317. Just passing
218             // shouldCacheRequest method to the component so, onStartRequest
219             // can decide whether to cache or not.
220             newListener.shouldCacheRequest = function(request)
221             {
222                 try {
223                     return Firebug.TabCacheModel.shouldCacheRequest(request)
224                 } catch (err) {}
225                 return false;
226             }
227 
228             // xxxHonza: this is a workaround for the tracing-listener to get the
229             // right context. Notice that if the window (parent browser) is closed
230             // during the response download the TabWatcher (used within this method)
231             // is undefined. But in such a case no cache is needed anyway.
232             // Another thing is that the context isn't available now, but will be
233             // as soon as this method is used from the stream listener.
234             newListener.getContext = function(win)
235             {
236                 try {
237                     return TabWatcher.getContextByWindow(win);
238                 } catch (err){}
239                 return null;
240             }
241         }
242         catch (err)
243         {
244             if (FBTrace.DBG_ERRORS)
245                 FBTrace.sysout("tabCache: Register Traceable Listener EXCEPTION", err);
246         }
247     },
248 
249     shouldCacheRequest: function(request)
250     {
251         request.QueryInterface(Ci.nsIHttpChannel);
252 
253         // Allow to customize caching rules.
254         if (dispatch2(this.fbListeners, "shouldCacheRequest", [request]))
255             return true;
256 
257         // Cache only text responses for now.
258         var contentType = request.contentType;
259         if (contentType)
260             contentType = contentType.split(";")[0];
261 
262         contentType = trim(contentType);
263         if (contentTypes[contentType])
264             return true;
265 
266         // Hack to work around application/octet-stream for js files (2063).
267         // Let's cache all files with js extensions.
268         var extension = getFileExtension(safeGetName(request));
269         if (extension == "js")
270             return true;
271 
272         if (FBTrace.DBG_CACHE)
273             FBTrace.sysout("tabCache.shouldCacheRequest; Request not cached: " +
274                 request.contentType + ", " + safeGetName(request));
275 
276         return false;
277     },

282 /**
283  * This cache object is intended to cache all responses made by a specific tab.
284  * The implementation is based on nsITraceableChannel interface introduced in
285  * Firefox 3.0.4. This interface allows to intercept all incoming HTTP data.
286  *
287  * This object replaces the SourceCache, which still exist only for backward
288  * compatibility.
289  *
290  * The object is derived from SourceCache so, the same interface and most of the
291  * implementation is used.
292  */
293 Firebug.TabCache = function(context)
294 {
295     if (FBTrace.DBG_CACHE)
296         FBTrace.sysout("tabCache.TabCache Created for: " + context.getName());
297 
298     Firebug.SourceCache.call(this, context);
299 };

301 Firebug.TabCache.prototype = extend(Firebug.SourceCache.prototype,
302 {
303     responses: [],       // responses in progress.
304 
305     storePartialResponse: function(request, responseText, win)
306     {
307         if (FBTrace.DBG_CACHE)
308             FBTrace.sysout("tabCache.storePartialResponse " + safeGetName(request),
309                 request.contentCharset);
310 
311         try
312         {
313             responseText = FBL.convertToUnicode(responseText, win.document.characterSet);
314         }
315         catch (err)
316         {
317             if (FBTrace.DBG_ERRORS || FBTrace.DBG_CACHE)
318                 FBTrace.sysout("tabCache.storePartialResponse EXCEPTION " +
319                     safeGetName(request), err);
320 
321             // Even responses that are not converted are stored into the cache.
322             // return false;
323         }
324 
325         var url = safeGetName(request);
326         var response = this.getResponse(request);
327 
328         // Size of each response is limited.
329         var limitNotReached = true;
330         if (response.size + responseText.length >= responseSizeLimit)
331         {
332             limitNotReached = false;
333             responseText = responseText.substr(0, responseSizeLimit - response.size);
334             FBTrace.sysout("tabCache.storePartialResponse Max size limit reached for: " + url);
335         }
336 
337         response.size += responseText.length;
338 
339         // Store partial content into the cache.
340         this.store(url, responseText);
341 
342         // Return false if furhter parts of this response should be ignored.
343         return limitNotReached;
344     },
345 
346     getResponse: function(request)
347     {
348         var url = safeGetName(request);
349         var response = this.responses[url];
350         if (!response)
351         {
352             this.invalidate(url);
353             this.responses[url] = response = {
354                 request: request,
355                 size: 0
356             };
357         }
358 
359         return response;
360     },
361 
362     storeSplitLines: function(url, lines)
363     {
364         if (FBTrace.DBG_CACHE)
365             FBTrace.sysout("tabCache.storeSplitLines: " + url, lines);
366 
367         var currLines = this.cache[url];
368         if (!currLines)
369             currLines = this.cache[url] = [];
370 
371         // Join the last line with the new first one so, the source code
372         // lines are properly formatted...
373         if (currLines.length && lines.length)
374         {
375             // ... but only if the last line isn't already completed.
376             var lastLine = currLines[currLines.length-1];
377             if (lastLine && lastLine.search(/\r|\n/) == -1)
378                 currLines[currLines.length-1] += lines.shift();
379         }
380 
381         // Append new lines (if any) into the array for specified url.
382         if (lines.length)
383             this.cache[url] = currLines.concat(lines);
384 
385         return this.cache[url];
386     },
387 
388     loadFromCache: function(url, method, file)
389     {
390         // The ancestor implementation (SourceCache) uses ioService.newChannel, which
391         // can result in additional request to the server (in case the response can't
392         // be loaded from the Firefox cache) - known as double-load problem.
393         // This new implementation (TabCache) uses nsITraceableChannel so, all responses
394         // should be already cached.
395 
396         // xxxHonza: TODO entire implementation of this method should be removed in Firebug 1.5
397         // xxxHonza: let's try to get the response from the cache till #449198 is fixed.
398         var stream;
399         var responseText;
400         try
401         {
402             if (!url)
403                 return responseText;
404 
405             var channel = ioService.newChannel(url, null, null);
406 
407             // These flag combination doesn't repost the request.
408             channel.loadFlags = Ci.nsIRequest.LOAD_FROM_CACHE |
409                 Ci.nsICachingChannel.LOAD_ONLY_FROM_CACHE |
410                 Ci.nsICachingChannel.LOAD_BYPASS_LOCAL_CACHE_IF_BUSY;
411 
412             var charset = "UTF-8";
413             var doc = this.context.window.document;
414             if (doc)
415                 charset = doc.characterSet;
416 
417             stream = channel.open();
418 
419             // The response doesn't have to be in the browser cache.
420             if (!stream.available())
421             {
422                 if (FBTrace.DBG_CACHE)
423                     FBTrace.sysout("tabCache.loadFromCache; Failed to load source for: " + url);
424 
425                 stream.close();
426                 return [$STR("message.Failed to load source for") + ": " + url];
427             }
428 
429             // Don't load responses that shouldn't be cached.
430             if (!Firebug.TabCacheModel.shouldCacheRequest(channel))
431             {
432                 if (FBTrace.DBG_CACHE)
433                     FBTrace.sysout("tabCache.loadFromCache; The resource from this URL is not text: " + url);
434 
435                 stream.close();
436                 return [$STR("message.The resource from this URL is not text") + ": " + url];
437             }
438 
439             responseText = readFromStream(stream, charset);
440 
441             if (FBTrace.DBG_CACHE)
442                 FBTrace.sysout("tabCache.loadFromCache (response coming from FF Cache) " +
443                     url, responseText);
444 
445             responseText = this.store(url, responseText);
446         }
447         catch (err)
448         {
449             if (FBTrace.DBG_ERRORS || FBTrace.DBG_CACHE)
450                 FBTrace.sysout("tabCache.loadFromCache EXCEPTION on url \'" + url +"\'", err);
451         }
452         finally
453         {
454             if (stream)
455                 stream.close();
456         }
457 
458         return responseText;
459     },
460 
461     // nsIStreamListener - callbacks from channel stream listener component.
462     onStartRequest: function(request, requestContext)
463     {
464         if (FBTrace.DBG_CACHE)
465             FBTrace.sysout("tabCache.channel.startRequest: " + safeGetName(request));
466 
467         // Make sure the response-entry (used to count total response size) is properly
468         // initialized (cleared) now. If no data is received, the response entry remains empty.
469         var response = this.getResponse(request);
470 
471         dispatch(Firebug.TabCacheModel.fbListeners, "onStartRequest", [this.context, request]);
472         dispatch(this.fbListeners, "onStartRequest", [this.context, request]);
473     },
474 
475     onDataAvailable: function(request, requestContext, inputStream, offset, count)
476     {
477         if (FBTrace.DBG_CACHE)
478             FBTrace.sysout("tabCache.channel.onDataAvailable: " + safeGetName(request));
479 
480         // If the stream is read a new one must be provided (the stream doesn't implement
481         // nsISeekableStream).
482         var stream = {
483             value: inputStream
484         };
485 
486         dispatch(Firebug.TabCacheModel.fbListeners, "onDataAvailable",
487             [this.context, request, requestContext, stream, offset, count]);
488         dispatch(this.fbListeners, "onDataAvailable", [this.context,
489             request, requestContext, stream, offset, count]);
490 
491         return stream.value;
492     },
493 
494     onStopRequest: function(request, requestContext, statusCode)
495     {
496         // The response is finally received so, remove the request from the list of
497         // current responses.
498         var url = safeGetName(request);
499         delete this.responses[url];
500 
501         var lines = this.cache[this.removeAnchor(url)];
502         var responseText = lines ? lines.join("") : "";
503 
504         if (FBTrace.DBG_CACHE)
505             FBTrace.sysout("tabCache.channel.stopRequest: " + safeGetRequestName(request), responseText);
506 
507         dispatch(Firebug.TabCacheModel.fbListeners, "onStopRequest", [this.context, request, responseText]);
508         dispatch(this.fbListeners, "onStopRequest", [this.context, request, responseText]);
509     }
510 });

521 ChannelListenerProxy.prototype =
522 {
523     /* nsIStreamListener */
524     onStartRequest: function(request, requestContext)
525     {
526         var context = this.getContext();
527         if (context)
528             context.sourceCache.onStartRequest(request, requestContext);

533         var context = this.getContext();

538     },

541     {
542         var context = this.getContext();
543         if (context)
544             context.sourceCache.onStopRequest(request, requestContext, statusCode);

547     /* nsISupports */
548     QueryInterface: function(iid)
549     {
550         if (iid.equals(Ci.nsIStreamListener) ||
551             iid.equals(Ci.nsISupportsWeakReference) ||
552             iid.equals(Ci.nsISupports))
553         {
554             return this;
555         }
556 
557         throw Components.results.NS_NOINTERFACE;
558     },
