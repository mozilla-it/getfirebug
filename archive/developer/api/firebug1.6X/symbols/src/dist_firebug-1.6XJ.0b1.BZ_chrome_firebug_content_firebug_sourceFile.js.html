 13 /*
 14  * SourceFile one for every compilation unit.
 15  * Unique URL for each. (href)
 16  * Unique outerScript, the statements outside of any function defintion
 17  * sourceCache keyed by href has source for this compilation unit
 18  * Stored by href in context.
 19  * Contains array of jsdIScript for functions (scripts) defined in this unit
 20  * May contain line table (for sources viewed)

 35     toString: function()
 36     {
 37         var str = (this.compilation_unit_type?this.compilation_unit_type+" ":"")+this.href+" script.tags( ";
 38         if (this.outerScript)
 39             str += (this.outerScript.isValid?this.outerScript.tag:"X") +"| ";
 40         if (this.innerScripts)
 41         {
 42             var numberInvalid = 0;
 43             for (var p in this.innerScripts)
 44             {
 45                 var script = this.innerScripts[p];
 46                 if (script.isValid)
 47                     str += p+" ";
 48                 else
 49                     numberInvalid++;
 50             }
 51         }
 52         str += ")"+(numberInvalid ? "("+numberInvalid+" invalid)" : "");
 53         return str;

 93      addToLineTable: function(script)
 94      {
 95          if (!script || !script.isValid)
 96          {
 97              if (FBTrace.DBG_ERRORS)
 98                  FBTrace.sysout("addToLineTable got invalid script "+(script?script.tag:"null")+"\n");
 99              return;
100          }
101 
102          // For outer scripts, a better algorithm would loop over PC, use pcToLine to mark the lines.
103          // This assumes there are fewer PCs in an outer script than lines, probably true for large systems.
104          // And now addToLineTable is only used for outerScripts (eval and top-level).
105          // But since we can't know the range of PC values we cannot use that approach.
106 
107          if (!this.outerScriptLineMap)
108              this.outerScriptLineMap = [];
109 
110          var lineCount = script.lineExtent + 1;
111          var offset = this.getBaseLineOffset();
112          if (FBTrace.DBG_LINETABLE)
113          {
114              FBTrace.sysout("lib.SourceFile.addToLineTable script.tag:"+script.tag+" lineExtent="+lineCount+" baseLineNumber="+script.baseLineNumber+" offset="+offset+" for "+this.compilation_unit_type+"\n");
115              var startTime = new Date().getTime();
116          }
117          if (lineCount > 100)
118              lineCount = 100; // isLineExecutable requires about 1ms per line, so it can only be called for toy programs
119 
120          for (var i = 0; i <= lineCount; i++)
121          {
122              var scriptLineNo = i + script.baseLineNumber;  // the max is (i + script.baseLineNumber + script.lineExtent)
123              var mapLineNo = scriptLineNo - offset;
124              try
125              {
126                  if (script.isLineExecutable(scriptLineNo, this.pcmap_type))
127                      this.outerScriptLineMap.push(mapLineNo);
128              }
129              catch (e)
130              {
131                  // I guess not...
132              }
133 
134              if (FBTrace.DBG_LINETABLE)
135              {
136                  var pcFromLine = script.lineToPc(scriptLineNo, this.pcmap_type);
137                  var lineFromPC = script.pcToLine(pcFromLine, this.pcmap_type);
138                  if (this.outerScriptLineMap.indexOf(mapLineNo) != -1)
139                      FBTrace.sysout("lib.SourceFile.addToLineTable ["+mapLineNo+"]="+script.tag+" for scriptLineNo="+scriptLineNo+" vs "+lineFromPC+"=lineFromPC; lineToPc="+pcFromLine+" with map="+(this.pcmap_type==PCMAP_PRETTYPRINT?"PP":"SOURCE")+"\n");
140                  else
141                      FBTrace.sysout("lib.SourceFile.addToLineTable not executable scriptLineNo="+scriptLineNo+" vs "+lineFromPC+"=lineFromPC; lineToPc="+pcFromLine+"\n");
142              }
143          }
144          if (FBTrace.DBG_LINETABLE)
145          {
146              var endTime = new Date().getTime();
147              var delta = endTime - startTime ;
148              if (delta > 0) FBTrace.sysout("SourceFile.addToLineTable processed "+lineCount+" lines in "+delta+" millisecs "+Math.round(lineCount/delta)+" lines per millisecond\n");
149              FBTrace.sysout("SourceFile.addToLineTable: "+this.toString()+"\n");
150          }
151      },

155          // This code is not called; it crashes FF3pre https://bugzilla.mozilla.org/show_bug.cgi?id=430205
156          if (!this.outerScriptLineMap)
157              this.outerScriptLineMap = {};

161          if (FBTrace.DBG_LINETABLE)

163              FBTrace.sysout("lib.SourceFile.addToLineTableByPCLoop script.tag:"+script.tag+" lineCount="+lineCount+" offset="+offset+" for "+this.compilation_unit_type+"\n");

165          }
166 
167          for (var i = 0; i <= 10*lineCount; i++)
168          {
169              var lineFromPC = script.pcToLine(i, this.pcmap_type);

171              this.outerScriptLineMap[lineFromPC] = script;

173          }
174 
175          if (FBTrace.DBG_LINETABLE)
176          {
177              FBTrace.sysout("SourceFile.addToLineTableByPCLoop: "+this.toString()+"\n");
178              var endTime = new Date().getTime();
179              var delta = endTime - startTime ;
180              if (delta > 0) FBTrace.sysout("SourceFileaddToLineTableByPCLoop processed "+lineCount+" lines in "+delta+" millisecs "+Math.round(lineCount/delta)+" lines per millisecond\n");
181          }

184      hasScriptAtLineNumber: function(lineNo, mustBeExecutableLine)
185      {
186          var offset = this.getBaseLineOffset();
187 
188          if (!this.innerScripts)
189              return; // eg URLOnly
190 
191          var targetLineNo = lineNo + offset;  // lineNo is user-viewed number, targetLineNo is jsd number
192 
193          var scripts = [];
194          for (var p in this.innerScripts)
195          {
196              var script = this.innerScripts[p];
197              if (mustBeExecutableLine && !script.isValid)
198                 continue;
199 
200              this.addScriptAtLineNumber(scripts, script, targetLineNo, mustBeExecutableLine, offset);
201 
202              if (scripts.length)
203                 return true;
204          }
205 
206          if (this.outerScript && !(mustBeExecutableLine && !this.outerScript.isValid) )
207              this.addScriptAtLineNumber(scripts, this.outerScript, targetLineNo, mustBeExecutableLine, offset);
208 
209          return (scripts.length > 0);
210      },

212      getScriptsAtLineNumber: function(lineNo, mustBeExecutableLine)
213      {
214          var offset = this.getBaseLineOffset();
215 
216          if (!this.innerScripts)
217              return; // eg URLOnly
218 
219          var targetLineNo = lineNo + offset;  // lineNo is user-viewed number, targetLineNo is jsd number
220 
221          var scripts = [];
222          for (var p in this.innerScripts)
223          {
224              var script = this.innerScripts[p];
225              if (mustBeExecutableLine && !script.isValid) continue;
226              this.addScriptAtLineNumber(scripts, script, targetLineNo, mustBeExecutableLine, offset);
227          }
228 
229          if (this.outerScript && !(mustBeExecutableLine && !this.outerScript.isValid) )
230              this.addScriptAtLineNumber(scripts, this.outerScript, targetLineNo, mustBeExecutableLine, offset);
231 
232          if (FBTrace.DBG_LINETABLE)
233          {
234              if (scripts.length < 1)
235              {
236                  FBTrace.sysout("lib.getScriptsAtLineNumber no targetScript at "+lineNo," for sourceFile:"+this.toString());
237                  return false;
238              }
239              else
240              {
241                  FBTrace.sysout("getScriptsAtLineNumber offset "+offset+" for sourcefile: "+this.toString()+"\n");
242              }
243          }
244 
245          return (scripts.length > 0) ? scripts : false;
246      },
247 
248      addScriptAtLineNumber: function(scripts, script, targetLineNo, mustBeExecutableLine, offset)
249      {
250          // script.isValid will be true.
251          if (FBTrace.DBG_LINETABLE)
252              FBTrace.sysout("addScriptAtLineNumber trying "+script.tag+", is "+script.baseLineNumber+" <= "+targetLineNo +" <= "+ (script.baseLineNumber + script.lineExtent)+"? using offset = "+offset+"\n");
253 
254          if (targetLineNo >= script.baseLineNumber)
255          {
256              if ( (script.baseLineNumber + script.lineExtent) >= targetLineNo)
257              {
258                  if (mustBeExecutableLine)
259                  {
260                      try
261                      {
262                          if (!script.isLineExecutable(targetLineNo, this.pcmap_type) )
263                          {
264                              if (FBTrace.DBG_LINETABLE)
265                                  FBTrace.sysout("getScriptsAtLineNumber tried "+script.tag+", not executable at targetLineNo:"+targetLineNo+" pcmap:"+this.pcmap_type+"\n");
266                              return;
267                          }
268                      }
269                      catch (e)
270                      {
271                          // Component returned failure code: 0x80040111 (NS_ERROR_NOT_AVAILABLE) [jsdIScript.isLineExecutable]
272                          return;
273                      }
274                  }
275                  scripts.push(script);
276                  if (FBTrace.DBG_LINETABLE)
277                  {
278                      var checkExecutable = "";
279                      if (mustBeExecutableLine)
280                          var checkExecutable = " isLineExecutable: "+script.isLineExecutable(targetLineNo, this.pcmap_type)+"@pc:"+script.lineToPc(targetLineNo, this.pcmap_type);
281                      FBTrace.sysout("getScriptsAtLineNumber found "+script.tag+", isValid: "+script.isValid+" targetLineNo:"+targetLineNo+checkExecutable+"\n");
282                  }
283              }
284          }
285      },

356 Firebug.SourceFile.summarizeSourceLineArray = function(sourceLines, size)
357 {
358     var buf  = "";
359     for (var i = 0; i < sourceLines.length; i++)
360      {
361          var aLine = sourceLines[i].substr(0,240);  // avoid huge lines
362          buf += aLine.replace(/\s/, " ", "g");
363          if (buf.length > size || aLine.length > 240)
364              break;
365      }
366      return buf.substr(0, size);
367 };

375 Firebug.SourceFile.NestedScriptAnalyzer.prototype =
376 {
377     // Adjust JSD line numbers based on origin of script
378     getSourceLineFromFrame: function(context, frame)
379     {
380         if (FBTrace.DBG_SOURCEFILES) FBTrace.sysout("NestedScriptAnalyzer in "+this.sourceFile.compilation_unit_type+": frame.line  - this.sourceFile.getBaseLineOffset()",
381              frame.line +" - "+this.sourceFile.getBaseLineOffset());

383         return frame.line - (this.sourceFile.getBaseLineOffset());
384     },
385     // Interpret frame to give fn(args)
386     getFunctionDescription: function(script, context, frame)
387     {
388         if (frame)
389         {
390             var name = frame.name;

392         }
393         else
394         {
395             var name = script.functionName;
396             var args = [];
397         }

402         }
403 
404         return {name: name, args: args};
405     },

409     {
410         var line = this.getBaseLineNumberByScript(script);
411         return new FBL.SourceLink(this.sourceFile.href, line, "js");
412     },

420 Firebug.SourceFile.addScriptsToSourceFile = function(sourceFile, outerScript, innerScripts)
421 {
422     // Attach the innerScripts for use later
423     if (!sourceFile.innerScripts)
424          sourceFile.innerScripts = {};
425 
426      var total = 0;
427      while (innerScripts.hasMoreElements())
428      {
429          var script = innerScripts.getNext();
430          if (!script || ( (script instanceof Ci.jsdIScript) && !script.tag) )
431          {
432              if (FBTrace.DBG_SOURCEFILES)
433                  FBTrace.sysout("addScriptsToSourceFile innerScripts.getNext FAILS "+sourceFile, script);
434              continue;
435          }
436          sourceFile.innerScripts[script.tag] = script;
437          if (FBTrace.DBG_SOURCEFILES)
438              total++;
439      }
440      if (FBTrace.DBG_SOURCEFILES)
441          FBTrace.sysout("addScriptsToSourceFile "+ total +" scripts, sourcefile="+sourceFile.toString(), sourceFile);
442 }

717 Firebug.SourceFile.getSourceFileByScript = function(context, script)
718 {
719     if (!context.sourceFileMap)
720          return null;
721 
722     // Other algorithms are possible:
723     //   We could store an index, context.sourceFileByTag
724     //   Or we could build a tree keyed by url, with SpiderMonkey script.fileNames at the top and our urls below
725     var lucky = context.sourceFileMap[script.fileName];  // we won't be lucky for file:/ urls, no normalizeURL applied
726     if (FBTrace.DBG_SOURCEFILES && lucky)
727         FBTrace.sysout("getSourceFileByScript trying to be lucky for "+
728             script.tag + " in "+lucky, script);
729 
730     if (lucky && lucky.hasScript(script))
731         return lucky;
732 
733     if (FBTrace.DBG_SOURCEFILES)
734         FBTrace.sysout("getSourceFileByScript looking for "+script.tag+"@"+script.fileName+" in "+
735             context.getName()+": ", context.sourceFileMap);
736 
737     for (var url in context.sourceFileMap)
738     {
739         var sourceFile = context.sourceFileMap[url];
740         if (sourceFile.hasScript(script))
741             return sourceFile;
742     }
743 };

745 Firebug.SourceFile.getScriptAnalyzer = function(context, script)
746 {
747     var sourceFile = Firebug.SourceFile.getSourceFileByScript(context, script);
748     if (FBTrace.DBG_STACK)
749          FBTrace.sysout("getScriptAnalyzer "+ (sourceFile?"finds sourceFile: ":"FAILS to find sourceFile"), sourceFile);
750      if (sourceFile)
751      {
752          var analyzer = sourceFile.getScriptAnalyzer(script);
753          if (FBTrace.DBG_STACK)
754              FBTrace.sysout("getScriptAnalyzer finds analyzer: ", analyzer);
755 
756          return analyzer;
757      }
758      return undefined;
759 };
