  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 
  7 // List of JSON content types.
  8 var contentTypes =
  9 {
 10     "text/plain": 1,
 11     "text/javascript": 1,
 12     "text/x-javascript": 1,
 13     "text/json": 1,
 14     "text/x-json": 1,
 15     "application/json": 1,
 16     "application/x-json": 1,
 17     "application/javascript": 1,
 18     "application/x-javascript": 1,
 19     "application/json-rpc": 1
 20 };
 21 
 22 // ************************************************************************************************
 23 // Model implementation
 24 
 25 Firebug.JSONViewerModel = extend(Firebug.Module,
 26 {
 27     dispatchName: "jsonViewer",
 28     initialize: function()
 29     {
 30         Firebug.NetMonitor.NetInfoBody.addListener(this);
 31 
 32         // Used by Firebug.DOMPanel.DirTable domplate.
 33         this.toggles = new ToggleBranch();
 34     },
 35 
 36     shutdown: function()
 37     {
 38         Firebug.NetMonitor.NetInfoBody.removeListener(this);
 39     },
 40 
 41     initTabBody: function(infoBox, file)
 42     {
 43         if (FBTrace.DBG_JSONVIEWER)
 44             FBTrace.sysout("jsonviewer.initTabBody", infoBox);
 45 
 46         // Let listeners to parse the JSON.
 47         dispatch(this.fbListeners, "onParseJSON", [file]);
 48 
 49         // The JSON is still no there, try to parse most common cases.
 50         if (!file.jsonObject)
 51         {
 52             if (this.isJSON(safeGetContentType(file.request), file.responseText))
 53                 file.jsonObject = this.parseJSON(file);
 54         }
 55 
 56         // The jsonObject is created so, the JSON tab can be displayed.
 57         if (file.jsonObject && hasProperties(file.jsonObject))
 58         {
 59             Firebug.NetMonitor.NetInfoBody.appendTab(infoBox, "JSON",
 60                 $STR("jsonviewer.tab.JSON"));
 61 
 62             if (FBTrace.DBG_JSONVIEWER)
 63                 FBTrace.sysout("jsonviewer.initTabBody; JSON object available " +
 64                     (typeof(file.jsonObject) != "undefined"), file.jsonObject);
 65         }
 66     },
 67 
 68     isJSON: function(contentType, data)
 69     {
 70         // Workaround for JSON responses without proper content type
 71         // Let's consider all responses starting with "{" as JSON. In the worst
 72         // case there will be an exception when parsing. This means that no-JSON
 73         // responses (and post data) (with "{") can be parsed unnecessarily,
 74         // which represents a little overhead, but this happens only if the request
 75         // is actually expanded by the user in the UI (Net & Console panels).
 76         var responseText = data ? trimLeft(data) : null;
 77         if (responseText && responseText.indexOf("{") == 0)
 78             return true;
 79 
 80         if (!contentType)
 81             return false;
 82 
 83         contentType = contentType.split(";")[0];
 84         contentType = trim(contentType);
 85         return contentTypes[contentType];
 86     },
 87 
 88     // Update listener for TabView
 89     updateTabBody: function(infoBox, file, context)
 90     {
 91         var tab = infoBox.selectedTab;
 92         var tabBody = infoBox.getElementsByClassName("netInfoJSONText").item(0);
 93         if (!hasClass(tab, "netInfoJSONTab") || tabBody.updated)
 94             return;
 95 
 96         tabBody.updated = true;
 97 
 98         if (file.jsonObject) {
 99             Firebug.DOMPanel.DirTable.tag.replace(
100                  {object: file.jsonObject, toggles: this.toggles}, tabBody);
101         }
102     },
103 
104     parseJSON: function(file)
105     {
106         var jsonString = new String(file.responseText);
107         return parseJSONString(jsonString, "http://" + file.request.originalURI.host);
108     },
109 });
110 
111 // ************************************************************************************************
112 // Registration
113 
114 Firebug.registerModule(Firebug.JSONViewerModel);
115 
116 // ************************************************************************************************
117 }});
