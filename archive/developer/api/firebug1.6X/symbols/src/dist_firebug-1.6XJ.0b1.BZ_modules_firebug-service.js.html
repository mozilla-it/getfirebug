642     {
643         if (url)
644         {

646             if (urlBreakpoints)
647             {
648                 for (var i = 0; i < urlBreakpoints.length; ++i)
649                 {
650                     var bp = urlBreakpoints[i];
651                     if (bp.type & BP_NORMAL && !(bp.type & BP_ERROR) )
652                     {
653                         if (bp.scriptsWithBreakpoint && bp.scriptsWithBreakpoint.length > 0)
654                         {
655                             var rc = cb.call.apply(bp, [url, bp.lineNo, bp, bp.scriptsWithBreakpoint]);
656                             if (rc)
657                                 return [bp];
658                         }
659                         else
660                         {
661                             var rc = cb.call.apply(bp, [url, bp.lineNo, bp]);
662                             if (rc)
663                                 return [bp];
664                         }
665                     }
666                 }
667             }
668         }
