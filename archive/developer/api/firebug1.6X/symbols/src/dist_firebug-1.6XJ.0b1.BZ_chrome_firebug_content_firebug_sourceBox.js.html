195     getSourceLinesFrom: function(selection)
196     {
197         // https://developer.mozilla.org/en/DOM/Selection
198         if (selection.isCollapsed)
199             return "";
200 
201         var anchorSourceRow = getAncestorByClass(selection.anchorNode, "sourceRow");
202         var focusSourceRow = getAncestorByClass(selection.focusNode, "sourceRow");
203         if (anchorSourceRow == focusSourceRow)
204         {
205             return selection.toString();// trivial case
206         }
207         var buf = this.getSourceLine(anchorSourceRow, selection.anchorOffset);
208 
209         var currentSourceRow = anchorSourceRow.nextSibling;
210         while(currentSourceRow && (currentSourceRow != focusSourceRow) && hasClass(currentSourceRow, "sourceRow"))
211         {
212             buf += this.getSourceLine(currentSourceRow);
213             currentSourceRow = currentSourceRow.nextSibling;
214         }
215         buf += this.getSourceLine(focusSourceRow, 0, selection.focusOffset);
216         return buf;

418     /* Select sourcebox with href, scroll lineNo into center, highlight lineNo with highlighter given
419      * @param href a URL, null means the selected sourcefile
420      * @param lineNo integer 1-maximumLineNumber
421      * @param highlighter callback, a function(sourceBox). sourceBox.centralLine will be lineNo
422      */
423     scrollToLine: function(href, lineNo, highlighter)
424     {
425         if (FBTrace.DBG_SOURCEFILES) FBTrace.sysout("SourceBoxPanel.scrollToLine: "+lineNo+"@"+href+"\n");
426 
427         if (this.context.scrollTimeout)
428         {
429             this.context.clearTimeout(this.context.scrollTimeout);
430             delete this.context.scrollTimeout
431         }
432 
433         if (href)
434         {
435             if (!this.selectedSourceBox || this.selectedSourceBox.repObject.href != href)
436             {
437                 var sourceFile = this.context.sourceFileMap[href];
438                 if (!sourceFile)
439                 {
440                     if(FBTrace.DBG_SOURCEFILES)
441                         FBTrace.sysout("scrollToLine FAILS, no sourceFile for href "+href, this.context.sourceFileMap);
442                     return;
443                 }
444                 this.navigate(sourceFile);
445             }
446         }
447 
448         this.context.scrollTimeout = this.context.setTimeout(bindFixed(function()
449         {
450             if (!this.selectedSourceBox)
451             {
452                 if (FBTrace.DBG_SOURCEFILES)
453                     FBTrace.sysout("SourceBoxPanel.scrollTimeout no selectedSourceBox");
454                 return;
455             }
456 
457             this.selectedSourceBox.targetedLine = lineNo;
458 
459             // At this time we know which sourcebox is selected but the viewport is not selected.
460             // We need to scroll, let the scroll handler set the viewport, then highlight any lines visible.
461             var skipScrolling = false;
462             if (this.selectedSourceBox.firstViewableLine && this.selectedSourceBox.lastViewableLine)
463             {
464                 var linesFromTop = lineNo - this.selectedSourceBox.firstViewableLine;
465                 var linesFromBot = this.selectedSourceBox.lastViewableLine - lineNo;
466                 skipScrolling = (linesFromTop > 3 && linesFromBot > 3);
467                 if (FBTrace.DBG_SOURCEFILES) FBTrace.sysout("SourceBoxPanel.scrollTimeout: skipScrolling: "+skipScrolling+" fromTop:"+linesFromTop+" fromBot:"+linesFromBot);
468             }
469             else  // the selectedSourceBox has not been built
470             {
471                 if (FBTrace.DBG_SOURCEFILES)
472                     FBTrace.sysout("SourceBoxPanel.scrollTimeout, no viewable lines", this.selectedSourceBox);
473             }
474 
475             if (highlighter)
476                  this.selectedSourceBox.highlighter = highlighter;
477 
478             if (!skipScrolling)
479             {
480                 var viewRange = this.getViewRangeFromTargetLine(this.selectedSourceBox, lineNo);
481                 this.selectedSourceBox.newScrollTop = this.getScrollTopFromViewRange(this.selectedSourceBox, viewRange);
482                 if (FBTrace.DBG_SOURCEFILES) FBTrace.sysout("SourceBoxPanel.scrollTimeout: newScrollTop "+this.selectedSourceBox.newScrollTop+" vs old "+this.selectedSourceBox.scrollTop+" for "+this.selectedSourceBox.repObject.href);
483                 this.selectedSourceBox.scrollTop = this.selectedSourceBox.newScrollTop; // *may* cause scrolling
484                 if (FBTrace.DBG_SOURCEFILES) FBTrace.sysout("SourceBoxPanel.scrollTimeout: scrollTo "+lineNo+" scrollTop:"+this.selectedSourceBox.scrollTop+ " lineHeight: "+this.selectedSourceBox.lineHeight);
485             }
486 
487             if (this.selectedSourceBox.highlighter)
488                 this.applyDecorator(this.selectedSourceBox); // may need to highlight even if we don't scroll
489 
490         }, this));
491     },

533     reView: function(sourceBox, clearCache)  // called for all scroll events, including any time sourcebox.scrollTop is set
534     {
535         if (sourceBox.targetedLine)
536         {
537             sourceBox.targetLineNumber = sourceBox.targetedLine;
538             var viewRange = this.getViewRangeFromTargetLine(sourceBox, sourceBox.targetedLine);
539             delete sourceBox.targetedLine;
540         }
541         else
542         {
543             var viewRange = this.getViewRangeFromScrollTop(sourceBox, sourceBox.scrollTop);
544         }
545 
546         if (clearCache)
547         {
548             this.clearSourceBox(sourceBox);
549         }
550         else if (sourceBox.scrollTop === sourceBox.lastScrollTop && sourceBox.clientHeight && sourceBox.clientHeight === sourceBox.lastClientHeight)
551         {
552             if (sourceBox.firstRenderedLine <= viewRange.firstLine && sourceBox.lastRenderedLine >= viewRange.lastLine)
553             {
554                 if (FBTrace.DBG_SOURCEFILES)
555                     FBTrace.sysout("reView skipping sourceBox "+sourceBox.scrollTop+"=scrollTop="+sourceBox.lastScrollTop+", "+ sourceBox.clientHeight+"=clientHeight="+sourceBox.lastClientHeight, sourceBox);
556                 // skip work if nothing changes.
557                 return;
558             }
559         }
560 
561         dispatch(this.fbListeners, "onBeforeViewportChange", [this]);  // XXXjjb TODO where should this be?
562         this.buildViewAround(sourceBox, viewRange);
563 
564         if (Firebug.uiListeners.length > 0)
565         {
566             var link = new SourceLink(sourceBox.repObject.href, sourceBox.centralLine, this.getSourceType());
567             dispatch(Firebug.uiListeners, "onViewportChange", [link]);
568         }
569 
570         sourceBox.lastScrollTop = sourceBox.scrollTop;
571         sourceBox.lastClientHeight = sourceBox.clientHeight;
572     },

574     buildViewAround: function(sourceBox, viewRange)
575     {
576         try
577         {
578             this.updateViewportCache(sourceBox, viewRange);
579         }
580         catch(exc)
581         {
582             if(FBTrace.DBG_ERRORS)
583                 FBTrace.sysout("buildViewAround updateViewportCache FAILS "+exc, exc);
584         }
585 
586         this.setViewportPadding(sourceBox, viewRange);
587 
588         sourceBox.centralLine = Math.floor( (viewRange.lastLine + viewRange.firstLine)/2 );
589 
590         this.applyDecorator(sourceBox);
591 
592         return;
593     },

595     updateViewportCache: function(sourceBox, viewRange)
596     {
597         var cacheHit = this.insertedLinesOverlapCache(sourceBox, viewRange);
598 
599         if (!cacheHit)
600         {
601             this.clearSourceBox(sourceBox);  // no overlap, remove old range
602             sourceBox.firstRenderedLine = viewRange.firstLine; // reset cached range
603             sourceBox.lastRenderedLine = viewRange.lastLine;
604         }
605         else  // cache overlap, expand range of cache
606         {
607             sourceBox.firstRenderedLine = Math.min(viewRange.firstLine, sourceBox.firstRenderedLine);
608             sourceBox.lastRenderedLine = Math.max(viewRange.lastLine, sourceBox.lastRenderedLine);
609         }
610         sourceBox.firstViewableLine = viewRange.firstLine;  // todo actually check that these are viewable
611         sourceBox.lastViewableLine = viewRange.lastLine;
612         sourceBox.numberOfRenderedLines = sourceBox.lastRenderedLine - sourceBox.firstRenderedLine + 1;
613 
614         if (FBTrace.DBG_SOURCEFILES)
615             FBTrace.sysout("buildViewAround viewRange: "+viewRange.firstLine+"-"+viewRange.lastLine+" rendered: "+sourceBox.firstRenderedLine+"-"+sourceBox.lastRenderedLine, sourceBox);
616     },
617 
618     /*
619      * Add lines from viewRange, but do not adjust first/lastRenderedLine.
620      * @return true if viewRange overlaps first/lastRenderedLine
621      */
622     insertedLinesOverlapCache: function(sourceBox, viewRange)
623     {
624         var topCacheLine = null;
625         var cacheHit = false;
626         for (var line = viewRange.firstLine; line <= viewRange.lastLine; line++)
627         {
628             if (line >= sourceBox.firstRenderedLine && line <= sourceBox.lastRenderedLine )
629             {
630                 cacheHit = true;
631                 continue;
632             }
633 
634             var lineHTML = this.getSourceLineHTML(sourceBox, line);
635 
636             var ref = null;
637             if (line < sourceBox.firstRenderedLine)   // prepend if we are above the cache
638             {
639                 if (!topCacheLine)
640                     topCacheLine = sourceBox.getLineNode(sourceBox.firstRenderedLine);
641                 ref = topCacheLine;
642             }
643 
644             var newElement = appendInnerHTML(sourceBox.viewport, lineHTML, ref);
645         }
646         return cacheHit;
647     },
648 
649     clearSourceBox: function(sourceBox)
650     {
651         if (sourceBox.firstRenderedLine)
652         {
653             var topMostCachedElement = sourceBox.getLineNode(sourceBox.firstRenderedLine);  // eg 1
654             var totalCached = sourceBox.lastRenderedLine - sourceBox.firstRenderedLine + 1;   // eg 20 - 1 + 1 = 19
655             if (topMostCachedElement && totalCached)
656                 this.removeLines(sourceBox, topMostCachedElement, totalCached);
657         }
658         sourceBox.lastRenderedLine = 0;
659         sourceBox.firstRenderedLine = 0;
660         sourceBox.numberOfRenderedLines = 0;
661     },

663     getSourceLineHTML: function(sourceBox, i)
664     {
665         var lineNo = sourceBox.decorator.getUserVisibleLineNumber(sourceBox, i);
666         var lineHTML = sourceBox.decorator.getLineHTML(sourceBox, i);
667         var lineId = sourceBox.decorator.getLineId(sourceBox, i);    // decorator lines may not have ids
668 
669         var lineNoText = this.getTextForLineNo(lineNo, sourceBox.maxLineNoChars);
670 
671         var theHTML =
672             '<div '
673                + (lineId ? ('id="' + lineId + '"') : "")
674                + ' class="sourceRow" role="presentation"><a class="'
675                +  'sourceLine' + '" role="presentation">'
676                + lineNoText
677                + '</a><span class="sourceRowText" role="presentation">'
678                + lineHTML
679                + '</span></div>';
680 
681         return theHTML;
682     },

684     getTextForLineNo: function(lineNo, maxLineNoChars)

687         var lineNoText = lineNo + "";

709     getViewRangeFromTargetLine: function(sourceBox, targetLineNumber)
710     {
711         var viewRange = {firstLine: 1, centralLine: targetLineNumber, lastLine: 1};
712 
713         var averageLineHeight = this.getAverageLineHeight(sourceBox);
714         var panelHeight = this.panelNode.clientHeight;
715         var linesPerViewport = Math.round((panelHeight / averageLineHeight) + 1);
716

718 
719         if (viewRange.firstLine <= 0)
720             viewRange.firstLine = 1;
721 
722         viewRange.lastLine = viewRange.firstLine + linesPerViewport;
723 
724         if (viewRange.lastLine > sourceBox.maximumLineNumber)
725             viewRange.lastLine = sourceBox.maximumLineNumber;

730     /*
731      * Use the average height of source lines in the cache to estimate where the scroll bar points based on scrollTop
732      */
733     getViewRangeFromScrollTop: function(sourceBox, scrollTop)
734     {
735         var viewRange = {};
736         var averageLineHeight = this.getAverageLineHeight(sourceBox);
737         viewRange.firstLine = Math.floor(scrollTop / averageLineHeight + 1);
738 
739         var panelHeight = this.panelNode.clientHeight;
740         var viewableLines = Math.ceil((panelHeight / averageLineHeight) + 1);
741         viewRange.lastLine = viewRange.firstLine + viewableLines;
742         if (viewRange.lastLine > sourceBox.maximumLineNumber)
743             viewRange.lastLine = sourceBox.maximumLineNumber;
744 
745         viewRange.centralLine = Math.floor((viewRange.lastLine - viewRange.firstLine)/2);
746 
747         if (FBTrace.DBG_SOURCEFILES)
748         {
749             FBTrace.sysout("getViewRangeFromScrollTop scrollTop:"+scrollTop+" viewRange: "+viewRange.firstLine+"-"+viewRange.lastLine);
750             if (!this.noRecurse)
751             {
752                 this.noRecurse = true;
753                 var testScrollTop = this.getScrollTopFromViewRange(sourceBox, viewRange);
754                 delete this.noRecurse;
755                 FBTrace.sysout("getViewRangeFromScrollTop "+((scrollTop==testScrollTop)?"checks":(scrollTop+"=!scrollTop!="+testScrollTop)));
756             }
757         }
758 
759         return viewRange;
760     },
761 
762     /*
763      * inverse of the getViewRangeFromScrollTop.
764      * If the viewRange was set by targetLineNumber, then this value become the new scroll top
765      *    else the value will be the same as the scrollbar's given value of scrollTop.
766      */
767     getScrollTopFromViewRange: function(sourceBox, viewRange)
768     {
769         var averageLineHeight = this.getAverageLineHeight(sourceBox);
770         var scrollTop = Math.floor(averageLineHeight * (viewRange.firstLine - 1));
771 
772         if (FBTrace.DBG_SOURCEFILES)
773         {
774             FBTrace.sysout("getScrollTopFromViewRange viewRange:"+viewRange.firstLine+"-"+viewRange.lastLine+" averageLineHeight: "+averageLineHeight+" scrollTop "+scrollTop);
775             if (!this.noRecurse)
776             {
777                 this.noRecurse = true;
778                 var testViewRange = this.getViewRangeFromScrollTop(sourceBox, scrollTop);
779                 delete this.noRecurse;
780                 var vrStr = viewRange.firstLine+"-"+viewRange.lastLine;
781                 var tvrStr = testViewRange.firstLine+"-"+testViewRange.lastLine;
782                 FBTrace.sysout("getScrollTopFromCenterLine "+((vrStr==tvrStr)? "checks" : vrStr+"=!viewRange!="+tvrStr));
783             }
784         }
785 
786         return scrollTop;
787     },

789     /*
790      * The virtual sourceBox height is the averageLineHeight * max lines
791      * @return float
792      */
793     getAverageLineHeight: function(sourceBox)
794     {
795         var averageLineHeight = sourceBox.lineHeight;  // fall back to single line height
796 
797         var renderedViewportHeight = sourceBox.viewport.clientHeight;
798         var numberOfRenderedLines = sourceBox.numberOfRenderedLines;
799         if (renderedViewportHeight && numberOfRenderedLines)
800             averageLineHeight = renderedViewportHeight / numberOfRenderedLines;
801 
802         return averageLineHeight;
803     },
804 
805     /*
806      * The virtual sourceBox = topPadding + sourceBox.viewport + bottomPadding
807      * The viewport grows as more lines are added to the cache
808      * The virtual sourceBox height is estimated from the average height lines in the viewport cache
809      */
810     getTotalPadding: function(sourceBox)
811     {
812         var numberOfRenderedLines = sourceBox.numberOfRenderedLines;
813         if (!numberOfRenderedLines)
814             return 0;
815 
816         var max = sourceBox.maximumLineNumber;
817         var averageLineHeight = this.getAverageLineHeight(sourceBox);
818         // total box will be the average line height times total lines
819         var virtualSourceBoxHeight = Math.floor(max * averageLineHeight);
820         if (virtualSourceBoxHeight < sourceBox.clientHeight)
821         {
822             var scrollBarHeight = sourceBox.offsetHeight - sourceBox.clientHeight;
823             // the total - view-taken-up - scrollbar
824             var totalPadding = sourceBox.clientHeight - sourceBox.viewport.clientHeight - 1;
825         }
826         else
827             var totalPadding = virtualSourceBoxHeight - sourceBox.viewport.clientHeight;
828 
829         if (FBTrace.DBG_SOURCEFILES)
830             FBTrace.sysout("getTotalPadding clientHeight:"+sourceBox.viewport.clientHeight+"  max: "+max+" gives total padding "+totalPadding);
831 
832         return totalPadding;
833     },

835     setViewportPadding: function(sourceBox, viewRange)
836     {
837         var firstRenderedLineElement = sourceBox.getLineNode(sourceBox.firstRenderedLine);
838         if (!firstRenderedLineElement)
839         {
840             if (FBTrace.DBG_ERRORS)
841                 FBTrace.sysout("setViewportPadding FAILS, no line at "+sourceBox.firstRenderedLine, sourceBox);
842             return;
843         }
844 
845         var firstRenderedLineOffset = firstRenderedLineElement.offsetTop;
846         var firstViewRangeElement = sourceBox.getLineNode(viewRange.firstLine);
847         var firstViewRangeOffset = firstViewRangeElement.offsetTop;
848         var topPadding = sourceBox.scrollTop - (firstViewRangeOffset - firstRenderedLineOffset);
849         // Because of rounding when converting from pixels to lines, topPadding can be +/- lineHeight/2, round up
850         var averageLineHeight = this.getAverageLineHeight(sourceBox);
851         var linesOfPadding = Math.floor( (topPadding + averageLineHeight)/ averageLineHeight);
852         var topPadding = (linesOfPadding - 1)* averageLineHeight;
853 
854         if (FBTrace.DBG_SOURCEFILES)
855             FBTrace.sysout("setViewportPadding sourceBox.scrollTop - (firstViewRangeOffset - firstRenderedLineOffset): "+sourceBox.scrollTop+"-"+"("+firstViewRangeOffset+"-"+firstRenderedLineOffset+")="+topPadding);
856         // we want the bottomPadding to take up the rest
857         var totalPadding = this.getTotalPadding(sourceBox);
858         if (totalPadding < 0)
859             var bottomPadding = Math.abs(totalPadding);
860         else
861             var bottomPadding = Math.floor(totalPadding - topPadding);
862 
863         if (bottomPadding < 0)
864             bottomPadding = 0;
865 
866         if(FBTrace.DBG_SOURCEFILES)
867         {
868             FBTrace.sysout("setViewportPadding viewport.offsetHeight: "+sourceBox.viewport.offsetHeight+" viewport.clientHeight "+sourceBox.viewport.clientHeight);
869             FBTrace.sysout("setViewportPadding sourceBox.offsetHeight: "+sourceBox.offsetHeight+" sourceBox.clientHeight "+sourceBox.clientHeight);
870             FBTrace.sysout("setViewportPadding scrollTop: "+sourceBox.scrollTop+" firstRenderedLine "+sourceBox.firstRenderedLine+" bottom: "+bottomPadding+" top: "+topPadding);
871         }
872         var view = sourceBox.viewport;
873 
874         // Set the size on the line number field so the padding is filled with same style as source lines.
875         view.previousSibling.style.height = topPadding + "px";
876         view.nextSibling.style.height = bottomPadding + "px";
877 
878         //sourceRow
879         view.previousSibling.firstChild.style.height = topPadding + "px";
880         view.nextSibling.firstChild.style.height = bottomPadding + "px";
881 
882         //sourceLine
883         view.previousSibling.firstChild.firstChild.style.height = topPadding + "px";
884         view.nextSibling.firstChild.firstChild.style.height = bottomPadding + "px";
885     },

887     applyDecorator: function(sourceBox)
888     {
889         if (this.context.sourceBoxDecoratorTimeout)
890         {
891             this.context.clearTimeout(this.context.sourceBoxDecoratorTimeout);
892             delete this.context.sourceBoxDecoratorTimeout;
893         }
894 
895         // Run source code decorating on 150ms timeout, which is bigger than
896         // the period in which scroll events are fired. So, if the user is moving
897         // scroll-bar thumb (or quickly clicking on scroll-arrows), the source code is
898         // not decorated (the timeout cleared by the code above) and the scrolling is fast.
899         this.context.sourceBoxDecoratorTimeout = this.context.setTimeout(
900             bindFixed(this.asyncDecorating, this, sourceBox), 150);
901 
902         if (this.context.sourceBoxHighlighterTimeout)
903         {
904             this.context.clearTimeout(this.context.sourceBoxHighlighterTimeout);
905             delete this.context.sourceBoxHighlighterTimeout;
906         }
907 
908         // Source code highlighting is using different timeout: 0ms. When searching
909         // within the Script panel, the user expects immediate response.
910         this.context.sourceBoxHighlighterTimeout = this.context.setTimeout(
911             bindFixed(this.asyncHighlighting, this, sourceBox));
912     },

915     {

917         {

920             if (Firebug.uiListeners.length > 0)

922 
923             if (FBTrace.DBG_SOURCEFILES)
924                 FBTrace.sysout("sourceBoxDecoratorTimeout "+sourceBox.repObject, sourceBox);
925         }
926         catch (exc)
927         {
928             if (FBTrace.DBG_ERRORS)
929                 FBTrace.sysout("sourcebox applyDecorator FAILS "+exc, exc);
930         }
931     },

950             if (FBTrace.DBG_ERRORS)
951                 FBTrace.sysout("sourcebox highlighter FAILS "+exc, exc);
952         }
