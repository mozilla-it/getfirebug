  2 
  3 FBTestApp.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 // Test Logger Implementation
  7 
  8 var Cc = Components.classes;
  9 var Ci = Components.interfaces;
 10 
 11 // ************************************************************************************************
 12 
 13 FBTestApp.TestLogger =
 14 {
 15     getDefaultFile: function(date)
 16     {
 17         function f(n, c) {
 18             if (!c) c = 2;
 19             var s = new String(n);
 20             while (s.length < c) s = "0" + s;
 21             return s;
 22         }
 23 
 24         var fileName = "firebug-" +
 25             Firebug.version + "-" +
 26             date.getFullYear() + "-" + f(date.getMonth()+1) + "-" + f(date.getDate()) +
 27             "+" + f(date.getHours()) + "-" + f(date.getMinutes());
 28 
 29         var file = this.getDefaultFolder();
 30         file.append(fileName + ".log");
 31         file.createUnique(Ci.nsIFile.NORMAL_FILE_TYPE, 0666);
 32 
 33         return file;
 34     },
 35 
 36     getOutputStream: function(date)
 37     {
 38         var file = this.getDefaultFile(date);
 39         var foStream = Cc["@mozilla.org/network/file-output-stream;1"]
 40             .createInstance(Ci.nsIFileOutputStream);
 41         foStream.init(file, 0x02 | 0x08 | 0x20, 0666, 0); // write, create, truncate
 42 
 43         var convertor = Cc["@mozilla.org/intl/converter-output-stream;1"]
 44             .createInstance(Ci.nsIConverterOutputStream);
 45 
 46         convertor.init(foStream, "UTF-8", 0, 0);
 47         return convertor;
 48     },
 49 
 50     getSystemInfo: function()
 51     {
 52         var appInfo = Cc["@mozilla.org/xre/app-info;1"].getService(Ci.nsIXULAppInfo);
 53         var currLocale = Firebug.getPref("general.useragent", "locale");
 54         var systemInfo = Cc["@mozilla.org/system-info;1"].getService(Ci.nsIPropertyBag);
 55         var osName = systemInfo.getProperty("name");
 56         var osVersion = systemInfo.getProperty("version");
 57 
 58         return {
 59             fbTest: FBTestApp.TestConsole.getVersion(),
 60             firebug: Firebug.version,
 61             appName: appInfo.name,
 62             appVersion: appInfo.version,
 63             appPlatform: appInfo.platformVersion,
 64             appBuildID: appInfo.appBuildID,
 65             locale: currLocale,
 66             osName: osName,
 67             osVersion: osVersion,
 68         };
 69     },
 70 
 71     getDefaultFolder: function()
 72     {
 73         var dir;
 74         var path = Firebug.getPref(FBTestApp.prefDomain, "defaultLogDir");
 75         if (!path)
 76         {
 77             // Create default folder for automated logs.
 78             var dirService = Cc["@mozilla.org/file/directory_service;1"].getService(Ci.nsIProperties);
 79             var dir = dirService.get("ProfD", Ci.nsILocalFile);
 80             dir.append("firebug");
 81             dir.append("fbtest");
 82             dir.append("logs");
 83         }
 84         else
 85         {
 86             dir = CCIN("@mozilla.org/file/local;1", "nsILocalFile");
 87             dir.initWithPath(path);
 88         }
 89 
 90         return dir;
 91     },
 92 };
 93 
 94 // ************************************************************************************************
 95 // Test Runner Progress Listener
 96 
 97 FBTestApp.TestLogger.ProgressListener = function(date)
 98 {
 99     this.startedTime = date;
100 }
101 
102 FBTestApp.TestLogger.ProgressListener.prototype =
103 {
104     onTestSuiteStart: function(tests)
105     {
106         if (FBTrace.DBG_FBTEST)
107             FBTrace.sysout("fbtrace.ProgressListener; onTestSuiteStart " + tests.length, tests);
108 
109         this.stream = FBTestApp.TestLogger.getOutputStream(this.startedTime);
110         if (!this.stream)
111             return;
112 
113         // Log system configuration
114         var info = FBTestApp.TestLogger.getSystemInfo();
115         this.stream.writeString("FIREBUG INFO | Firebug: " + info.firebug + "\n");
116         this.stream.writeString("FIREBUG INFO | FBTest: " + info.fbTest + "\n");
117         this.stream.writeString("FIREBUG INFO | App Name: " + info.appName + "\n");
118         this.stream.writeString("FIREBUG INFO | App Version: " + info.appVersion + "\n");
119         this.stream.writeString("FIREBUG INFO | App Platform: " + info.appPlatform + "\n");
120         this.stream.writeString("FIREBUG INFO | App Build ID: " + info.appBuildID + "\n");
121         this.stream.writeString("FIREBUG INFO | Locale: " + info.locale + "\n");
122         this.stream.writeString("FIREBUG INFO | OS Name: " + info.osName + "\n");
123         this.stream.writeString("FIREBUG INFO | OS Version: " + info.osVersion + "\n");
124         this.stream.writeString("FIREBUG INFO | Export Date: " + this.startedTime.toGMTString() + "\n");
125         this.stream.writeString("FIREBUG INFO | Test Suite: " + FBTestApp.TestConsole.testListPath + "\n");
126         this.stream.writeString("FIREBUG INFO | Total Tests: " + tests.length + "\n");
127         this.stream.writeString("\n");
128     },
129 
130     onTestSuiteDone: function(canceled)
131     {
132         if (FBTrace.DBG_FBTEST)
133             FBTrace.sysout("fbtrace.ProgressListener; onTestSuiteDone " + canceled);
134 
135         if (!this.stream)
136             return;
137 
138         var passingTests = FBTestApp.TestSummary.passingTests;
139 
140         // Log summary
141         this.stream.writeString("\n");
142         this.stream.writeString("FIREBUG INFO | Test Suite Finished: " + (new Date()).toGMTString() + "\n");
143         this.stream.writeString("FIREBUG INFO | Passing: " + passingTests.passing + "\n");
144         this.stream.writeString("FIREBUG INFO | Failing: " + passingTests.failing + "\n");
145 
146         this.stream.close();
147     },
148 
149     onTestStart: function(test)
150     {
151         if (FBTrace.DBG_FBTEST)
152             FBTrace.sysout("fbtrace.ProgressListener; onTestStart: " + test.uri, test);
153 
154         if (!this.stream)
155             return;
156 
157         var testName = test.testPage ? test.testPage : test.uri;
158 
159         // Log Start of the test.
160         this.stream.writeString("FIREBUG INFO | " + testName +
161             " | [START] " + test.desc + "\n");
162     },
163 
164     onTestDone: function(test)
165     {
166         if (FBTrace.DBG_FBTEST)
167             FBTrace.sysout("fbtrace.ProgressListener; onTestDone", test);
168 
169         if (!this.stream)
170             return;
171 
172         var moniker = "TEST-PASS";
173         if (test.error)
174             moniker = (test.category == "fails") ? "TEST-KNOWN-FAIL" : "TEST-UNEXPECTED-FAIL";
175 
176         var testName = test.testPage ? test.testPage : test.uri;
177 
178         // Report test messages for failures.
179         if (test.error)
180         {
181             for (var i=0; i<test.results.length; i++)
182             {
183                 var result = test.results[i];
184                 this.stream.writeString("FIREBUG INFO | " + testName +
185                     " | " + (result.pass ? "[OK]" : "[ERROR]") + " " + result.msg + "\n");
186             }
187         }
188 
189         // Log result of the test.
190         this.stream.writeString("FIREBUG " + moniker + " | " + testName + " | [DONE]\n");
191     },
192 };
193 
194 // ************************************************************************************************
195 }});
