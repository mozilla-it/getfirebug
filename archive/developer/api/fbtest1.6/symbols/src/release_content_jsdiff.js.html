 24 function diffString( o, n ) {
 25   o = o.replace(/\s+$/, '');
 26   n = n.replace(/\s+$/, '');
 27 
 28   var out = diff(o == "" ? [] : o.split(/\s+/), n == "" ? [] : n.split(/\s+/) );
 29   var str = "";
 30 
 31   var oSpace = o.match(/\s+/g);
 32   if (oSpace == null) {
 33     oSpace = ["\n"];
 34   } else {
 35     oSpace.push("\n");
 36   }
 37   var nSpace = n.match(/\s+/g);
 38   if (nSpace == null) {
 39     nSpace = ["\n"];
 40   } else {
 41     nSpace.push("\n");
 42   }
 43 
 44   if (out.n.length == 0) {
 45       for (var i = 0; i < out.o.length; i++) {
 46         str += '<del>' + escapeText(out.o[i]) + oSpace[i] + "</del>";
 47       }
 48   } else {
 49     if (out.n[0].text == null) {
 50       for (n = 0; n < out.o.length && out.o[n].text == null; n++) {
 51         str += '<del>' + escapeText(out.o[n]) + oSpace[n] + "</del>";
 52       }
 53     }
 54 
 55     for ( var i = 0; i < out.n.length; i++ ) {
 56       if (out.n[i].text == null) {
 57         str += '<ins>' + escapeText(out.n[i]) + nSpace[i] + "</ins>";
 58       } else {
 59         var pre = "";
 60 
 61         for (n = out.n[i].row + 1; n < out.o.length && out.o[n].text == null; n++ ) {
 62           pre += '<del>' + escapeText(out.o[n]) + oSpace[n] + "</del>";
 63         }
 64         str += " " + out.n[i].text + nSpace[i] + pre;
 65       }
 66     }
 67   }
 68   
 69   return str;
 70 }

 74                     (Math.random() * 100) + "%, " + 
 75                     (Math.random() * 100) + "%)";

 77 function diffString2( o, n ) {
 78   o = o.replace(/\s+$/, '');
 79   n = n.replace(/\s+$/, '');
 80 
 81   var out = diff(o == "" ? [] : o.split(/\s+/), n == "" ? [] : n.split(/\s+/) );
 82 
 83   var oSpace = o.match(/\s+/g);
 84   if (oSpace == null) {
 85     oSpace = ["\n"];
 86   } else {
 87     oSpace.push("\n");
 88   }
 89   var nSpace = n.match(/\s+/g);
 90   if (nSpace == null) {
 91     nSpace = ["\n"];
 92   } else {
 93     nSpace.push("\n");
 94   }
 95 
 96   var os = "";
 97   var colors = new Array();
 98   for (var i = 0; i < out.o.length; i++) {
 99       colors[i] = randomColor();
100 
101       if (out.o[i].text != null) {
102           os += '<span style="background-color: ' +colors[i]+ '">' + 
103                 escapeText(out.o[i].text) + oSpace[i] + "</span>";
104       } else {
105           os += "<del>" + escapeText(out.o[i]) + oSpace[i] + "</del>";
106       }
107   }
108 
109   var ns = "";
110   for (var i = 0; i < out.n.length; i++) {
111       if (out.n[i].text != null) {
112           ns += '<span style="background-color: ' +colors[out.n[i].row]+ '">' + 
113                 escapeText(out.n[i].text) + nSpace[i] + "</span>";
114       } else {
115           ns += "<ins>" + escapeText(out.n[i]) + nSpace[i] + "</ins>";
116       }
117   }
118 
119   return { o : os , n : ns };
120 }

122 function diff( o, n ) {
123   var ns = new Object();
124   var os = new Object();
125   
126   for ( var i = 0; i < n.length; i++ ) {
127     if ( ns[ n[i] ] == null )
128       ns[ n[i] ] = { rows: new Array(), o: null };
129     ns[ n[i] ].rows.push( i );
130   }
131   
132   for ( var i = 0; i < o.length; i++ ) {
133     if ( os[ o[i] ] == null )
134       os[ o[i] ] = { rows: new Array(), n: null };
135     os[ o[i] ].rows.push( i );
136   }
137   
138   for ( var i in ns ) {
139     if ( ns[i].rows.length == 1 && typeof(os[i]) != "undefined" && os[i].rows.length == 1 ) {
140       n[ ns[i].rows[0] ] = { text: n[ ns[i].rows[0] ], row: os[i].rows[0] };
141       o[ os[i].rows[0] ] = { text: o[ os[i].rows[0] ], row: ns[i].rows[0] };
142     }
143   }
144   
145   for ( var i = 0; i < n.length - 1; i++ ) {
146     if ( n[i].text != null && n[i+1].text == null && n[i].row + 1 < o.length && o[ n[i].row + 1 ].text == null && 
147          n[i+1] == o[ n[i].row + 1 ] ) {
148       n[i+1] = { text: n[i+1], row: n[i].row + 1 };
149       o[n[i].row+1] = { text: o[n[i].row+1], row: i + 1 };
150     }
151   }
152   
153   for ( var i = n.length - 1; i > 0; i-- ) {
154     if ( n[i].text != null && n[i-1].text == null && n[i].row > 0 && o[ n[i].row - 1 ].text == null && 
155          n[i-1] == o[ n[i].row - 1 ] ) {
156       n[i-1] = { text: n[i-1], row: n[i].row - 1 };
157       o[n[i].row-1] = { text: o[n[i].row-1], row: i - 1 };
158     }
159   }
160   
161   return { o: o, n: n };
162 }
