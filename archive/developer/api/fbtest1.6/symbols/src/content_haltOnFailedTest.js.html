  3 FBTestApp.ns(function() { with (FBL) {
  4 
  5 // ************************************************************************************************
  6 // Halt On Failed Test Implementation
  7 
  8 var Cc = Components.classes;
  9 var Ci = Components.interfaces;
 10 
 11 // Services
 12 var loader = Cc["@mozilla.org/moz/jssubscript-loader;1"].getService(Ci.mozIJSSubScriptLoader);
 13 var observerService = Cc["@mozilla.org/observer-service;1"].getService(Ci.nsIObserverService);
 14 
 15 // ************************************************************************************************
 16 
 17 FBTestApp.TestWindowLoader.HaltOnFailedTest =
 18 {
 19     initialize: function()
 20     {
 21         // Localize strings in XUL (using string bundle).
 22         this.internationalizeUI();
 23 
 24         FBTestApp.TestWindowLoader.HaltOnFailedTest.enabled = Firebug.getPref(FBTestApp.prefDomain, "haltOnFailedTest");
 25         this.setHaltOnFailedTestButton();
 26     },
 27 
 28     internationalizeUI: function()
 29     {
 30         var buttons = ["haltOnFailedTest"];
 31 
 32         for (var i=0; i<buttons.length; i++)
 33         {
 34             var element = $(buttons[i]);
 35             FBL.internationalize(element, "label");
 36             FBL.internationalize(element, "tooltiptext");
 37             FBL.internationalize(element, "pickerTooltiptext");
 38             FBL.internationalize(element, "barTooltiptext");
 39         }
 40     },
 41 
 42     setHaltOnFailedTestButton: function()
 43     {
 44         $('haltOnFailedTest').setAttribute('checked', FBTestApp.TestWindowLoader.HaltOnFailedTest.enabled?'true':'false');
 45     },
 46 
 47     onToggleHaltOnFailedTest: function()
 48     {
 49         FBTestApp.TestWindowLoader.HaltOnFailedTest.enabled = !FBTestApp.TestWindowLoader.HaltOnFailedTest.enabled;
 50         Firebug.setPref(FBTestApp.prefDomain, "haltOnFailedTest", FBTestApp.TestWindowLoader.HaltOnFailedTest.enabled);
 51         FBTestApp.TestWindowLoader.HaltOnFailedTest.setHaltOnFailedTestButton();
 52     },
 53 
 54     onFailure: function()
 55     {
 56         if (!FBTestApp.TestWindowLoader.HaltOnFailedTest.enabled)
 57             return;
 58 
 59         if (FBTrace.DBG_FBTEST)
 60             FBTrace.sysout("FBTestApp.TestWindowLoader.HaltOnFailedTest.onFailure ");
 61 
 62         FBTestApp.TestRunner.clearTestTimeout();
 63         Firebug.Debugger.halt(function breakOnFailure(frame)
 64         {
 65             var dropFrames = 7;
 66 
 67             if (FBTrace.DBG_FBTEST)
 68                 FBTrace.sysout("FBTestApp.TestWindowLoader.HaltOnFailedTest.onFailure.breakOnFailure dropping "+dropFrames, frame);
 69 
 70             for (var i = 0; frame && frame.isValid && i < dropFrames; i++)
 71                 frame = frame.callingFrame;
 72 
 73             Firebug.Debugger.breakAsIfDebugger(frame);
 74         });
 75     },
 76 
 77     /* nsIObserve */
 78     observe: function(subject, topic, data)
 79     {
 80         try
 81         {
 82             if (topic == "fbtest")
 83             {
 84                 if (data === "shutdown")
 85                     observerService.removeObserver(FBTestApp.TestWindowLoader.HaltOnFailedTest, "fbtest");
 86 
 87                 if (data in FBTestApp.TestWindowLoader.HaltOnFailedTest)
 88                 {
 89                     FBTestApp.TestWindowLoader.HaltOnFailedTest[data]();
 90                 }
 91                 else
 92                 {
 93                     if (FBTrace.DBG_FBTEST)
 94                         FBTrace.sysout("FBTestApp.TestWindowLoader.HaltOnFailedTest no method for "+data);
 95                 }
 96             }
 97         }
 98         catch (e)
 99         {
100             dump("FBTestApp.TestWindowLoader.observe; EXCEPTION " + e, e);
101         }
102     },
103 };
104 
105 // ************************************************************************************************
106 // Registration
107 
108 /**
109  * Listen to events fired by {@link FBTestApp.TestConsole}.
110  */
111 observerService.addObserver(FBTestApp.TestWindowLoader.HaltOnFailedTest, "fbtest", false);
112 
113 // ************************************************************************************************
114 }});
