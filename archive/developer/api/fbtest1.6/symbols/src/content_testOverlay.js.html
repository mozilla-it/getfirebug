  4 // Test Console Overlay Implementation

  6 /**
  7  * This overlay is intended to append a new menu-item into the Firebug's icon menu.
  8  * This menu is used to open the Test Console (test runner window).
  9  */
 10 var FBTestFirebugOverlay = {};
 11 
 12 (function() {
 13 
 14 var Cc = Components.classes;
 15 var Ci = Components.interfaces;
 16 
 17 var cmdLineHandler = Cc["@mozilla.org/commandlinehandler/general-startup;1?type=FBTest"].getService(Ci.nsICommandLineHandler);
 18 
 19 this.initialize = function()
 20 {
 21     window.removeEventListener("load", FBTestFirebugOverlay.initialize, false);
 22 
 23     // abandon ship if we are loaded by chromebug
 24     var winURL = window.location.toString();
 25     if (winURL == "chrome://chromebug/content/chromebug.xul")
 26         return;
 27 
 28     // Open console if the command line says so of if the pref says so.
 29     var cmd = cmdLineHandler.wrappedJSObject;
 30     if (cmd.runFBTests)
 31         FBTestFirebugOverlay.open(cmd.testListURI);
 32     else if (Firebug.getPref(Firebug.prefDomain, "alwaysOpenTestConsole"))
 33         FBTestFirebugOverlay.open();
 34 };
 35 
 36 this.open = function(testListURI)
 37 {
 38     var consoleWindow = null;
 39     FBL.iterateBrowserWindows("FBTestConsole", function(win) {
 40         consoleWindow = win;
 41         return true;
 42     });
 43 
 44     var args = {
 45         firebugWindow: window,
 46         testListURI: testListURI
 47     };
 48 
 49     // Try to connect an existing trace-console window first.
 50     if (consoleWindow)
 51     {
 52         if ("initWithParams" in consoleWindow)
 53             consoleWindow.initWithParams(args);
 54         consoleWindow.focus();
 55         return;
 56     }
 57 
 58     consoleWindow = window.openDialog(
 59         "chrome://fbtest/content/testConsole.xul",
 60         "FBTestConsole",
 61         "chrome,resizable,scrollbars=auto,minimizable,dialog=no",
 62         args);
 63 
 64     if (FBTrace.DBG_FBTEST)
 65         FBTrace.sysout("fbtest.TestConsoleOverlay.open on FirebugWindow: " +
 66             window.location);
 67 };
 68 
 69 // Register load listener for command line arguments handling.
 70 window.addEventListener("load", FBTestFirebugOverlay.initialize, false);
 71 
 72 }).apply(FBTestFirebugOverlay);
