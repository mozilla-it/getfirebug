 15     onUpload: function()

 22         {
 23             // xxxHonza: localization

 27

 30

 33         var options =
 34         {
 35             success: function(headerResp)
 36             {
 37                 self.onHeaderUploaded(headerResp, header);
 38             },
 39             error: function(status, error, reason)
 40             {
 41                 if (FBTrace.DBG_FBTEST || FBTrace.DBG_ERRORS)
 42                     FBTrace.sysout("fbtest.TestCouchUploader.onUpload; ERROR Can't upload test results" +
 43                         status + ", " + error + ", " + reason);

 47         }
 48         CouchDB.saveDoc(header, options);

 52     {
 53         if (FBTrace.DBG_FBTEST)
 54             FBTrace.sysout("fbtest.TestCouchUploader.onUploaonHeaderUploaded; " +
 55                 "Header uploaded OK", headerResp);
 56

 58         var self = this;
 59         var results = {docs: []};

112     getHeaderDoc: function(extInfo)
113     {
114         var appInfo = Cc["@mozilla.org/xre/app-info;1"].getService(Ci.nsIXULAppInfo);
115         var currLocale = Firebug.getPref("general.useragent", "locale");
116         var systemInfo = Cc["@mozilla.org/system-info;1"].getService(Ci.nsIPropertyBag);
117         var application = Cc["@mozilla.org/fuel/application;1"].getService(Ci.fuelIApplication);
118 
119         var header = {type: "user-header"};
120         header["App Build ID"] = appInfo.appBuildID;
121         //header["App Changeset"] =
122         header["App Name"] = appInfo.name;
123         header["App Platform"] = appInfo.platformVersion;
124         header["App Version"] = appInfo.version;
125         //header["CPU Architecture"] =
126         header["Export Date"] = (new Date()).toGMTString();
127         header["FBTest"] = FBTestApp.TestConsole.getVersion();
128         header["Firebug"] = Firebug.version;
129         header["Locale"] = currLocale;
130         header["OS Detailed Name"] = ""; //xxxHonza todo
131         header["OS Name"] = systemInfo.getProperty("name");
132         header["OS Version"] = systemInfo.getProperty("version");
133         header["Test Suite"] = FBTestApp.TestConsole.testListPath;
134         header["Total Tests"] = this.getTotalTests().toString();
135 
136         if (extInfo && application.extensions)
137         {
138             // Put together a list of installed extensions.
139             var extensions = [];
140             for (var i=0; i<application.extensions.all.length; i++)
141             {
142                 var ext = application.extensions.all[i];
143                 extensions.push({
144                     name: ext.name,
145                     id: ext.id,
146                     enabled: ext.enabled
147                 });
148             }
149             header["Extensions"] = extensions;
150         }
151 
152         return header;
153     },
154 
155     getResultDoc: function(test)
156     {
157         var result = extend(this.getHeaderDoc(false), {type: "user-result"});
158 
159         result.description = test.desc;
160         result.file = test.testPage ? test.testPage : test.uri;
161         result.result = test.error ? (test.category == "fails" ? "TEST-KNOWN-FAIL" :
162             "TEST-UNEXPECTED-FAIL") : "TEST-PASS"
163 
164         if (test.error)
165         {
166             var progress = "";
167             for (var resultIdx in test.results)
168             {
169                 var testResult = test.results[resultIdx];
170                 progress += (testResult.pass ? "[OK]" : "[ERROR]") + " " + testResult.msg + "\n";
171             }
172             result.progress = progress;
173         }
174 
175         return result;
176     },

195 var CouchDB =
196 {
197     uri: "http://brasstacks.mozilla.com/couchdb/firebug/",
198 
199     saveDoc: function(doc, options)
200     {
201         options = options || {};
202         this.ajax({
203             type: "POST",
204             url: this.uri,
205             contentType: "application/json",
206             data: toJSON(doc),
207             complete: function(req)
208             {
209                 var resp = parseJSON(req.responseText);
210                 if (req.status == 201)
211                 {
212                     doc._id = resp.id;
213                     doc._rev = resp.rev;
214                     if (options.success)
215                         options.success(resp);
216                 }
217                 else if (options.error)
218                 {
219                     options.error(req.status, resp.error, resp.reason);
220                 }
221                 else
222                 {
223                     alert("The document could not be saved: " + resp.reason);
224                 }
225             }
226         });
227     },
228 
229     bulkSave: function(docs, options)
230     {
231         extend(options, {successStatus: 201});
232         this.ajax({
233             type: "POST",
234             url: this.uri + "_bulk_docs",
235             contentType: "application/json",
236             data: toJSON(docs),
237             complete: function(req)
238             {
239                 var resp = parseJSON(req.responseText);
240                 if (req.status == 201)
241                 {
242                     if (options.success)
243                         options.success(resp);
244                 }
245                 else if (options.error)
246                 {
247                     options.error(req.status, resp.error, resp.reason);
248                 }
249                 else
250                 {
251                     alert("The document could not be saved: " + resp.reason);
252                 }
253             },
254         });
255     },
256 
257     ajax: function(options)
258     {
259         try
260         {
261             var request = CCIN("@mozilla.org/xmlextras/xmlhttprequest;1", "nsIXMLHttpRequest");
262             request.open(options.type, options.url, true);
263             request.setRequestHeader("Content-Type", options.contentType);
264             request.onreadystatechange = function()
265             {
266                 if (request.readyState === 4)
267                 {
268                     if (options.complete)
269                         options.complete(request);
270                 }
271             }
272             request.send(options.data);
273         }
274         catch (e)
275         {
276             if (FBTrace.DBG_FBTEST || FBTrace.DBG_ERRORS)
277                 FBTrace.sysout("fbtest.TestCouchUploader; ajax EXCEPTION " + e, e);
278         }
279     }
