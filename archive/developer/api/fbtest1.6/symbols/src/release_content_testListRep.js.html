516 FBTestApp.TestGroup = function(name)
517 {
518     this.name = name;
519     this.tests = [];
520 }
521 
522 FBTestApp.TestGroup.prototype =
523 {
524     getErrors: function(includeMessages)
525     {
526         var text = "";
527         for (var i=0; i<this.tests.length; i++)
528         {
529             var test = this.tests[i];
530             var errors = test.getErrors(includeMessages);
531             if (errors)
532                 text += errors + "\n";
533         }
534         return text;
535     },
536 
537     getFailingTests: function()
538     {
539         var tests = [];
540         for (var i=0; i<this.tests.length; i++)
541         {
542             var test = this.tests[i];
543             if (!test.error || test.category == "fails")
544                 continue;
545             tests.push(test);
546         }
547         return tests;
548     },
549 
550     update: function()
551     {
552         var error = false;
553         for (var i=0; i<this.tests.length; i++)
554         {
555             var test = this.tests[i];
556             if (test.error && test.category != "fails")
557             {
558                 error = true;
559                 break;
560             }
561         }
562 
563         if (error)
564             setClass(this.row, "error");
565         else
566             removeClass(this.row, "error");
567     }
568 }

573 FBTestApp.Test = function(group, uri, desc, category, testPage)
574 {
575     if (category != "passes" && category != "fails")
576     {
577         if (FBTrace.DBG_ERRORS || FBTrace.DBG_FBTEST)
578             FBTrace.sysout("fbrace.FTestApp.Test; Wrong category for a test: " +
579                 category + ", " + uri);
580     }
581 
582     // Test definition.
583     this.group = group;
584     this.uri = uri;
585     this.desc = desc;
586     this.category = category;
587     this.testPage = testPage;
588 
589     // Used by the test runner.
590     this.results = [];
591     this.error = false;
592     this.row = null;
593     this.path = null;
594 
595     // Timing
596     this.start = 0;
597     this.end = 0;
598 
599     this.disabled = false;
600 }
601 
602 FBTestApp.Test.prototype =
603 {
604     appendResult: function(testResult)
605     {
606         this.results.push(testResult);
607 
608         // If it's an error update test so, it's reflecting an error state.
609         if (!testResult.pass)
610         {
611             setClass(this.row, "error");
612             this.error = true;
613         }
614     },
615 
616     onStartTest: function(baseURI)
617     {
618         this.path = baseURI + this.uri;
619         this.results = [];
620         this.error = false;
621 
622         setClass(this.row, "running");
623         removeClass(this.row, "results");
624         removeClass(this.row, "error");
625 
626         // Remove previous results from the UI.
627         if (hasClass(this.row, "opened"))
628         {
629             var infoBody = this.row.nextSibling;
630             clearNode(FBL.getElementByClass(infoBody, "testBodyCol"));
631         }
632 
633         // Clear time info
634         var timeNode = FBL.getElementByClass(this.row, "statusIcon");
635         clearNode(timeNode);
636         timeNode.removeAttribute("title");
637     },
638 
639     onTestDone: function()
640     {
641         removeClass(this.row, "running");
642         if (this.results.length > 0)
643             setClass(this.row, "results");
644 
645         var timeNode = FBL.getElementByClass(this.row, "statusIcon");
646         var elapsedTime = this.end - this.start;
647         timeNode.innerHTML = "(" + formatTime(elapsedTime) + ")";
648         timeNode.setAttribute("title", elapsedTime + "ms");
649 
650         // Update group error flag.
651         this.group.update();
652     },
653 
654     onManualVerify: function(verifyMsg, instructions)
655     {
656         removeClass(this.row, "running");
657     },
658 
659     getErrors: function(includeMessages)
660     {
661         if (!this.error || this.category == "fails")
662             return "";
663 
664         var text = "[FAILED] " + this.uri + ": " + this.desc;
665         if (!includeMessages)
666             return text;
667 
668         text += "\n";
669 
670         for (var i=0; i<this.results.length; i++)
671         {
672             var testResult = this.results[i];
673             text += "- " + testResult.msg + (testResult.pass ? "" : " [ERROR]") + "\n";
674         }
675         return text;
676     }
