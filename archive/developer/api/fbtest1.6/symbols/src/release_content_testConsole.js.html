401     refreshTestList: function()
402     {
403         if (!this.groups)
404         {
405             FBTrace.sysout("fbtest.refreshTestList; ERROR There are no tests.");
406             return;
407         }
408 
409         var browser = $("taskBrowser");
410         var doc = browser.contentDocument;
411         var testListNode = $("testList", doc);
412         if (!testListNode)
413         {
414             var iframed = $("FBTest", doc);
415             if (iframed)
416             {
417                 doc = iframed.contentDocument;
418                 testListNode = $("testList", doc);
419             }
420             if (!testListNode)
421             {
422                 testListNode = doc.createElement("div");
423                 testListNode.setAttribute("id", "testList");
424                 var body = getBody(doc);
425                 if (!body)
426                 {
427                     FBTrace.sysout("fbtest.refreshTestList; ERROR There is no <body> element.");
428                     return;
429                 }
430                 body.appendChild(testListNode);
431             }
432         }
433 
434         eraseNode(testListNode);
435 
436         // Generate UI (domplate).
437         var GroupList = FBTestApp.GroupList;
438         this.table = GroupList.tableTag.replace({}, testListNode);
439 
440         var tbody = this.table.firstChild;
441         var location = "";
442         for (var i=0; i<this.groups.length; i++)
443         {
444             var group = this.groups[i];
445 
446             if (location != group.testListPath)
447             {
448                 location = group.testListPath;
449 
450                 // Insert group separator. The 'extension' field is only set if the console displays
451                 // tests from multiple extensions. So, use it and only display the separator
452                 // in such cases.
453                 if (group.extension)
454                     GroupList.groupSeparatorTag.insertRows({group: group},
455                         tbody.lastChild ? tbody.lastChild : tbody);
456             }
457 
458             group.row = GroupList.groupRowTag.insertRows({group: group},
459                 tbody.lastChild ? tbody.lastChild : tbody)[0];
460         }

463     autoRun: function()
464     {
465         if (!cmdLineHandler.wrappedJSObject.runFBTests)
466         {
467             // Check pref if the auto logger should be registered. Useful for
468             // knowing, which test caused crash.
469             var enableTestLogger = Firebug.getPref(FBTestApp.prefDomain, "enableTestLogger");
470             if (enableTestLogger)
471             {
472                 var listener = new FBTestApp.TestLogger.ProgressListener(new Date());
473                 FBTestApp.TestRunner.addListener(listener);
474             }
475 
476             return;
477         }
478 
479         // The auto run is done just the first time the test-console is opened.
480         cmdLineHandler.wrappedJSObject.runFBTests = false;
481 
482         if (FBTrace.DBG_FBTEST)
483             FBTrace.sysout("fbtest.autoRun; defaultTestList: " + FBTestApp.defaultTestList +
484                 ", defaultTest: " + FBTestApp.defaultTest);
485 
486         // Run all asynchronously so, callstack is correct.
487         setTimeout(function()
488         {
489             // If a test is specified on the command line, run it. Otherwise
490             // run entire test suite.
491             if (FBTestApp.defaultTest)
492             {
493                 var test = FBTestApp.TestConsole.getTest(FBTestApp.defaultTest);
494                 if (test)
495                 {
496                     FBTestApp.TestRunner.runTests([test]);
497                 }
498                 else
499                 {
500                     throw new Error("fbtest.autoRun; Test from command line doesn't exist: " +
501                         FBTestApp.defaultTest);
502                 }
503             }
504             else
505             {
506                 // Register a listener that continuously logs test results so,
507                 // in case of a crash there is at least part of the log.
508                 var listener = new FBTestApp.TestLogger.ProgressListener(new Date());
509                 FBTestApp.TestRunner.addListener(listener);
510 
511                 FBTestApp.TestConsole.onRunAll(function(canceled)
512                 {
513                     // Don't forget to remove the logger listener now.
514                     FBTestApp.TestRunner.removeListener(listener);
515 
516                     // Quit Firefox now.
517                     if (!canceled)
518                         goQuitApplication();
519                 });
520             }
521         }, 100);
522     },

678     // Directories
679     chromeToPath: function (aPath)
680     {
681         try
682         {
683             if (!aPath || !(/^chrome:/.test(aPath)))
684                 return this.urlToPath( aPath );
685 
686             var ios = Cc['@mozilla.org/network/io-service;1'].getService(Ci["nsIIOService"]);
687             var uri = ios.newURI(aPath, "UTF-8", null);
688             var cr = Cc['@mozilla.org/chrome/chrome-registry;1'].getService(Ci["nsIChromeRegistry"]);
689             var rv = cr.convertChromeURL(uri).spec;
690 
691             if (/content\/$/.test(aPath)) // fix bug  in convertToChromeURL
692             {
693                 var m = /(.*\/content\/)/.exec(rv);
694                 if (m)
695                     rv = m[1];
696             }
697 
698             if (/^file:/.test(rv))
699                 rv = this.urlToPath(rv);
700             else
701                 rv = this.urlToPath("file://"+rv);
702 
703             return rv;
704         }
705         catch (err)
706         {
707             if (FBTrace.DBG_FBTEST)

715     {
716         try

719                 return;
720 
721             return Cc["@mozilla.org/network/protocol;1?name=file"]
722                       .createInstance(Ci.nsIFileProtocolHandler)
723                       .getFileFromURLSpec(aPath);
724         }
725         catch (e)

728         }

731     chromeToUrl: function (aPath, aDir)
732     {
733         try
734         {
735             if (!aPath || !(/^chrome:/.test(aPath)))
736                 return this.pathToUrl(aPath);
737 
738             var uri = ios.newURI(aPath, "UTF-8", null);
739             var rv = chromeRegistry.convertChromeURL(uri).spec;
740             if (aDir)
741                 rv = rv.substr(0, rv.lastIndexOf("/") + 1);
742 
743             if (/content\/$/.test(aPath)) // fix bug  in convertToChromeURL
744             {
745                 var m = /(.*\/content\/)/.exec(rv);
746                 if (m)
747                     rv = m[1];
748             }
749 
750             if (!/^file:/.test(rv))
751                 rv = this.pathToUrl(rv);
752 
753             return rv;
754         }
755         catch (err)
756         {
757             if (FBTrace.DBG_FBTEST)
758                 FBTrace.sysout("fbtest.chromeToUrl EXCEPTION", err);
759         }
760 
761         return null;
762     },

765     {

772             return Cc["@mozilla.org/network/protocol;1?name=file"]

774                 .getURLSpecFromFile(uri).spec;
775         }
776         catch (e)
777         {
778             throw new Error("urlToPath fails for "+aPath+ " because of "+e);
779         }
