  3 const PrefService = Components.classes["@mozilla.org/preferences-service;1"];
  4 const nsIPrefBranch2 = Components.interfaces.nsIPrefBranch2;
  5 const prefs = PrefService.getService(nsIPrefBranch2);
  6 const nsIPrefService = Components.interfaces.nsIPrefService;
  7 const prefService = PrefService.getService(nsIPrefService);
  8 
  9 var ChromeBugOpener =
 10 {
 11     openNow: function()
 12     {
 13         var opener = this.getCommandLineHandler().wrappedJSObject;
 14         if (opener)
 15             return opener.openNow(window);
 16         else
 17             window.dump("ChromeBugOpener: no wrappedJSObject in command line handler\n");
 18     },
 19 
 20     getCommandLineHandler: function()
 21     {
 22         if (!this.ChromeBugCommandLineHandler)
 23         {
 24             this.ChromeBugCommandLineHandler = Components.classes['@mozilla.org/commandlinehandler/general-startup;1?type=chromebug'].
 25                 getService(Components.interfaces.nsICommandLineHandler);
 26         }
 27 
 28         return this.ChromeBugCommandLineHandler;
 29     },
 30 
 31     setMenuByPref: function()
 32     {
 33         var menuitem = document.getElementById("menu_OpenChromeBugAlways");
 34         if (menuitem)
 35         {
 36             var openalways = Boolean(prefs.getBoolPref("extensions.chromebug.openalways"));
 37             menuitem.setAttribute("checked", openalways.toString() );
 38             //alert("set menuitem.checked to "+menuitem.getAttribute("checked"));
 39         }
 40         else
 41             window.dump("ChromeBugOpener: no element with id='menu_OpenChromeBugAlways'\n");
 42     },
 43 
 44     observe: function(subject, topic, data)
 45     {
 46         if (data == "extensions.chromebug.openalways")
 47             ChromeBugOpener.setMenuByPref();
 48     }
 49 }
 50 
 51 function ChromeBugOpenerOnLoad(event)
 52 {
 53     ChromeBugOpener.setMenuByPref();
 54     prefs.addObserver("extensions.chromebug", ChromeBugOpener, false);
 55     window.removeEventListener("load", ChromeBugOpenerOnLoad, false);
 56 }
 57 
 58 window.addEventListener("load", ChromeBugOpenerOnLoad, false);
