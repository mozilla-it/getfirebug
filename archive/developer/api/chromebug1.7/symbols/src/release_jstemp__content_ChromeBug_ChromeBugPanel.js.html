  3 /*
  4  * This file has a lot of experimental code, partly because I am unsure about the model and partly because so
  5  * much of this is unknown to anyone.
  6  *
  7  * For the most part Chromebug is a wrapper around Firebug. You can see from the UI it is Firebug with an extra
  8  * top toolbar.
  9  *
 10  * The biggest difference is that Firebug deals with neatly contained windows whose life time in contained with in
 11  * browser.xul. Chromebug deals multiple XUL windows and components, some with lifetimes similar to Chromebug itself.
 12  *
 13  * Firebug creates a meta-data object called "context" for each web page. (So it incorrectly mixes iframe info)
 14  * Chromebug creates a context for each nsIDOMWondow plus one for components.
 15  *
 16  * Most of Chromebug works on any XUL application. A few uses of DTD or CSS files from 'browser' causes problems
 17  * however.
 18  *
 19  */

340     /**
341      * This method is executed using setInterval at the end of Chromebug.initialize.
342      * It's purpose is to restore UI state, which must be done asynchronously and

345     restoreState: function()  // TODO context+file
346     {
347         // The restoration logic has limited number of attempts.
348         if (--this.retryRestoreCounter <= 0)
349         {
350             if (FBTrace.DBG_INITIALIZE)
351                 FBTrace.sysout("restoreState Stop restoration, we did our best.");
352 
353             this.stopRestoration();
354             return;
355         }
356 
357         var lastTry = this.retryRestoreCounter == 1;
358         var previousStateJSON = prefs.getCharPref("extensions.chromebug.previousContext");
359 
360         // Bail out if there is no previous state or it's an empty JSON object.
361         if (!previousStateJSON || previousStateJSON.length < 3)
362         {
363             if (FBTrace.DBG_INITIALIZE)
364                 FBTrace.sysout("restoreState NO previousStateJSON ");
365 
366             this.stopRestoration(); // no reason to beat our head against the wall...
367             return;
368         }
369 
370         // Also bail out if the parsing fails.
371         var previousState = parseJSONString(previousStateJSON, window.location.toString());
372         if (!previousState)
373         {
374             if (FBTrace.DBG_INITIALIZE)
375                 FBTrace.sysout("restoreState could not parse previousStateJSON "+previousStateJSON);
376 
377             this.stopRestoration();
378             return;
379         }
380 
381         if (FBTrace.DBG_INITIALIZE)
382             FBTrace.sysout("restoreState parse previousStateJSON "+previousStateJSON, previousState);
383 
384         // If the return value is non null, the context has been properly restored.
385         var context = this.restoreContext(previousState);
386         if (!context)
387             return;
388 
389         // Restore filter (toolbar popup menu).
390         var pkg = this.restoreFilter(previousState, context);
391 
392         // The file must exist in order to restore the file and it's scroll position.
393         if (previousState.sourceLink)
394         {
395             var sourceFile = getSourceFileByHref(previousState.sourceLink.href, context);
396             if (!sourceFile && !lastTry)
397                 return;
398         }
399 
400         // Chromebug UI seems to be ready for restoration so, remove the asynchronous interval.
401         this.stopRestoration();
402 
403         // Create real instance of the source link so, it has the correct type.
404         var sourceLink = previousState.sourceLink ? new SourceLink(previousState.sourceLink.href,
405             previousState.sourceLink.line, previousState.sourceLink.type) : null;
406 
407         var panelName = previousState.panelName;
408 
409         if (sourceLink)
410             FirebugChrome.select(sourceLink, panelName);
411         else if (panelName)
412             FirebugChrome.selectPanel(panelName);
413         else
414             FirebugChrome.selectPanel('trace');
415 
416         // Keep trying, this method will be asynchronously called again.
417     },

579     createContext: function(global, name)
580     {
581         var persistedState = null; // TODO
582         // domWindow in fbug is browser.contentWindow type nsIDOMWindow.
583         // docShell has a nsIDOMWindow interface
584         if (global instanceof Ci.nsIDOMWindow)
585         {
586             if (global.closed)
587                 return null;
588 
589             // When a XUL window is destroyed the destructor functions from XBL run after the unload event.
590             // We delete the context in the unload event, then the destructor can then trigger an new context creation.
591             // To avoid this we maintain a list of the windows we just destroyed then clean them up on a setTimeout
592             if (safeGetWindowLocation(global) in this.blockedWindowLocations)
593                 return null;
594 
595             var browser = Firebug.Chromebug.createBrowser(global, name);  // I guess this has side effects we need
596             var context = Firebug.TabWatcher.watchTopWindow(global, safeGetWindowLocation(global), true);
597 
598             // we want to write window.console onto the actual window, not the wrapper
599             if (global.wrappedJSObject && !global.wrappedJSObject.console)
600                 global.wrappedJSObject.console = Firebug.Console.createConsole(context, global);
601             else // we don't know what we are doing
602                 global.console = Firebug.Console.createConsole(context, global);
603 
604             var url = safeToString(global ? global.location : null);
605             if (isDataURL(url))
606             {
607                 var lines = context.sourceCache.load(url);
608                 var props = splitDataURL(url);
609                 if (props.fileName)
610                     context.sourceCache.storeSplitLines(props.fileName, lines);
611                 FBTrace.sysout("createContext data url stored in to context under "+(props.fileName?props.fileName+ " & ":"just dataURL ")+url);
612             }
613         }
614         else
615         {
616             var browser = Firebug.Chromebug.createBrowser(global, name);
617             var context = Firebug.TabWatcher.createContext(global, browser, Chromebug.DomWindowContext);
618         }
