 40     getDocShellByDOMWindow: function(domWindow)

 48            catch (exc)

191     eachDocShell: function(docShell, content_chrome_OR_all, iterator)
192     {
193         var treeItemType = Ci.nsIDocShellTreeItem.typeAll;
194         if (content_chrome_OR_all == "chrome")
195             treeItemType = Ci.nsIDocShellTreeItem.typeChrome;
196         else if (content_chrome_OR_all == "content")
197             treeItemType = Ci.nsIDocShellTreeItem.typeContent;
198         // From inspector@mozilla.org inspector.js appendContainedDocuments
199         // Load all the window's content docShells
200         var containedDocShells = docShell.getDocShellEnumerator(treeItemType,
201                                           Ci.nsIDocShell.ENUMERATE_FORWARDS);
202         while (containedDocShells.hasMoreElements())
203         {
204             try
205             {
206                 var childDocShell = containedDocShells.getNext().QueryInterface(Ci.nsIDocShell);
207                 iterator(childDocShell);
208             }
209             catch (exc)
210             {
211                 FBTrace.sysout("XULAppModule.eachDocShell FAILED", exc);
212             }
213         }

245     getDOMWindowTreeByDocShell: function(docShell)
246     {
247         var domWindow = Chromebug.XULAppModule.getDOMWindowByDocShell(docShell);
248         var url = safeGetWindowLocation(domWindow);
249 
250         var domWindowsByURL = {};
251 
252         this.eachDocShell(docShell, "all", function visitSubShells(childShell)
253         {
254             if (childShell == docShell)
255             {
256                 domWindowsByURL[url] = domWindow;
257                 return;
258             }
259 
260             if (!domWindowsByURL)
261                 domWindowsByURL = {};
262 
263             var childDOMWindow = Chromebug.XULAppModule.getDOMWindowByDocShell(childShell);
264 
265             var key = Chromebug.XULAppModule.getKeyByDOMWindow(childDOMWindow, domWindowsByURL);
266 
267             var familyTree = Chromebug.XULAppModule.getDOMWindowTreeByDocShell(childShell);  // recurse
268 
269             domWindowsByURL[key] = familyTree; // either a window or a table
270         });
271 
272         return domWindowsByURL;

275     getKeyByDOMWindow: function(domWindow, domWindowsByURL)
276     {
277         var url = safeGetWindowLocation(domWindow);
278         var title = domWindow.document.title;
279         var key = url +" - "+title;
280         if (!key)
281                 key = "(no location)";
282 
283         while (key in domWindowsByURL)
284                 key = key +".";
285 
286         return key;

289     getDOMWindowTree: function()
290     {
291         var domWindowsByURL = {};
292 
293         var enumerator = windowMediator.getXULWindowEnumerator(null);  // null means all
294         while(enumerator.hasMoreElements())
295         {
296              var xul_window = enumerator.getNext();
297              if (xul_window instanceof Ci.nsIXULWindow)
298              {
299                  if (xul_window.docShell)
300                  {
301                      var domWindow = this.getDOMWindowByXULWindow(xul_window);
302                      var familyTree = this.getDOMWindowTreeByXULWindow(xul_window);  // recurse
303 
304                      var key = this.getKeyByDOMWindow(domWindow, domWindowsByURL);
305                      domWindowsByURL[key] = familyTree; // either a window or a table
306                  }
307                  else
308                      FBTrace.sysout("A XUL Window without a docShell??", xul_window);
309              }
310              else
311              {
312                  FBTrace.sysout("getXULWindowEnumerator gave element that was not nsIXULWindow!", xul_window);
313              }
314          }
315         return domWindowsByURL;
316     },

411         if (FBTrace.DBG_CHROMEBUG)
412             FBTrace.sysout("Chromebug.XULAppModule.addXULWindow "+outerDOMWindow.location.href+ " complete length="+this.xulWindows.length, " index="+this.getXULWindowIndex(xul_window));

461     cleanUpXULWindow: function(xul_win)
462     {
463         try
464         {
465             if (xul_win instanceof Ci.nsIXULWindow)
466             {
467                 var mark = this.getXULWindowIndex(xul_win);
468                 if (mark == -1)   // A window closed but we don't know which one.
469                 {
470                     // https://bugzilla.mozilla.org/show_bug.cgi?id=325636
471                     var SIP=Components.Constructor("@mozilla.org/supports-interface-pointer;1",
472                         Ci.nsISupportsInterfacePointer);
473                     for (var i = 0; i < this.xulWindows.length; i++)
474                     {
475                         var ptr = new SIP;
476                         ptr.data = xul_win;
477                         if (ptr.data == this.xulWindows[i])
478                         {
479                             mark = i;
480                             if (FBTrace.DBG_CHROMEBUG)
481                                 FBTrace.sysout("XULAppModule.onclose: timeless nsISupportsInterfacePointer found mark="+mark+"\n");
482                             break;
483                         }
484                     }
485                 }
486                 if (mark != -1)
487                 {
488                     if (FBTrace.DBG_CHROMEBUG)
489                         FBTrace.sysout("XULAppModule.onclose: removing getXULWindowIndex="+mark);
490 
491                     var tag = this.xulWindowTags[mark];
492                     this.xulWindows.splice(mark,1);
493                     this.xulWindowTags.splice(mark,1);
494                     var outerDOMWindow = this.getDOMWindowByXULWindow(xul_win);
495                     FBTrace.sysout("XULAppModule.onclose: removing getXULWindowIndex="+mark+" with outerDOMWindow "+outerDOMWindow);
496                     Firebug.TabWatcher.unwatchTopWindow(outerDOMWindow);
497                 }
498                 else
499                 {
500                     var outerDOMWindow = this.getDOMWindowByXULWindow(xul_win);
501                     var url = new String(outerDOMWindow.location);
502                     if (reChromebug.test(url))
503                         return; // ignore self
504                     FBTrace.sysout("XULAppModule.onclose: xul_window is unknown to us at location "+outerDOMWindow.location);
505                 }
506              }
507              else
508                  FBTrace.sysout("XULAppModule.onclose: not a nsIXULWindow");
509         }
510         catch(e)
511         {
512             FBTrace.sysout("XULAppModule.onClose fails ", e);
513             throw "NO, do not exit";
514         }
515     },

540     reloadWindow: function(xul_window)
541     {
542         var outerDOMWindow = Chromebug.XULAppModule.getDOMWindowByXULWindow(xul_window);
543         if (outerDOMWindow && outerDOMWindow instanceof Ci.nsIDOMWindow)
544         {
545             try
546             {
547                 if (!this.seesionStore)
548                 {
549                     this.sessionStore = Components.classes["@mozilla.org/browser/sessionstore;1"].
550                         getService(Ci.nsISessionStore);
551                 }
552                 var storedState = sessionStore.getWindowState(outerDOMWindow);
553                 var ss = sessionStore;
554                 // save until the window is ready for state
555                 this.stateReloader = function(event)
556                 {
557                     var windowToBeRestored = event.currentTarget;
558                     windowToBeRestored.dump("setWindowState for "+windowToBeRestored.location+" to "+storedState+"\n");
559                     windowToBeRestored.removeEventListener("DOMContentLoaded", Chromebug.XULAppModule.stateReloader, "true");
560                     sessionStore.setWindowState(windowToBeRestored, storedState, true);
561                     delete Chromebug.XULAppModule.stateReloader;
562                 }
563             }
564             catch (exc)
565             {
566                 var ssEnabled = prefs.getBoolPref("browser.sessionstore.enabled");
567                 FBTrace.sysout("Firebug.Chromebug.reloadWindow FAILS with browser.sessionstore.enabled= "+ssEnabled, exc);
568             }
569 
570             FBTrace.sysout("ChromeBug reloadWindow closing outerDOMWindow\n");
571             outerDOMWindow.close();
572             FBTrace.sysout("ChromeBug reloadWindow opening new window\n");
573             var ff = window.open();
574             return ff;
575         }
576         else
577             FBTrace.sysout("XULAppModule.reload, no domWindow for xul_window_tag:"+xul_window_tag+"\n");
578         return false;
579     },
