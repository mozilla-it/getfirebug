 20 Chromebug.parseWebURI = function(uri)
 21 {
 22     var m = reWeb.exec(uri);
 23     if(m)
 24     {
 25         var split = FBL.splitURLBase(uri);
 26         return {path: m[1], name: split.path+'/'+split.name, kind:"web", pkgName: m[1]};
 27     }

 39 Chromebug.parseNoWindowURI = function(uri)
 40 {
 41     if (uri.indexOf('noWindow')==0)
 42     {
 43         var sandbox = uri.indexOf('Sandbox');
 44         if (sandbox > 0)
 45             return {path: "Sandbox", name: uri.substr(11), kind: "sandbox", pkgName: "Sandbox"};
 46 
 47         return {path: uri.substr(0,9), name: uri.substr(11), kind: "noWindow", pkgName: "noWindow" }
 48     }

 51 Chromebug.parseDataURI = function(URI)
 52 {
 53     if (isDataURL(URI))
 54     {
 55         var split = splitURLBase(URI);

 58             FBTrace.sysout("parseDataURI "+URI, split);
 59         return {path: "data:", name: split.path+'/'+split.name, kind:"data", pkgName: "data:"};
 60     }

 63 Chromebug.parseComponentURI = function(URIString)
 64 {
 65     var m = reComponents.exec(URIString);
 66     if (m)

 68         return { path: "components", pkgName: m[1], name: new String(URIString), href: URIString, kind: 'component' };
 69     }
 70     else

 91 Chromebug.parseExtensionURI = function(URIString)
 92 {
 93     const appURLStem = Firebug.Chromebug.getPlatformStringURL("resource:app");
 94 
 95     var m = FBL.reChrome.exec(URIString) || reExtensionInFileURL.exec(URIString) || reResource.exec(URIString);
 96     var pkgName, remainder;
 97     if (m)
 98     {
 99         pkgName = m[1];
100         remainder = m[0].length;
101     }
102     else
103     {
104         if (URIString && URIString.indexOf(appURLStem) == 0)
105         {
106             pkgName = "application";
107             remainder = appURLStem.length;
108         }
109         // else not one of ours
110         return null;
111     }
112     return {path: pkgName, name: new String(URIString.substr(remainder)), pkgName: pkgName, href: URIString, kind:'extension'};
113 };

116 Chromebug.parseURI = function(URI)
117 {
118     if (!URI || Firebug.Chromebug.isChromebugURL(URI))
119         return null;
120 
121     var description = null;
122     if (!description)
123         description = Chromebug.parseNoWindowURI(URI);
124     if (!description)
125         description = Chromebug.parseComponentURI(URI);
126     if (!description)
127         description = Chromebug.parseExtensionURI(URI);
128     if (!description)
129         description = Chromebug.parseModuleURI(URI);
130     if (!description)
131         description = Chromebug.parseSystemURI(URI);
132     if (!description)
133         description = Chromebug.parseWebURI(URI);
134     if (!description)
135         description = Chromebug.parseDataURI(URI);
136 
137     if (!description)
138     {
139         if (FBTrace.SOURCEFILES)
140             FBTrace.sysout("Chromebug.parseURI: no match for "+URI);
141         description = {path:"mystery", name:URI, kind: "mystery", pkgName: "unparsable"};
142     }
143 
144     return description;
145 }

156     getDescription: function(sourceFile)

179     getPackageNames: function()
180     {
181         var slots = {};
182         this.eachSourceFileDescription(function extractPackageNames(d)
183         {
184             slots[d.pkgName] = 1;
185         });
186         var list = [];
187         for (var p in slots)
188             if (slots.hasOwnProperty(p))
189                 list.push(p);
190         return list;

196         var rc = Firebug.Chromebug.eachSourceFile(function visitSourceFiles(sourceFile)
197         {
198             var d = getDescription(sourceFile);
199             if ( d )
200             {
201                 var rc = fnTakesSourceFileDescription(d);
202                 if (rc)
203                     return rc;
204             }
205         });
206         return rc;
207     },

313             this.setCurrentLocation(description);
314         }
315         else
316             FBTrace.sysout("AllFilesList.onSelectLocation no context in description"+description, description);
317     }
