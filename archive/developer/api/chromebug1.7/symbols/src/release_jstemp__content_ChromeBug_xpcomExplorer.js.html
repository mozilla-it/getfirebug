 13 XPCOMExplorer.categoryListLocator = function(xul_element)
 14 {
 15     if (!XPCOMExplorer.categoryList)
 16     {
 17         XPCOMExplorer.categoryList = {
 18             elementBoundTo: xul_element,
 19 
 20             getLocationList: function()
 21             {
 22                 if (!this.catman)
 23                     this.catman = Components.classes["@mozilla.org/categorymanager;1"].getService(Components.interfaces.nsICategoryManager);
 24 
 25                 var list = [];
 26                 var categories = this.catman.enumerateCategories();
 27                 while( categories.hasMoreElements() )
 28                 {
 29                     var category =  categories.getNext().QueryInterface(nsISupportsCString);
 30                     var entries = this.catman.enumerateCategory(category);
 31                     while( entries.hasMoreElements() )
 32                     {
 33                         var entry = entries.getNext().QueryInterface(nsISupportsCString);
 34                         list.push(category + "/" + entry);
 35                     }
 36                 }
 37                 return list;
 38             },
 39 
 40             getDefaultLocation: function()
 41             {
 42                 var locations = this.getLocationList();
 43                 if (locations && locations.length > 0) return locations[0];
 44             },
 45 
 46             getObjectLocation: function(categorySlashEntry)
 47             {
 48                 return categorySlashEntry;
 49             },
 50 
 51             getObjectDescription: function(categorySlashEntry)
 52             {
 53                 var c = categorySlashEntry.indexOf('/');
 54                 if (c != -1)
 55                     return { path: categorySlashEntry.substr(0, c), name: categorySlashEntry.substr(c+1)};
 56                 else
 57                     return { path: "error", name: categorySlashEntry};
 58             },
 59 
 60             onSelectLocation: function(event)
 61             {
 62                 var categorySlashEntry = event.currentTarget.repObject;
 63                 if (categorySlashEntry)
 64                 {
 65                     var d = XPCOMExplorer.categoryList.getObjectDescription(categorySlashEntry);
 66                     Firebug.Console.log(d);
 67                     var value = XPCOMExplorer.categoryList.catman.getCategoryEntry(d.path, d.name);
 68                     Firebug.Console.log(categorySlashEntry+": "+value); // category/entry
 69                 }
 70                 else
 71                     FBTrace.sysout("onSelectLocation FAILED, no repObject in currentTarget", event.currentTarget);
 72             }
 73         }
 74         xul_element.addEventListener("selectObject", XPCOMExplorer.categoryList.onSelectLocation, false);
 75     }
 76     return XPCOMExplorer.categoryList;

222     getLocationList: function()

230         var locations = this.getLocationList();
231         if (locations && locations.length > 0) return locations[0];

233 
234     getObjectLocation: function(jscontext)
235     {
236         if (!jscontext)
237             FBTrace.sysout("getObjectLocation for nothing ");
238         var global = jscontext.globalObject ? new XPCNativeWrapper(jscontext.globalObject.getWrappedValue()) : null;
239         if (global)
240         {
241             var document = global.document;
242             if (document)
243             {
244                 var location = document.location.toString();
245                 var rootWindow = getRootWindow(global);
246                 if (rootWindow && rootWindow.location)
247                     return location +" < "+rootWindow.location.toString();
248                 else
249                     return location;
250             }
251             else
252                 return "noDocument://"+jscontext.tag;
253             }
254         else
255             return "noGlobal://"+jscontext.tag;
256     },
257 
258     getObjectDescription: function(jscontext)
259     {
260         var URI = this.getObjectLocation(jscontext);
261         if (!URI)
262             return {path: "no URI", name: "no URI"};
263         if (Firebug.Chromebug.isChromebugURL(URI))
264             var d = {path:"avoided chromebug", name: URI};
265         if (!d)
266             var d = Chromebug.parseURI( URI );
267         if (!d)
268             d = {path:"unparsable", name: this.getObjectLocation(jscontext)};
269         d.name = d.name +" ("+jscontext.tag+")";
270         return d;
271     },
272 
273     getContextByLocation: function(location)  // TODO MOVE to context list
274     {
275         for (var i = 0; i < Chromebug.XULAppModule.contexts.length; ++i)
276         {
277             var context = Chromebug.XULAppModule.contexts[i];
278             try
279             {
280                 if (context.window.location == location)
281                     return context;
282             }
283             catch(e)
284             {
285                 //? Exception... "Unexpected error arg 0 [nsIDOMWindowInternal.location]"
286                 FBTrace.sysout("ChromeBugPanel.getContextByLocation: ignoring ", e);
287             }
288         }
289     },
290 
291     onSelectLocation: function(event)
292     {
293         var object = event.currentTarget.repObject;
294         if (object)
295         {
296             var jscontext = object;
297             var global = new XPCNativeWrapper(jscontext.globalObject.getWrappedValue());
298             if (global)
299             {
300                 var context = Firebug.Chromebug.getContextByGlobal(global)
301                 if (context)

303                     Firebug.Chromebug.selectContext(context);
304                 }
305                 else
306                 {
307                     FBTrace.sysout("onSelectLocation no context, showing global");
308                     FirebugChrome.select(global, "DOM", null, true);
309                 }
310             }
311             else
312             {
313                 FBTrace.sysout("XPCOMExplorer.jsContextList onSelectLocation: FAILED to no globalObject in jscontext\n");
314                 FirebugChrome.select(object, "script", null, true);  // jscontext
315             }
316         }
317         else
