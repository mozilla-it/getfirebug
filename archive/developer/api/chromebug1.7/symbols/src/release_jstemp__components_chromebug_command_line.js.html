 41     openWindow: function(opener, windowType, url, w, h, params)
 42     {
 43         var wm = Components.classes["@mozilla.org/appshell/window-mediator;1"].
 44                     getService(Components.interfaces.nsIWindowMediator);
 45 
 46         var win = windowType ? wm.getMostRecentWindow(windowType) : null;
 47 
 48         if (win)
 49         {
 50             if ("initWithParams" in win)
 51                   win.initWithParams(params);
 52             win.focus();
 53             Components.utils.reportError("ChromebugCommandLineHandler reused a window");
 54         }
 55         else
 56         {
 57             if (!w)
 58                 w = opener.screen.availWidth;
 59             if (!h)
 60                 h = opener.screen.availHeight;
 61 
 62             if (!w || w < 200) // see issue 2576
 63                 w = 400;
 64             if (!h || h < 200)
 65                 h = 300;
 66 
 67             var features = "outerWidth="+w+","+"outerHeight="+h;
 68 
 69             var winFeatures = "resizable,dialog=no,centerscreen" + (features != "" ? ("," + features) : "");
 70             if (commandLineHandler.debug)
 71             {
 72                 Components.utils.reportError("chromebug_command_line opening window with features: "+features);
 73             }
 74             //opener.dump("========= open "+url+" with "+params+"\n");
 75             //for (var p in params)
 76             //    opener.dump("========= "+p+" = "+params[p]+"\n");
 77             win = opener.openDialog(url, "_blank", winFeatures, params);
 78         }
 79         return win;
 80     },

 82     openChromebug: function(window)
 83     {
 84         var inType = "chromebug:ui"; // MUST BE windowType on chromebug.xul
 85         var url = "chrome://chromebug/content/chromebug.xul";
 86         var title = "ChromeBug";
 87 
 88         var prefedWidth = prefs.getIntPref("extensions.chromebug.outerWidth");
 89         var prefedHeight = prefs.getIntPref("extensions.chromebug.outerHeight");
 90 
 91         var releaser =
 92         {
 93             url: "chrome://chromebug/content/blocker.xul",
 94             unblock: null, // set by blocker.xul, called by chromebug.xul
 95         }
 96 
 97         var chromeBugWindow = this.openWindow(window, inType, url, prefedWidth, prefedHeight, releaser);
 98         chromeBugWindow.document.title = title;
 99 
100         if (commandLineHandler.debug)
101         {
102             var chromeURI = iosvc.newURI(url, null, null);
103             var localURI = chromeReg.convertChromeURL(chromeURI);
104             Components.utils.reportError(title+" maps "+url+' to '+localURI.spec);
105             Components.utils.reportError("ChromeBug x,y,w,h = ["+chromeBugWindow.screenX+","+chromeBugWindow.screenY+","+
106                chromeBugWindow.width+","+chromeBugWindow.height+"]");
107         }
108 
109         var winFeatures = "modal,resizable,dialog=no,centerscreen";
110 
111         win = window.openDialog(releaser.url, "_blank", winFeatures, releaser);
112 
113         return chromeBugWindow;
114     },

120     handle : function clh_handle(cmdLine)
121     {
122         window = appShellService.hiddenDOMWindow;
123         try
124         {
125             window.dump("Chromebug Command Line Handler taking arguments from state:"+cmdLine.state+"\n");
126             for (var i = 0; i < cmdLine.length; i++)
127                 window.dump("Chromebug Command Line Handler arguments on cmdLine: "+cmdLine.length+"."+i+")"+cmdLine.getArgument(i)+"\n");
128 
129             if (cmdLine.state == cmdLine.STATE_REMOTE_AUTO) // FF is already running
130             {
131                 var skipChrome = cmdLine.handleFlagWithParam("chromebug", false); // take ourselves out
132                 window.dump("Chromebug Command Line Handler removing chrome arguments from command line:"+skipChrome+"\n");
133                 var noProfile = cmdLine.handleFlagWithParam("p", false); // remove annoying messages about -p
134                 Components.utils.reportError("Chromebug detected that the application is already running.");
135             }
136             else  // New chromebug that may launch FF
137             {
138                 try
139                 {
140                     var launchChromebug = cmdLine.handleFlag("chromebug", false);
141                     if (launchChromebug)
142                     {
143                         prefs.setBoolPref("extensions.firebug.filterSystemURLs", false);  // See firebug-service.js
144                         prefs.setBoolPref("extensions.chromebug.launch", true);
145                         Components.utils.reportError("Chromebug command line sees -chromebug");
146                         commandLineHandler.openChromebug(window);
147                     }
148                     else
149                     {
150                         prefs.setBoolPref("extensions.chromebug.launch", false);
151                         Components.utils.reportError("Chromebug command no -chromebug");
152                     }
153                 }
154                 catch (e)
155                 {
156                     Components.utils.reportError("Failed to process command line because of: "+e);
157                 }
158             }
159         }
160         catch (e)
161         {
162             Components.utils.reportError("Chromebug Command Line Handler FAILS: "+e);
163             return;
164         }
165     },

217 function getTmpFile()
218 {
219     var file = Components.classes["@mozilla.org/file/directory_service;1"].
220         getService(Components.interfaces.nsIProperties).
221         get("TmpD", Components.interfaces.nsIFile);
222     file.append("fbs.tmp");
223     file.createUnique(Components.interfaces.nsIFile.NORMAL_FILE_TYPE, 0666);
224     appShellService.hiddenDOMWindow.dump("cbcl opened tmp file "+file.path+"\n");
225     return file;
226 }

228 function getTmpStream(file)
229 {
230     // file is nsIFile, data is a string
231     var foStream = Components.classes["@mozilla.org/network/file-output-stream;1"].
232                              createInstance(Components.interfaces.nsIFileOutputStream);
233 
234     // use 0x02 | 0x10 to open file for appending.
235     foStream.init(file, 0x02 | 0x08 | 0x20, 0666, 0);
236     // write, create, truncate
237     // In a c file operation, we have no need to set file mode with or operation,
238     // directly using "r" or "w" usually.
239 
240     return foStream;
241 }

245 function tmpout(text)
246 {
247     if (!fbs.foStream)
248         fbs.foStream = getTmpStream(getTmpFile());
249 
250     fbs.foStream.write(text, text.length);
