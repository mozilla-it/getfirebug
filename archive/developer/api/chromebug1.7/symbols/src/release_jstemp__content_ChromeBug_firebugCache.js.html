 31         if (FBTrace.DBG_FBCACHE)

 36     {
 37         if (FBTrace.DBG_FBCACHE)
 38             FBTrace.sysout("fbcache.initContext; " + context.name);

 42     addStyleSheet: function()
 43     {
 44         var panelType = Firebug.getPanelType(panelName);
 45         var doc = FirebugChrome.getPanelDocument(panelType);
 46 
 47         // Make sure the stylesheet isn't appended twice.
 48         if ($("cbFBCache", doc))
 49             return;
 50 
 51         var styleSheet = createStyleSheet(doc, "chrome://chromebug/skin/firebugCache.css");
 52         styleSheet.setAttribute("id", "cbFBCache");
 53         addStyleSheet(doc, styleSheet);
 54     },
 55 
 56     // UI Commands
 57     refresh: function(context)
 58     {
 59         var panel = context.getPanel(panelName, false);
 60         panel.refresh();
 61     }

 70 Firebug.Chromebug.FBCachePanel = function() {}

 72 Firebug.Chromebug.FBCachePanel.prototype = extend(Firebug.Panel,
 73 {
 74     name: panelName,
 75     title: "Firebug Cache",
 76     searchable: false,
 77     editable: false,
 78 
 79     initializeNode: function(panelNode)
 80     {
 81         if (FBTrace.DBG_FBCACHE)
 82             FBTrace.sysout("fbcache.FBCachePanel.initializeNode; " + this.context.name);
 83 
 84         // Refresh panel content.
 85         this.refresh();
 86     },
 87 
 88     refresh: function()
 89     {
 90         if (FBTrace.DBG_FBCACHE)
 91             FBTrace.sysout("fbcache.FBCachePanel.refresh; " + this.context.name, this.context);
 92 
 93         clearNode(this.panelNode);
 94 
 95         // Append custom stylesheet for this panel.
 96         Module.addStyleSheet();
 97 
 98         // If the current browser window (associated with the context) has TabWatcher
 99         // display its cache entries. Otherwise bail out.
100         var tabWatcher = this.context.browser.contentWindow.Firebug.TabWatcher;
101         if (!tabWatcher)
102         {
103             if (FBTrace.DBG_FBCACHE)
104                 FBTrace.sysout("fbcache.FBCachePanel.refresh; NO TAB-WATCHER " +
105                     this.context.name, this.context);
106 
107             // Display info about no Firebug in the current window instance.
108             Module.Message.render(this.panelNode, "Firebug is not embedded in: " +
109                 this.context.name + " - " + this.context.getTitle()); //xxxHonza: localization
110             return;
111         }
112 
113         // Get all caches from the browser window (there is one cache per tab).
114         var cacheList = [];
115         tabWatcher.iterateContexts(function(context)
116         {
117             if (FBTrace.DBG_FBCACHE)
118                 FBTrace.sysout("fbcache.refresh; context found: " + context.name, context);
119 
120             cacheList.push(new Module.TabCache(context.name, context.sourceCache));
121         });
122 
123         // The cache is empty.
124         if (!cacheList.length)
125         {
126             if (FBTrace.DBG_FBCACHE)
127                 FBTrace.sysout("fbcache.refresh; empty cache: " + this.context.name, this.context);
128 
129             // Display info about empty cache in the current window instance.
130             Module.Message.render(this.panelNode, "Firebug cache is empty for: " +
131                 this.context.name + " - " + this.context.getTitle()); //xxxHonza: localization
132             return;
133         }
134 
135         // Insert all existing caches into the UI.
136         Module.CacheList.tableTag.replace({cacheList: cacheList}, this.panelNode);
137 
138         // Append also a summary info.
139         var table = this.panelNode.firstChild;
140         var tbody = table.firstChild;
141         Module.CacheList.summaryTag.insertRows({cacheList: cacheList}, tbody.lastChild);
142     },
143 
144     destroyNode: function()
145     {
146         if (FBTrace.DBG_FBCACHE)
147             FBTrace.sysout("fbcache.FBCachePanel.destroyNode;");
148     },
149 
150     show: function(state)
151     {
152         if (FBTrace.DBG_FBCACHE)
153             FBTrace.sysout("fbcache.FBCachePanel.show;", state);
154 
155         this.showToolbarButtons("cbFBCacheButtons", true);
156     },
157 
158     hide: function()
159     {
160         if (FBTrace.DBG_FBCACHE)
161             FBTrace.sysout("fbcache.FBCachePanel.hide;");
162 
163         this.showToolbarButtons("cbFBCacheButtons", false);
164     },
165 
166     getOptionsMenuItems: function()
167     {
168          if (this.controller)
169             return this.controller.getOptionsMenuItems();
170 
171          return [];
172     },
173 
174     onPrefChange: function(optionName, optionValue)
175     {
176         if (FBTrace.DBG_FBCACHE)
177             FBTrace.sysout("fbcache.FBCachePanel.onPrefChange; " + optionName + "=" + optionValue);
178     },

219 /**
220  * This template represents list of caches in associated browser window.
221  * Each tab in that window maintains its own cache.
222  */
223 Firebug.Chromebug.FBCacheModule.CacheList = domplate(Firebug.Rep,
224 {
225     tableTag:
226         TABLE({"class": "cacheListTable", cellpadding: 0, cellspacing: 0, onclick: "$onClick"},
227             TBODY(
228                 FOR("cache", "$cacheList",
229                     TR({"class": "cacheListRow", _repObject: "$cache"},
230                         TD({"class": "cacheNameCol cacheListCol"},
231                             SPAN({"class": "cacheName"},
232                                 "$cache|getCacheName"
233                             ),
234                             SPAN({"class": "cacheSize"},
235                                 "$cache|getCacheSize"
236                             )
237                         )
238                     )
239                 )
240             )
241         ),
242 
243     cacheBodyTag:
244         TR({"class": "cacheBodyRow", _repObject: "$cache"},
245             TD({"class": "cacheBodyCol", colspan: 1})
246         ),
247 
248     summaryTag:
249         TR({"class": "summaryRow"},
250             TD({"class": "totalSize summaryCol"},
251                 SPAN("Total Size: $cacheList|getTotalSize")
252             )
253         ),
254 
255     getCacheName: function(cache)
256     {
257         return cache.name;
258     },
259 
260     getCacheSize: function(cache)
261     {
262         if (!cache.entries)
263         {
264             cache.size = 0;
265             cache.entries = [];
266             for (var url in cache.sourceCache.cache)
267             {
268                 var entry = new Module.CacheEntry(url, cache.sourceCache.cache[url]);
269                 cache.size += entry.size;
270                 cache.entries.push(entry);
271             }
272         }
273 
274         return formatSize(cache.size);
275     },
276 
277     getTotalSize: function(cacheList)
278     {
279         var totalSize = 0;
280         for (var i=0; i<cacheList.length; i++)
281             totalSize += cacheList[i].size;
282         return formatSize(totalSize);
283     },
284 
285     onClick: function(event)
286     {
287         if (isLeftClick(event))
288         {
289             var row = getAncestorByClass(event.target, "cacheListRow");
290             if (row)
291             {
292                 this.toggleRow(row);
293                 cancelEvent(event);
294             }
295         }
296     },
297 
298     expandGroup: function(row)
299     {
300         if (hasClass(row, "cacheListRow"))
301             this.toggleRow(row, true);
302     },
303 
304     collapseGroup: function(row)
305     {
306         if (hasClass(row, "cacheListRow") && hasClass(row,"opened"))
307             this.toggleRow(row);
308     },
309 
310     toggleRow: function(row, forceOpen)
311     {
312         var opened = hasClass(row, "opened");
313         if (opened && forceOpen)
314             return;
315 
316         toggleClass(row, "opened");
317         if (hasClass(row, "opened"))
318         {
319             var cache = row.repObject;
320             var infoBodyRow = this.cacheBodyTag.insertRows({cache: cache}, row)[0];
321             infoBodyRow.repObject = cache;
322             this.initBody(infoBodyRow);
323         }
324         else
325         {
326             var infoBodyRow = row.nextSibling;
327             row.parentNode.removeChild(infoBodyRow);
328         }
329     },
330 
331     initBody: function(infoBodyRow)
332     {
333         var tabCache = infoBodyRow.repObject;
334 
335         var entries = [];
336         for (var url in tabCache.sourceCache.cache)
337             entries.push(new Module.CacheEntry(url, tabCache.sourceCache.cache[url]));
338 
339         if (!entries.length)
340         {
341             infoBodyRow.firstChild.innerHTML = "There are no cache entries";
342             return;
343         }
344 
345         // Create basic structure for list of cache entries
346         // and insert all cache entries into it.
347         var table = Module.CacheTable.tableTag.replace({}, infoBodyRow.firstChild);
348         Module.CacheTable.resultTag.insertRows({entries: entries}, table.firstChild);
349     }
350 });

358 Firebug.Chromebug.FBCacheModule.CacheTable = domplate(Firebug.Rep,
359 {
360     tableTag:
361         TABLE({"class": "fbCacheTable", cellpadding: 0, cellspacing: 0, onclick: "$onClick"},
362             TBODY()
363         ),
364 
365     resultTag:
366         FOR("entry", "$entries",
367             TR({"class": "cacheEntryRow", _repObject: "$entry"},
368                 TD({"class": "cacheEntryCol", width: "100%"},
369                     SPAN({"class": "cacheEntryURL cacheEntryLabel"},
370                         "$entry|getURL"
371                     ),
372                     SPAN({"class": "cacheEntrySize cacheEntryLabel"},
373                         "$entry|getSize"
374                     )
375                 )
376             )
377         ),
378 
379     resultInfoTag:
380         TR({"class": "cacheEntryInfoRow", _repObject: "$entry"},
381             TD({"class": "cacheEntryInfoCol", colspan: 2})
382         ),
383 
384     getURL: function(entry)
385     {
386         return entry.url;
387     },
388 
389     getSize: function(entry)
390     {
391         try
392         {
393             // xxxHonza: localization
394             var numLines = entry.lines.length;
395             return formatSize(entry.size) + " (" + numLines + " " +
396                 (numLines > 1 ? "lines" : "line") + ")";
397         }
398         catch (err)
399         {
400             if (FBTrace.DBG_FBCACHE)
401                 FBTrace.sysout("fbcache.CacheList.getSize; ", err);
402         }
403     },
404 
405     onClick: function(event)
406     {
407         if (isLeftClick(event))
408         {
409             var row = getAncestorByClass(event.target, "cacheEntryRow");
410             if (row)
411             {
412                 this.toggleResultRow(row);
413                 cancelEvent(event);
414             }
415         }
416     },
417 
418     toggleResultRow: function(row)
419     {
420         var entry = row.repObject;
421 
422         toggleClass(row, "opened");
423         if (hasClass(row, "opened"))
424         {
425             var infoBodyRow = this.resultInfoTag.insertRows({entry: entry}, row)[0];
426             this.initInfoBody(infoBodyRow);
427         }
428         else
429         {
430             var infoBodyRow = row.nextSibling;
431             row.parentNode.removeChild(infoBodyRow);
432         }
433     },
434 
435     initInfoBody: function(infoBodyRow)
436     {
437         var entry = infoBodyRow.repObject;
438         var TabView = Firebug.Chromebug.FBCacheModule.CacheEntryTabView;
439         var tabViewNode = TabView.viewTag.replace({entry: entry}, infoBodyRow.firstChild, TabView);
440 
441         // Select default tab.
442         TabView.selectTabByName(tabViewNode, "Data");
443     },
444 });

451 Firebug.Chromebug.FBCacheModule.CacheEntryTabView = domplate(Firebug.Rep,

453     listeners: [],

456         TABLE({"class": "tabView", cellpadding: 0, cellspacing: 0},

458                 TR({"class": "tabViewRow"},

466     tabList:
467         DIV({"class": "tabViewBody"},

469             TAG("$tabBodies")
470         ),
471 
472     // List of tabs
473     tabBar:
474         DIV({"class": "tabBar"},
475             A({"class": "DataTab tab", onclick: "$onClickTab",
476                 view: "Data", $collapsed: "$entry|hideDataTab"},
477                     $STR("fbcache.tab.Data")
478             )
479         ),
480 
481     // List of tab bodies
482     tabBodies:
483         DIV({"class": "tabBodies"},
484             DIV({"class": "tabDataBody tabBody"})
485         ),
486 
487     hideDataTab: function(result)
488     {
489         return false;
490     },
491 
492     onClickTab: function(event)
493     {
494         this.selectTab(event.target);
495     },
496 
497     selectTabByName: function(tabView, tabName)
498     {
499         var tab = getElementByClass(tabView, tabName + "Tab");
500         if (tab)
501             this.selectTab(tab);
502     },
503 
504     selectTab: function(tab)
505     {
506         var view = tab.getAttribute("view");
507         var viewBody = getAncestorByClass(tab, "tabViewBody");
508 
509         // Deactivate current tab.
510         if (viewBody.selectedTab)
511         {
512             viewBody.selectedTab.removeAttribute("selected");
513             viewBody.selectedBody.removeAttribute("selected");
514         }
515 
516         // Store info about new active tab. Each tab has to have a body,
517         // which is identified by class.
518         var tabBody = getElementByClass(viewBody, "tab" + view + "Body");
519         viewBody.selectedTab = tab;
520         viewBody.selectedBody = tabBody;
521 
522         // Activate new tab.
523         viewBody.selectedTab.setAttribute("selected", "true");
524         viewBody.selectedBody.setAttribute("selected", "true");
525 
526         this.updateTabBody(viewBody, view);
527     },
528 
529     updateTabBody: function(viewBody, tabName)
530     {
531         var tab = viewBody.selectedTab;
532         var infoRow = getAncestorByClass(viewBody, "cacheEntryInfoRow");
533         var entry = infoRow.repObject;
534 
535         // Update Stack tab content
536         var tabDataBody = getElementByClass(viewBody, "tabDataBody");
537         if (tabName == "Data" && !tabDataBody.updated)
538         {
539             tabDataBody.updated = true;
540 
541             var text = entry.lines.join("");
542             if (text && text.length > 10*1024*1024)
543                 text = text.substr(0, 10*1024*1024) + "...";
544 
545             var prevWidth = Firebug.textWrapWidth;
546             try
547             {
548                 Firebug.textWrapWidth = 100000;
549                 insertWrappedText(text, tabDataBody);
550             }
551             catch (e)
552             {
553                 if (FBTrace.DBG_FBCACHE)
554                     FBTrace.sysout("fbcache.CacheEntryTabView.updateTabBody; ", e);
555             }
556             finally
557             {
558                 // textWrapWidth should be always set to the previous value.
559                 Firebug.textWrapWidth = prevWidth;
560             }
561         }
562     },
563 });

567 /**
568  * Helper object for source cache (for one tab in Firefox).
569  */
570 Firebug.Chromebug.FBCacheModule.TabCache = function(name, sourceCache)
571 {
572     this.name = name;
573     this.sourceCache = sourceCache;
574     this.size = 0; // Computed in getCacheSize
575 }
576 
577 /**
578  * Helper object for cache entries.
579  */
580 Firebug.Chromebug.FBCacheModule.CacheEntry = function(url, lines)
581 {
582     this.url = url;
583     this.lines = lines;
584     this.size = (lines && lines.length) ? lines.join("").length : 0;
585 }
