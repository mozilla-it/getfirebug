 18 // ***********************************************************************************
 19 // TraceConsole Module
 20 var tracePanel = null;
 21 
 22 Firebug.Chromebug.TraceConsoleModule = extend(Firebug.Module,
 23 {
 24     dispatchName: "traceConsole",
 25 
 26     initialize: function(prefDomain, prefNames)  // the prefDomain is from the app, eg chromebug
 27     {
 28         if (FBTrace.DBG_CB_CONSOLE)
 29             FBTrace.sysout("cb.TraceConsoleModule.initialize, prefDomain (ignored): " + prefDomain);
 30 
 31         traceService.addObserver(this, "firebug-trace-on-message", false);
 32     },
 33 
 34     shutdown: function()
 35     {
 36         if (FBTrace.DBG_CB_CONSOLE)
 37             FBTrace.sysout("cb.TraceConsoleModule.shutdown, prefDomain: " + prefDomain);
 38 
 39         traceService.removeObserver(this, "firebug-trace-on-message");
 40     },
 41 
 42     initContext: function(context)
 43     {
 44         if (FBTrace.DBG_CB_CONSOLE)
 45             FBTrace.sysout("cb.TraceConsoleModule.initContext; " +
 46                 context.getName() + " - " + context.getTitle());
 47 
 48     },
 49 
 50     destroyContext: function(context)
 51     {
 52         if (FBTrace.DBG_CB_CONSOLE)
 53             FBTrace.sysout("cb.TraceConsoleModule.destroyContext; " +
 54                 context.getName() + " - " + context.getTitle());
 55         return true; // do not remove
 56     },
 57 
 58     // nsIObserver
 59     observe: function(subject, topic, data)
 60     {
 61         if (topic == "firebug-trace-on-message")
 62         {
 63             // Display messages only with "extensions.firebug" type.
 64             var messageInfo = subject.wrappedJSObject;
 65 
 66             // type is controlled by FBTrace.prefDomain in the XULWindow that sent the trace message
 67             if (messageInfo.type != "extensions.firebug")  // TODO selectable
 68                 return;
 69 
 70             if (tracePanel)
 71             {
 72                 tracePanel.dump(new Firebug.TraceModule.TraceMessage(
 73                         messageInfo.type, data, messageInfo.obj, messageInfo.scope));
 74                 return false;
 75             }
 76             return false;
 77         }
 78     },
 79 
 80     clearPanel: function(context)
 81     {
 82         if (tracePanel)
 83             tracePanel.clear();
 84         else
 85             FBTrace("tracePanel.clearPanel no tracePanel");
 86     },
 87 });

 90 // ************************************************************************************************
 91 // Trace Panel
 92 
 93 Firebug.Chromebug.TraceConsolePanel = function() {}
 94 
 95 Firebug.Chromebug.TraceConsolePanel.prototype = extend(Firebug.Panel,
 96 {
 97     name: "trace",
 98     title: "Trace",
 99     searchable: true,
100     editable: false,
101 
102     initializeNode: function(myPanelNode)
103     {
104         this.controller = new Firebug.TraceOptionsController("extensions.firebug", this.onPrefChange);
105         this.logs = Firebug.TraceModule.MessageTemplate.createTable(myPanelNode);
106         Firebug.TraceModule.onLoadConsole(window, myPanelNode);
107         this.unwrapper = bind(this.unWrapMessage, this);
108 
109         if (tracePanel)  // share the panelNode across all contexts
110         {
111             this.panelNode.parentNode.removeChild(this.panelNode);
112             this.panelNode = tracePanel.panelNode;
113         }
114         else
115         {
116             var doc = this.panelNode.ownerDocument;
117             var stylesheet = createStyleSheet(doc, "chrome://firebug/skin/traceConsole.css");
118             addStyleSheet(doc, stylesheet);
119             tracePanel = this;
120         }
121         if (FBTrace.DBG_CB_CONSOLE)
122             FBTrace.sysout("TraceConsolePanel initializeNode in panel "+this.name);
123     },
124 
125     destroyNode: function()
126     {
127         //if (FBTrace.DBG_CB_CONSOLE)
128             FBTrace.sysout("TraceConsolePanel destroyNode");
129     },
130 
131     destroy: function()
132     {
133         FBTrace.sysout("TraceConsolePanel.destory no-op, panelNode: "+this.panelNode);
134         return true;  // we are in charge of destruction here.
135     },
136 
137     show: function(state)
138     {
139         if (FBTrace.DBG_CB_CONSOLE)
140             FBTrace.sysout("cb.TraceConsolePanel.show ", state);
141 
142         this.showToolbarButtons("cbTraceButtons", true);
143         var buttons = Firebug.chrome.$("cbTraceButtons");
144         if (!buttons)
145         {
146             buttons = document.getElementById("cbTraceButtons");
147             if (buttons)
148                 FBTrace.sysout("tracePanel fails with Firebug.chrome.$ but succeeds with document.getElementById");
149             else
150                 FBTrace.sysout("tracePanel fails with document.getElementById "+window.location);
151         }
152 
153         if (this.lastScrollTop && this.logs)
154             this.logs.scrollTop = this.lastScrollTop;
155     },
156 
157     hide: function()
158     {
159         this.showToolbarButtons("cbTraceButtons", false);
160 
161         if (FBTrace.DBG_CB_CONSOLE)
162             FBTrace.sysout("TraceFirebug.panel hide", this);
163 
164         this.lastScrollTop = this.logs.scrollTop;
165     },
166 
167     clear: function()
168     {
169         if (FBTrace.DBG_CB_CONSOLE)
170             FBTrace.sysout("cb.TraceConsolePanel.clear");
171 
172         if (this.panelNode)
173             clearNode(this.panelNode);
174 
175         this.logs = Firebug.TraceModule.MessageTemplate.createTable(this.panelNode);
176     },
177 
178     getOptionsMenuItems: function()
179     {
180          var items = this.controller.getOptionsMenuItems();
181          items.push("-");
182          items.push(this.getAllOffOptionMenuItem());
183          return items;
184     },
185 
186     getAllOffOptionMenuItem: function()
187     {
188         var self = this;
189         return {
190             label: "chromebug.AllOptionsOff",
191             command: function allOff()
192             {
193                 if (FBTrace.DBG_OPTOINS)
194                     FBTrace.sysout("getAllOffOptionMenuItem ", self.controller);
195                 self.controller.clearOptions();
196             }
197         };
198     },
199 
200     onPrefChange: function(optionName, optionValue)
201     {
202         FBTrace.sysout("tracePanel.onPrefChange: "+optionName+"="+optionValue);
203     },
204 
205     search: function(text)
206     {
207         if (!text)
208             return;
209 
210         // Make previously visible nodes invisible again
211         if (this.matchSet)
212         {
213             for (var i in this.matchSet)
214                 removeClass(this.matchSet[i], "matched");
215         }
216 
217         this.matchSet = [];
218 
219         var a_row = getElementByClass(this.logs, "messageRow");
220         if (!a_row)
221             return false; // too early
222 
223         var rows = a_row.parentNode;
224 
225         function findRow(node) {var mr = getAncestorByClass(node, "messageRow");
226         FBTrace.sysout("findRow ",mr);return mr;}
227         var search = new TextSearch(rows, findRow);
228         FBTrace.sysout("tracePanel search #nodes"+rows.childNodes.length, search );
229         var logRow = search.find(text);
230         if (!logRow)
231             return false;
232 
233         for (; logRow; logRow = search.findNext())
234         {
235             setClass(logRow, "matched");
236             FBTrace.sysout("tracePanel search FOUND:\'"+text+"\'", logRow);
237             FBTrace.sysout("tracePanel search #nodes"+rows.childNodes.length, search );
238             this.matchSet.push(logRow);
239         }
240 
241         return true;
242     },
243     // ********************************************************************************************
244     // Message dump
245     dump: function(message)
246     {
247         // Notify listeners
248         Firebug.TraceModule.onDump(message);
249 
250         if (!this.logs)
251             return;
252 
253         var scrolledToBottom = isScrolledToBottom(this.getScrollingNode());
254 
255         var index = message.text.indexOf("ERROR");
256         if (index != -1)
257             message.type = "DBG_ERRORS";
258 
259         index = message.text.indexOf("EXCEPTION");
260         if (index != -1)
261             message.type = "DBG_ERRORS";
262 
263         // The wrapper could a domplate but then the domplate expansion would be in the domplate and confuse me.
264         var wrapper = this.logs.ownerDocument.createElement("tr");
265         wrapper.className = "messageRow " + message.type;
266 
267         var wrapperNumberCell =  this.logs.ownerDocument.createElement("td");
268         wrapperNumberCell.className = "messageNameCol messageCol";
269 
270         var wrapperNumber =  this.logs.ownerDocument.createElement("div");
271         message.wrapperIndex = this.logs.firstChild.childNodes.length + 1;
272         wrapperNumber.innerHTML = message.wrapperIndex +"";
273         wrapperNumber.className = "messageNameLabel messageLabel";
274         wrapperNumberCell.appendChild(wrapperNumber);
275 
276         var wrapperTime = this.logs.ownerDocument.createElement("td");
277         wrapperTime.className = "messageTimeCol messageCol";
278         var timeLabel =  this.logs.ownerDocument.createElement("div");
279         timeLabel.className = "messageTimeLabel messageLabel";
280         timeLabel.innerHTML = message.time;
281         wrapperTime.appendChild(timeLabel);
282 
283         var wrapperData = this.logs.ownerDocument.createElement("td");
284         wrapperData.innerHTML = message.getLabel(-1);
285         wrapperData.className = "messageBodyCol messageCol messageLabel";
286 
287         wrapper.message = message;
288         wrapper.addEventListener('click', this.unwrapper, true);
289 
290         wrapper.appendChild(wrapperNumberCell);
291         wrapper.appendChild(wrapperTime);
292         wrapper.appendChild(wrapperData);
293         this.logs.firstChild.appendChild(wrapper);
294         if (scrolledToBottom)
295             scrollToBottom(this.getScrollingNode());
296     },
297 
298     unWrapMessage: function(event)
299     {
300         var wrapper  = event.currentTarget;
301         wrapper.removeEventListener('click', this.unwrapper, true);
302         var message = wrapper.message;
303         var index = message.wrapperIndex - 1; // somewhere in the domplate expansion a one is added.
304 
305         // expand the domplate into the end of parent
306         var self = this;
307         var outputNodes =
308         {
309             getScrollingNode: function()
310             {
311                 return self.panelNode;
312             },
313             getTargetNode: function()
314             {
315                 return wrapper.parentNode;
316             },
317         }
318 
319 
320         var result = Firebug.TraceModule.MessageTemplate.dump(message, outputNodes, index);
321 
322         // move the new row over the wrapper
323 
324         try
325         {
326             var theUnwrapped = message.row;
327             var old = wrapper.parentNode.replaceChild(theUnwrapped, wrapper);
328         }
329         catch (exc)
330         {
331             FBTrace.sysout("tracePanel.unWrapMessage replaceChild", exc);
332         }
333         Firebug.TraceModule.MessageTemplate.toggleRow(theUnwrapped);
334     },
335 
336 
337     // ********************************************************************************************
338     // Interface to the output nodes, going by the name outputNodes
339 
340     getScrollingNode: function()
341     {
342         return this.panelNode;
343     },
344 
345     getTargetNode: function()
346     {
347         return this.logs.firstChild;
348     },
349     // ********************************************************************************************
350 
351 });
