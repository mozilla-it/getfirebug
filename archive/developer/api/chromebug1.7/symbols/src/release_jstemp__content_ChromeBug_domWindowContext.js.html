 13 Chromebug.DomWindowContext = function(global, browser, chrome, persistedState)
 14 {
 15     var tabContext = new Firebug.TabContext(global, browser, Firebug.chrome, persistedState);
 16     for (var n in tabContext)
 17     {
 18          if (typeof(this[n]) != 'function')
 19              this[n] = tabContext[n];
 20     }
 21 
 22     this.isChromeBug = true;
 23     this.loaded = true;
 24     this.detached = window;  // the window containing firebug for the context is chromebug window
 25     this.originalChrome = null;
 26 
 27 
 28     this.global = global;
 29 
 30     if (global instanceof Ci.nsIDOMWindow)
 31         this.window = global;
 32     else
 33     {
 34         delete this.window;
 35         if (global && global.location)
 36             var name = global.location; // special case for jetpack
 37         else if (global)
 38             var name = Firebug.Rep.getTitle(global);
 39         else
 40             var name ="mystery";
 41 
 42         if (name == "Sandbox")
 43         {
 44             if (browser.currentURI.spec)
 45             {
 46                 var parts = browser.currentURI.spec.split('/');
 47                 name += '/'+ parts.splice(-3).join('/');
 48                 this.setName(browser.currentURI.spec);
 49             }
 50             else
 51                 name += '/'+ browser.currentURI;
 52         }
 53         else
 54         {
 55             this.setName("noWindow://"+name);
 56         }
 57     }
 58 
 59     var persistedState = FBL.getPersistedState(this, "script");
 60     if (!persistedState.enabled)  // for now default all chromebug window to enabled.
 61         persistedState.enabled = "enable";
 62 
 63     if (FBTrace.DBG_CHROMEBUG)
 64         FBTrace.sysout("Chromebug.domWindowContext "+(this.window?"has window ":" no window ")+(this.global?" ":"NULL global ")+" and name "+this.getName(), {global: this.global, window: this.window});
 65 }
 66 
 67 Chromebug.DomWindowContext.prototype = extend(Firebug.TabContext.prototype,
 68 {
 69     setName: function(name)
 70     {
 71         this.name = new String(name);
 72     },
 73 
 74     clearName: function()
 75     {
 76         delete this.name;
 77     },
 78 
 79     getGlobalScope: function()
 80     {
 81         return this.global;  // override Firebug's getGlobalScope; same iff global == domWindow
 82     },
 83     // *************************************************************************************************
 84 
 85     loadHandler: function(event)  // this is bound to a XULWindow's outerDOMWindow (only)
 86     {
 87         // We've just loaded all of the content for an outer nsIDOMWindow for a XUL window. We need to create a context for it.
 88         var outerDOMWindow = event.currentTarget; //Reference to the currently registered target for the event.
 89         var domWindow = event.target.defaultView;
 90 
 91         if (domWindow == outerDOMWindow)
 92         {
 93             var oldName = this.getName();
 94             this.clearName(); // remove the cache
 95 
 96             if (FBTrace.DBG_CHROMEBUG)
 97                 FBTrace.sysout("context.domWindowWatcher found outerDOMWindow "+outerDOMWindow.location+" tried context rename: "+oldName+" -> "+this.getName());
 98             return;
 99         }
100 
101         if (FBTrace.DBG_CHROMEBUG)
102             FBTrace.sysout("context.domWindowWatcher, new "+Chromebug.XULAppModule.getDocumentTypeByDOMWindow(domWindow)+" window "+safeGetWindowLocation(domWindow)+" in outerDOMWindow "+ outerDOMWindow.location+" event.orginalTarget: "+event.originalTarget.documentURI);
103 
104         var context = Firebug.Chromebug.getContextByGlobal(domWindow);
105         if (context)
106         {
107             // then we had one, say from a Frame
108             var oldName = context.getName();
109             this.clearName();  // remove name cache
110             if (FBTrace.DBG_CHROMEBUG)
111                 FBTrace.sysout("ChromeBugPanel.domWindowWatcher found context with id="+context.uid+" and outerDOMWindow.location.href="+outerDOMWindow.location.href+"\n");
112             if (FBTrace.DBG_CHROMEBUG)
113                 FBTrace.sysout("ChromeBugPanel.domWindowWatcher rename context with id="+context.uid+" from "+oldName+" to "+context.getName()+"\n");
114             if (FBTrace.DBG_CHROMEBUG)
115                 FBTrace.sysout("loadHandler found context with sourceFileMap ", context.sourceFileMap);
116         }
117         else
118         {
119             var context = Firebug.Chromebug.getOrCreateContext(domWindow, safeGetWindowLocation(domWindow)); // subwindow
120 
121             if (!context.onUnload)
122                 context.onUnload = bind(context.unloadHandler, context)
123             domWindow.addEventListener("unload", context.onUnload, true);
124 
125             if (FBTrace.DBG_CHROMEBUG) FBTrace.sysout("ChromeBugPanel.domWindowWatcher created context with id="+context.uid+" and outerDOMWindow.location.href="+outerDOMWindow.location.href+"\n");
126         }
127     },
128 
129     unloadHandler: function(event)
130     {
131         try
132         {
133             //FBTrace.sysout("DOMWindowContext.unLoadHandler event.currentTarget.location: "+event.currentTarget.location, event);
134             var outerDOMWindow = event.currentTarget; //Reference to the currently registered target for the event.
135             var domWindow = event.target.defaultView;
136 
137             if (!domWindow)
138             {
139                 FBTrace.sysout("ChromeBug unloadHandler found no DOMWindow for event.target", event.target);
140                 return;
141             }
142 
143             if (! domWindow instanceof Ci.nsIDOMWindow)
144             {
145                 FBTrace.sysout("ChromeBug unloadHandler domWindow not nsIDOMWindow event.currentTarget.location"+event.currentTarget.location, domWindow);
146                 return;
147             }
148 
149             var context = Firebug.Chromebug.getContextByGlobal(domWindow);
150             if (context)
151             {
152                 if (domWindow == outerDOMWindow)
153                 {
154                     FBTrace.sysout("Firebug.Chromebug.unloadHandler found outerDOMWindow with id="+context.uid+" name " +context.getName()+" and domWindow.location.href="+domWindow.location.href+"\n");
155                     /*var xul_win = Chromebug.XULAppModule.getXULWindowByRootDOMWindow(outerDOMWindow);
156 
157                     Chromebug.XULAppModule.addCloser(xul_win, function delayUntilOnCloseWindow()
158                     {
159                         Firebug.TabWatcher.unwatchTopWindow(domWindow);
160                     });
161                     */
162                 }
163                 else
164                 {
165                     FBTrace.sysout("Firebug.Chromebug.unloadHandler found context with id="+context.uid+" name " +context.getName()+" and domWindow.location.href="+domWindow.location.href+"\n");
166                     Firebug.TabWatcher.unwatchTopWindow(domWindow);
167                 }
168 
169             }
170             else
171             {
172                 // Seems like the outer XUL window DOMWindow is cleaned up by onCloseWindow in xulapp
173                 FBTrace.sysout("ChromeBug unloadHandler found no context for domWindow:"+domWindow.location);
174             }
175         }
176         catch(exc)
177         {
178             if (FBTrace)
179                 FBTrace.sysout("domwindowContext.unloadHandler FAILS "+exc, exc);
180         }
181     },
182 
183 });
