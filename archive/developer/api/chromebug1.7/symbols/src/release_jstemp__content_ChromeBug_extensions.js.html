  2 
  3 FBL.ns(function() { with (FBL) {
  4 
  5 // ***********************************************************************************
  6 // Shorcuts and Services
  7 
  8 const Cc = Components.classes;
  9 const Ci = Components.interfaces;
 10 
 11 //************************************************************************************************
 12 //Chromebug Extensions Panel
 13 
 14 Chromebug.ExtensionsPanel = function() {}
 15 
 16 //************************************************************************************************
 17 
 18 
 19 Chromebug.ExtensionsPanel.prototype = extend(Firebug.DOMPanel.prototype,
 20 {
 21     name: "extensions",
 22     title: "Extensions",
 23     searchable: false,
 24     editable: false,
 25 
 26     supportsObject: function(object)
 27     {
 28         if (object instanceof Ci.nsIInstallLocation)
 29             return 10;
 30         else
 31             return 0;
 32     },
 33 
 34     getDefaultSelection: function()
 35     {
 36         this.refresh();
 37         return this.extensions;
 38     },
 39 
 40     refresh: function()
 41     {
 42         if (!this.extmgr)
 43             this.extmgr = Cc["@mozilla.org/extensions/manager;1"].getService(Ci.nsIExtensionManager);
 44 
 45         this.extensions = {};
 46         var enumerateLocations = this.extmgr.installLocations;
 47         while(enumerateLocations.hasMoreElements())
 48         {
 49             var installLocation = enumerateLocations.getNext();
 50             if (installLocation instanceof Ci.nsIInstallLocation)
 51             {
 52                 var enumerateItemLocations = installLocation.itemLocations;
 53                 if (enumerateItemLocations instanceof Ci.nsIDirectoryEnumerator)
 54                 {
 55                     while (itemLocation = enumerateItemLocations.nextFile)
 56                     {
 57                         var id = installLocation.getIDForLocation(itemLocation);
 58                         if (id) // else not an extension
 59                         {
 60                             var updateItem = this.extmgr.getItemForID(id);
 61 
 62                             this.extensions[id] =
 63                             {
 64                                 itemLocation: itemLocation,
 65                                 installLocation: installLocation,
 66                                 external: installLocation.itemIsManagedIndependently(id),
 67                                 updateItem: (updateItem instanceof Ci.nsIUpdateItem ? updateItem : null),
 68                             };
 69                         }
 70 
 71                     }
 72                 }
 73             }
 74         }
 75 
 76     },
 77 
 78     show: function(state)
 79     {
 80         this.selection = null;
 81         if (FBTrace.DBG_EXTENSIONS)
 82             FBTrace.sysout("Chromebug.ExtensionsPanel.show;", state);
 83 
 84         this.showToolbarButtons("cbExtensionButtons", true);
 85         Firebug.DOMPanel.prototype.show.apply(this, arguments);
 86     },
 87 
 88     hide: function()
 89     {
 90         if (FBTrace.DBG_CBWINDOW)
 91             FBTrace.sysout("Chromebug.ExtensionsPanel.hide;");
 92 
 93         this.showToolbarButtons("cbExtensionButtons", false);
 94         Firebug.DOMPanel.prototype.hide.apply(this, arguments);
 95     },
 96 
 97     getContextMenuItems: function()
 98     {
 99          var items = [];
100          items.push(
101                  {label: "Add New Extension Link",
102                      command: bindFixed(this.addNewExtensionLink, this) }
103              );
104          return items;
105     },
106 
107     onPrefChange: function(optionName, optionValue)
108     {
109         if (FBTrace.DBG_CBWINDOW)
110             FBTrace.sysout("Chromebug.ExtensionsPanel.onPrefChange; " + optionName + "=" + optionValue);
111     },
112     
113     addNewExtensionLink: function()
114     {
115         try
116         {
117             FBTrace.sysout("extensions.addNewExtensionLink ");
118             // ask the user for the source folder
119             var filePicker = Cc["@mozilla.org/filepicker;1"].createInstance(Ci.nsIFilePicker);
120             filePicker.init(window, "Select the install.rdf file in the source folder you want to link", Ci.nsIFilePicker.modeOpen);
121             filePicker.appendFilter("install.rdf", "install.rdf"); 
122 
123             var rv = filePicker.show();
124             
125             if (rv == Ci.nsIFilePicker.returnOK || rv == Ci.nsIFilePicker.returnReplace) 
126             {
127                 var file = filePicker.file;
128                     FBTrace.sysout("extensions.addNewExtensionLink "+file.path);
129                     Components.utils.import("resource://firebug/storageService.js");
130                     var installRDF = TextService.readText(file);
131                     FBTrace.sysout("extensions.addNewExtensionLink read "+installRDF);
132                     var installRDF = installRDF.split('\n').join('');
133                   FBTrace.sysout("extensions.addNewExtensionLink remove newlines "+installRDF);
134                   
135                     var reTargetApplication = /<[^>]*?targetApplication>.*?<[^>]*?targetApplication>/;
136                     var cleanup = installRDF.replace(reTargetApplication, "", 'gim');
137                     FBTrace.sysout("extensions.addNewExtensionLink remove targetApplication "+cleanup, cleanup.split(">"));
138                     
139                     var reId = /<[^>]*?id>(.*?)<[^>]*?id>/;
140                     var m = reId.exec(cleanup);
141                     if (m)
142                         var id = m[1];
143                      
144                     if (!id)
145                     {
146                         window.alert("Did not find id in install.rdf, sorry");
147                         FBTrace.sysout("extensions.addNewExtensionLink FAILS to find id in "+cleanup, m);
148                     }
149                     FBTrace.sysout("extensions.addNewExtensionLink found id "+id);
150                     // Link files is just path to install.rdf directory in a file named by |id|
151                     var dir = TextService.getProfileDirectory();
152                     var link = TextService.getFileInDirectory(dir, "extensions/"+id);
153                   var result = TextService.writeText(link, file.parent.path);
154                   window.alert("wrote "+file.parent.path+" into file "+result);
155             }
156             
157         }
158         catch(exc)
159         {
160             FBTrace.sysout("extensions.addNewExtensionLink FAILS "+exc, exc);
161         }
162         
163     },
164 });
165 
166 
167 Firebug.registerPanel(Chromebug.ExtensionsPanel);
168 
169 }});
