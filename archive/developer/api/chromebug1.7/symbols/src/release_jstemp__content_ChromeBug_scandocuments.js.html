  3 FBL.ns(function() { with (FBL) {
  4 
  5 Chromebug.DocumentScanner = extend(Firebug.Module,
  6 {
  7     dispatchName: "documentScanner",
  8     scanningDocuments: false,
  9 
 10     initialize: function()
 11     {
 12         this.onScanningDocumentsMouseOver = bind(this.onScanningDocumentsMouseOver, this);
 13         this.onScanningDocumentsMouseDown = bind(this.onScanningDocumentsMouseDown, this);
 14         this.onScanningDocumentsClick = bind(this.onScanningDocumentsClick, this);
 15     },
 16 
 17    // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
 18 
 19     onScanningDocumentsMouseOver: function(event)
 20     {
 21         if (FBTrace.DBG_INSPECT)
 22            FBTrace.sysout("onScanningDocuments mouse over "+FBL.getElementCSSSelector(event.target)+ " original: "+FBL.getElementCSSSelector(event.originalTarget), event);
 23         if (!this.scanDocuments(event.originalTarget))
 24         {
 25             cancelEvent(event);
 26             if (FBTrace.DBG_INSPECT)
 27                 FBTrace.sysout("onScanningDocuments canceled mouseover Event", event);
 28         }
 29         // else let it go to inspect
 30     },
 31 
 32     onScanningDocumentsMouseDown: function(event)
 33     {
 34         if (FBTrace.DBG_INSPECT)
 35            FBTrace.sysout("onScanningDocuments mouse down", event);
 36         this.stopScanningDocuments(false, true);
 37         cancelEvent(event);  // inspector sees nothing or mousedown and stopInspecting but not inspecting
 38         Firebug.Inspector.stopInspecting(false, false);  // not canceled and don't wait, I'll wait for you.
 39     },
 40 
 41     onScanningDocumentsClick: function(event)
 42     {
 43         if (FBTrace.DBG_INSPECT)
 44            FBTrace.sysout("onScanningDocuments click", event);
 45         var win = event.currentTarget.defaultView;
 46         if (win)
 47         {
 48             win = getRootWindow(win);
 49             this.detachClickInspectListeners(win);
 50         }
 51         else
 52             FBTrace.sysout("onScanningDocuments no defaultView", event.currentTarget);
 53         cancelEvent(event);  // inspector sees nothing or if on same target, 'click' and redundant detach occurs
 54     },
 55 
 56     hasContentDocument: ["tabbrowser", "browser", "iframe"],
 57 
 58     scanDocuments: function(node)
 59     {
 60         if (node && node.nodeType != 1)
 61             node = node.parentNode;
 62 
 63         if (node && node.firebugIgnore)
 64             return;
 65 
 66         if (!node)
 67             return;
 68 
 69         var nodeAsScanned = node;
 70         var tagName = node.tagName;
 71         FBTrace.sysout("scanDocuments tag:"+tagName+" scanning "+(this.scanningDOMWindow?this.scanningDOMWindow.location:"none"));
 72 
 73         if (this.hasContentDocument.indexOf(tagName) != -1) // then the tag is a container of content
 74         {
 75             var doc = node.contentDocument;
 76             node = doc.documentElement;  // switch to the inside of the container
 77         }
 78 
 79         var domWindow = node.ownerDocument.defaultView;
 80         if (domWindow == this.scanningDOMWindow)  // we did not cross windows,
 81         {
 82             if (FBTrace.DBG_INSPECT)
 83                 FBTrace.sysout("scanDocuments tag:"+tagName+" still scanning window "+(this.scanningDOMWindow?this.scanningDOMWindow.location:"none"));
 84             return true;                          // carry on with inspect
 85         }
 86 
 87         var context = Firebug.Chromebug.getContextByGlobal(domWindow);
 88         if (!context)
 89         {
 90              if (FBTrace.DBG_INSPECT)
 91                  FBTrace.sysout("scanDocuments tag:"+tagName+" no context! "+domWindow.location);
 92 
 93             //this.scanningContext.hoverNode = nodeAsScanned;
 94             return false;  // we crossed windows, don't inspect here
 95         }
 96         if (this.scanningContext != context)
 97             Firebug.Chromebug.selectContext(context);
 98 
 99         this.scanningContext = context;
100 
101         if (FBTrace.DBG_INSPECT)
102             FBTrace.sysout("scanDocuments tag:"+tagName+" new context, scanning "+this.scanningContext.getName());
103 
104                  Firebug.Inspector.stopInspecting(false, false);  // on old window
105 
106         this.scanningDOMWindow = domWindow;
107         context.hoverNode = node;  // tell inspect where to start
108 
109         Firebug.Inspector.startInspecting(context);  // on new window
110 
111         if (FBTrace.DBG_INSPECT)
112                 FBTrace.sysout("ScanDocuments startInspecting new context "+context.window.location);
113 
114         return true; // do inspect
115     },
116 
117     //*****************************************************************************
118     toggleScanningDocuments: function(context)
119     {
120         try
121         {
122             if (FBTrace.DBG_INSPECT)
123                 FBTrace.sysout("ChromeBug startScanning with "+context?context.getName():"NULL CONTEXT!"+" in "+window.location);
124             if (this.scanningDocuments)
125                 this.stopScanningDocuments();
126             else
127                 this.startScanningDocuments(context);
128         }
129         catch(exc)
130         {
131             FBTrace.sysout("toggleScanningDocuments fails "+exc, exc);
132         }
133     },
134 
135     startScanningDocuments: function(context)
136     {
137         this.scanningDocuments = true;
138         this.scanningContext = context;
139 
140         if (FBTrace.DBG_INSPECT)
141             FBTrace.sysout("ChromeBug startScanning with "+context?context.getName():"NULL CONTEXT!"+" in "+window.location);
142         Firebug.chrome.setGlobalAttribute("cmd_toggleScanningDocuments", "checked", "true");
143         Firebug.chrome.setGlobalAttribute("cmd_toggleInspecting", "checked", "true");
144 
145         this.attachScanListeners();
146 
147         // Remember the previous panel and bar state so we can revert if the user cancels
148         this.previousContext = context;
149         this.previousPanelName = context.panelName;
150         this.previousSidePanelName = context.sidePanelName;
151         this.previouslyCollapsed = $("fbContentBox").collapsed;
152         this.previouslyFocused = context.detached && Firebug.chrome.isFocused();
153 
154         var htmlPanel = Firebug.chrome.selectPanel("html");
155         this.previousObject = htmlPanel.selection;
156 
157         htmlPanel.panelNode.focus();
158         htmlPanel.startInspecting();
159 
160         if (context.hoverNode)
161             this.scanDocuments(context.hoverNode);
162     },
163 
164     stopScanningDocuments: function(cancelled, waitForClick)
165     {
166         if (FBTrace.DBG_INSPECT)
167             FBTrace.sysout("ChromeBug stopScanning  scanning:"+this.scanningDocuments+"\n");
168         if (!this.scanningDocuments)
169             return;
170 
171         var context = this.scanningContext;
172 
173         this.detachInspectListeners();
174         if (!waitForClick)
175             this.detachClickInspectListeners();
176 
177         Firebug.chrome.setGlobalAttribute("cmd_toggleScanningDocuments", "checked", "false");
178         Firebug.chrome.setGlobalAttribute("cmd_toggleInspecting", "checked", "false");
179 
180         var htmlPanel = Firebug.chrome.unswitchToPanel(context, "html", cancelled);
181 
182         if (this.previouslyFocused)
183             Firebug.chrome.focus();
184 
185         if (cancelled)
186         {
187             if (this.previouslyCollapsed)
188                 Firebug.showBar(false);
189 
190             if (this.previousPanelName == "html")
191                 Firebug.chrome.select(this.previousObject);
192             else
193                 Firebug.chrome.selectPanel(this.previousPanelName, this.previousSidePanelName);
194         }
195 
196         //htmlPanel.stopInspecting(htmlPanel.selection, cancelled);
197 
198         delete this.previousObject;
199         delete this.previousPanelName;
200         delete this.previousSidePanelName;
201         delete this.scanningContext;
202         this.scanningDocuments = false;
203         delete this.scanningDOMWindow;
204         delete context.hoverNode;
205 
206     },
207 
208     attachScanListeners: function()
209     {
210         this.keyListeners =
211         [
212             FirebugChrome.keyCodeListen("RETURN", null, bindFixed(this.stopScanningDocuments, this)),
213             FirebugChrome.keyCodeListen("ESCAPE", null, bindFixed(this.stopScanningDocuments, this, true)),
214             FirebugChrome.keyCodeListen("UP", isControl, bindFixed(this.inspectNodeBy, this, "up"), true),
215             FirebugChrome.keyCodeListen("DOWN", isControl, bindFixed(this.inspectNodeBy, this, "down"), true),
216         ];
217 
218         Chromebug.XULAppModule.iterateOuterDOMWindows( bind(function(subWin)
219         {
220             var context = Firebug.Chromebug.getContextByGlobal(subWin);
221             if (!context)  // don't attach to windows we are not watching
222                  return;
223             // If you change the bubbling/capture option on these, do so on the removeEventListener as well.
224             subWin.document.addEventListener("mouseover", this.onScanningDocumentsMouseOver, true); // trigger on capture
225             subWin.document.addEventListener("mousedown", this.onScanningDocumentsMouseDown, true);
226             subWin.document.addEventListener("click", this.onScanningDocumentsClick, true);
227             if (FBTrace.DBG_INSPECT)
228                 FBTrace.sysout("scanDocuments attachListeners to "+subWin.location+"\n");
229         }, this));
230     },
231 
232     detachInspectListeners: function()
233     {
234         if (this.keyListeners)  // XXXjjb for some reason this is null some times...
235         {
236             for (var i = 0; i < this.keyListeners.length; ++i)
237                 FirebugChrome.keyIgnore(this.keyListeners[i]);
238             delete this.keyListeners;
239         }
240 
241         Chromebug.XULAppModule.iterateOuterDOMWindows( bind(function(subWin)
242         {
243             try
244             {
245                 subWin.document.removeEventListener("mouseover", this.onScanningDocumentsMouseOver, true);
246                 subWin.document.removeEventListener("mousedown", this.onScanningDocumentsMouseDown, true);
247             }
248             catch (exc)
249             {
250                 if (FBTrace.DBG_ERRORS)
251                     FBTrace.sysout("scandocuments detachInspectListeners fails for subWin.document "+subWin.document.location+" because: "+exc, exc);
252             }
253         }, this));
254     },
255 
256     detachClickInspectListeners: function()
257     {
258         // We have to remove the click listener in a second phase because if we remove it
259         // after the mousedown, we won't be able to cancel clicked links
260         Chromebug.XULAppModule.iterateOuterDOMWindows( bind(function(subWin)
261         {
262             subWin.document.removeEventListener("click", this.onScanningDocumentsClick, true);
263         }, this));
264     },
265 });
266 
267 Firebug.registerModule(Chromebug.DocumentScanner);
268
