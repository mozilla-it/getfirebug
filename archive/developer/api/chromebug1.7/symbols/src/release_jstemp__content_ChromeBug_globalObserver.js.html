 16 Chromebug.globalObserver =
 17 {
 18     observe: function(subject, topic, data)
 19     {
 20 
 21         var FirebugTrace = traceConsoleService.getTracer("extensions.firebug");
 22 
 23         // Log info into the Firebug tracing console.
 24         var shout = (Chromebug.globalObserver.shoutOptionValue?"GGGGGGGGGGGGGGGGGGGGGGGGGGGGGGGGGGGGG globalObserver.":"");
 25         FirebugTrace.sysout(shout+"observe: "+topic, {subject:subject, data: data});
 26 
 27         if (topic == 'domwindowopened')
 28         {
 29             try
 30             {
 31                 if (subject instanceof Ci.nsIDOMWindow)
 32                 {
 33                     if (FBTrace.DBG_CHROMEBUG || FBTrace.DBG_WINDOWS)
 34                         FBTrace.sysout("Chromebug.globalObserver found domwindowopened "+subject.location+"\n");
 35                 }
 36             }
 37             catch(exc)
 38             {
 39                 FBTrace.sysout("Chromebug.globalObserver notify console opener FAILED ", exc);
 40             }
 41         }
 42         else if (topic == 'domwindowclosed') // Apparently this event comes before the unload event on the DOMWindow
 43         {
 44             if (subject instanceof Ci.nsIDOMWindow)
 45             {
 46                 if (FBTrace.DBG_WINDOWS)
 47                     FBTrace.sysout("Chromebug.globalObserver found domwindowclosed "+subject.location+getStackDump());
 48                 if (subject.location.toString() == "chrome://chromebug/content/chromebug.xul")
 49                     throw new Error("Chromebug.globalObserver should not find chromebug.xul");
 50             }
 51         }
 52         else if (topic == 'dom-window-destroyed')  // subject appears to be the nsIDOMWindow with a location that is invalid and closed == true; data null
 53         {
 54             if (FBTrace.DBG_WINDOWS)
 55                 FBTrace.sysout("Chromebug.globalObserver found dom-window-destroyed subject:", subject);
 56         }
 57     },
 58 
 59     shoutOptionName: "shoutAboutObserverEvents",
 60     shoutOptionLabel: "Shout About Observer Events",
 61     shoutOptionValue: true,
 62 };

229     forEachTopicElement: function(fn)
230     {
231         var topicDivs = getElementsByClass(this.panelNode, "traceOption")
232         for (var i = 0; i < topicDivs.length; i++)
233         {
234             var topicElt = topicDivs[i];
235             fn(topicElt);
236         }

239     updateTopics: function()
240     {
241         return;
242         var self = this;
243         this.forEachTopicElement(function setChecked(topicElt)
244         {
245             var topic = topicElt.innerHTML;
246             if (self.isObserved(topicElt))
247                 topicElt.setAttribute("checked", "true");
248             else
249                 topicElt.removeAttribute("checked");
250         });
251     },
252 
253     allTopics: function(add)
254     {
255         var self = this;
256         this.forEachTopicElement(function addIfTrue(topicElt)
257         {
258             try
259             {
260                 if (add)
261                     self.addTopic(topicElt);
262                 else
263                     self.removeTopic(topicElt);
264 
265                 if (FBTrace.DBG_CHROMEBUG)
266                     FBTrace.sysout("AllTopics add: "+add+" isObserved "+self.isObserved(topicElt)+" "+topicElt.innerHTML);
267             }
268             catch(exc)
269             {
270                 FBTrace.sysout("globalObserver allTopics fails for "+topicElt.innerHTML+" "+exc, exc);
271             }
272         });
273     },
274 
275     isObserved: function(topicElt)
276     {
277         var topic = topicElt.innerHTML;
278         var observers = observerService.enumerateObservers(topic);
279         if (!observers)
280         {
281             if (FBTrace.DBG_CHROMEBUG)
282                 FBTrace.sysout("isObserved no observers of topic "+topic);
283 
284             return false;
285         }
286         while (observers.hasMoreElements())
287         {
288             var x = observers.getNext();
289             if (x.wrappedJSObject == Chromebug.globalObserver)
290             {
291                 if (FBTrace.DBG_CHROMEBUG)
292                     FBTrace.sysout("isObserved found Chromebug.globaleObserver of topic "+topic, x);
293                 return true;
294             }
295         }
296         if (FBTrace.DBG_CHROMEBUG)
297             FBTrace.sysout("isObserved no observers of topic "+topic+" match Chromebug.globalObserver");
298         return false
299     },
300 
301     toggleTopic: function(event)
302     {
303         var topicElt = event.target;
304         if (FBTrace.DBG_CHROMEBUG)
305                 FBTrace.sysout("globalObserver panel onclick "+topicElt.innerHTML);
306         if (!this.isObserved(topicElt))
307             this.addTopic(topicElt)
308         else
309             this.removeTopic(topicElt);
310         if (FBTrace.DBG_CHROMEBUG)
311             FBTrace.sysout("globalObserver panel onclick isObserved:"+this.isObserved(topicElt));
312     },
313 
314     addTopic: function(node)
315     {
316         var topic = node.innerHTML;
317         observerService.addObserver(Chromebug.globalObserver, topic, false);
318         node.setAttribute("checked", "true");
319         if (FBTrace.DBG_CHROMEBUG)
320             FBTrace.sysout("addObserver "+topic);
321     },
322 
323     removeTopic: function(node)
324     {
325         var topic = node.innerHTML;
326         observerService.removeObserver(Chromebug.globalObserver, topic);
327         node.removeAttribute("checked");
328         if (FBTrace.DBG_CHROMEBUG)
329             FBTrace.sysout("removeObserver "+topic);
330     },
