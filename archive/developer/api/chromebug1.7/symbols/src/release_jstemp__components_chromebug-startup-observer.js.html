226

228

230                {
231                    var cb = jsdState._chromebug;
232                    if (!cb || !cb.breakpointedScripts)

253        jsd.breakpointHook =
254        {
255            onExecute: function(frame, type, val)
256            {
257                frame.script.clearBreakpoint(0);
258                var script = frame.script;
259                //if (trace) jsdState.dump("breakpointHook script "+script.tag+"\n");
260                var cb = jsdState._chromebug;
261                var broken = cb.breakpointedScripts[script.tag];
262                if (broken)
263                {
264                    delete cb.breakpointedScripts[script.tag];
265                }
266 
267                if ( reXUL.test(script.fileName) ) // we set a bp in every script with these names
268                {
269                   // here we are treating the first bp that hits as the outer script
270                    var scripts = cb.xulScriptsByURL[script.fileName];
271                    for (var i = 0; i < scripts.length; i++)
272                    {
273                        var s = scripts[i];
274                        s.clearBreakpoint(0);
275                        delete cb.breakpointedScripts[s.tag];
276                        if (s.tag === script.tag)
277                            scripts.splice(i,1); // remove the 'outerscript' so the inner ones remain
278                    }
279                    var saveInner = [];
280                    for (var i = 0; i < cb.innerScripts.length; i++)
281                        saveInner[i] = cb.innerScripts[i];
282 
283                    cb.innerScripts = scripts;
284                    analyzeScope(cb, frame, jsdState);
285                    cb.innerScripts = saveInner;
286                }
287                if (!frame.callingFrame) // then top-level
288                {
289                    if (frame.scope)
290                        analyzeScope(cb, frame, jsdState);
291                    else // no scope, looks like this is where command line handlers end up
292                        if (trace) Components.utils.reportError("no callingFrame and no executionContext for "+frame.script.fileName+"\n");
293                }
294 
295                return Ci.jsdIExecutionHook.RETURN_CONTINUE;
296            }

370         }
371         else if (topic === "command-line-startup")
372         {
373             return;
374             Components.utils.reportError("Chromebug "+topic+" command-line-startup");
375              if (subject instanceof Ci.nsICommandLine)

389                  //subject.preventDefault = true;
390                  Components.utils.reportError("Chromebug subject.preventDefault"+ subject.preventDefault);
391              }
392         }

482 function getLocationSafe(global)
483 {
484     try
485     {
486         if (global && global.location)  // then we have a window, it will be an nsIDOMWindow, right?
487             return global.location.toString();
488         else if (global && global.tag)
489             return "global_tag_"+global.tag;
490     }
491     catch (exc)
492     {
493         // FF3 gives (NS_ERROR_INVALID_POINTER) [nsIDOMLocation.toString]
494     }
495     return null;
496 }

500 function getGlobalName(global)
501 {
502     var label = safeToString(global);          // eg [object XPCWrappedNative [object foo
503     var m = reObject.exec(label);
504     var n = null;
505     if (m)
506         n = reObject.exec(m[1]);
507     if (n)
508         return n[1];
509 
510     return m ? m[1] : label;
511 }

513 function safeToString(ob)
514 {
515     try
516     {
517         if (ob && (typeof (ob['toString']) == "function") )
518             return ob.toString();
519     }
520     catch (exc)
521     {
522     }
523     return "[object has no toString() function]";

525 var fbs = {
526         getOutermostScope: function(frame)
527         {
528             var scope = frame.scope;
529             if (scope)
530             {
531                 while(scope.jsParent)
532                     scope = scope.jsParent;
533 
534                 // These are just determined by trial and error.
535                 if (scope.jsClassName == "Window" || scope.jsClassName == "ChromeWindow" || scope.jsClassName == "ModalContentWindow")
536                 {
537                     lastWindowScope = wrapIfNative(scope.getWrappedValue());
538                     return  lastWindowScope;
539                 }
540 
541               if (scope.jsClassName == "Sandbox")
542                 {
543                     var proto = scope.jsPrototype;
544                     if (proto.jsClassName == "XPCNativeWrapper")  // this is the path if we have web page in a sandbox
545                     {
546                         proto = proto.jsParent;
547                         if (proto.jsClassName == "Window")
548                             return wrapIfNative(proto.getWrappedValue());
549                     }
550                     else
551                     {
552                         return wrapIfNative(scope.getWrappedValue());
553                     }
554                 }
555 
556                 return wrapIfNative(scope.getWrappedValue());  // not a window or a sandbox
557             }
558             else
559             {
560                 return null;
561             }
562         },
563 
564 };

580 function analyzeScope(cb, frame, jsdState)
581 {
582     var scope = frame.scope;
583     while(scope.jsParent) // walk to the oldest scope
584         scope = scope.jsParent;
585 
586     var frameGlobal = fbs.getOutermostScope(frame);
587     var frameGlobalTag = cb.globals.indexOf(frameGlobal);
588     if (frameGlobalTag < 0)
589     {
590         cb.globals.push(frameGlobal);
591         frameGlobalTag = cb.globals.indexOf(frameGlobal);
592     }
593 
594     var scopeName = getLocationSafe(frameGlobal);
595     if (!scopeName)
596     {
597         var globalName = getGlobalName(frameGlobal);
598         if (globalName)
599             scopeName = "noWindow://"+globalName;
600     }
601     if (false) //jsdState.avoidSelf(scopeName))
602     {
603         //gStartupObserverSingleton.trackFiles.drop(frame.script.fileName, scopeName);
604 
605         if (trace) Components.utils.reportError("dropping "+frameGlobalTag+" with location "+scopeName+"\n");
606     }
607     else
608     {
609         if (trace) Components.utils.reportError("assigning "+frameGlobalTag+" to "+frame.script.fileName+"\n");
610 
611         cb.globalTagByScriptTag[frame.script.tag] = frameGlobalTag;
612         cb.globalTagByScriptFileName[frame.script.fileName] = frameGlobalTag;
613         // add the unassigned innerscripts
614         for (var i = 0; i < cb.innerScripts.length; i++)
615         {
616             var script = cb.innerScripts[i];
617             if (script.fileName != frame.script.fileName)  // so what frameGlobalTag then?
618                 if (trace) Components.utils.reportError("innerscript "+script.fileName+" mismatch "+frame.script.fileName+"\n");
619             cb.globalTagByScriptTag[script.tag] = frameGlobalTag;
620         }
621         //gStartupObserverSingleton.trackFiles.def(frame.script.fileName, scopeName+" with "+cb.innerScripts.length, frame.script.tag, cb.globalTagByScriptTag[frame.script.tag]);
622     }
623     cb.innerScripts = [];
624 }
