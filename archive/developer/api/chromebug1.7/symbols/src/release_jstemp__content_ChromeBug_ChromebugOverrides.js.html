 28     // Override Firebug.HTMLPanel.prototype
 29     getTreeWalker: function(node)
 30     {
 31         if (!this.treeWalker || this.treeWalker.currentNode !== node)
 32         {
 33             // Apparently you cannot reset the currentNode on this treeWalker
 34             //http://mxr.mozilla.org/comm-central/source/mozilla/extensions/inspector/resources/content/viewers/dom/dom.js#819
 35             this.treeWalker = CCIN("@mozilla.org/inspector/deep-tree-walker;1", "inIDeepTreeWalker");
 36             this.treeWalker.showAnonymousContent = true;
 37             this.treeWalker.showSubDocuments = true; // does not matter, we don't visit children, only siblings
 38 
 39             try
 40             {
 41                 this.treeWalker.init(node, Components.interfaces.nsIDOMNodeFilter.SHOW_ALL);
 42             }
 43             catch(exc)
 44             {
 45                 FBTrace.sysout("Falling back to DOM Tree walker, inIDeepTreeWalker in chromebug requires FF 3.6 or higher "+exc, exc);
 46                 this.treeWalker = node.ownerDocument.createTreeWalker(
 47                         node, NodeFilter.SHOW_ALL, null, false);
 48             }
 49         }
 50         return this.treeWalker;
 51     },

 56         if (FBTrace.DBG_HTML)
 57             FBTrace.sysout("Chromebug getFirstChild("+FBL.getElementCSSSelector(node)+") = "+FBL.getElementCSSSelector(child));
 58         return child;

 61     getNextSibling: function(node)
 62     {
 63         // the Mozilla XBL tree walker cannot be initialized then called for nextSibling.
 64         // So we have to go up one, hunt around in the children to find ourselves, then get the next sibling.
 65 
 66         if (node === null)
 67             return null;
 68 
 69         var nextSibling = null;
 70         if (this.treeWalker && this.treeWalker.currentNode === node)
 71             nextSibling = this.treeWalker.nextSibling();
 72 
 73         if (nextSibling === null)
 74         {
 75             var parent = this.getParentNode(node);
 76             if (parent)  // then we are not root,
 77             {
 78                 // look for siblings by first finding ourselves in parent list
 79                 var walker = this.getTreeWalker(parent);
 80 
 81                 var child = walker.firstChild();
 82                 while (child !== null && child !== node)
 83                     child = walker.nextSibling();
 84 
 85                 if (child === null)   // then we did not find our selves in our parents children
 86                 {
 87                     var tag = node.ownerDocument.location + "";
 88                     if (this.embeddedBrowserParents && this.embeddedBrowserParents[tag] === parent) // then we are an embedded browser
 89                     {
 90                          var walker = this.getTreeWalker(parent);
 91                          nextSibling = walker.firstChild();  // it's a lie we are repeating here.
 92                     }
 93                     else
 94                     {
 95                         FBTrace.sysout("Chromebug getNextSibling FAILS "+FBL.getElementCSSSelector(node)+" not a child of parent "+FBL.getElementCSSSelector(parent));
 96                     }
 97                 }
 98                 else
 99                 {
100                     nextSibling = walker.nextSibling();
101                 }
102 
103             }
104             // else we are root and our nextSibling is null
105         }
106         if (FBTrace.DBG_HTML)
107             FBTrace.sysout("Chromebug getNextSibling("+FBL.getElementCSSSelector(node)+") = "+FBL.getElementCSSSelector(nextSibling));
108         return nextSibling;
109     },

179     /*
180      * Map the frame to a context
181      * @param frame jsdIStackFrame
182      * @param global option previously computed from FBS
183      * @return a context
184      */
185     getContextByFrame: function(frame, global)
186     {
187         var context = null;
188         if (frame && frame.isValid)
189         {
190             if (Firebug.Chromebug.isChromebugURL(frame.script.fileName))
191                 return null;
192 
193             if (!global)
194                 global = Firebug.Chromebug.getGlobalByFrame(frame);
195 
196             if (global)
197             {
198                 var name = safeGetWindowLocation(global);
199                 if (Firebug.Chromebug.isChromebugURL(name)) // if the frame was compiled in chromebug, ignore it
200                     return null;
201 
202                 context = Firebug.Chromebug.getOrCreateContext(global, frame.script.fileName);
203             }
204         }
205         else
206         {
207             throw new Error("no frame isValid");
208         }
209         if (context)
210         {
211             if (FBTrace.DBG_TOPLEVEL)
212                 FBTrace.sysout("getContextByFrame "+normalizeURL(frame.script.fileName)+": frame.scope gave existing context "+context.getName());
213         }
214         else
215         {
216              FBTrace.sysout("getContextByFrame "+normalizeURL(frame.script.fileName)+": no context!");
217         }
218         return context;
219     },
